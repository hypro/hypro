98  break;
99  }
100  }

102  if ( allMatched ) {

115  }

118  // Find corresponding rows, then compare relevant entries.

120  Eigen::Index rowPos = 0;
121  //std::cout << "Search for: " << rhsVars[rowI] << std::endl;

129  //std::cout << "reset matrix entries do not match." << std::endl;
130  return false;
131  }
132  }
133 
134  // compare constant part (b)
135  if ( this->getReset().getVector()( rowPos ) != rhs.getReset().getVector()( rowI ) ) {
136  //std::cout << "constant parts do not match." << std::endl;
137  return false;
138  }
139  }

141  return true;
