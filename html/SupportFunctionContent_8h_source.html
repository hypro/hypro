44 struct trafoContent {
45  std::shared_ptr<SupportFunctionContent<Number, Setting>> origin;
46  std::shared_ptr<const lintrafoParameters<Number>> parameters;
47  unsigned currentExponent;
48  std::size_t successiveTransformations;
49  // 2^power defines the max. number of successive lin.trans before reducing the SF
50 
51  trafoContent( const std::shared_ptr<SupportFunctionContent<Number, Setting>>& _origin, const matrix_t<Number>& A, const vector_t<Number>& b )
52  : origin( _origin )
53  , currentExponent( 1 ) {
54  // Determine, if we need to create new parameters or if this matrix and vector pair has already been used (recursive).
55  parameters = std::make_shared<const lintrafoParameters<Number>>( A, b );
56  // in case this transformation has already been performed, parameters will be updated.
57  auto trafoInDepth = _origin->hasTrafo( parameters, A, b, 0 ); // returns a pair of bool and depth in which the node was found -> this saves some comparisons later, as we know if the second value is 1, we can reduce.
58  assert( trafoInDepth.second != 1 || ( ( origin->type() == SF_TYPE::LINTRAFO ) && ( *origin->linearTrafoParameters()->parameters == *parameters ) && trafoInDepth.first ) );
59  if ( Setting::USE_LIN_TRANS_REDUCTION ) {
60  DEBUG( "hypro.representations.supportFunction", "Reduction of sf with depth: " << origin->depth() + 1 );
61  // best points for reduction are powers of 2 thus we only use these points for possible reduction points
62  bool reduced;
63  do {
64  reduced = false;
65  if ( trafoInDepth.first && trafoInDepth.second == 1 && origin->linearTrafoParameters()->currentExponent == currentExponent ) {
66  successiveTransformations = origin->linearTrafoParameters()->successiveTransformations + 1;
67  } else {
68  successiveTransformations = 0;
69  }
70  TRACE( "hypro.representations.supportFunction", "successiveTransformations with exponent " << currentExponent << ": " << successiveTransformations );
71  if ( successiveTransformations == unsigned( carl::pow( 2, parameters->power ) - 1 ) ) {
72  TRACE( "hypro.representations.supportFunction", "Update origin pointer as a result of lintrafo reduction." );
73  reduced = true;
74  currentExponent = currentExponent * ( carl::pow( 2, parameters->power ) );
75  for ( std::size_t i = 0; i < unsigned( carl::pow( 2, parameters->power ) - 1 ); i++ ) {
76  origin = origin->linearTrafoParameters()->origin;
77  }
78  // Note: The following assertion does not hold in combination with the current reduction techniques.
79  //assert(origin->type() != SF_TYPE::LINTRAFO || (origin->linearTrafoParameters()->parameters == this->parameters && origin->linearTrafoParameters()->currentExponent >= currentExponent) );
80 
81  // update information whether another trafo follows/is the origin.
82  // TODO: If the next operation is not a linTrafo we can directly quit.
83  trafoInDepth = origin->hasTrafo( parameters, A, b, 0 );
84  TRACE( "hypro.representations.supportFunction", "Next linear trafo at depth: " << trafoInDepth.second );
85  TRACE( "hypro.representations.supportFunction", "Next node type: " << origin->type() );
86  }
87  } while ( reduced == true );
88  DEBUG( "hypro.representations.supportFunction", "Reduced sf to depth: " << origin->depth() + 1 );
89  assert( origin->checkTreeValidity() );
90  }
91  }

101  std::pair<matrix_t<Number>, vector_t<Number>> reduceLinTrans( const matrix_t<Number>& _a, const vector_t<Number>& _b, std::size_t _power ) {
102  std::size_t powerOfTwo = carl::pow( 2, _power );
103  // first compute the new b
104  vector_t<Number> bTrans = _b;
105  matrix_t<Number> aTrans = _a;
106  for ( std::size_t i = 1; i < powerOfTwo; i++ ) {
107  bTrans = _a * bTrans + _b;
108  }
109  // now compute a^i efficiently
110  for ( std::size_t i = 0; i < _power; i++ ) {
111  aTrans = aTrans * aTrans;
112  }
113  return std::make_pair( aTrans, bTrans );
114  }

378  bool checkTreeValidity() const {
379 #ifndef NDEBUG
380  using Node = std::shared_ptr<SupportFunctionContent<Number, Setting>>;
381  using Res = bool;
382  std::vector<Node> callStack;
383  std::vector<std::pair<int, std::vector<Res>>> resultStack; // The first value is an iterator to the calling frame
384 
385  callStack.push_back( getThis() );
386  resultStack.push_back( std::make_pair( -1, std::vector<Res>() ) );
387 
388  while ( !callStack.empty() ) {
389  Node cur = callStack.back();
390 
391  if ( cur->originCount() == 0 ) {
392  // Do computation and write results in case recursion ends.
393 
394  std::pair<int, std::vector<Res>> currentResult = resultStack.back();
395 
396  // update result
397  // special case: When the node is a leaf, we directly return the result.
398  if ( currentResult.first == -1 ) {
399  // first item is also top.
400  return true;
401  }
402 
403  switch ( cur->type() ) {
404  case SF_TYPE::INFTY_BALL:
405  case SF_TYPE::TWO_BALL:
406  case SF_TYPE::POLY:
407  case SF_TYPE::BOX:
408  case SF_TYPE::ELLIPSOID: {
409  resultStack.at( currentResult.first ).second.push_back( true );
410  break;
411  }
412  default:
413  assert( false );
414  }
415 
416  // leave recursive call.
417  callStack.pop_back();
418  resultStack.pop_back();
419 
420  } else {
421  // Detect, if this call is finished or new.
422  if ( resultStack.back().second.size() == cur->originCount() ) {
423  // the call is finished, perform accumulating operations and forward result.
424 
425  // accumulate results
426  Res accumulatedResult = 0;
427  switch ( cur->type() ) {
428  case SF_TYPE::INTERSECT:
429  case SF_TYPE::SUM:
430  case SF_TYPE::LINTRAFO:
431  case SF_TYPE::UNITE:
432  case SF_TYPE::SCALE:
433  case SF_TYPE::PROJECTION:
434  accumulatedResult = true;
435  for ( auto val : resultStack.back().second ) {
436  if ( !val ) {
437  accumulatedResult = false;
438  break;
439  }
440  }
441  break;
442  case SF_TYPE::NONE: {
443  std::cout << __func__ << ": SF Type not properly initialized!" << std::endl;
444  assert( false );
445  break;
446  }
447  default:
448  std::cout << __func__ << ": SF Type unexpected!" << std::endl;
449  assert( false );
450  }
451 
452  if ( resultStack.back().first == -1 ) {
453  // we reached the top
454  return accumulatedResult;
455  }
456 
457  resultStack.at( resultStack.back().first ).second.push_back( accumulatedResult );
458 
459  // delete result frame and close recursive call
460  callStack.pop_back();
461  resultStack.pop_back();
462 
463  } else {
464  // this is the branch for calling recursively
465 
466  // here we create the new stack levels. As the parameters are all the same, we do not care for their order (could be extended).
467  int callingFrame = int( callStack.size() ) - 1;
468 
469  switch ( cur->type() ) {
470  case SF_TYPE::SUM: {
471  callStack.push_back( cur->summands()->rhs );
472  callStack.push_back( cur->summands()->lhs );
473  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
474  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
475  break;
476  }
477  case SF_TYPE::INTERSECT: {
478  callStack.push_back( cur->intersectionParameters()->rhs );
479  callStack.push_back( cur->intersectionParameters()->lhs );
480  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
481  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
482  break;
483  }
484  case SF_TYPE::LINTRAFO: {
485  callStack.push_back( cur->linearTrafoParameters()->origin );
486  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
487  break;
488  }
489  case SF_TYPE::SCALE: {
490  callStack.push_back( cur->scaleParameters()->origin );
491  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
492  break;
493  }
494  case SF_TYPE::UNITE: {
495  for ( unsigned i = 0; i < cur->unionParameters()->items.size(); ++i ) {
496  callStack.push_back( cur->unionParameters()->items.at( i ) );
497  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
498  }
499  break;
500  }
501  case SF_TYPE::POLY:
502  case SF_TYPE::INFTY_BALL:
503  case SF_TYPE::TWO_BALL:
504  case SF_TYPE::ELLIPSOID:
505  case SF_TYPE::BOX:
506  case SF_TYPE::ZONOTOPE: {
507  assert( false );
508  break;
509  }
510  case SF_TYPE::PROJECTION: {
511  callStack.push_back( cur->projectionParameters()->origin );
512  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
513  break;
514  }
515  case SF_TYPE::NONE:
516  default:
517  assert( false );
518  }
519  }
520  }
521  }
522 
523  assert( false );
524  std::cout << "THIS SHOULD NOT HAPPEN." << std::endl;
525  return false;
526 #endif
527  // in case this function is not put into an assertion, always return true in productive code.
528  return true;
529  }

697  std::pair<bool, std::size_t> hasTrafo( std::shared_ptr<const lintrafoParameters<Number>>& resNode, const matrix_t<Number>& A, const vector_t<Number>& b, std::size_t depth ) const {
698  switch ( mType ) {
699  case SF_TYPE::SUM: {
700  std::pair<bool, std::size_t> res = summands()->lhs->hasTrafo( resNode, A, b, depth + 1 );
701  if ( !res.first ) {
702  res = summands()->rhs->hasTrafo( resNode, A, b, depth + 1 );
703  }
704  return res;
705  }
706  case SF_TYPE::INTERSECT: {
707  std::pair<bool, std::size_t> res = intersectionParameters()->lhs->hasTrafo( resNode, A, b, depth + 1 );
708  if ( !res.first ) {
709  res = intersectionParameters()->rhs->hasTrafo( resNode, A, b, depth + 1 );
710  }
711  return res;
712  }
713  case SF_TYPE::LINTRAFO: {
714  if ( linearTrafoParameters()->parameters->matrix() == A && linearTrafoParameters()->parameters->vector() == b ) {
715  resNode = linearTrafoParameters()->parameters;
716  return std::make_pair( true, depth + 1 );
717  }
718  return linearTrafoParameters()->origin->hasTrafo( resNode, A, b, depth + 1 );
719  }
720  case SF_TYPE::SCALE: {
721  return scaleParameters()->origin->hasTrafo( resNode, A, b, depth + 1 );
722  }
723  case SF_TYPE::UNITE: {
724  for ( const auto& item : unionParameters()->items ) {
725  auto tmp = item->hasTrafo( resNode, A, b, depth + 1 );
726  if ( tmp.first ) {
727  return tmp;
728  }
729  }
730  return std::make_pair( false, 0 );
731  }
732  case SF_TYPE::POLY:
733  case SF_TYPE::INFTY_BALL:
734  case SF_TYPE::TWO_BALL:
735  case SF_TYPE::ELLIPSOID:
736  case SF_TYPE::BOX:
737  case SF_TYPE::ZONOTOPE: {
738  return std::make_pair( false, 0 );
739  }
740  case SF_TYPE::PROJECTION: {
741  return projectionParameters()->origin->hasTrafo( resNode, A, b, depth + 1 );
742  break;
743  }
744  case SF_TYPE::NONE:
745  default:
746  assert( false );
747  return std::make_pair( false, 0 );
748  }
749  }

751  std::vector<SF_TYPE> collectLevelEntries( unsigned level ) const {
752  //std::cout << __func__ << ": level: " << level << std::endl;
753  std::vector<SF_TYPE> items;
754  if ( level == 0 ) {
755  items.push_back( this->mType );
756  //std::cout << "items push " << this->mType << std::endl;
757  //std::cout << "items size: " << items.size() << std::endl;
758  return items;
759  }
760 
761  // Note that at this point we can omit terminal cases, as the level is larger than the depht of this subtree.
762  switch ( mType ) {
763  case SF_TYPE::LINTRAFO: {
764  //std::cout << "Current: lintrafor" << std::endl;
765  std::vector<SF_TYPE> tmpItems = linearTrafoParameters()->origin->collectLevelEntries( level - 1 );
766  items.insert( items.end(), tmpItems.begin(), tmpItems.end() );
767  return items;
768  }
769  case SF_TYPE::SCALE: {
770  //std::cout << "Current: scale" << std::endl;
771  std::vector<SF_TYPE> tmpItems = scaleParameters()->origin->collectLevelEntries( level - 1 );
772  items.insert( items.end(), tmpItems.begin(), tmpItems.end() );
773  return items;
774  }
775  case SF_TYPE::PROJECTION: {
776  //std::cout << "Current: projection" << std::endl;
777  std::vector<SF_TYPE> tmpItems = projectionParameters()->origin->collectLevelEntries( level - 1 );
778  items.insert( items.end(), tmpItems.begin(), tmpItems.end() );
779  return items;
780  }
781  case SF_TYPE::SUM: {
782  //std::cout << "Current: sum" << std::endl;
783  std::vector<SF_TYPE> lhsItems = summands()->lhs->collectLevelEntries( level - 1 );
784  std::vector<SF_TYPE> rhsItems = summands()->rhs->collectLevelEntries( level - 1 );
785  items.insert( items.end(), lhsItems.begin(), lhsItems.end() );
786  items.insert( items.end(), rhsItems.begin(), rhsItems.end() );
787  return items;
788  }
789  case SF_TYPE::UNITE: {
790  //std::cout << "Current: union" << std::endl;
791  std::vector<SF_TYPE> res;
792  for ( const auto& item : unionParameters()->items ) {
793  std::vector<SF_TYPE> tmp = item->collectLevelEntries( level - 1 );
794  res.insert( res.end(), tmp.begin(), tmp.end() );
795  }
796  return res;
797  }
798  case SF_TYPE::INTERSECT: {
799  //std::cout << "Current: intersect" << std::endl;
800  std::vector<SF_TYPE> lhsItems = intersectionParameters()->lhs->collectLevelEntries( level - 1 );
801  std::vector<SF_TYPE> rhsItems = intersectionParameters()->rhs->collectLevelEntries( level - 1 );
802  items.insert( items.end(), lhsItems.begin(), lhsItems.end() );
803  items.insert( items.end(), rhsItems.begin(), rhsItems.end() );
804  return items;
805  }
806  default:
807  return std::vector<SF_TYPE>();
808  }

811  std::string printLevel( unsigned l, std::string separator = "\t" ) const {
