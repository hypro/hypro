306  template <typename N = Number, carl::DisableIf<carl::is_float<N>> = carl::dummy>
307  void reduceNumberRepresentation( std::size_t limit = fReach_DENOMINATOR ) const {
308  if ( !mVertices.empty() ) {
309  // determine barycenter to set rounding directions
310  std::size_t dimension = std::size_t( mVertices.begin()->rawCoordinates().rows() );
311  vector_t<Number> barycenter = vector_t<Number>::Zero( dimension );
312  for ( const auto& vertex : mVertices ) {
313  barycenter = barycenter + ( vertex.rawCoordinates() / Number( mVertices.size() ) );
314  }
315  for ( auto& vertex : mVertices ) {
316  // distance vector to the barycenter - each component now determines its own rounding direction.
317  vector_t<Number> roundingDirections = vertex.rawCoordinates() - barycenter;
318  for ( std::size_t d = 0; d < dimension; ++d ) {
319  assert( d < vertex.dimension() );
320  // determine, which is larger: numerator or denominator
321  Number denom = carl::getDenom( vertex.at( d ) );
322  Number num = carl::getNum( vertex.at( d ) );
323  if ( denom > limit && carl::abs( num ) > limit ) {
324  if ( denom > num ) {
325  Number newNum = num / denom * limit;
326  if ( roundingDirections( d ) >= 0 ) { // round up
327  newNum = carl::ceil( newNum );
328  assert( newNum / limit >= vertex.at( d ) );
329  } else { // round down
330  newNum = carl::floor( newNum );
331  assert( newNum / limit <= vertex.at( d ) );
332  }
333  vertex[d] = newNum / limit;
334  } else { // num >= denom
335  Number newDenom = denom / num * limit;
336  if ( roundingDirections( d ) >= 0 ) { // round up
337  newDenom = carl::floor( newDenom );
338  assert( limit / newDenom >= vertex.at( d ) );
339  } else { // round down
340  newDenom = carl::ceil( newDenom );
341  assert( limit / newDenom <= vertex.at( d ) );
342  }
343  vertex[d] = limit / newDenom;
344  }
345  }
346  }
347  }
348  }
