46  public:
47  Settings( PlottingSettings plotting, FixedAnalysisParameters fixedParameters, std::vector<AnalysisParameters> strategy )
48  : mPlotting( plotting )
49  , mFixedParameters( fixedParameters )
50  , mStrategy( strategy ) {
51  if ( strategy.empty() ) return;
52  // calculate gcd of the strategies timesteps
53  mpz_class numerator_gcd = mStrategy.front().timeStep.get_num();
54  mpz_class denominator_lcm = mStrategy.front().timeStep.get_den();
55 
56  for ( AnalysisParameters const& params : mStrategy ) {
57  numerator_gcd = gcd( numerator_gcd, params.timeStep.get_num() );
58  denominator_lcm = lcm( denominator_lcm, params.timeStep.get_den() );
59  }
60 
61  mFixedParameters.fixedTimeStep = mpq_class( numerator_gcd, denominator_lcm );
62 
63  for ( AnalysisParameters& params : mStrategy ) {
64  mpq_class timeStepFactor = params.timeStep / mFixedParameters.fixedTimeStep;
65  assert( timeStepFactor.get_den() == 1 );
66  assert( timeStepFactor.get_num().fits_sint_p() );
67  params.timeStepFactor = int( timeStepFactor.get_num().get_si() );
68  }
69  }
