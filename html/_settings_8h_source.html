45  Settings( PlottingSettings plotting, FixedAnalysisParameters fixedParameters, std::vector<AnalysisParameters> strategy )
46  : mPlotting( plotting )
47  , mFixedParameters( fixedParameters )
48  , mStrategy( strategy ) {
49  if ( strategy.empty() ) return;
50  // calculate gcd of the strategies timesteps
51 
52  mpz_class numerator_gcd = mStrategy.front().timeStep.get_num();
53  mpz_class denominator_lcm = mStrategy.front().timeStep.get_den();
54 
55  TRACE( "hypro.settings", "numerator_gcd " << numerator_gcd )
56  TRACE( "hypro.settings", "denominator_lcm " << denominator_lcm );
57 
58  for ( AnalysisParameters const& params : mStrategy ) {
59  numerator_gcd = gcd( numerator_gcd, params.timeStep.get_num() );
60  denominator_lcm = lcm( denominator_lcm, params.timeStep.get_den() );
61 
62  TRACE( "hypro.settings", "params.timeStep.get_num() " << params.timeStep.get_num() )
63  TRACE( "hypro.settings", "params.timeStep.get_den() " << params.timeStep.get_den() );
64  TRACE( "hypro.settings", "numerator_gcd " << numerator_gcd )
65  TRACE( "hypro.settings", "denominator_lcm " << denominator_lcm );
66  }
67 
68  mFixedParameters.fixedTimeStep = mpq_class( numerator_gcd, denominator_lcm );
69 
70  TRACE( "hypro.settings", "fixedTimeStep " << mFixedParameters.fixedTimeStep );
71 
72  for ( AnalysisParameters& params : mStrategy ) {
73  mpq_class timeStepFactor = params.timeStep / mFixedParameters.fixedTimeStep;
74  TRACE( "hypro.settings", "timeStepFactor " << timeStepFactor );
75  assert( timeStepFactor.get_den() == 1 );
