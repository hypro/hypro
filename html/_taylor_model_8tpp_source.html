529 template <typename Number>
530 TaylorModel<Number> TaylorModel<Number>::substitute( const std::map<carl::Variable, TaylorModel<Number>> &substitutions,
531  Domain<Number> &domain, const exponent order ) const {
532  TaylorModel<Number> result( *this );
533 
534  if ( isConstant() || substitutions.empty() ) {
535  return result;
536  }
537 
538  MultivariatePolynomial<carl::Interval<Number>> polyZero;
539 
540  // Substitute the variables, which have to be replaced by 0, beforehand,
541  // as this could significantly simplify this multivariate polynomial.
542  for ( auto sub = substitutions.begin(); sub != substitutions.end(); ++sub ) {
543  if ( sub->second.isZero() ) {
544  result.expansion.substituteIn( sub->first, polyZero );
545  if ( result.isConstant() ) {
546  return result;
547  }
548  }
549  }
550 
551  TermsType &mTerms = result.expansion.getTerms();
552  carl::Interval<Number> intOne( 1 );
553  TaylorModel<Number> tmOne( intOne );
554 
555  // Find and sort all exponents occurring for all variables to substitute as
556  // basis.
557  std::map<std::pair<carl::Variable, exponent>, TaylorModel<Number>> expResults;
558  for ( auto term : mTerms ) {
559  if ( term.monomial() ) {
560  const Monomial &m = *( term.monomial() );
561 
562  for ( unsigned i = 0; i < m.nrVariables(); ++i ) {
563  if ( m[i].second > 1 && substitutions.find( m[i].first ) != substitutions.end() ) {
564  expResults[m[i]] = tmOne;
565  }
566  }
567  }
568  }
569 
570  // Calculate the exponentiation of the multivariate polynomial to substitute
571  // the
572  // for variables for, reusing the already calculated exponentiations.
573  if ( !expResults.empty() ) {
574  auto expResultA = expResults.begin();
575  auto expResultB = expResultA;
576  auto sub = substitutions.begin();
577  while ( sub->first != expResultB->first.first ) {
578  assert( sub != substitutions.end() );
579  ++sub;
580  }
581 
582  expResultB->second = sub->second.pow( expResultB->first.second, domain, order );
583  ++expResultB;
584 
585  while ( expResultB != expResults.end() ) {
586  if ( expResultA->first.first != expResultB->first.first ) {
587  ++sub;
588  assert( sub != substitutions.end() );
589  // Go to the next variable.
590  while ( sub->second.isZero() ) {
591  assert( sub != substitutions.end() );
592  ++sub;
593  }
594 
595  expResultB->second = sub->second.pow( expResultB->first.second, domain, order );
596  } else {
597  TaylorModel<Number> tmTemp =
598  sub->second.pow( expResultB->first.second - expResultA->first.second, domain, order );
599  expResultB->second = expResultA->second.multiply( tmTemp, domain, order );
600  }
601 
602  ++expResultA;
603  ++expResultB;
604  }
605  }
606 
607  carl::Interval<Number> intZero( 0 );
608  TaylorModel<Number> resultB;
609 
610  // Substitute the variable for which all occurring exponentiations are
611  // calculated.
612  for ( auto term : mTerms ) {
613  TaylorModel<Number> termResult( term.coeff() );
614  if ( term.monomial() ) {
615  const Monomial &m = *( term.monomial() );
616 
617  for ( unsigned i = 0; i < m.nrVariables(); ++i ) {
618  if ( m[i].second == 1 ) {
619  auto iter = substitutions.find( m[i].first );
620  if ( iter != substitutions.end() ) {
621  termResult.multiply_assign( iter->second, domain, order );
622  } else {
623  termResult.multiply_assign( m[i].first, domain, order );
624  }
625  } else {
626  auto iter = expResults.find( m[i] );
627  if ( iter != expResults.end() ) {
628  termResult.multiply_assign( iter->second, domain, order );
629  } else {
630  termResult.multiply_assign( Term<carl::Interval<Number>>( intOne, m[i].first, m[i].second ), domain,
631  order );
632  }
633  }
634  }
635  }
636 
637  resultB += termResult;
638  }
639 
640  resultB.remainder += remainder;
641 
642  return resultB;
643 }
