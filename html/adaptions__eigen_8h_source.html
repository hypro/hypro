173 template <typename Number>
174 Number norm( const hypro::vector_t<Number>& in, bool roundUp = true ) {
175  Number squares = 0;
176  for ( size_t i = 0, size = in.size(); i < size; ++i ) {
177  squares = squares + carl::pow( *( in.data() + i ), 2 );
178  }
179  assert( squares >= 0 );
180  if ( squares == 0 ) {
181  return 0;
182  }
183 
184  std::pair<Number, Number> dist = carl::sqrt_safe( squares );
185  if ( roundUp ) {
186  return dist.second;
187  } else {
188  return dist.first;
189  }
190 }

279 template <typename Number>
280 int effectiveDimension( const std::vector<vector_t<Number>>& vertices ) {
281  if ( !vertices.empty() ) {
282  if ( vertices.size() == 1 ) {
283  return 0;
284  }
285  long maxDim = vertices.begin()->rows();
286  matrix_t<Number> matr = matrix_t<Number>( vertices.size() - 1, maxDim );
287  // use first vertex as origin, start at second vertex
288  unsigned rowIndex = 0;
289  for ( auto vertexIt = ++vertices.begin(); vertexIt != vertices.end(); ++vertexIt, ++rowIndex ) {
290  matr.row( rowIndex ) = ( *vertexIt - *vertices.begin() ).transpose();
291  }
292  return int( matr.fullPivLu().rank() );
293  }
294  return -1;

303 template <typename Number>
304 inline std::pair<bool, Number> linearDependent( const vector_t<Number>& lhs, const vector_t<Number>& rhs ) {
305  unsigned firstNonZeroPos = 0;
306  if ( lhs.nonZeros() == 0 || rhs.nonZeros() == 0 ) {
307  return std::make_pair( true, 0 );
308  }
309 
310  // both are non-Zero vectors
311  while ( firstNonZeroPos != lhs.rows() && lhs( firstNonZeroPos ) == 0 ) {
312  ++firstNonZeroPos;
313  }
314  if ( firstNonZeroPos == lhs.rows() ) {
315  return std::make_pair( false, 0 );
316  }
317  if ( rhs( firstNonZeroPos ) == 0 ) {
318  return std::make_pair( false, 0 );
319  }
320  Number scalar = lhs( firstNonZeroPos ) / rhs( firstNonZeroPos );
321  if ( lhs == rhs * scalar ) {
322  //std::cout << __func__ << ": scalar: " << scalar << std::endl;
323  return std::make_pair( true, scalar );
324  }
325  return std::make_pair( false, 0 );
326 }
328 template <typename Number, carl::DisableIf<std::is_same<Number, double>> = carl::dummy>
329 vector_t<Number> reduceNumberRepresentation( const vector_t<Number>& in ) {
330  vector_t<Number> res = in;
331  if ( in.rows() > 1 ) {
332  // collect common denominator to obtain integer vector
333  Number commonDenominator = carl::getDenom( in( 0 ) );
334  for ( unsigned i = 1; i < in.rows(); ++i ) {
335  commonDenominator *= carl::getDenom( in( i ) );
336  }
337  // make the vector integer
338  res *= commonDenominator;
339 
340  // find gcd for the numerators
341  Number gcd = carl::gcd( res( 0 ), res( 1 ) );
342  for ( unsigned i = 2; i < res.rows(); ++i ) {
343  gcd = carl::gcd( gcd, res( i ) );
344  }
345  assert( gcd != Number( 0 ) );
346  res = res / gcd;
347  }
348  return res;
349 }
