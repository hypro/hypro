174  Number squares = 0;
175  for ( size_t i = 0, size = in.size(); i < size; ++i ) {
176  squares = squares + carl::pow( *( in.data() + i ), 2 );
177  }
178  assert( squares >= 0 );
179  if ( squares == 0 ) {
180  return 0;
181  }
182 
183  std::pair<Number, Number> dist = carl::sqrt_safe( squares );
184  if ( roundUp ) {
185  return dist.second;
186  } else {
187  return dist.first;
188  }

294 inline std::pair<bool, Number> linearDependent( const vector_t<Number>& lhs, const vector_t<Number>& rhs ) {
295  unsigned firstNonZeroPos = 0;
296  if ( lhs.nonZeros() == 0 || rhs.nonZeros() == 0 ) {
297  return std::make_pair( true, 0 );
298  }
299 
300  // both are non-Zero vectors
301  while ( firstNonZeroPos != lhs.rows() && lhs( firstNonZeroPos ) == 0 ) {
302  ++firstNonZeroPos;
303  }
304  if ( firstNonZeroPos == lhs.rows() ) {
305  return std::make_pair( false, 0 );
306  }
307  if ( rhs( firstNonZeroPos ) == 0 ) {
308  return std::make_pair( false, 0 );
309  }
310  Number scalar = lhs( firstNonZeroPos ) / rhs( firstNonZeroPos );
311  if ( lhs == rhs * scalar ) {
312  //std::cout << __func__ << ": scalar: " << scalar << std::endl;
313  return std::make_pair( true, scalar );
314  }
315  return std::make_pair( false, 0 );
316 }
