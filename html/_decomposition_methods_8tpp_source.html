45 template <typename Number>
46 bool isTimedSubspace( const Location<Number> &loc, size_t index ) {
47  TRACE( "hypro.decisionEntity", "Investigating " << loc.getName() << ", subspace " << index );
48  // check if flow is of the form
49  // 0 ... 0 1
50  // .........
51  // 0 ... 0 1
52  // 0 ....0 0
53 
54  if ( !( loc.getLinearFlow( index ) ).isTimed() ) {
55  TRACE( "hypro.decisionEntity", "Flow is not timed." );
56  return false;
57  }
58 
59  // check if the constraints of the invariant set only contain 0s and one entry 1/-1 at most
60  if ( loc.getInvariant().size() > 0 ) {
61  if ( !loc.getInvariant().isAxisAligned( index ) ) {
62  TRACE( "hypro.decisionEntity", "Invariant is not axis-aligned." );
63  return false;
64  }
65  }
66 
67  for ( const auto &transition : loc.getTransitions() ) {
68  TRACE( "hypro.decisionEntity", "Investigating " << transition->getSource()->getName() << " -> " << transition->getTarget()->getName() );
69 
70  // for each transitions check if the constraints of the guard set only only contain 0s and one entry 1/-1 at most
71 
72  if ( transition->getGuard().size() > 0 ) {
73  if ( !transition->getGuard().isAxisAligned( index ) ) {
74  TRACE( "hypro.decisionEntity", "Guard is not axis-aligned." );
75  return false;
76  }
77  }
78 
79  // check if for each transition the reset function is of the form
80  //
81  // 0/1 0 ... 0 | 0
82  // 0 0/1 0 ... 0 | 0
83  // ..................
84  // 0 ...... 0 0/1 | 0
85  if ( transition->getReset().size() > 0 ) {
86  matrix_t<Number> reset = transition->getReset().getMatrix( index );
87  for ( int i = 0; i < reset.rows(); i++ ) {
88  matrix_t<Number> expected0 = matrix_t<Number>::Zero( 1, reset.cols() );
89  matrix_t<Number> expected1 = matrix_t<Number>::Zero( 1, reset.cols() );
90  expected1( 0, i ) = Number( 1.0 );
91  if ( !( reset.row( i ) == expected0 || reset.row( i ) == expected1 ) ) {
92  TRACE( "hypro.decisionEntitiy", "Reset matrix rows do not match." );
93  return false;
94  }
95  }
96  }
97 
98  if ( transition->getReset().size() > 0 ) {
99  matrix_t<Number> reset = transition->getReset().getVector( index );
100  if ( reset != matrix_t<Number>::Zero( reset.rows(), 1 ) ) {
101  TRACE( "hypro.decisionEntitiy", "Reset vector rows do not reset to zero." );
102  return false;
103  }
104  }
105  }
106  TRACE( "hypro.decisionEntitiy", "Is timed." );
107  return true;
108 }

318 template <typename Number>
319 Decomposition getSubspaceDecomposition( const HybridAutomaton<Number> &automaton ) {
320  if ( checkDecomposed( automaton ) ) {
321  // return empty decomposition
322  return Decomposition();
323  }
324  typedef boost::adjacency_list<boost::vecS, boost::vecS, boost::undirectedS> Graph;
325  Graph G( automaton.dimension() - 1 );
326 
327  //check flow and invariant of locations
328  for ( auto locPtr : automaton.getLocations() ) {
329  for ( std::size_t i = 0; i < locPtr->getNumberFlow(); ++i ) {
330  detail::addEdgesForAffineTrafo( locPtr->getLinearFlow( i ).getFlowMatrix(), G );
331  detail::addEdgesForRectMap( locPtr->getRectangularFlow( i ).getFlowIntervals(), G );
332  }
333  // TODO: add further flow types
334  detail::addEdgesForCondition( locPtr->getInvariant(), G );
335  }
336 
337  //check reset and guards of transitions
338  for ( const auto &transition : automaton.getTransitions() ) {
339  detail::addEdgesForLinTrafo( transition->getReset().getMatrix(), G );
340  detail::addEdgesForRectTrafo( transition->getReset().getIntervals(), G );
341  detail::addEdgesForCondition( transition->getGuard(), G );
342  }
343 
344  //check local bad states
345  for ( const auto &[locPtr, condition] : automaton.getLocalBadStates() ) {
346  detail::addEdgesForCondition( condition, G );
347  }
348  //check global bad states
349  for ( const auto &condition : automaton.getGlobalBadStates() ) {
350  detail::addEdgesForCondition( condition, G );
351  }
352 
353  std::vector<int> component( num_vertices( G ) );
354  int num = boost::connected_components( G, &component[0] );
355  // components now holds an array. It has one entry for each variable specifying its component
356  // we parse this to a vector of vectors. each vector contains the variable indices for one component
357  // and can consequently can be given to project.
358  Decomposition res;
359  res.subspaces = std::vector<std::vector<size_t>>( num );
360  res.subspaceTypes = std::vector<DynamicType>( num, DynamicType::undefined );
361  std::vector<size_t>::size_type i;
362  for ( i = 0; i != component.size(); i++ ) {
363  res.subspaces[component[i]].push_back( i );
364  }
365  return res;
366 }

370  // compute decomposition
371  Decomposition decomposition = getSubspaceDecomposition( automaton );
372
