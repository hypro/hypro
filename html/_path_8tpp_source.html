35 std::pair<Transition<Number>*, carl::Interval<tNumber>> Path<Number, tNumber>::getTransitionToJumpDepth( unsigned depth ) const {
36  TRACE( "hypro.datastructures", "Get transition for depth " << depth << " in path " << *this );
37  if ( depth == 0 ) {
38  return std::make_pair( nullptr, carl::Interval<tNumber>::unboundedInterval() );
39  }
40 
41  unsigned pos = 0;
42  while ( pos < mPath.size() && depth > 0 ) {
43  if ( mPath.at( pos ).isDiscreteStep() ) {
44  depth--;
45  if ( depth > 0 )
46  ++pos;
47  } else {
48  pos++;
49  }
50  }
51  if ( pos == mPath.size() ) {
52  TRACE( "hypro.datastructures", "Did not find appropriate transition." );
53  return std::make_pair( nullptr, carl::Interval<tNumber>::emptyInterval() );
54  }
55  return std::make_pair( mPath.at( pos ).transition, mPath.at( pos ).timeInterval );
56 }

135 template <typename Number, typename tNumber>
136 tNumber Path<Number, tNumber>::maximalTimeSpan( typename std::deque<TPathElement<Number, tNumber>>::const_iterator start, typename std::deque<TPathElement<Number, tNumber>>::const_iterator end ) const {
137  tNumber timespan = 0;
138  bool validPath = true;
139  auto currentPos = start;
140  //std::cout << __func__ << ": start loop." << std::endl;
141  while ( validPath ) {
142  if ( currentPos == end ) {
143  if ( !currentPos->isDiscreteStep() ) {
144  timespan += currentPos->timeInterval.upper();
145  }
146  //std::cout << "Found end." << std::endl;
147  assert( validPath );
148  break;
149  } else if ( currentPos == mPath.end() ) {
150  validPath = false;
151  //std::cout << "did not find end - invalid path." << std::endl;
152  break;
153  } else {
154  assert( validPath );
155  if ( !( currentPos->isDiscreteStep() ) ) {
156  //std::cout << "increase timespan by " << currentPos->timeInterval.upper() << std::endl;
157  timespan += currentPos->timeInterval.upper();
158  }
159  }
160  ++currentPos;
161  }
162  //std::cout << "Exited loop." << std::endl;
163  if ( !validPath ) {
164  return -1;
165  }
166  return timespan;
167 }

197 template <typename Number, typename tNumber>
198 bool Path<Number, tNumber>::hasChatteringZeno() const {
199  // find all cycles first and store potential cycles
200  // std::cout << __func__ << ": checking path: " << *this << std::endl;
201  std::vector<std::vector<Transition<Number>*>> potentialCycles;
202  for ( auto startElemIt = mPath.begin(); startElemIt != mPath.end(); ++startElemIt ) {
203  // start measuring the time from a timestep -> if the startElement is not a timestep, increase it.
204  if ( !startElemIt->isDiscreteStep() ) {
205  ++startElemIt;
206  }
207  if ( startElemIt == mPath.end() ) {
208  break;
209  }
210  //std::cout << __func__ << ": Starting point is: " << *startElemIt << std::endl;
211  // if the startElement iterator is not pointing to the end of the path (this is the case when the last item is a transition)
212  // detect cycle by finding the same location.
213  if ( startElemIt != mPath.end() ) {
214  auto nextElem = startElemIt;
215  nextElem++;
216  if ( ( nextElem ) == mPath.end() ) {
217  break;
218  }
219  const Location<Number>* startLoc = ( startElemIt )->transition->getSource();
220  // std::cout << "Dimension: " << (startElemIt)->transition->getReset().getContinuousResetMatrix().cols() << std::endl;
221  std::vector<Transition<Number>*> transitionSequence;
222  transitionSequence.push_back( ( startElemIt )->transition );
223  //std::cout << __func__ << ": Starting location is " << startLoc->getId() << std::endl;
224  for ( ; nextElem != mPath.end(); ++nextElem ) {
225  //std::cout << __func__ << ": next element is " << *(nextElem) << std::endl;
226  //std::cout << __func__ << ": current maximal timespan is " << maximalTimeSpan(startElemIt,nextElem) << std::endl;
227  if ( nextElem->isDiscreteStep() &&
228  nextElem->transition->getTarget() == startLoc &&
229  maximalTimeSpan( startElemIt, nextElem ) == 0 ) {
230  transitionSequence.push_back( nextElem->transition );
231  // std::cout << "Found potential path." << std::endl;
232  // std::cout << "Transition<Number> sequence length: " << transitionSequence.size() << std::endl;
233  potentialCycles.push_back( transitionSequence );
234  break;
235  } else if ( nextElem->isDiscreteStep() ) {
236  transitionSequence.push_back( nextElem->transition );
237  }
238  }
239  }
240  }
241 
242  // std::cout << __func__ << ": Exited loop, now verifying resets. Number of potential cycles: " << potentialCycles.size() << std::endl;
243 
244  // for each of the location sequences collect the resets - if they form the identity we have found a simple case of chattering Zeno behavior.
245  // TODO: Extend this for discrete and clock resets!
246  // TODO: Extend this for the reset vector!
247  for ( auto& transitionSequence : potentialCycles ) {
248  assert( ( *transitionSequence.begin() )->getSource() == transitionSequence.back()->getTarget() );
249  assert( !transitionSequence.empty() );
250  auto transitionIt = transitionSequence.begin();
251  // std::cout << "Reset: " << (*transitionIt)->getReset() << std::endl;
252  // std::cout << "Dimension: " << (*transitionIt)->getReset().getContinuousResetMatrix().cols() << std::endl;
253  hypro::matrix_t<Number> resetMatrix = ( *transitionIt )->getReset().getMatrix();
254  ++transitionIt;
255  for ( ; transitionIt != transitionSequence.end(); ++transitionIt ) {
256  // std::cout << "Reset: " << (*transitionIt)->getReset() << std::endl;
257  resetMatrix = ( *transitionIt )->getReset().getMatrix() * resetMatrix;
258  }
259  //std:cout << __func__ << ": Resulting reset matrix: " << resetMatrix << std::endl;
260  if ( resetMatrix == hypro::matrix_t<Number>::Identity( resetMatrix.rows(), resetMatrix.cols() ) ) {
261  // std::cout << __func__ << ": Found chattering Zeno path: " << *this << std::endl;
262  return true;
263  }
264  }
265  return false;
266 }
