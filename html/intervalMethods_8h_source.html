13  * @param intervals The intervals as a vector (call by reference).
14  * @param[in] hsp The halfspace

16  * @tparam Number The used number type.
17  *
18  * @return True, if one of the resulting intervals is empty, False otherwise.

21  bool icpIntersectHalfspace(std::vector<carl::Interval<Number>>& intervals, const Halfspace<Number>& hsp) {
22  assert(Eigen::Index(intervals.size()) == hsp.normal().rows());
23 
24  // contraction variable cVar, other variables (for that run) oVar.
25  for(Eigen::Index cVar = 0; cVar < hsp.normal().rows(); ++cVar) {
26  if(hsp.normal()(cVar) != 0) {
27  // initialize with constant part.
28  carl::Interval<Number> newIntv = Number(-1) * carl::Interval<Number>(hsp.offset(),carl::BoundType::INFTY,hsp.offset(),carl::BoundType::WEAK);
29  // update interval using interval arithmetic (ICP-style: propagation).
30  for(Eigen::Index oVar = 0; oVar < hsp.normal().rows(); ++oVar) {
31  if(cVar != oVar && hsp.normal()(oVar) != 0) {
32  newIntv += hsp.normal()(oVar)*intervals[oVar];
33  }
34  }
35  newIntv /= Number(-hsp.normal()(cVar));
36  intervals[cVar] = set_intersection(intervals[cVar], newIntv);
37  if(intervals[cVar].isEmpty()) {
38  return true;
39  }
40  }
41 
42  }
43  return false;
44  }
