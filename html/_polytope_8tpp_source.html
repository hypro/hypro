150 template <typename Number, typename Converter, class Setting>
151 void PolytopeT<Number, Converter, Setting>::calculateFan() {
152  std::vector<Facet<Number>> facets = convexHull( mPoints );
153  std::set<Point<Number>> preresult;
154  for ( unsigned i = 0; i < facets.size(); i++ ) {
155  for ( unsigned j = 0; j < facets[i].vertices().size(); j++ ) {
156  preresult.insert( facets[i].vertices().at( j ) );
157  }
158  }
159  Fan<Number> fan;
160  for ( auto& point : preresult ) {
161  Cone<Number>* cone = new Cone<Number>();
162  for ( unsigned i = 0; i < facets.size(); i++ ) {
163  for ( unsigned j = 0; j < facets[i].vertices().size(); j++ ) {
164  if ( point == facets[i].vertices().at( j ) ) {
165  std::vector<Ridge<Number>> ridges = getRidges( facets[i] );
166  for ( unsigned m = 0; m < ridges.size(); m++ ) {
167  if ( checkInsideRidge( ridges[m], point ) ) {
168  std::vector<Facet<Number>> conefacets = shareRidge( facets, ridges[m] );
169 
170  matrix_t<Number> matrix = matrix_t<Number>( conefacets.size(), point.size() );
171  for ( unsigned k = 1; k < conefacets.size(); k++ ) {
172  for ( unsigned l = 0; l < conefacets[k].getNormal().size(); l++ ) {
173  matrix( k, l ) = conefacets[k].getNormal()( l );
174  }
175  }
176 
177  for ( unsigned j = 0; j < point.size(); j++ ) {
178  matrix( 0, j ) = 1;
179  if ( matrix.fullPivLu().rank() == point.size() ) {
180  break;
181  } else {
182  matrix( 0, j ) = 0;
183  }
184  }
185  vector_t<Number> b = vector_t<Number>::Zero( conefacets.size() );
186  b( 0 ) = 1;
187  vector_t<Number> result = matrix.fullPivHouseholderQr().solve( b );
188 
189  cone->add( std::shared_ptr<Halfspace<Number>>(
190  new Halfspace<Number>( result, result.dot( point.rawCoordinates() ) ) ) );
191  // cone->add(std::make_shared<Halfspace<Number>>(Halfspace<Number>(result,
192  // result.dot(point.rawCoordinates()))));
193  }
194  }
195  }
196  }
197  }
198  fan.add( cone );
199  }
200  mFan = fan;
201 }

834 template <typename Number, typename Converter, class Setting>
835 Point<Number> PolytopeT<Number, Converter, Setting>::localSearch( Point<Number>& _vertex, Point<Number>& _sinkMaximizerTarget ) {
836  if ( Setting::fukuda_DEBUG ) {
837  std::cout << "-------------------------" << std::endl;
838  std::cout << "in the following: Local Search for Vertex " << _vertex << std::endl;
839  std::cout << "-------------------------" << std::endl;
840  }
841 
842  // compute the maximizer vector of the currently considered vertex
843  Point<Number> maximizerTarget;
844  vector_t<Number> maximizerVector = polytope::computeMaximizerVector( maximizerTarget, _vertex );
845 
846  // compute the ray direction (a vector)
847  vector_t<Number> ray = polytope::computeEdge( maximizerTarget, _sinkMaximizerTarget );
848 
849  if ( Setting::fukuda_DEBUG ) {
850  std::cout << "Starting Point of Ray: " << maximizerTarget << std::endl;
851  std::cout << "End Point of Ray: " << _sinkMaximizerTarget << std::endl;
852  }
853 
854  // compute the normal cone of _vertex
855  Cone<Number>* cone = polytope::computeCone( _vertex, maximizerVector );
856 
857  // iterate through all planes and check which one intersects with the ray
858  Number factor;
859  Point<Number> origin = cone->origin();
860  Halfspace<Number> intersectedPlane;
861 
862  std::vector<Halfspace<Number>*> planes = cone->get();
863 
864  if ( Setting::fukuda_DEBUG ) {
865  std::cout << "-----------------" << std::endl;
866  std::cout << "Ray: " << ray << std::endl;
867  }
868 
869  for ( typename std::vector<Halfspace<Number>*>::iterator it = planes.begin(); it != planes.end(); ++it ) {
870  if ( ( *it )->intersection( factor, ray ) ) {
871  if ( Setting::fukuda_DEBUG ) {
872  std::cout << "Intersection found " << std::endl;
873  }
874  intersectedPlane = *( *it );
875  break;
876  }
877  }
878 
879  Point<Number> secondOrigin;
880 
881  if ( Setting::fukuda_DEBUG ) {
882  std::cout << "-----------------" << std::endl;
883  std::cout << "Normal of Intersection Plane: " << intersectedPlane.normal() << std::endl;
884  std::cout << "Offset of Intersection Plane: " << intersectedPlane.offset() << std::endl;
885  std::cout << "-----------------" << std::endl;
886  }
887 
888  std::vector<vector_t<Number>> decompositionEdges = polytope::computeEdgeSet( _vertex );
889 
890  for ( unsigned i = 0; i < decompositionEdges.size(); i++ ) {
891  Number dotProduct = intersectedPlane.normal().dot( decompositionEdges.at( i ) );
892  Number normFactor = intersectedPlane.normal().norm() * decompositionEdges.at( i ).norm();
893 
894  // has to be done...
895  dotProduct = std::round( dotProduct.toDouble() * 1000000 );
896  normFactor = std::round( normFactor.toDouble() * 1000000 );
897 
898  if ( Setting::fukuda_DEBUG ) {
899  std::cout << "Dot Product: " << dotProduct << std::endl;
900  std::cout << "Norm Factor: " << normFactor << std::endl;
901  std::cout << "Parallelism Factor: " << dotProduct / normFactor << std::endl;
902  std::cout << "Value of the if condition: " << ( dotProduct / normFactor == 1 ) << std::endl;
903  }
904 
905  if ( ( dotProduct / normFactor == 1 + EPSILON ) || ( dotProduct / normFactor == 1 - EPSILON ) ||
906  ( dotProduct / normFactor == -1 + EPSILON ) || ( dotProduct / normFactor == -1 - EPSILON ) ||
907  ( dotProduct / normFactor == -1 ) || ( dotProduct / normFactor == 1 ) ) {
908  if ( Setting::fukuda_DEBUG ) {
909  std::cout << "Parallel Edge found" << std::endl;
910  std::cout << "-----------------" << std::endl;
911  }
912 
913  // we have to find out from which decomposition our edge came from
914  // and accordingly initialize a counter that describes the direction
915  unsigned half = decompositionEdges.size() / 2;
916 
917  std::pair<int, int> counter;
918  if ( i + 1 <= half ) {
919  counter.first = 1;
920  counter.second = i + 1;
921  } else {
922  counter.first = 2;
923  counter.second = i + 1 - half;
924  }
925 
926  // now we have to retrieve the new vertex in this edge direction, using
927  // the adjacency oracle
928  // the result is stored in secondOrigin
929  bool res = polytope::adjOracle( secondOrigin, _vertex, counter );
930  if ( res ) {
931  break;
932  }
933  }
934  }
935 
936  // add this normal cone to fan of polytope
937  this->mFan.add( cone );
938 
939  if ( Setting::fukuda_DEBUG ) {
940  std::cout << "-------------------------" << std::endl;
941  std::cout << "Local Search result: " << secondOrigin << std::endl;
942  std::cout << "-------------------------" << std::endl;
943  }
944 
945  return secondOrigin;
946 }
