295  template<typename N = Number, carl::DisableIf< std::is_same<N, double> > = carl::dummy>
296  void reduceNumberRepresentation(const std::vector<Point<Number>>& _vertices = std::vector<Point<Number>>(), unsigned limit = fReach_DENOMINATOR) const {
297  //#ifdef REDUCE_NUMBERS
298  if(Setting::REDUCE_NUMBERS == true){
299  TRACE("hypro.hPolytope","Attempt to reduce numbers.");
300  std::vector<Point<Number>> originalVertices;
301  if(_vertices.empty()) {
302  TRACE("hypro.hPolytope","No passed vertices, computed vertices.");
303  originalVertices = this->vertices();
304  } else {
305  TRACE("hypro.hPolytope","Use passed vertices.");
306  originalVertices = _vertices;
307  }
308  TRACE("hypro.hPolytope","Vertices empty: " << originalVertices.empty());
309 
310  if(!this->empty()){
311  // normal reduction
312  for(unsigned planeIndex = 0; planeIndex < mHPlanes.size(); ++planeIndex){
313  #ifdef HPOLY_DEBUG_MSG
314  std::cout << "Original: " << mHPlanes.at(planeIndex) << std::endl;
315  #endif
316  // find maximal value
317  Number largest = Number(0);
318  mHPlanes.at(planeIndex).makeInteger();
319  #ifdef HPOLY_DEBUG_MSG
320  std::cout << "As Integer: " << mHPlanes.at(planeIndex) << std::endl;
321  #endif
322  largest = carl::abs(mHPlanes.at(planeIndex).offset());
323  for(unsigned i = 0; i < mDimension; ++i){
324  if(carl::abs(mHPlanes.at(planeIndex).normal()(i)) > largest){
325  largest = carl::abs(mHPlanes.at(planeIndex).normal()(i));
326  }
327  }
328 
329  // reduce, if reduction is required
330  if(largest > (limit*limit)) {
331  #ifdef HPOLY_DEBUG_MSG
332  std::cout << "Actual reduction" << std::endl;
333  #endif
334  vector_t<Number> newNormal(mDimension);
335  for(unsigned i = 0; i < mDimension; ++i){
336  newNormal(i) = carl::floor(Number((mHPlanes.at(planeIndex).normal()(i)/largest)*Number(limit)));
337  assert(carl::abs(Number(mHPlanes.at(planeIndex).normal()(i)/largest)) <= Number(1));
338  assert(carl::isInteger(newNormal(i)));
339  assert(newNormal(i) <= Number(limit));
340  }
341  mHPlanes.at(planeIndex).setNormal(newNormal);
342  Number newOffset = mHPlanes.at(planeIndex).offset();
343  newOffset = carl::ceil(Number((newOffset/largest)*Number(limit)));
344  for(const auto& vertex : originalVertices) {
345  Number tmp = newNormal.dot(vertex.rawCoordinates());
346  if(tmp > newOffset){
347  newOffset = newOffset + (tmp-newOffset);
348  assert(newNormal.dot(vertex.rawCoordinates()) <= newOffset);
349  }
350  assert(Halfspace<Number>(newNormal,newOffset).contains(vertex));
351  }
352  newOffset = carl::ceil(newOffset);
353  #ifdef HPOLY_DEBUG_MSG
354  std::cout << "Reduced to " << convert<Number,double>(newNormal).transpose() << " <= " << carl::toDouble(newOffset) << std::endl;
355  #endif
356  mHPlanes.at(planeIndex).setOffset(newOffset);
357  }
358  #ifdef HPOLY_DEBUG_MSG
359  std::cout << "Reduced: " << mHPlanes.at(planeIndex) << std::endl;
360  #endif
361  }
362  #ifdef HPOLY_DEBUG_MSG
363  //std::cout << "After Reduction: " << *this << std::endl;
364  #endif
365  }
366  }
367  //#endif
368  }
