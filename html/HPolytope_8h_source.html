297  template<typename N = Number, carl::DisableIf< std::is_same<N, double> > = carl::dummy>
298  void reduceNumberRepresentation(const std::vector<Point<Number>>& _vertices = std::vector<Point<Number>>(), unsigned limit = fReach_DENOMINATOR) const {
299  //#ifdef REDUCE_NUMBERS
300  if(Setting::REDUCE_NUMBERS == true){
301  TRACE("hypro.hPolytope","Attempt to reduce numbers.");
302  std::vector<Point<Number>> originalVertices;
303  if(_vertices.empty()) {
304  TRACE("hypro.hPolytope","No passed vertices, computed vertices.");
305  originalVertices = this->vertices();
306  } else {
307  TRACE("hypro.hPolytope","Use passed vertices.");
308  originalVertices = _vertices;
309  }
310  TRACE("hypro.hPolytope","Vertices empty: " << originalVertices.empty());
311 
312  if(!this->empty()){
313  // normal reduction
314  for(unsigned planeIndex = 0; planeIndex < mHPlanes.size(); ++planeIndex){
315  #ifdef HPOLY_DEBUG_MSG
316  std::cout << "Original: " << mHPlanes.at(planeIndex) << std::endl;
317  #endif
318  // find maximal value
319  Number largest = Number(0);
320  mHPlanes.at(planeIndex).makeInteger();
321  #ifdef HPOLY_DEBUG_MSG
322  std::cout << "As Integer: " << mHPlanes.at(planeIndex) << std::endl;
323  #endif
324  largest = carl::abs(mHPlanes.at(planeIndex).offset());
325  for(unsigned i = 0; i < mDimension; ++i){
326  if(carl::abs(mHPlanes.at(planeIndex).normal()(i)) > largest){
327  largest = carl::abs(mHPlanes.at(planeIndex).normal()(i));
328  }
329  }
330 
331  // reduce, if reduction is required
332  if(largest > (limit*limit)) {
333  #ifdef HPOLY_DEBUG_MSG
334  std::cout << "Actual reduction" << std::endl;
335  #endif
336  vector_t<Number> newNormal(mDimension);
337  for(unsigned i = 0; i < mDimension; ++i){
338  newNormal(i) = carl::floor(Number((mHPlanes.at(planeIndex).normal()(i)/largest)*Number(limit)));
339  assert(carl::abs(Number(mHPlanes.at(planeIndex).normal()(i)/largest)) <= Number(1));
340  assert(carl::isInteger(newNormal(i)));
341  assert(newNormal(i) <= Number(limit));
342  }
343  mHPlanes.at(planeIndex).setNormal(newNormal);
344  Number newOffset = mHPlanes.at(planeIndex).offset();
345  newOffset = carl::ceil(Number((newOffset/largest)*Number(limit)));
346  for(const auto& vertex : originalVertices) {
347  Number tmp = newNormal.dot(vertex.rawCoordinates());
348  if(tmp > newOffset){
349  newOffset = newOffset + (tmp-newOffset);
350  assert(newNormal.dot(vertex.rawCoordinates()) <= newOffset);
351  }
352  assert(Halfspace<Number>(newNormal,newOffset).contains(vertex));
353  }
354  newOffset = carl::ceil(newOffset);
355  #ifdef HPOLY_DEBUG_MSG
356  std::cout << "Reduced to " << convert<Number,double>(newNormal).transpose() << " <= " << carl::toDouble(newOffset) << std::endl;
357  #endif
358  mHPlanes.at(planeIndex).setOffset(newOffset);
359  }
360  #ifdef HPOLY_DEBUG_MSG
361  std::cout << "Reduced: " << mHPlanes.at(planeIndex) << std::endl;
362  #endif
363  }
364  #ifdef HPOLY_DEBUG_MSG
365  //std::cout << "After Reduction: " << *this << std::endl;
366  #endif
367  }
368  }
369  //#endif
370  }
