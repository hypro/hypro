322  template <typename N = Number, carl::DisableIf<std::is_same<N, double>> = carl::dummy>
323  void reduceNumberRepresentation( const std::vector<Point<Number>>& _vertices = std::vector<Point<Number>>(), unsigned limit = fReach_DENOMINATOR ) const {
324  //#ifdef REDUCE_NUMBERS
325  if ( Setting::REDUCE_NUMBERS == true ) {
326  TRACE( "hypro.hPolytope", "Attempt to reduce numbers." );
327  std::vector<Point<Number>> originalVertices;
328  if ( _vertices.empty() ) {
329  TRACE( "hypro.hPolytope", "No passed vertices, computed vertices." );
330  originalVertices = this->vertices();
331  } else {
332  TRACE( "hypro.hPolytope", "Use passed vertices." );
333  originalVertices = _vertices;
334  }
335  TRACE( "hypro.hPolytope", "Vertices empty: " << originalVertices.empty() );
336 
337  if ( !this->empty() ) {
338  // normal reduction
339  for ( unsigned planeIndex = 0; planeIndex < mHPlanes.size(); ++planeIndex ) {
340 #ifdef HPOLY_DEBUG_MSG
341  std::cout << "Original: " << mHPlanes.at( planeIndex ) << std::endl;
342 #endif
343  // find maximal value
344  Number largest = Number( 0 );
345  mHPlanes.at( planeIndex ).makeInteger();
346 #ifdef HPOLY_DEBUG_MSG
347  std::cout << "As Integer: " << mHPlanes.at( planeIndex ) << std::endl;
348 #endif
349  largest = carl::abs( mHPlanes.at( planeIndex ).offset() );
350  for ( unsigned i = 0; i < mDimension; ++i ) {
351  if ( carl::abs( mHPlanes.at( planeIndex ).normal()( i ) ) > largest ) {
352  largest = carl::abs( mHPlanes.at( planeIndex ).normal()( i ) );
353  }
354  }
355 
356  // reduce, if reduction is required
357  if ( largest > ( limit * limit ) ) {
358 #ifdef HPOLY_DEBUG_MSG
359  std::cout << "Actual reduction" << std::endl;
360 #endif
361  vector_t<Number> newNormal( mDimension );
362  for ( unsigned i = 0; i < mDimension; ++i ) {
363  newNormal( i ) = carl::floor( Number( ( mHPlanes.at( planeIndex ).normal()( i ) / largest ) * Number( limit ) ) );
364  assert( carl::abs( Number( mHPlanes.at( planeIndex ).normal()( i ) / largest ) ) <= Number( 1 ) );
365  assert( carl::isInteger( newNormal( i ) ) );
366  assert( newNormal( i ) <= Number( limit ) );
367  }
368  mHPlanes.at( planeIndex ).setNormal( newNormal );
369  Number newOffset = mHPlanes.at( planeIndex ).offset();
370  newOffset = carl::ceil( Number( ( newOffset / largest ) * Number( limit ) ) );
371  for ( const auto& vertex : originalVertices ) {
372  Number tmp = newNormal.dot( vertex.rawCoordinates() );
373  if ( tmp > newOffset ) {
374  newOffset = newOffset + ( tmp - newOffset );
375  assert( newNormal.dot( vertex.rawCoordinates() ) <= newOffset );
376  }
377  assert( Halfspace<Number>( newNormal, newOffset ).contains( vertex ) );
378  }
379  newOffset = carl::ceil( newOffset );
380 #ifdef HPOLY_DEBUG_MSG
381  std::cout << "Reduced to " << convert<Number, double>( newNormal ).transpose() << " <= " << carl::toDouble( newOffset ) << std::endl;
382 #endif
383  mHPlanes.at( planeIndex ).setOffset( newOffset );
384  }
385 #ifdef HPOLY_DEBUG_MSG
386  std::cout << "Reduced: " << mHPlanes.at( planeIndex ) << std::endl;
387 #endif
388  }
389 #ifdef HPOLY_DEBUG_MSG
390 //std::cout << "After Reduction: " << *this << std::endl;
391 #endif
392  }
393  }
394  //#endif
395  }
