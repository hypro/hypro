293  template<typename N = Number, carl::DisableIf< std::is_same<N, double> > = carl::dummy>
294  void reduceNumberRepresentation(const std::vector<Point<Number>>& _vertices = std::vector<Point<Number>>(), unsigned limit = fReach_DENOMINATOR) const {
295  #ifdef REDUCE_NUMBERS
296  TRACE("Attempt to reduce numbers.");
297  std::vector<Point<Number>> originalVertices;
298  if(_vertices.empty()) {
299  TRACE("No passed vertices, computed vertices.");
300  originalVertices = this->vertices();
301  } else {
302  TRACE("Use passed vertices.");
303  originalVertices = _vertices;
304  }
305  TRACE("Vertices empty: " << originalVertices.empty());
306 
307  if(!this->empty()){
308  // normal reduction
309  for(unsigned planeIndex = 0; planeIndex < mHPlanes.size(); ++planeIndex){
310  #ifdef HPOLY_DEBUG_MSG
311  std::cout << "Original: " << mHPlanes.at(planeIndex) << std::endl;
312  #endif
313  // find maximal value
314  Number largest = Number(0);
315  mHPlanes.at(planeIndex).makeInteger();
316  #ifdef HPOLY_DEBUG_MSG
317  std::cout << "As Integer: " << mHPlanes.at(planeIndex) << std::endl;
318  #endif
319  largest = carl::abs(mHPlanes.at(planeIndex).offset());
320  for(unsigned i = 0; i < mDimension; ++i){
321  if(carl::abs(mHPlanes.at(planeIndex).normal()(i)) > largest){
322  largest = carl::abs(mHPlanes.at(planeIndex).normal()(i));
323  }
324  }
325 
326  // reduce, if reduction is required
327  if(largest > (limit*limit)) {
328  #ifdef HPOLY_DEBUG_MSG
329  std::cout << "Actual reduction" << std::endl;
330  #endif
331  vector_t<Number> newNormal(mDimension);
332  for(unsigned i = 0; i < mDimension; ++i){
333  newNormal(i) = carl::floor(Number((mHPlanes.at(planeIndex).normal()(i)/largest)*Number(limit)));
334  assert(carl::abs(Number(mHPlanes.at(planeIndex).normal()(i)/largest)) <= Number(1));
335  assert(carl::isInteger(newNormal(i)));
336  assert(newNormal(i) <= Number(limit));
337  }
338  mHPlanes.at(planeIndex).setNormal(newNormal);
339  Number newOffset = mHPlanes.at(planeIndex).offset();
340  newOffset = carl::ceil(Number((newOffset/largest)*Number(limit)));
341  for(const auto& vertex : originalVertices) {
342  Number tmp = newNormal.dot(vertex.rawCoordinates());
343  if(tmp > newOffset){
344  newOffset = newOffset + (tmp-newOffset);
345  assert(newNormal.dot(vertex.rawCoordinates()) <= newOffset);
346  }
347  assert(Halfspace<Number>(newNormal,newOffset).contains(vertex));
348  }
349  newOffset = carl::ceil(newOffset);
350  #ifdef HPOLY_DEBUG_MSG
351  std::cout << "Reduced to " << convert<Number,double>(newNormal).transpose() << " <= " << carl::toDouble(newOffset) << std::endl;
352  #endif
353  mHPlanes.at(planeIndex).setOffset(newOffset);
354  }
355  #ifdef HPOLY_DEBUG_MSG
356  std::cout << "Reduced: " << mHPlanes.at(planeIndex) << std::endl;
357  #endif
358  }
359  #ifdef HPOLY_DEBUG_MSG
360  std::cout << "After Reduction: " << *this << std::endl;
361  #endif
362  #endif
363  }
364  }
