171  //transition x enabled segments, segment ind
172  std::vector<EnabledSets<Representation>> enabledSegments{};
173 
174  for ( const auto& transition : loc->getTransitions() ) {
175  auto& currentSucc = enabledSegments.emplace_back( EnabledSets<Representation>{ transition.get() } );
176 
177  SegmentInd cnt = 0;
178  for ( auto const& valuationSet : flowpipe ) {
179  auto [containment, intersected] = intersect( valuationSet, transition->getGuard() );
180 
181  if ( containment != CONTAINMENT::NO ) {
182  currentSucc.valuationSets.push_back( { intersected, cnt } );
183  }
184  ++cnt;
185  }
186  }
187 
188  std::vector<JumpSuccessor<Representation>> successors{};
189 
190  // aggregation
191  // for each transition
192  for ( auto& [transition, valuationSets] : enabledSegments ) {
193  // no aggregation
194  std::size_t blockSize = 1;
195  if ( mSettings.aggregation == AGG_SETTING::AGG ) {
196  if ( mSettings.clustering > 0 ) {
197  blockSize = ( valuationSets.size() + mSettings.clustering ) / mSettings.clustering; //division rounding up
198  } else {
199  blockSize = valuationSets.size();
200  }
201 
202  } else if ( mSettings.aggregation == AGG_SETTING::MODEL && transition->getAggregation() != Aggregation::none ) {
203  if ( transition->getAggregation() == Aggregation::clustering ) {
204  blockSize = ( blockSize + transition->getClusterBound() ) / transition->getClusterBound(); //division rounding up
205  }
206  }
207  successors.emplace_back( JumpSuccessor<Representation>{ transition, aggregate( blockSize, valuationSets ) } );
208  }
209 
210  // applyReset
211  for ( auto& [transition, valuationSets] : successors ) {
212  for ( auto it = valuationSets.begin(); it != valuationSets.end(); ) {
213  it->valuationSet = applyReset( it->valuationSet, transition->getReset() );
214  CONTAINMENT containment;
215  std::tie( containment, it->valuationSet ) = intersect( it->valuationSet, transition->getTarget()->getInvariant() );
216  if ( containment == CONTAINMENT::NO ) {
217  it = valuationSets.erase( it );
218  } else {
219  it->valuationSet.reduceRepresentation();
220  ++it;
221  }
222  }
223  }
