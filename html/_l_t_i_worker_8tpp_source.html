169 std::vector<JumpSuccessor<Representation>> LTIWorker<Representation>::computeJumpSuccessors( std::vector<Representation> const& flowpipe, Location<Number> const* loc ) const {
170  //transition x enabled segments, segment ind
171  std::vector<EnabledSets<Representation>> enabledSegments{};
172 
173  for ( const auto& transition : loc->getTransitions() ) {
174  auto& currentSucc = enabledSegments.emplace_back( EnabledSets<Representation>{ transition.get() } );
175 
176  SegmentInd cnt = 0;
177  for ( auto const& valuationSet : flowpipe ) {
178  auto [containment, intersected] = intersect( valuationSet, transition->getGuard() );
179 
180  if ( containment != CONTAINMENT::NO ) {
181  currentSucc.valuationSets.push_back( { intersected, cnt } );
182  }
183  ++cnt;
184  }
185  }
186 
187  std::vector<JumpSuccessor<Representation>> successors{};
188 
189  // aggregation
190  // for each transition
191  for ( auto& [transition, valuationSets] : enabledSegments ) {
192  // no aggregation
193  std::size_t blockSize = 1;
194  if ( mSettings.aggregation == AGG_SETTING::AGG ) {
195  if ( mSettings.clustering > 0 ) {
196  blockSize = ( valuationSets.size() + mSettings.clustering ) / mSettings.clustering; //division rounding up
197  } else {
198  blockSize = valuationSets.size();
199  }
200 
201  } else if ( mSettings.aggregation == AGG_SETTING::MODEL && transition->getAggregation() != Aggregation::none ) {
202  if ( transition->getAggregation() == Aggregation::clustering ) {
203  blockSize = ( blockSize + transition->getClusterBound() ) / transition->getClusterBound(); //division rounding up
204  }
205  }
206  successors.emplace_back( JumpSuccessor<Representation>{ transition, aggregate( blockSize, valuationSets ) } );
207  }
208 
209  // applyReset
210  for ( auto& [transition, valuationSets] : successors ) {
211  for ( auto it = valuationSets.begin(); it != valuationSets.end(); ) {
212  it->valuationSet = applyReset( it->valuationSet, transition->getReset() );
213  CONTAINMENT containment;
214  std::tie( containment, it->valuationSet ) = intersect( it->valuationSet, transition->getTarget()->getInvariant() );
215  if ( containment == CONTAINMENT::NO ) {
216  it = valuationSets.erase( it );
217  } else {
218  it->valuationSet.reduceRepresentation();
219  ++it;
220  }
221  }
222  }
223

225 }
