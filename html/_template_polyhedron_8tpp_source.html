154 bool TemplatePolyhedronT<Number, Converter, Setting>::empty() const {
155  //Quick checks
156  if ( mMatrixPtr == nullptr ) return true;
157 
158  //Check emptiness cache
159  if ( mEmptyState != SETSTATE::UNKNOWN ) return ( mEmptyState == SETSTATE::EMPTY );
160 
161  //Linear time quick checks:
162  //If all coeffs negative -> all directions point towards origin -> unbounded polyhedron infeasible therefore empty
163  //If all coeffs positive -> all directions point away from origin -> bounded polyhedron not empty
164  bool allNegative = true;
165  bool allPositive = true;
166  for ( int i = 0; i < mVector.rows(); i++ ) {
167  if ( mVector( i ) < 0 ) {
168  allPositive = false;
169  } else if ( mVector( i ) > 0 ) {
170  allNegative = false;
171  }
172  }
173  if ( allPositive ) return false;
174  if ( allNegative ) return true;
175 
176  //If no quick check triggered: Solve LP and cache result
177  //std::cout << "TemplatePolyhedron::empty, mOptimizer mat: \n" << mOptimizer.matrix() << "mOptimizer vec: \n" << mOptimizer.vector() << std::endl;
178  mEmptyState = !mOptimizer.checkConsistency() ? SETSTATE::EMPTY : SETSTATE::NONEMPTY;
179  //std::cout << "TemplatePolyhedron::empty, is empty? " << (mEmptyState == SETSTATE::EMPTY) << std::endl;
180  TRACE( "hypro.representations.TPolytope", "Optimizer result: " << mEmptyState );
181  return ( mEmptyState == SETSTATE::EMPTY );

278 void TemplatePolyhedronT<Number, Converter, Setting>::removeRedundancy() {
279  //HPoly method: optimize, then get redundant constraints, then remove the redundant constraints
280  //Time: LP + LP + linear
281  //std::cout << "TemplatePolyhedron::removeRedundancy, mNonRedundant? " << mNonRedundant << std::endl;
282  if ( !mNonRedundant && !empty() ) {
283  std::vector<std::size_t> redundant = mOptimizer.redundantConstraints();
284  std::sort( redundant.begin(), redundant.end() );
285  //std::cout << "TemplatePolyhedron::removeRedundancy, redundant indices: {";
286  //for(const auto& r : redundant){
287  // //std::cout << r << ",";
288  //}
289  //std::cout << "}" << std::endl;
290  if ( !redundant.empty() ) {
291  matrix_t<Number> nonRedundantMatrix = matrix_t<Number>::Zero( mMatrixPtr->rows() - redundant.size(), mMatrixPtr->cols() );
292  vector_t<Number> nonRedundantVector = vector_t<Number>::Zero( mMatrixPtr->rows() - redundant.size() );
293  int topUnfilledRow = 0;
294  auto it = redundant.begin();
295  for ( int i = 0; i < mMatrixPtr->rows(); ++i ) {
296  //std::cout << "TemplatePolyhedron::removeRedundancy, currentRow: " << i << " redundant.s: " << redundant.size() << " lastUnfilledRow: " << topUnfilledRow << std::endl;
297  //std::cout << "TemplatePolyhedron::removeRedundancy, nonRedundantMatrix: \n " << nonRedundantMatrix << "nonRedundantVector: \n" << nonRedundantVector << std::endl;
298  if ( it == redundant.end() || std::size_t( i ) != *it ) {
299  //std::cout << "TemplatePolyhedron::removeRedundancy, i: " << i << " != it: " << *it << ", add row!" << std::endl;
300  nonRedundantMatrix.row( topUnfilledRow ) = mMatrixPtr->row( i );
301  nonRedundantVector( topUnfilledRow ) = mVector( i );
302  topUnfilledRow++;
303  } else {
304  //std::cout << "TemplatePolyhedron::removeRedundancy, i: " << i << " == it: " << *it << ", pop back!" << std::endl;
305  ++it;
306  }
307  }
308  assert( it == redundant.end() );
309  mMatrixPtr = std::move( std::make_shared<matrix_t<Number>>( nonRedundantMatrix ) );
310  mVector = std::move( nonRedundantVector );
311  mOptimizer.setMatrix( *mMatrixPtr );
312  mOptimizer.setVector( mVector );
313  }
314  mNonRedundant = true;
315  //Emptiness should stay the same
316  }

318  //Time:
319  //If a plane is only a tangent plane to a point, then it is redundant either
320  //Assume polytope is convex and bounded: For each plane, shift it only a bit away, if point defined by intersection of three planes still inside, then plane redundant
321  //But there are still redundant planes left

334 template <typename Number, typename Converter, typename Setting>
335 std::pair<bool, bool> TemplatePolyhedronT<Number, Converter, Setting>::checkFullInsideFullOutside( const vector_t<Number>& normal, const Number& offset ) const {
336  //Quick check:
337  //If all coefficients of directions similar to hspace normal smaller than hspace offset, then current obj completely in hspace
338  //If all coefficients of directions not similar to hspace normal bigger than hspace offset, then current obj completely not in hspace
339  bool fullyInside = true;
340  bool fullyOutside = true;
341  for ( int i = 0; i < mMatrixPtr->rows(); ++i ) {
342  if ( normal.dot( mMatrixPtr->row( i ) ) > 0 ) {
343  if ( mVector( i ) > offset ) {
344  fullyInside = false;
345  }
346  } else {
347  if ( -mVector( i ) <= offset ) {
348  fullyOutside = false;
349  }
350  }
351  if ( !fullyInside && !fullyOutside ) break;
352  }
353  assert( !( fullyInside && fullyOutside ) );
354  return std::make_pair( fullyInside, fullyOutside );
355 }

460 TemplatePolyhedronT<Number, Converter, Setting> TemplatePolyhedronT<Number, Converter, Setting>::linearTransformation( const matrix_t<Number>& A ) const {
461  //Other Idea: Effectively, a linear transformation is only a scaling of the coeff vector, but how to find scaling factors?
462

467  assert( A.cols() == (int)dimension() );
468  assert( A.cols() == mMatrixPtr->cols() );
469  assert( A.rows() == mMatrixPtr->cols() );
470  if ( A.isApprox( matrix_t<Number>::Identity( dimension(), dimension() ) ) ) {
471  return *this;
472  }
473 
474  //Evaluate in the transformed directions
475  matrix_t<Number> dirsRotatedInverse = ( *mMatrixPtr ) * A;
476  //std::cout << "TemplatePolyhedron::linearTransformation, dirsRotatedInverse: \n" << dirsRotatedInverse << std::endl;
477  assert( dirsRotatedInverse.rows() == mMatrixPtr->rows() );
478  assert( dirsRotatedInverse.cols() == mMatrixPtr->cols() );
479  for ( int i = 0; i < dirsRotatedInverse.rows(); i++ ) {
480  COUNT( "Affine Transformation Evaluate" );
481  }
482  std::vector<EvaluationResult<Number>> evalInInvRotatedDirs = multiEvaluate( dirsRotatedInverse, false );
483  //std::vector<EvaluationResult<Number>> evalInInvRotatedDirs = multiEvaluate(dirsRotatedInverse, true);
484  //std::cout << "TemplatePolyhedron::linearTransformation, evalInInvRotatedDirs: {";
485  //for(auto& e : evalInInvRotatedDirs){
486  // //std::cout << e << ", ";
487  //}
488  //std::cout << "}" << std::endl;
489  vector_t<Number> newVector = vector_t<Number>::Zero( mVector.rows() );
490  //std::cout << "TemplatePolyhedron::linearTransformation, newVector before filling: \n" << newVector << std::endl;
491  assert( evalInInvRotatedDirs.size() == std::size_t( newVector.rows() ) );
492  assert( evalInInvRotatedDirs.size() == std::size_t( mVector.rows() ) );
493  for ( std::size_t i = 0; i < evalInInvRotatedDirs.size(); ++i ) {
494  assert( evalInInvRotatedDirs.at( i ).errorCode == SOLUTION::FEAS );

522 TemplatePolyhedronT<Number, Converter, Setting> TemplatePolyhedronT<Number, Converter, Setting>::minkowskiSum( const TemplatePolyhedronT<Number, Converter, Setting>& rhs ) const {
523  //Emptiness check is very costly and usually does not happen during reachability computation
524  if ( empty() ) return rhs;
525  if ( rhs.empty() ) return *this;
526 
527  //In case rhs has a different matrix than this, overapproximate the minkowskisum of both via template directions
528  //Costly, but shouldn't happen too often
529  assert( mMatrixPtr->cols() == rhs.matrix().cols() );
530  if ( mMatrixPtr->rows() != rhs.matrix().rows() || !mMatrixPtr->isApprox( rhs.matrix() ) ) {
531  auto rhsHPoly = typename Converter::HPolytope( rhs.matrix(), rhs.vector() );
532  auto thisHPoly = typename Converter::HPolytope( this->matrix(), this->vector() );
533  auto summed = thisHPoly.minkowskiSum( rhsHPoly );
534  //auto evalInDirs = summed.multiEvaluate(combineRows(computeTemplate<Number>(dimension(), 8)));
535  for ( int i = 0; i < mMatrixPtr->rows(); i++ ) {
536  COUNT( "Minkowski Sum Evaluate" );
537  }
538  auto evalInDirs = summed.multiEvaluate( *mMatrixPtr, false );
539  vector_t<Number> newVector = vector_t<Number>::Zero( evalInDirs.size() );
540  for ( std::size_t i = 0; i < evalInDirs.size(); ++i ) {
541  if ( evalInDirs.at( i ).errorCode == SOLUTION::FEAS ) {
542  newVector( i ) = evalInDirs.at( i ).supportValue;
543  } else {
544  //INFTY should not be possible since minkowskiSum of convex objects is also convex and therefore evaluable in all directions.
545  //An infeasible set could occur from minkowskiSum when both sets are empty, in which case we just return Empty()
546  assert( evalInDirs.at( i ).errorCode == SOLUTION::INFEAS );
547  return TemplatePolyhedronT<Number, Converter, Setting>::Empty( dimension() );
548  }
549  }
550  return TemplatePolyhedronT<Number, Converter, Setting>( mMatrixPtr, newVector, getNonRedundant() );
551  }
552 
553  //This only works if both TPolys have the same; if MinkowskiSum with a box, use smth else
554  vector_t<Number> newVector = vector_t<Number>::Zero( mMatrixPtr->rows() );
555  for ( int i = 0; i < mMatrixPtr->rows(); i++ ) {
556  newVector( i ) = this->vector()( i ) + rhs.vector()( i );
557  }
558  return TemplatePolyhedronT<Number, Converter, Setting>( mMatrixPtr, newVector, getNonRedundant() );
