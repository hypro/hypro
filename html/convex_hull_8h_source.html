42 template <typename Number>
43 static void initConvexHull( const std::vector<Point<Number>>& points, std::vector<std::shared_ptr<Facet<Number>>>& facets ) {
44  unsigned dimCheck = effectiveDimension( points );
45  unsigned dimension = points.at( 0 ).dimension();
46 
47  if ( dimCheck == dimension ) {
48  std::vector<Point<Number>> initialPoints;
49  unsigned minIndex = 0, maxIndex = 0; // determine min and max of first value
50 
51  for ( unsigned j = 0; j < points.size(); j++ ) {
52  if ( points.at( minIndex ).rawCoordinates()( 0 ) > points.at( j ).rawCoordinates()( 0 ) ) {
53  minIndex = j;
54  }
55  if ( points.at( maxIndex ).rawCoordinates()( 0 ) < points.at( j ).rawCoordinates()( 0 ) ) {
56  maxIndex = j;
57  }
58  }
59 
60  initialPoints.push_back( points.at( minIndex ) );
61  initialPoints.push_back( points.at( maxIndex ) );
62 
63  // search rest for rank d accordingly for each dimension
64  for ( unsigned d = 1; d < dimension; d++ ) {
65  for ( unsigned i = 0; i < points.size(); i++ ) {
66  if ( std::find( initialPoints.begin(), initialPoints.end(), points.at( i ) ) == initialPoints.end() ) {
67  if ( maxRank( initialPoints, points.at( i ) ) ) {
68  initialPoints.push_back( points.at( i ) );
69  break;
70  }
71  }
72  }
73  }
74 
75  // search for unused point as last member for initConvexhull
76  unsigned index = 0;
77  while ( initialPoints.size() < dimension + 1 && index < points.size() ) {
78  if ( std::find( initialPoints.begin(), initialPoints.end(), points.at( index ) ) == initialPoints.end() ) {
79  initialPoints.push_back( points.at( index ) );
80  }
81  index++;
82  }
83 
84  // inti facets
85  for ( unsigned i = 0; i < dimension + 1; i++ ) {
86  facets.emplace_back( std::make_shared<Facet<Number>>( Facet<Number>() ) );
87  }
88  for ( unsigned i = 0; i < dimension + 1; ++i ) {
89  std::vector<Point<Number>> points_for_facet;
90  for ( unsigned j = 0; j < dimension + 1; ++j ) {
91  if ( i != j ) {
92  points_for_facet.push_back( initialPoints.at( j ) );
93  }
94  }
95  facets.at( i )->setPoints( std::move( points_for_facet ), initialPoints.at( i ) );
96  }
97 
98  // Add neighbors
99  for ( unsigned i = 0; i < facets.size(); ++i ) {
100  for ( unsigned j = 0; j < facets.size(); ++j ) {
101  if ( i != j ) {
102  facets.at( i )->addNeighbor( facets.at( j ) );
103  }
104  }
105  }
106 
107  } else {
108  std::cout << __func__ << " : " << __LINE__ << "Error: Inconsistent point set " << std::endl;
109  assert( false );
110  facets = std::vector<std::shared_ptr<Facet<Number>>>();
111  }
112 }

486 template <typename Number>
487 static std::pair<std::vector<std::shared_ptr<Facet<Number>>>, std::map<Point<Number>, std::set<Point<Number>>>>
488 convexHull( const std::vector<Point<Number>>& pts ) {
489  // initialization: remove duplicate points
490  auto points = pts;
491  std::sort( points.begin(), points.end() );
492  std::unique( points.begin(), points.end() );
493 
494  if ( points.size() >= points.at( 0 ).dimension() + 1 ) {
495  std::vector<std::shared_ptr<Facet<Number>>> facets;
496  std::vector<Point<Number>> unassignedPoints, assignedPoints;
497 
498  // init facets and points
499  initConvexHull( points, facets );
500  pointsNotContainedInFacets( points, facets, unassignedPoints, assignedPoints ); // Determine points which belong to a facet
501 
502  for ( auto facet : facets ) {
503  for ( auto point : unassignedPoints ) {
504  if ( facet->isAbove( point ) ) { // isAbove
505  facet->addPointToOutsideSet( point );
506  }
507  }
508  }
509 
510  std::queue<std::shared_ptr<Facet<Number>>> workingSet;
511  removeBorderFacets( facets, workingSet ); // extract the facets to be examined
512 
513  /*
514  * ------------------------------------------------------------------------------
515  * MAIN BODY
516  * ------------------------------------------------------------------------------
517  */
518 
519  while ( !workingSet.empty() ) {
520  std::shared_ptr<Facet<Number>> currentFacet = workingSet.front(); // next facet
521  Point<Number> currentPoint = currentFacet->furthest_Point(); // next point
522 
523  std::vector<std::shared_ptr<Facet<Number>>> currentVisibleFacets;
524  currentVisibleFacets.push_back( currentFacet );
525  std::queue<std::shared_ptr<Facet<Number>>> newVisibleFacets;
526  bool changed = true;
527 
528  // determin horizon into currentVisibleFacets
529  while ( changed ) {
530  changed = false;
531  for ( const auto& facet : currentVisibleFacets ) {
532  assert( !facet->neighbors().empty() );
533  for ( const auto& neighbor : facet->neighbors() ) {
534  if ( neighbor->isAbove( currentPoint ) ) {
535  newVisibleFacets.push( neighbor ); // update new visible facets
536  assert( !neighbor->neighbors().empty() );
537  }
538  }
539  }
540  while ( !newVisibleFacets.empty() ) {
541  if ( std::find( currentVisibleFacets.begin(), currentVisibleFacets.end(), *newVisibleFacets.front() ) == currentVisibleFacets.end() ) {
542  currentVisibleFacets.push_back( newVisibleFacets.front() ); // insert the new visible facets into the current visible facets - if they are no duplicates
543  changed = true;
544  }
545  newVisibleFacets.pop(); // remove the new visible facet
546  }
547  }
548 
549  /* Stefan: Note that here, we have a set of visible facets - we need to get the ridges, which are between a
550  * visible facet and the next invisible
551  * facet (Horizon ridges). This means, that we have to take into account all ridges, which are built of one
552  * facet out of the visible facet set and one facet out of
553  * the neighbors set.
554  */
555  std::vector<std::shared_ptr<Facet<Number>>> newFacets;
556 
557  // Create new facets
558  for ( auto facet : currentVisibleFacets ) {
559  for ( auto ridge : getRidges( facet ) ) {
560  for ( auto neighbor : ridge.neighbors() ) {
561  if ( std::find( currentVisibleFacets.begin(), currentVisibleFacets.end(), *neighbor ) == currentVisibleFacets.end() ) { // we have a neighbor of this ridge, which is not visible -> horizon
562  //std::cout << "Found a candidate for a newFacet" << std::endl;
563  // ridge, create new facet
564  Point<Number> insidePoint = findInsidePoint( ridge, facet );
565 
566  std::shared_ptr<Facet<Number>> newFacet = std::make_shared<Facet<Number>>( ridge.vertices(), currentPoint, insidePoint );
567 
568  newFacet->addNeighbor( neighbor );
569  neighbor->addNeighbor( newFacet );
570 
571  newFacets.emplace_back( std::move( newFacet ) );
572  }
573  }
574  }
575  }
576 
577  // at this point we created all new facets from any horizon ridge to the current considered point
578  // (currentPoint). We need to set up the neighborhood
579  // relations in between the new facets.
580  // std::cout << __func__ << " New Facets before neighborhood: " << newFacets << std::endl;
581 
582  determineNeighbors( newFacets );
583 
584  // std::vector<Point<Number>> outsidePoints =
585  // polytope::currentPoints_outside_of_currentVisibleFacets(currentVisibleFacets);
586 
587  // update unassignedPoints -> assignedPoints
588  if ( std::find( unassignedPoints.begin(), unassignedPoints.end(), currentPoint ) != unassignedPoints.end() ) {
589  assignedPoints.push_back( currentPoint ); // assign currentPoint if not yet
590 
591  for ( unsigned i = 0; i < unassignedPoints.size(); i++ ) {
592  if ( unassignedPoints.at( i ) == currentPoint ) {
593  unassignedPoints.erase( unassignedPoints.begin() + i ); // erase currentPoint in unassignedPoints
594  break;
595  }
596  }
597  }
598 
599  // update outside sets of the new facets.
600  for ( auto facet : newFacets ) {
601  for ( auto point : unassignedPoints ) {
602  if ( facet->isAbove( point ) ) {
603  facet->addPointToOutsideSet( point );
604  }
605  }
606  }
