5 FormulaT<Number> FourierMotzkinQE<Number>::eliminateQuantifiers() {
6  DEBUG( "hypro.algorithms.qe", "Eliminate in: " << mFormula );
7  // iterate over query
8  for ( const auto& QuantifierVariablesPair : mQuery ) {
9  // we are ignoring the quantifier type
10  assert( QuantifierVariablesPair.first == QuantifierType::EXISTS );
11 
12  // eliminate one variable after the other
13  for ( const auto& var : QuantifierVariablesPair.second ) {
14  DEBUG( "hypro.algorithms.qe", "Elimiate " << var );
15 
16  auto bounds = findBounds( var );
17  // combine all lower-upper bound pairs.
18  FormulasT<Number> newConstraints;
19  if ( !bounds[2].empty() ) {
20  DEBUG( "hypro.algorithms.qe", "Found equation(s), substitute." );
21  newConstraints = substituteEquations( bounds, var );
22  } else {
23  DEBUG( "hypro.algorithms.qe", "Create lower-upper bound pairs." );
24  newConstraints = createNewConstraints( bounds, var );
25  }
26 
27  // add all constraints which are not containing var to newConstraints
28  for ( const auto formulaIt : bounds[3] ) {
29  assert( ( formulaIt ).getType() == carl::FormulaType::CONSTRAINT );
30  newConstraints.emplace_back( formulaIt );
31  }
32 
33  // assemble new formula
34  mFormula = FormulaT<Number>( carl::FormulaType::AND, newConstraints );
35  DEBUG( "hypro.algorithms.qe", "Done elimiating " << var << ", new formula size: " << mFormula.size() );
36  }
37  }
38 
39  DEBUG( "hypro.algorithms.qe", "Formula after elimination: " << mFormula );
40 
41  return mFormula;
42 }

44 template <typename Number>
45 typename FourierMotzkinQE<Number>::FormulaPartition FourierMotzkinQE<Number>::findBounds(
46  const carl::Variable& variable ) {
47  // result vector initialized with three subsets
48  typename FourierMotzkinQE<Number>::FormulaPartition res{ 4, std::vector<FormulaT<Number>>() };
49 
50  // if the formula only contains one constraint, check for occurence of the variable.
51  if ( mFormula.getType() == carl::FormulaType::CONSTRAINT ) {
52  if ( !mFormula.constraint().hasVariable( variable ) ) {
53  res[3].push_back( mFormula );
54  } else {
55  if ( mFormula.constraint().relation() == carl::Relation::EQ ) {
56  res[2].push_back( mFormula );
57  } else if ( isLinearLowerBound( mFormula.constraint(), variable ) ) {
58  res[0].push_back( mFormula );
59  } else {
60  res[1].push_back( mFormula );
61  }
62  }
63  return res;
64  }
65 
66  // More than one constaint: search formula to find bounds
67  for ( auto formulaIt = mFormula.begin(); formulaIt != mFormula.end(); ++formulaIt ) {
68  assert( ( *formulaIt ).getType() == carl::FormulaType::CONSTRAINT );
69  if ( ( *formulaIt ).constraint().hasVariable( variable ) ) {
70  if ( ( *formulaIt ).constraint().relation() == carl::Relation::EQ ) {
71  res[2].push_back( *formulaIt );
72  } else if ( isLinearLowerBound( ( *formulaIt ).constraint(), variable ) ) {
73  res[0].push_back( *formulaIt );
74  } else {
75  res[1].push_back( *formulaIt );
76  }
77  } else {
78  res[3].push_back( *formulaIt );
79  }
80  }
81 
82  return res;
83 }

86 FormulasT<Number> FourierMotzkinQE<Number>::createNewConstraints(
87  const typename FourierMotzkinQE<Number>::FormulaPartition& bounds, carl::Variable v ) {
88  FormulasT<Number> constraints;
89 
90  // combine all pairs of lower and upper bounds.
91  for ( const auto& lowerBnd : bounds[0] ) {
92  for ( const auto& upperBnd : bounds[1] ) {
93  TRACE( "hypro.algorithms.qe", "Combine " << ( lowerBnd ) << " and " << ( upperBnd ) );
94 
95  PolyT<Number> lhs = getRemainder( lowerBnd.constraint(), v, true );
96 
97  TRACE( "hypro.algorithms.qe", "Lhs is " << lhs );
98 
99  PolyT<Number> rhs = getRemainder( upperBnd.constraint(), v, false );
100 
101  TRACE( "hypro.algorithms.qe", "Rhs is " << rhs );
102 
103  constraints.emplace_back( FormulaT<Number>( ConstraintT<Number>( lhs - rhs, carl::Relation::LEQ ) ) );
104 
105  TRACE( "hypro.algorithms.qe", "Created new constraint: " << constraints.back() );
106  }
107  }
108 
109  // copy the rest
110  for ( const auto& c : bounds[2] ) {
111  constraints.emplace_back( c );
112  }
113  for ( const auto& c : bounds[3] ) {
114  constraints.emplace_back( c );
115  }
116 
117  return constraints;

120 template <typename Number>
121 FormulasT<Number> FourierMotzkinQE<Number>::substituteEquations(
122  const typename FourierMotzkinQE<Number>::FormulaPartition& bounds, carl::Variable v ) {
123  assert( !bounds[2].empty() );
124  FormulasT<Number> constraints;
125 
126  // check if equations are pairwise equal - if not return false, otherwise use one of the equations.
127  // Todo: This test is not correct: having x = 3 + y and x = 3 would fail this test but still is satisfiable when y
128  // is existentially quantified or if it is all-quantified it is after x.
129  FormulasT<Number> substitutes;
130  if ( bounds[2].size() == 1 ) {
131  substitutes.push_back( FormulaT<Number>( bounds[2].front() ) );
132  } else {
133  for ( auto f = bounds[2].begin(); f != bounds[2].end(); ++f ) {
134  assert( f->getType() == carl::FormulaType::CONSTRAINT );
135  assert( f->constraint().relation() == carl::Relation::EQ );
136 
137  for ( auto g = std::next( f ); g != bounds[2].end(); ++g ) {
138  assert( g->getType() == carl::FormulaType::CONSTRAINT );
139  assert( g->constraint().relation() == carl::Relation::EQ );
140  FormulaT<Number> newEq =
141  FormulaT<Number>( f->constraint().lhs() - g->constraint().lhs(), carl::Relation::EQ );
142 
143  // 1st case: equalities contradict each other - quit substitution.
144  if ( newEq.getType() == carl::FormulaType::FALSE ) {
145  constraints.clear();
146  constraints.emplace_back( FormulaT<Number>( carl::FormulaType::FALSE ) );
147  return constraints;
148  } else if ( newEq.getType() == carl::FormulaType::TRUE ) {
149  // both equations are completely equal, just take one.
150  substitutes.push_back( FormulaT<Number>( *f ) );
151  } else {
152  // the equations are no equal but involve different variables, e.g. [eliminate x]: x = y AND x = z
153  substitutes.push_back( FormulaT<Number>( *f ) );
154  substitutes.push_back( FormulaT<Number>( *g ) );
155  // Todo: We could add the constraint y = z.
156  }
157  }
158  }
159  }
160 
161  // substitute
162  for ( const auto& f : substitutes ) {
163  // std::cout << "Substitute: " << f << std::endl;
164  assert( f.getType() == carl::FormulaType::CONSTRAINT );
165  PolyT<Number> substitute = -( f.constraint().lhs() - f.constraint().coefficient( v, 1 ) * v );
166  // lower bounds
167  for ( auto fc : bounds[0] ) {
168  assert( fc.getType() == carl::FormulaType::CONSTRAINT );
169  // constraints.emplace_back( carl::substitute( fc.constraint().lhs(), v, substitute ),
170  constraints.emplace_back( fc.constraint().lhs().substitute( v, substitute ),
171  fc.constraint().relation() );
172  // std::cout << "substitute lower bound to " << constraints.back() << std::endl;
173  }
174  // upper bounds
175  for ( auto fc : bounds[1] ) {
176  assert( fc.getType() == carl::FormulaType::CONSTRAINT );
177  // constraints.emplace_back( carl::substitute( fc.constraint().lhs(), v, substitute ),
178  constraints.emplace_back( fc.constraint().lhs().substitute( v, substitute ),
179  fc.constraint().relation() );
180  // std::cout << "substitute upper bound to " << constraints.back() << std::endl;
181  }
182  }
183 
184  return constraints;
185 }

208 template <typename Number>
209 bool FourierMotzkinQE<Number>::isLinearLowerBound( const ConstraintT<Number>& c, carl::Variable v ) {
210  assert( c.hasVariable( v ) );
211  assert( c.coefficient( v, 1 ).isNumber() );
212 
213  // is linear lower bound when the coefficient is > 0 and the relation is LEQ or LESS, or if the coefficient is < 0
214  // and the relation is GEQ or GREATER.
215  if ( ( ( c.relation() == carl::Relation::LEQ || c.relation() == carl::Relation::LESS ) &&
216  c.coefficient( v, 1 ) < 0 ) ||
217  ( ( c.relation() == carl::Relation::GEQ || c.relation() == carl::Relation::GREATER ) &&
218  c.coefficient( v, 1 ) > 0 ) ) {
219  // std::cout << c << " is lower bound." << std::endl;
220  return true;
221  }
222  // std::cout << c << " is no lower bound." << std::endl;
223  return false;
