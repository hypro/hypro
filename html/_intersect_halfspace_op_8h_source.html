94  //Method described in LeGuernic's and Girard's paper about support functions, but extended to halfspaces and not only hyperplanes
95  std::vector<EvaluationResult<Number>> aggregate( std::vector<std::vector<EvaluationResult<Number>>>& resultStackBack, const matrix_t<Number>& directions ) const {
96  COUNT( "IntersectHalfspaceOp::aggregate" );
97  TRACE( "hypro.representations.supportFunction", ": INTERSECT HSPACE, accumulate results." );
98  assert( resultStackBack.size() == 1 );
99  std::vector<EvaluationResult<Number>> accumulatedRes;
100  for ( int i = 0; i < directions.rows(); ++i ) {
101  //If direction not similar to halfspace normal, then use halfspace evaluate
102  if ( hspace.normal().dot( directions.row( i ) ) < 0 ) {
103  accumulatedRes.push_back( resolveEvaluationResults( resultStackBack.front().at( i ), directions.row( i ) ) );
104  } else {
105  accumulatedRes.push_back( leGuernicFast( vector_t<Number>( directions.row( i ).transpose() ) ) );
106  }
107  }
108  return accumulatedRes;
109  }

111  //Take the smaller evaluation result between the one from child and the hspace
112  EvaluationResult<Number> resolveEvaluationResults( const EvaluationResult<Number>& other, const vector_t<Number>& dir ) const {
113  COUNT( "IntersectHalfspaceOp::resolveEvaluationResults" );
114  auto hspaceEval = hspace.evaluate( dir, true );
115  if ( other.errorCode == SOLUTION::FEAS ) {
116  //Since the first result we compare with can still be infty
117  if ( hspaceEval.errorCode == SOLUTION::INFTY ) {
118  hspaceEval.errorCode = SOLUTION::FEAS;
119  hspaceEval.supportValue = other.supportValue;
120  hspaceEval.optimumValue = other.optimumValue;
121  } else {
122  hspaceEval.errorCode = SOLUTION::FEAS;
123  hspaceEval.supportValue = other.supportValue < hspaceEval.supportValue ? other.supportValue : hspaceEval.supportValue;
124  hspaceEval.optimumValue = other.optimumValue < hspaceEval.optimumValue ? other.optimumValue : hspaceEval.optimumValue;
125  }
126  }
127  return hspaceEval;
128  }

130  //Computes the support value of the intersection between our child and the member halfspace in direction "direction"
131  EvaluationResult<Number> leGuernic( const vector_t<Number>& direction ) const {
132  COUNT( "IntersectHalfspaceOp::leGuernic" );
133 
134  //Projection matrix - normalize columns then put them into matrix
135  //NOTE: This is the already transposed matrix M^T
136  matrix_t<Number> projMat = matrix_t<Number>::Zero( direction.rows(), 2 );
137  projMat.col( 0 ) = hspace.normal().transpose();
138  projMat.col( 1 ) = direction.transpose();
139 
140  //Wiggle vector and wiggle angle
141  Number wiggleAngle = 0.0;
142  vector_t<Number> wiggleVec = vector_t<Number>::Zero( 2 );
143  wiggleVec( 0 ) = Number( std::cos( carl::toDouble( wiggleAngle ) ) );
144  wiggleVec( 1 ) = Number( std::sin( carl::toDouble( wiggleAngle ) ) );
145 
146  //Golden section search for the value f
147  Number maximumAngle = PI_UP;
148  Number tolerance = 0.001;
149  Number goldenRatio = ( std::sqrt( 5 ) + 1 ) / 2;
150  Number upper = maximumAngle - ( ( maximumAngle - wiggleAngle ) / goldenRatio );
151  Number lower = wiggleAngle + ( ( maximumAngle - wiggleAngle ) / goldenRatio );
152  EvaluationResult<Number> evalInProjectedWiggleDir;
153  SupportFunctionNewT<Number, Converter, Setting> childSF( this->getChildren().at( 0 ) );
154 
155  //Get absolute value of difference
156  Number absDiff = upper - lower;
157  if ( absDiff <= 0 ) {
158  absDiff *= Number( -1 );
159  }
160 
161  while ( absDiff > tolerance ) {
162  COUNT( "IntersectHalfspaceOp::wobbleIterations" );
163 
164  //Compute evaluation value for upper bound
165  wiggleVec( 0 ) = Number( std::cos( carl::toDouble( upper ) ) );
166  wiggleVec( 1 ) = Number( std::sin( carl::toDouble( upper ) ) );
167  evalInProjectedWiggleDir = childSF.evaluate( projMat * wiggleVec, true );
168  COUNT( "IntersectHalfspaceOp::leGuernic::evaluate" );
169  Number heightOfUpper = ( evalInProjectedWiggleDir.supportValue - hspace.offset() * Number( std::cos( carl::toDouble( upper ) ) ) ) / Number( std::sin( carl::toDouble( upper ) ) );
170 
171  //Compute evaluation value for lower bound
172  wiggleVec( 0 ) = Number( std::cos( carl::toDouble( lower ) ) );
173  wiggleVec( 1 ) = Number( std::sin( carl::toDouble( lower ) ) );
174  evalInProjectedWiggleDir = childSF.evaluate( projMat * wiggleVec, true );
175  COUNT( "IntersectHalfspaceOp::leGuernic::evaluate" );
176  Number heightOfLower = ( evalInProjectedWiggleDir.supportValue - hspace.offset() * Number( std::cos( carl::toDouble( lower ) ) ) ) / Number( std::sin( carl::toDouble( lower ) ) );
177 
178  //Reduce search space to where minimum of evaluation values seems to be
179  if ( heightOfLower < heightOfUpper ) {
180  wiggleAngle = upper;
181  } else {
182  maximumAngle = lower;
183  }
184 
185  //Update upper and lower bound
186  upper = maximumAngle - ( ( maximumAngle - wiggleAngle ) / goldenRatio );
187  lower = wiggleAngle + ( ( maximumAngle - wiggleAngle ) / goldenRatio );
188  absDiff = upper - lower;
189  if ( absDiff <= 0 ) {
190  absDiff *= Number( -1 );
191  }
192  }
193  //Resulting angle is the average of maximumAngle and wiggleAngle
194  Number resultAngle = ( maximumAngle + wiggleAngle ) / 2;
195  assert( 0 <= resultAngle && resultAngle <= PI_UP );
196  wiggleVec( 0 ) = Number( std::cos( carl::toDouble( resultAngle ) ) );
197  wiggleVec( 1 ) = Number( std::sin( carl::toDouble( resultAngle ) ) );
198  evalInProjectedWiggleDir = childSF.evaluate( projMat * wiggleVec, true );
199  COUNT( "IntersectHalfspaceOp::leGuernic::evaluate" );
200  Number heightOfResult = ( evalInProjectedWiggleDir.supportValue - hspace.offset() * Number( std::cos( carl::toDouble( resultAngle ) ) ) ) / Number( std::sin( carl::toDouble( resultAngle ) ) );
201  //NOTE: This evaluation result does not return an optimal value
202  return EvaluationResult<Number>( heightOfResult, SOLUTION::FEAS );
203  }

205  //Golden section search where the amount of evaluations made is halved
206  //NOTE: algorithm idea is from wikipedia "Golden section search", the variable names in the comments are mappings to the variable names in the article
207  EvaluationResult<Number> leGuernicFast( const vector_t<Number>& direction ) const {
208  COUNT( "IntersectHalfspaceOp::leGuernic" );
209 
210  //Projection matrix - normalize columns then put them into matrix
211  //NOTE: This is the already transposed matrix M^T
212  matrix_t<Number> projMat( direction.rows(), 2 );
213  projMat.col( 0 ) = hspace.normal().transpose();
214  projMat.col( 1 ) = direction.transpose();
215 
216  //Wiggle vector and wiggle angle
217  Number wiggleAngle = 0.0;
218  vector_t<Number> wiggleVec( 2 );
219 
220  //Golden section search for the value f
221  Number maximumAngle = PI_UP;
222  const Number tolerance = 0.001;
223  const Number goldenRatioInv = ( std::sqrt( 5 ) - 1 ) / 2;
224  const Number goldenRatioInv2 = ( 3 - std::sqrt( 5 ) ) / 2;
225  assert( wiggleAngle < maximumAngle );
226  carl::Interval<Number> resInterval( wiggleAngle, maximumAngle ); //upper = b, lower = a
227 
228  //Determine difference
229  Number h = PI_UP;
230 
231  //Compute steps needed to acquire tolerance
232  unsigned steps = static_cast<unsigned>( carl::ceil( std::log( carl::toDouble( tolerance ) / carl::toDouble( h ) ) / std::log( carl::toDouble( goldenRatioInv ) ) ) );
233 
234  //Compute upper (c) and lower bound (d)
235  Number upper = resInterval.lower() + goldenRatioInv2 * h;
236  Number lower = resInterval.lower() + goldenRatioInv * h;
237 
238  //Compute upper height yc
239  EvaluationResult<Number> evalInProjectedWiggleDir;
240  SupportFunctionNewT<Number, Converter, Setting> childSF( this->getChildren().at( 0 ) );
241  wiggleVec( 0 ) = Number( std::cos( carl::toDouble( upper ) ) );
242  wiggleVec( 1 ) = Number( std::sin( carl::toDouble( upper ) ) );
243  evalInProjectedWiggleDir = childSF.evaluate( projMat * wiggleVec, true );
244  COUNT( "IntersectHalfspaceOp::leGuernic::evaluate" );
245  Number heightOfUpper = ( evalInProjectedWiggleDir.supportValue - hspace.offset() * Number( std::cos( carl::toDouble( upper ) ) ) ) / Number( std::sin( carl::toDouble( upper ) ) );
246 
247  //Compute lower height yd
248  wiggleVec( 0 ) = Number( std::cos( carl::toDouble( lower ) ) );
249  wiggleVec( 1 ) = Number( std::sin( carl::toDouble( lower ) ) );
250  evalInProjectedWiggleDir = childSF.evaluate( projMat * wiggleVec, true );
251  COUNT( "IntersectHalfspaceOp::leGuernic::evaluate" );
252  Number heightOfLower = ( evalInProjectedWiggleDir.supportValue - hspace.offset() * Number( std::cos( carl::toDouble( lower ) ) ) ) / Number( std::sin( carl::toDouble( lower ) ) );
253 
254  //Update interval
255  for ( unsigned i = 0; i < steps; ++i ) {
256  COUNT( "IntersectHalfspaceOp::wobbleIterations" );
257  if ( heightOfUpper < heightOfLower ) {
258  resInterval.setUpper( lower );
259  lower = upper;
260  heightOfLower = heightOfUpper;
261  h = goldenRatioInv * h;
262  upper = resInterval.lower() + goldenRatioInv2 * h;
263  wiggleVec( 0 ) = Number( std::cos( carl::toDouble( upper ) ) );
264  wiggleVec( 1 ) = Number( std::sin( carl::toDouble( upper ) ) );
265  evalInProjectedWiggleDir = childSF.evaluate( projMat * wiggleVec, true );
266  COUNT( "IntersectHalfspaceOp::leGuernic::evaluate" );
267  heightOfUpper = ( evalInProjectedWiggleDir.supportValue - hspace.offset() * Number( std::cos( carl::toDouble( upper ) ) ) ) / Number( std::sin( carl::toDouble( upper ) ) );
268  } else {
269  resInterval.setLower( upper );
270  upper = lower;
271  heightOfUpper = heightOfLower;
272  h = goldenRatioInv * h;
273  lower = resInterval.lower() + goldenRatioInv * h;
274  wiggleVec( 0 ) = Number( std::cos( carl::toDouble( lower ) ) );
275  wiggleVec( 1 ) = Number( std::sin( carl::toDouble( lower ) ) );
276  evalInProjectedWiggleDir = childSF.evaluate( projMat * wiggleVec, true );
277  COUNT( "IntersectHalfspaceOp::leGuernic::evaluate" );
278  heightOfLower = ( evalInProjectedWiggleDir.supportValue - hspace.offset() * Number( std::cos( carl::toDouble( lower ) ) ) ) / Number( std::sin( carl::toDouble( lower ) ) );
279  }
280  }
281 
282  return EvaluationResult<Number>( ( ( heightOfLower + heightOfUpper ) / 2 ), vector_t<Number>::Zero( getDimension() ), SOLUTION::FEAS );
283  }

286  bool empty( const std::vector<bool>& childrenEmpty ) const {
287  assert( childrenEmpty.size() == 1 );
288  if ( RootGrowNode<Number, Converter, Setting>::mEmptyState != SETSTATE::UNKNOWN ) {
289  return ( RootGrowNode<Number, Converter, Setting>::mEmptyState == SETSTATE::EMPTY );
290  }
291  if ( childrenEmpty.front() ) {
292  RootGrowNode<Number, Converter, Setting>::mEmptyState = SETSTATE::EMPTY;
293  return true;
294  }
295  SupportFunctionNewT<Number, Converter, Setting> child( this->getChildren().at( 0 ) );
296  auto posNormalEval = child.evaluate( hspace.normal(), true );
297  auto negNormalEval = child.evaluate( -hspace.normal(), true );
298  if ( -negNormalEval.supportValue <= hspace.offset() && hspace.offset() <= posNormalEval.supportValue ) {
299  RootGrowNode<Number, Converter, Setting>::mEmptyState = SETSTATE::NONEMPTY;
300  return false;
301  }
302  RootGrowNode<Number, Converter, Setting>::mEmptyState = SETSTATE::EMPTY;
303  return true;
