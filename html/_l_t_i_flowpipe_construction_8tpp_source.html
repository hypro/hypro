64 ErrorBoxes<Number> constructErrorBoxes( Number const& delta, matrix_t<Number> const& flowMatrix, Representation const& initialSet, matrix_t<Number> const& trafoMatrix, Box<Number> const& externalInput ) {
65  if ( initialSet.empty() ) return {};
66 
67  assert( initialSet.dimension() == externalInput.dimension() );
68 
69  size_t dimension = flowMatrix.cols();
70  matrix_t<Number> matrixBlock = matrix_t<Number>::Zero( 3 * dimension, 3 * dimension );
71 
72  matrixBlock.block( 0, 0, dimension, dimension ) = flowMatrix.cwiseAbs(); //Upper left block
73  matrixBlock.block( 0, dimension, dimension, dimension ) = matrix_t<Number>::Identity( dimension, dimension ); //Upper middle block
74  matrixBlock.block( dimension, 2 * dimension, dimension, dimension ) = matrix_t<Number>::Identity( dimension, dimension ); //Middle right block
75 
76  matrixBlock *= delta;
77 
78  matrix_t<double> matrixBlockDouble = convert<Number, double>( matrixBlock );
79  matrixBlockDouble = matrixBlockDouble.exp();
80  matrixBlock = convert<double, Number>( matrixBlockDouble );
81 
82  matrix_t<Number> AMatrix = flowMatrix * ( matrix_t<Number>::Identity( dimension, dimension ) - trafoMatrix );
83 
84  //Transform initial set by AMatrix
85  Representation transformedInitialSetA = initialSet.affineTransformation( AMatrix.block( 0, 0, dimension - 1, dimension - 1 ), AMatrix.block( 0, dimension - 1, dimension - 1, 1 ) );
86 
87  Box<Number> boundingBoxA = Converter<Number>::template toBox<BoxLinearOptimizationOn>( transformedInitialSetA );
88 
89  if ( boundingBoxA.empty() ) return {};
90 
91  //Augment box by a dimension for constants
92  boundingBoxA.insert( { 1, 1 } );
93 
94  boundingBoxA = boundingBoxA.makeSymmetric();
95  assert( boundingBoxA.isSymmetric() );
96 
97  //Transform by upper middle block
98  boundingBoxA = boundingBoxA.linearTransformation( matrixBlock.block( 0, dimension, dimension, dimension ) );
99 
100  matrix_t<Number> BMatrix = flowMatrix * flowMatrix * trafoMatrix;
101 
102  Representation transformedInitialSetB = initialSet.affineTransformation( AMatrix.block( 0, 0, dimension - 1, dimension - 1 ), AMatrix.block( 0, dimension - 1, dimension - 1, 1 ) );
103 
104  Box<Number> boundingBoxB = Converter<Number>::template toBox<BoxLinearOptimizationOn>( transformedInitialSetB );
105 
106  //Augment box by a dimension for constants
107  boundingBoxB.insert( { 1, 1 } );
108 
109  boundingBoxB = boundingBoxB.makeSymmetric();
110  assert( boundingBoxB.isSymmetric() );
111 
112  //Transform by upper right block
113  boundingBoxB = boundingBoxB.linearTransformation( matrixBlock.block( 0, 2 * dimension, dimension, dimension ) );
114 
115  Box<Number> errorBoxX0 = boundingBoxA.minkowskiSum( boundingBoxB );
116 
117  //Remove augmentation via projection
118  std::vector<size_t> dims( errorBoxX0.dimension() - 1 );
119  std::iota( dims.begin(), dims.end(), 0 ); //fill with 0,1,...
120  errorBoxX0 = errorBoxX0.projectOn( dims );

123

129

137 std::vector<TimedValuationSet<Representation>> aggregate( size_t segmentsPerBlock, std::vector<IndexedValuationSet<Representation>>& enabled ) {
138  std::vector<TimedValuationSet<Representation>> aggregated{};
139 
140  if ( enabled.empty() ) return aggregated;
141 
142  size_t blockCount = 0;
143  for ( auto& [set, ind] : enabled ) {
144  if ( blockCount == 0 ) {
145  //beginning of block
146  aggregated.emplace_back( TimedValuationSet<Representation>{ set, carl::Interval{ ind, ind } } );
147  } else {
148  //in a block
149  aggregated.back().valuationSet = aggregated.back().valuationSet.unite( set );
150  }
151  blockCount += 1;
152 
153  //block is now full, finish previous and start next one
154  if ( blockCount == segmentsPerBlock ) {
155  aggregated.back().time.setUpper( ind );
156  blockCount = 0;
157  }
158  }
159  // closure of last block which might be incomplete
160  aggregated.back().time.setUpper( enabled.back().index );
161 
162  return aggregated;
163 }
