76 template <typename Number>
77 std::optional<EventTimingContainer<Number>> EventTimingProvider<Number>::getTimings( const Path<Number, tNumber>& path ) const {
78  TRACE( "hypro.datastructures.timing", "Path: " << path );
79 
80  // abort if we do not have a tree yet
81  if ( mRoot->getChildren().size() == 0 ) {
82  TRACE( "hypro.datastructures.timing", "No child nodes, abort." );
83  return std::nullopt;
84  }
85 
86  // pointers to nodes on the path
87  std::vector<std::set<EventTimingNode<Number>*>> workingSets;
88  // stores candidates which match the timeStep
89  std::vector<std::set<EventTimingNode<Number>*>> transitionCandidates;
90 
91  // fill working set with initial nodes
92  for ( const auto nPtr : mRoot->getChildren() ) {
93  workingSets.emplace_back( std::set<EventTimingNode<Number>*>{} );
94  workingSets.back().emplace( nPtr );
95  // prepare structure of 2nd set
96  transitionCandidates.emplace_back( std::set<EventTimingNode<Number>*>{} );
97  }
98 
99  // iterate over path / follow the path
100  auto pathIt = path.begin();
101  while ( pathIt != path.end() ) {
102  assert( workingSets.size() == transitionCandidates.size() );
103  TRACE( "hypro.datastructures.timing", "Follow time step, have " << workingSets.size() << " path-candidates." );
104  // remove all nodes from working set whose invariant timings do not contain the time interval in the path
105  assert( !pathIt->isDiscreteStep() );
106  // verify against invariant timings.
107  // Two-phase approach: Collect all which satisfy the invariant somehow,
108  // then merge invariant intervals and check whether the invariant was satisfied the whole time.
109 
110  // use flag to exit loop early
111  bool allEmpty = true;
112 
113  for ( std::size_t setIndex = 0; setIndex < workingSets.size(); ++setIndex ) {
114  TRACE( "hypro.datastructures.timing", "Set " << setIndex );
115  for ( auto nodeIt = workingSets[setIndex].begin(); nodeIt != workingSets[setIndex].end(); ) {
116  allEmpty = false;
117  // check for full containment
118  if ( ( *nodeIt )->getTimings().satisfiedInvariant( pathIt->getTimestamp() ) ) {
119  TRACE( "hypro.datastructures.timing", "Time step matching node " << *nodeIt );
120  transitionCandidates[setIndex].insert( *nodeIt );
121  nodeIt = workingSets[setIndex].erase( nodeIt );
122  } else if ( ( *nodeIt )->getTimings().getInvariantTimings().intersectsEntry( pathIt->getTimestamp(), CONTAINMENT::FULL ) ||
123  ( *nodeIt )->getTimings().getInvariantTimings().intersectsEntry( pathIt->getTimestamp(), CONTAINMENT::PARTIAL ) ||
124  ( *nodeIt )->getTimings().getInvariantTimings().intersectsEntry( pathIt->getTimestamp(), CONTAINMENT::YES ) ) {
125  // check for partial containment
126  TRACE( "hypro.datastructures.timing", "Time step matching node (partial) " << *nodeIt );
127  transitionCandidates[setIndex].insert( *nodeIt );
128  nodeIt = workingSets[setIndex].erase( nodeIt );
129  } else {
130  assert( !( *nodeIt )->getTimings().hasInvariantEvent( pathIt->getTimestamp(), CONTAINMENT::YES ) );
131  assert( !( *nodeIt )->getTimings().hasInvariantEvent( pathIt->getTimestamp(), CONTAINMENT::PARTIAL ) );
132  assert( !( *nodeIt )->getTimings().hasInvariantEvent( pathIt->getTimestamp(), CONTAINMENT::FULL ) );
133  // the node does not satisfy the invariant, erase
134  nodeIt = workingSets[setIndex].erase( nodeIt );
135  }
136  }
137  assert( workingSets[setIndex].empty() );
138  }
139 
140  assert( workingSets.size() == transitionCandidates.size() );
141 
142  if ( allEmpty ) {
143  return std::nullopt;
144  }
145 
146  // merge invariant intervals to detect whether the invariant interval is fully covered
147  for ( auto& subset : transitionCandidates ) {
148  if ( !subset.empty() ) {
149  TRACE( "hypro.datastructures.timing", "Merge invariant intervals of " << subset.size() << " candidates." );
150  TRACE( "hypro.datastructures.timing", "Merge invariant of node " << ( *subset.begin() ) );
151  // we want to unify the intervals - this is realized by reversing the order.
152  auto currentOrder = ( *subset.begin() )->getTimings().getInvariantTimings().getTypeOrder();
153  auto reversedOrder = currentOrder;
154  // reverse order to obtain weakest information by using merge(..)
155  std::reverse( ++reversedOrder.end(), reversedOrder.end() );
156 
157  HierarchicalIntervalVector<CONTAINMENT, tNumber> mergedHIV( ( *subset.begin() )->getTimings().getInvariantTimings() );
158  mergedHIV.setTypeOrder( reversedOrder );
159  // merge all invariant information
160  for ( auto it = ++subset.begin(); it != subset.end(); ++it ) {
161  TRACE( "hypro.datastructures.timing", "Merge invariant of node " << ( *it ) );
162  auto tmp = ( *it )->getTimings().getInvariantTimings();
163  tmp.setTypeOrder( reversedOrder );
164  mergedHIV = merge( {mergedHIV, tmp} );
165  }
166  //restore order
167  mergedHIV.setTypeOrder( currentOrder );
168 
169  TRACE( "hypro.datastructures.timing", "Invariant HIV after merge: " << mergedHIV );
170 
171  // if the invariant cannot be fully covered (information missing), remove subset (clear).
172  if ( !positiveCover( mergedHIV, pathIt->getTimestamp() ) ) {
173  subset.clear();
174  TRACE( "hypro.datastructures.timing", "Clear subset, no full cover detected." );
175  assert( subset.empty() );
176  }
177  }
178  }
179 
180  assert( workingSets.size() == transitionCandidates.size() );
181 
182  // follow the discrete jump, if applicable
183  ++pathIt;
184  if ( pathIt != path.end() ) {
185  // set up new working set
186  TRACE( "hypro.datastructures.timing", "Follow discrete step, have " << transitionCandidates.size() << " sets." );
187 
188  // try to follow the transition
189  assert( pathIt->isDiscreteStep() );
190  for ( std::size_t setIndex = 0; setIndex < transitionCandidates.size(); ++setIndex ) {
191  assert( workingSets[setIndex].empty() );
192  TRACE( "hypro.datastructures.timing", "Consider " << transitionCandidates[setIndex].size() << " nodes." );
193  for ( const auto nPtr : transitionCandidates[setIndex] ) {
194  TRACE( "hypro.datastructures.timing", "Search children of node " << nPtr );
195  for ( const auto chPtr : nPtr->getChildren() ) {
196  if ( chPtr->getEntryTransition() == pathIt->transition && carl::set_have_intersection( chPtr->getEntryTimestamp(), pathIt->getTimestamp() ) ) {
197  workingSets[setIndex].insert( chPtr );
198  TRACE( "hypro.datastructures.timing", "Disrete step matching node " << chPtr << "(now have " << workingSets[setIndex].size() << ")." );
199  }
200  }
201  }
202  // prepare for next iteration.
203  transitionCandidates[setIndex].clear();
204  }
205  // increase pathIt to the next time step or the end of the path
206  ++pathIt;
207  } else {
208  // the last element in the path was a time step, move nodes to workingSet
209  TRACE( "hypro.datastructures.timing", "Copy transitionCandidates to workingSet." );
210  workingSets = transitionCandidates;
211  }
212  }
213  assert( workingSets.size() == transitionCandidates.size() );
214  assert( workingSets.size() == mRoot->getChildren().size() );
215 
216  TRACE( "hypro.datastructures.timing", "After loop, have " << workingSets.size() << " subsets." );
217 
218  // at this point workingSet should contain all nodes which can be reached following the path. Merge/intersect those and get the most precise.
219  bool allEmpty = true;
220  std::vector<EventTimingContainer<Number>> candidates;
221  for ( const auto& set : workingSets ) {
222  if ( !set.empty() ) {
223  TRACE( "hypro.datastructures.timing", "Unite timings of " << set.size() << " containers." );
224  allEmpty = false;
225  auto nodeIt = set.begin();
226  EventTimingContainer<Number> res = ( *nodeIt )->getTimings();
227  ++nodeIt;
228  for ( ; nodeIt != set.end(); ++nodeIt ) {
229  res = unite( {res, ( *nodeIt )->getTimings()} );
230  }
231  TRACE( "hypro.datastructures.timing", "Merge result: " << res );
232  candidates.emplace_back( std::move( res ) );
233  }
234  }
235 
236  if ( allEmpty ) {
237  TRACE( "hypro.datastructures.timing", "All sets are empty, return nullopt." );
238  return std::nullopt;
239  }
240 
241  TRACE( "hypro.datastructures.timing", "Intersect candidates." );
242  return intersect( candidates );
243 }

282 template <typename Number>
283 EventTimingNode<Number>* EventTimingProvider<Number>::findNode( const Path<Number, tNumber>& path, std::size_t level ) const {
284  TRACE( "hypro.datastructures.timing", "Get timings for path " << path << ", level " << level );
285 
286  // find the first initial node separately, since the root-node is not part of the path
287  // Todo: We ignore several initial states for once.
288  if ( mRoot->getChildren().size() == 0 ) {
289  TRACE( "hypro.datastructures.timing", "No child nodes, abort." );
290  return nullptr;
291  }
292 
293  // stores the node candidates for the current step.
294  std::vector<EventTimingNode<Number>*> workingSet;
295  // initialize working set with nodes from the passed level
296  for ( const auto nPtr : mRoot->getChildren() ) {
297  if ( nPtr->getLevel() == level ) {
298  TRACE( "hypro.datastructures.timing", "Initialize working set with node " << nPtr );
299  workingSet.emplace_back( nPtr );
300  }
301  }
302 
303  auto pathIt = path.begin();
304  while ( pathIt != path.end() ) {
305  assert( !pathIt->isDiscreteStep() );
306  // time step
307  for ( auto nodeIt = workingSet.begin(); nodeIt != workingSet.end(); ) {
308  if ( !positiveCover( ( *nodeIt )->getTimings().getInvariantTimings(), pathIt->timeInterval ) ) {
309  nodeIt = workingSet.erase( nodeIt );
310  } else {
311  ++nodeIt;
312  }
313  }
314 
315  TRACE( "hypro.datastructures.timing", "Have " << workingSet.size() << " nodes which match the invariant timings." );
316 
317  ++pathIt;
318 
319  if ( pathIt != path.end() ) {
320  assert( pathIt->isDiscreteStep() );
321  // discrete step
322  std::vector<EventTimingNode<Number>*> transitionCandidates;
323  for ( auto nodeIt = workingSet.begin(); nodeIt != workingSet.end(); ++nodeIt ) {
324  if ( ( *nodeIt )->getTimings().hasTransition( pathIt->transition ) && positiveCover( ( *nodeIt )->getTimings().getTransitionTimings( pathIt->transition ), pathIt->timeInterval ) ) {
325  for ( const auto chPtr : ( *nodeIt )->getChildren() ) {
326  if ( chPtr->getEntryTransition() == pathIt->transition && chPtr->getEntryTimestamp() == pathIt->timeInterval ) {
327  transitionCandidates.emplace_back( chPtr );
328  }
329  }
330  }
331  }
332  workingSet = transitionCandidates;
333 
334  TRACE( "hypro.datastructures.timing", "Have " << workingSet.size() << " nodes which match the transition timings." );
335  ++pathIt;
336  }
337  }
338 
339  TRACE( "hypro.datastructures.timing", "Have " << workingSet.size() << " potential nodes, return node " << workingSet.front() );
340  assert( workingSet.size() == 1 );
341  return workingSet.front();
342 }
