242 //Actual traverse function. Result type and Param type not void
243 template <typename Number, typename Converter, typename Setting>
244 template <typename Result, typename... Rargs>
245 Result SupportFunctionNewT<Number, Converter, Setting>::traverse(
246  std::function<Parameters<Rargs...>( RootGrowNode<Number, Converter, Setting>*, Parameters<Rargs...>& )>&& transform,
247  std::function<Result( RootGrowNode<Number, Converter, Setting>*, Parameters<Rargs...>& )>&& compute,
248  std::function<Result( RootGrowNode<Number, Converter, Setting>*, std::vector<Result>&, Parameters<Rargs...>& )>&& aggregate,
249  Parameters<Rargs...>&& initParams ) const {
250  //Usings
251  using Node = RootGrowNode<Number, Converter, Setting>*;
252  using Param = Parameters<Rargs...>;
253  using Res = Result;
254 
255  //Prepare Stacks
256  std::vector<Node> callStack;
257  std::vector<Param> paramStack;
258  std::vector<std::pair<int, std::vector<Res>>> resultStack;
259  callStack.push_back( mRoot.get() );
260  paramStack.push_back( initParams );
261  resultStack.push_back( std::make_pair( -1, std::vector<Res>() ) );
262 
263  //Traversal loop
264  while ( !callStack.empty() ) {
265  Node cur = callStack.back();
266  Param currentParam = paramStack.back();
267 
268  if ( cur->getOriginCount() == 0 ) {
269  std::pair<int, std::vector<Res>> currentResult = resultStack.back();
270 
271  //If leaf and end of stack is reached
272  if ( currentResult.first == -1 ) {
273  return std::apply( compute, std::forward_as_tuple( cur, currentParam ) );
274 
275  //If leaf and not end of stack is reached
276  } else {
277  resultStack.at( currentResult.first ).second.push_back( std::apply( compute, std::forward_as_tuple( cur, currentParam ) ) );
278  }
279 
280  // delete result frame and close recursive call
281  callStack.pop_back();
282  paramStack.pop_back();
283  resultStack.pop_back();
284 
285  } else {
286  //If enough arguments for operation of node and #arguments != 0
287  if ( resultStack.back().second.size() >= cur->getOriginCount() ) {
288  Res accumulatedResult = std::apply( aggregate, std::forward_as_tuple( cur, resultStack.back().second, currentParam ) );
289 
290  // we reached the top, exit
291  if ( resultStack.back().first == -1 ) {
292  return accumulatedResult;
293  }
294 
295  // forward result.
296  resultStack.at( resultStack.back().first ).second.emplace_back( accumulatedResult );
297 
298  // delete result frame and close recursive call
299  callStack.pop_back();
300  paramStack.pop_back();
301  resultStack.pop_back();
302 
303  //Every other case (recursive call)
304  } else {
305  // here we create the new stack levels.
306  std::size_t callingFrame = callStack.size() - 1;
307  for ( auto& c : cur->rGetChildren() ) {
308  callStack.push_back( c.get() );
309  paramStack.push_back( std::apply( transform, std::forward_as_tuple( cur, currentParam ) ) );
310  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
311  }
312  }
313  }
314  }
315  std::cout << "THIS SHOULD NOT HAPPEN." << std::endl;
316  return Res();
317 }

323 template <typename Number, typename Converter, typename Setting>
324 bool SupportFunctionNewT<Number, Converter, Setting>::empty() const {
325  if ( mRoot == nullptr ) return true;
326 
327  if ( mEmptyState != SETSTATE::UNKNOWN ) return ( mEmptyState == SETSTATE::EMPTY );
328 
329  //first function - parameters are not transformed
330  std::function<void( RootGrowNode<Number, Converter, Setting>* )> doNothing = []( RootGrowNode<Number, Converter, Setting>* ) {};
331 
332  //if leaf - call empty function of representation
333  std::function<bool( RootGrowNode<Number, Converter, Setting>* )> leafEmpty =
334  []( RootGrowNode<Number, Converter, Setting>* n ) -> bool {
335  return n->empty();
336  };
337 
338  //if not leaf - not empty if all children not empty
339  std::function<bool( RootGrowNode<Number, Converter, Setting>*, std::vector<bool>& )> childrenEmpty =
340  []( RootGrowNode<Number, Converter, Setting>* n, std::vector<bool>& childrenEmpty ) -> bool {
341  return n->empty( childrenEmpty );
342  };
343 
344  mEmptyState = traverse( std::move( doNothing ), std::move( leafEmpty ), std::move( childrenEmpty ) ) ? SETSTATE::EMPTY : SETSTATE::NONEMPTY;
345  assert( mEmptyState != SETSTATE::UNKNOWN );
346  return ( mEmptyState == SETSTATE::EMPTY );
347 }

397 template <typename Number, typename Converter, typename Setting>
398 std::vector<EvaluationResult<Number>> SupportFunctionNewT<Number, Converter, Setting>::multiEvaluate( const matrix_t<Number>& _directions, bool useExact ) const {
399  if ( mRoot == nullptr ) {
400  return std::vector<EvaluationResult<Number>>();
401  }
402  if ( _directions.isApprox( matrix_t<Number>::Zero( _directions.rows(), _directions.cols() ) ) ) {
403  return std::vector<EvaluationResult<Number>>();
404  }
405 
406  using Matrix = Parameters<matrix_t<Number>>;
407  using RGNPtr = RootGrowNode<Number, Converter, Setting>*;
408  using EvalVec = std::vector<EvaluationResult<Number>>;
409 
410  //Define lambda functions that will call the functions transform, compute and aggregate dependent on the current node type
411  std::function<Matrix( RGNPtr, Matrix& )> trans =
412  [&]( RGNPtr n, Matrix& param ) -> Matrix {
413  return n->transform( std::get<0>( param.args ) );
414  };
415 
416  std::function<EvalVec( RGNPtr, Matrix& )> comp =
417  [&]( RGNPtr n, Matrix& dir ) -> EvalVec {
418  return n->compute( std::get<0>( dir.args ), useExact );
419  };
420 
421  std::function<EvalVec( RGNPtr, std::vector<EvalVec>&, Matrix& )> agg =
422  [&]( RGNPtr n, std::vector<EvalVec>& resultStackBack, Matrix& currentParam ) -> EvalVec {
423  return n->aggregate( resultStackBack, std::get<0>( currentParam.args ) );
424  };
425 
426  //Traverse the underlying SupportFunctionNewT with the three functions and given directions as initial parameters.
427  Parameters<matrix_t<Number>> params = Parameters<matrix_t<Number>>( _directions );
428  return traverse( std::move( trans ), std::move( comp ), std::move( agg ), std::move( params ) );
429 }

496 template <typename Number, typename Converter, typename Setting>
497 std::size_t SupportFunctionNewT<Number, Converter, Setting>::size() const {
498  if ( mRoot == nullptr ) return std::size_t( 0 );
499 
500  //first function - parameters are not transformed
501  std::function<void( RootGrowNode<Number, Converter, Setting>* )> doNothing = []( RootGrowNode<Number, Converter, Setting>* ) {};
502 
503  //leaves compute their storage size
504  std::function<std::size_t( RootGrowNode<Number, Converter, Setting>* )> sizeofLeaf =
505  []( RootGrowNode<Number, Converter, Setting>* n ) -> std::size_t {
506  assert( n->getType() == SFNEW_TYPE::LEAF );
507  return sizeof( *n );
508  };
509 
510  //operations compute their storage size and add it to the ones of their children
511  std::function<std::size_t( RootGrowNode<Number, Converter, Setting>*, std::vector<std::size_t>& )> sizeofOp =
512  []( RootGrowNode<Number, Converter, Setting>* n, std::vector<std::size_t>& v ) -> std::size_t {
513  std::size_t storage = 0;
514  for ( const auto& childStorageSize : v ) {
515  storage += childStorageSize;
516  }
517  return storage + sizeof( *n );
518  };
519 
520  return traverse( std::move( doNothing ), std::move( sizeofLeaf ), std::move( sizeofOp ) );
521 }

531  return std::make_pair( CONTAINMENT::BOT, *this );
532  }
533  if ( this->empty() ) {
534  return std::make_pair( CONTAINMENT::NO, *this );
535  }
536  //Check for emptiness not needed here since satisfiesHalfspace() is only called via satisfiedHalfspaces(), which already checks emptiness
537  bool limiting = false;
538  EvaluationResult<Number> planeEvalRes = this->evaluate( rhs.normal(), false );
539  //std::cout << "SFN::satisfiesHalfspace, planeEvalRes: " << planeEvalRes << " rhs offset: " << rhs.offset() << std::endl;

800  std::vector<vector_t<Number>> templateDirections = computeTemplate<Number>( this->dimension(), directions );
801  for ( const auto& direction : templateDirections ) {
802  EvaluationResult<Number> thisRes = this->evaluate( direction, true );
803  EvaluationResult<Number> rhsRes = rhs.evaluate( direction, true );
804  assert( thisRes.errorCode != SOLUTION::INFTY && rhsRes.errorCode != SOLUTION::INFTY );
805  if ( thisRes.errorCode == SOLUTION::FEAS ) {

815  //If this is unbound and rhs bound, then rhs is still contained on this side
816  //If both are unbound, then both SFs are halfspaces and are only contained in each other, if the normals point to the same direction
817  //while the offset of rhs is smaller than of this.
818  }
819  return true;
820 }
