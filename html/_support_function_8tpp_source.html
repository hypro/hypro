452 std::pair<CONTAINMENT, SupportFunctionT<Number, Converter, Setting>> SupportFunctionT<Number, Converter, Setting>::satisfiesHalfspace( const Halfspace<Number>& rhs ) const {
453  //std::cout << __func__ << ": " << _mat << std::endl << " <= " << _vec << std::endl;
454 
455  // catch zero-constraints separately
456  if ( rhs.normal() == vector_t<Number>::Zero( rhs.normal().rows() ) ) {
457  return rhs.offset() <= 0 ? std::make_pair( CONTAINMENT::FULL, *this ) : std::make_pair( CONTAINMENT::NO, *this );
458  }
459 
460  bool limiting = false;
461  EvaluationResult<Number> planeEvalRes = content->evaluate( rhs.normal(), false );
462  if ( planeEvalRes.errorCode == SOLUTION::INFEAS ) {
463  //std::cout << "Is infeasible (should not happen)." << std::endl;
464  //std::cout << "Set is (Hpoly): " << std::endl << Converter::toHPolytope(*this) << std::endl;
465  assert( Converter::toHPolytope( *this ).empty() );
466  return std::make_pair( CONTAINMENT::NO, *this );
467  } else if ( planeEvalRes.supportValue > rhs.offset() ) {
468  //std::cout << "Object will be limited. " << std::endl;
469  // the actual object will be limited by the new plane
470  limiting = true;
471  // std::cout << "evaluate(" << convert<Number,double>(-(_mat.row(rowI))) << ") <= " << -(_vec(rowI)) << ": " << content->evaluate(-(_mat.row(rowI))).supportValue << " <= " << -(_vec(rowI)) << std::endl;

473  if ( content->evaluate( -( rhs.normal() ), false ).supportValue < -( rhs.offset() ) ) {
474  //std::cout << "fullyOutside" << std::endl;
475  // the object lies fully outside one of the planes -> return false
476  return std::make_pair( CONTAINMENT::NO, this->intersectHalfspace( rhs ) );
477  }
478  }
479 
480  if ( limiting ) {
481  auto tmp = this->intersectHalfspace( rhs );
482  assert( !tmp.empty() );
483  return std::make_pair( CONTAINMENT::PARTIAL, tmp );
484  } else {

489 template <typename Number, typename Converter, typename Setting>
490 std::pair<CONTAINMENT, SupportFunctionT<Number, Converter, Setting>> SupportFunctionT<Number, Converter, Setting>::satisfiesHalfspaces( const matrix_t<Number>& _mat, const vector_t<Number>& _vec ) const {
491  DEBUG( "hypro.representations.supportFunction", "Matrix: " << _mat << std::endl
492  << " <= " << _vec );
493  if ( _mat.rows() == 0 ) {
494  return std::make_pair( CONTAINMENT::FULL, *this );
495  }
496  assert( _mat.rows() == _vec.rows() );
497 
498  // maintain potential result
499  std::optional<SupportFunctionT<Number, Converter, Setting>> res;
500  std::vector<Eigen::Index> intersectingHalfspaces;
501  for ( Eigen::Index rowI = 0; rowI < _mat.rows(); ++rowI ) {
502  // catch zero-constraints separately
503  if ( _mat.row( rowI ).isZero() ) {
504  if ( _vec( rowI ) > 0 ) {
505  DEBUG( "hypro.representations.supportFunction", "Row " << rowI << " failed." );
506  return std::make_pair( CONTAINMENT::NO, *this );
507  }
508  continue;
509  }
510 #ifdef HYPRO_LOGGING
511  if ( std::find( intersectingHalfspaces.begin(), intersectingHalfspaces.end(), rowI ) != intersectingHalfspaces.end() ) {
512  TRACE( "hypro.representations.supportFunction", "Re-evaluation against plane " << rowI );
513  } else {
514  TRACE( "hypro.representations.supportFunction", "First-time evaluation against plane " << rowI );
515  }
516 
517 #endif
518 
519  EvaluationResult<Number> planeEvalRes;
520  if ( res ) {
521  planeEvalRes = res.value().evaluate( _mat.row( rowI ), false );
522  } else {
523  planeEvalRes = content->evaluate( _mat.row( rowI ), false );
524  }
525 
526  if ( planeEvalRes.errorCode == SOLUTION::INFEAS ) {
527  DEBUG( "hypro.representations.supportFunction", "Row " << rowI << " failed." );
528  return std::make_pair( CONTAINMENT::NO, *this );
529  } else if ( !carl::AlmostEqual2sComplement( planeEvalRes.supportValue, _vec( rowI ) ) && planeEvalRes.supportValue > _vec( rowI ) ) {
530  // Case where the object will be limited, i.e., the support from the set is larger than the constant part of the constraint in the chosen direction.
531 
532  TRACE( "hypro.representations.supportFunction", "Object will be limited, as " << planeEvalRes.supportValue << " > " << _vec( rowI ) );
533 
534  // to check for emptiness, compute support in opposite direction.
535  Number invDirVal;
536  if ( res ) {
537  invDirVal = res.value().evaluate( -( _mat.row( rowI ) ), false ).supportValue;
538  } else {
539  invDirVal = content->evaluate( -( _mat.row( rowI ) ), false ).supportValue;
540  }
541 
542  TRACE( "hypro.representations.supportFunction", "Potentially fully empty, if " << invDirVal << " < " << -( _vec( rowI ) ) );
543 
544  // case where the result is empty, i.e., when the supporting hyperplane in the opposite direction lies outside the current half-space.
545  if ( invDirVal < -( _vec( rowI ) ) ) {
546  // exact verification in case the values are close to each other
547  if ( carl::AlmostEqual2sComplement( Number( -invDirVal ), planeEvalRes.supportValue, 16 ) ) {
548  EvaluationResult<Number> secndPosEval;
549  if ( res ) {
550  secndPosEval = res.value().evaluate( _mat.row( rowI ), true );
551  } else {
552  secndPosEval = content->evaluate( _mat.row( rowI ), true );
553  }
554  if ( secndPosEval.supportValue > _vec( rowI ) ) {
555  EvaluationResult<Number> secndNegEval;
556  if ( res ) {
557  secndNegEval = res.value().evaluate( -( _mat.row( rowI ) ), true );
558  } else {
559  secndNegEval = content->evaluate( -( _mat.row( rowI ) ), true );
560  }
561  if ( secndNegEval.supportValue < -( _vec( rowI ) ) ) {
562  TRACE( "hypro.representations.supportFunction", "fullyOutside" );
563  // the object lies fully outside one of the planes -> return false
564  DEBUG( "hypro.representations.supportFunction", "Row " << rowI << " failed -- fully outside." );
565  return std::make_pair( CONTAINMENT::NO, this->intersectHalfspaces( _mat, _vec ) );
566  }
567  }
568  } else {
569  // the values are far enough away from each other to make this result a false negative.
570  TRACE( "hypro.representations.supportFunction", "fullyOutside, as " << invDirVal << " >= " << -( _vec( rowI ) ) );
571  // the object lies fully outside one of the planes -> return false
572  DEBUG( "hypro.representations.supportFunction", "Row " << rowI << " failed." );
573  return std::make_pair( CONTAINMENT::NO, this->intersectHalfspaces( _mat, _vec ) );
574  }
575  } else {
576  // as we do multiple checks, only add if not already contained.
577  if ( std::find( intersectingHalfspaces.begin(), intersectingHalfspaces.end(), rowI ) == intersectingHalfspaces.end() ) {
578  // update cache
579  intersectingHalfspaces.push_back( rowI );
580  // object is not empty. Create intersection object, and use this to add all limiting planes.
581  if ( res ) {
582  // the result, whose right-hand-side are the added planes
583  intersectionContent<Number, Setting>* intersection = res.value().rGetContent()->rIntersectionParameters();
584  // add constraint to right-hand-side
585  appendRow( intersection->rhs->rPolytope()->rMatrix(), vector_t<Number>( _mat.row( rowI ) ) );
586  appendRow( intersection->rhs->rPolytope()->rVector(), _vec( rowI ) );
587 
588  TRACE( "hypro.representations.supportFunction", "Add plane at index " << rowI << " to intersection object." );
589  } else {
590  TRACE( "hypro.representations.supportFunction", "Initialize intersection object with plane at index " << rowI );
591  // use override reduction to enforce, that the resulting support function is of type polytope to be able to add halfspaces, this is currently not supported for boxes.
592  matrix_t<Number> constraint = _mat.row( rowI );
593  vector_t<Number> constant = _vec.row( rowI );
594  res = SupportFunctionT<Number, Converter, Setting>( content->intersect( SupportFunctionT<Number, Converter, Setting>( constraint, constant, true ).content ) );
595  }
596 
597  // finally, reset rowI to zero (to re-check all other constraints), as the limitation might have rendered the object emtpy in combination with other constraints.
598  rowI = 0;
599  }
600  }
601  }
602  }
603 
604  // case: object is fully contained
605  if ( !res.has_value() ) {
606  return std::make_pair( CONTAINMENT::FULL, *this );
607  } else {
608 //assert( !res.value().empty() );
609 //return std::make_pair( CONTAINMENT::PARTIAL, res.value() );
610 #ifdef HYPRO_LOGGING
611  intersectionContent<Number, Setting>* intersection = res.value().rGetContent()->rIntersectionParameters();
612  TRACE( "hypro.representations.supportFunction", "Final intersection constraints " << intersection->rhs->rPolytope()->constraints() );
613  TRACE( "hypro.representations.supportFunction", "Final intersection constants " << intersection->rhs->rPolytope()->constants() );
614 #endif
615  CONTAINMENT containmentState = res.value().empty() ? CONTAINMENT::NO : CONTAINMENT::PARTIAL;
616  return std::make_pair( containmentState, res.value() );
617  }
618 }

663 template <typename Number, typename Converter, typename Setting>
664 const SupportFunctionT<Number, Converter, Setting>& SupportFunctionT<Number, Converter, Setting>::evaluateTemplate( std::size_t directionCount, bool force ) const {
665  if ( !mTemplateSet || force ) {
666  std::vector<vector_t<Number>> templateDirections = computeTemplate<Number>( this->dimension(), directionCount );
667 
668  matrix_t<Number> templateDirectionMatrix = combineRows( templateDirections );
669 
670  //lets the support function evaluate the offset of the halfspaces for each direction
671  std::vector<EvaluationResult<Number>> offsets = this->multiEvaluate( templateDirectionMatrix );
672 
673  std::vector<std::size_t> boundedConstraints;
674  for ( unsigned offsetIndex = 0; offsetIndex < offsets.size(); ++offsetIndex ) {
675  //std::cout << "Result: " << offsets[offsetIndex] << std::endl;
676  if ( offsets[offsetIndex].errorCode != SOLUTION::INFTY ) {
677  boundedConstraints.push_back( offsetIndex );
678  }
679  }
680  matrix_t<Number> constraints = matrix_t<Number>( boundedConstraints.size(), this->dimension() );
681  vector_t<Number> constants = vector_t<Number>( boundedConstraints.size() );
682  unsigned pos = boundedConstraints.size() - 1;
683  while ( !boundedConstraints.empty() ) {
684  constraints.row( pos ) = templateDirectionMatrix.row( boundedConstraints.back() );
685  constants( pos ) = offsets[boundedConstraints.back()].supportValue;
686  boundedConstraints.pop_back();
687  --pos;
688  }
689 
690  mMatrix = constraints;
691  mVector = constants;
692  mTemplateSet = true;
693  }
694  return *this;
695 }
