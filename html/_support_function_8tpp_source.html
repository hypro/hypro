356 template <typename Number, typename Converter, typename Setting>
357 std::pair<CONTAINMENT, SupportFunctionT<Number, Converter, Setting>> SupportFunctionT<Number, Converter, Setting>::satisfiesHalfspace( const Halfspace<Number>& rhs ) const {
358  // catch zero-constraints separately
359  if ( rhs.normal() == vector_t<Number>::Zero( rhs.normal().rows() ) ) {
360  return rhs.offset() <= 0 ? std::make_pair( CONTAINMENT::FULL, *this ) : std::make_pair( CONTAINMENT::NO, *this );
361  }
362 
363  bool limiting = false;
364  EvaluationResult<Number> planeEvalRes = content->evaluate( rhs.normal(), false );
365  if ( planeEvalRes.errorCode == SOLUTION::INFEAS ) {
366  assert( Converter::toHPolytope( *this ).empty() );
367  return std::make_pair( CONTAINMENT::NO, *this );
368  } else if ( planeEvalRes.supportValue > rhs.offset() ) {
369  // the actual object will be limited by the new plane
370  limiting = true;
371  if ( content->evaluate( -( rhs.normal() ), false ).supportValue < -( rhs.offset() ) ) {
372  // the object lies fully outside one of the planes -> return false
373  return std::make_pair( CONTAINMENT::NO, this->intersectHalfspace( rhs ) );
374  }
375  }
376 
377  if ( limiting ) {
378  auto tmp = this->intersectHalfspace( rhs );
379  assert( !tmp.empty() );
380  return std::make_pair( CONTAINMENT::PARTIAL, tmp );
381  } else {
382  return std::make_pair( CONTAINMENT::FULL, *this );
383  }
384 }

386 template <typename Number, typename Converter, typename Setting>
387 std::pair<CONTAINMENT, SupportFunctionT<Number, Converter, Setting>> SupportFunctionT<Number, Converter, Setting>::satisfiesHalfspaces( const matrix_t<Number>& _mat, const vector_t<Number>& _vec ) const {
388  DEBUG( "hypro.representations.supportFunction", "Matrix: " << _mat << std::endl
389  << " <= " << _vec );
390  if ( _mat.rows() == 0 ) {
391  return std::make_pair( CONTAINMENT::FULL, *this );
392  }
393  assert( _mat.rows() == _vec.rows() );
394 
395  // maintain potential result
396  std::optional<SupportFunctionT<Number, Converter, Setting>> res;
397  std::vector<Eigen::Index> intersectingHalfspaces;
398  for ( Eigen::Index rowI = 0; rowI < _mat.rows(); ++rowI ) {
399  // catch zero-constraints separately
400  if ( _mat.row( rowI ).isZero() ) {
401  if ( _vec( rowI ) > 0 ) {
402  DEBUG( "hypro.representations.supportFunction", "Row " << rowI << " failed." );
403  return std::make_pair( CONTAINMENT::NO, *this );
404  }
405  continue;
406  }
407 #ifdef HYPRO_LOGGING
408  if ( std::find( intersectingHalfspaces.begin(), intersectingHalfspaces.end(), rowI ) != intersectingHalfspaces.end() ) {
409  TRACE( "hypro.representations.supportFunction", "Re-evaluation against plane " << rowI );
410  } else {
411  TRACE( "hypro.representations.supportFunction", "First-time evaluation against plane " << rowI );
412  }
413 
414 #endif
415 
416  EvaluationResult<Number> planeEvalRes;
417  if ( res ) {
418  planeEvalRes = res.value().evaluate( _mat.row( rowI ), false );
419  } else {
420  planeEvalRes = content->evaluate( _mat.row( rowI ), false );
421  }
422 
423  if ( planeEvalRes.errorCode == SOLUTION::INFEAS ) {
424  DEBUG( "hypro.representations.supportFunction", "Row " << rowI << " failed." );
425  return std::make_pair( CONTAINMENT::NO, *this );
426  } else if ( !carl::AlmostEqual2sComplement( planeEvalRes.supportValue, _vec( rowI ) ) && planeEvalRes.supportValue > _vec( rowI ) ) {
427  // Case where the object will be limited, i.e., the support from the set is larger than the constant part of the constraint in the chosen direction.
428 
429  TRACE( "hypro.representations.supportFunction", "Object will be limited, as " << planeEvalRes.supportValue << " > " << _vec( rowI ) );
430 
431  // to check for emptiness, compute support in opposite direction.
432  Number invDirVal;
433  if ( res ) {
434  invDirVal = res.value().evaluate( -( _mat.row( rowI ) ), false ).supportValue;
435  } else {
436  invDirVal = content->evaluate( -( _mat.row( rowI ) ), false ).supportValue;
437  }
438 
439  TRACE( "hypro.representations.supportFunction", "Potentially fully empty, if " << invDirVal << " < " << -( _vec( rowI ) ) );
440 
441  // case where the result is empty, i.e., when the supporting hyperplane in the opposite direction lies outside the current half-space.
442  if ( invDirVal < -( _vec( rowI ) ) ) {
443  // exact verification in case the values are close to each other
444  if ( carl::AlmostEqual2sComplement( Number( -invDirVal ), planeEvalRes.supportValue, 16 ) ) {
445  EvaluationResult<Number> secndPosEval;
446  if ( res ) {
447  secndPosEval = res.value().evaluate( _mat.row( rowI ), true );
448  } else {
449  secndPosEval = content->evaluate( _mat.row( rowI ), true );
450  }
451  if ( secndPosEval.supportValue > _vec( rowI ) ) {
452  EvaluationResult<Number> secndNegEval;
453  if ( res ) {
454  secndNegEval = res.value().evaluate( -( _mat.row( rowI ) ), true );
455  } else {
456  secndNegEval = content->evaluate( -( _mat.row( rowI ) ), true );
457  }
458  if ( secndNegEval.supportValue < -( _vec( rowI ) ) ) {
459  TRACE( "hypro.representations.supportFunction", "fullyOutside" );
460  // the object lies fully outside one of the planes -> return false
461  DEBUG( "hypro.representations.supportFunction", "Row " << rowI << " failed -- fully outside." );
462  return std::make_pair( CONTAINMENT::NO, this->intersectHalfspaces( _mat, _vec ) );
463  }
464  }
465  } else {
466  // the values are far enough away from each other to make this result a false negative.
467  TRACE( "hypro.representations.supportFunction", "fullyOutside, as " << invDirVal << " >= " << -( _vec( rowI ) ) );
468  // the object lies fully outside one of the planes -> return false
469  DEBUG( "hypro.representations.supportFunction", "Row " << rowI << " failed." );
470  return std::make_pair( CONTAINMENT::NO, this->intersectHalfspaces( _mat, _vec ) );
471  }
472  } else {
473  // as we do multiple checks, only add if not already contained.
474  if ( std::find( intersectingHalfspaces.begin(), intersectingHalfspaces.end(), rowI ) == intersectingHalfspaces.end() ) {
475  // update cache
476  intersectingHalfspaces.push_back( rowI );
477  // object is not empty. Create intersection object, and use this to add all limiting planes.
478  if ( res ) {
479  // the result, whose right-hand-side are the added planes
480  intersectionContent<Number, Setting>* intersection = res.value().rGetContent()->rIntersectionParameters();
481  // add constraint to right-hand-side
482  appendRow( intersection->rhs->rPolytope()->rMatrix(), vector_t<Number>( _mat.row( rowI ) ) );
483  appendRow( intersection->rhs->rPolytope()->rVector(), _vec( rowI ) );
484 
485  TRACE( "hypro.representations.supportFunction", "Add plane at index " << rowI << " to intersection object." );
486  } else {
487  TRACE( "hypro.representations.supportFunction", "Initialize intersection object with plane at index " << rowI );
488  // use override reduction to enforce, that the resulting support function is of type polytope to be able to add halfspaces, this is currently not supported for boxes.
489  matrix_t<Number> constraint = _mat.row( rowI );
490  vector_t<Number> constant = _vec.row( rowI );
491  res = SupportFunctionT<Number, Converter, Setting>( content->intersect( SupportFunctionT<Number, Converter, Setting>( constraint, constant, true ).content ) );
492  }
493 
494  // finally, reset rowI to zero (to re-check all other constraints), as the limitation might have rendered the object emtpy in combination with other constraints.
495  rowI = 0;
496  }
497  }
498  }
499  }
500 
501  // case: object is fully contained
502  if ( !res.has_value() ) {
503  return std::make_pair( CONTAINMENT::FULL, *this );
504  } else {
505 #ifdef HYPRO_LOGGING
506  intersectionContent<Number, Setting>* intersection = res.value().rGetContent()->rIntersectionParameters();
507  TRACE( "hypro.representations.supportFunction", "Final intersection constraints " << intersection->rhs->rPolytope()->constraints() );
508  TRACE( "hypro.representations.supportFunction", "Final intersection constants " << intersection->rhs->rPolytope()->constants() );
509 #endif
510  CONTAINMENT containmentState = res.value().empty() ? CONTAINMENT::NO : CONTAINMENT::PARTIAL;
511  return std::make_pair( containmentState, res.value() );
512  }
513 }

559 const SupportFunctionT<Number, Converter, Setting>& SupportFunctionT<Number, Converter, Setting>::evaluateTemplate( std::size_t directionCount, bool force ) const {
560  if ( !mTemplateSet || force ) {
561  std::vector<vector_t<Number>> templateDirections = computeTemplate<Number>( this->dimension(), directionCount );
562 
563  matrix_t<Number> templateDirectionMatrix = combineRows( templateDirections );
564 
565  //lets the support function evaluate the offset of the halfspaces for each direction
566  std::vector<EvaluationResult<Number>> offsets = this->multiEvaluate( templateDirectionMatrix );
567 
568  std::vector<std::size_t> boundedConstraints;
569  for ( unsigned offsetIndex = 0; offsetIndex < offsets.size(); ++offsetIndex ) {
570  if ( offsets[offsetIndex].errorCode != SOLUTION::INFTY ) {
571  boundedConstraints.push_back( offsetIndex );
572  }
573  }
574  matrix_t<Number> constraints = matrix_t<Number>( boundedConstraints.size(), this->dimension() );
575  vector_t<Number> constants = vector_t<Number>( boundedConstraints.size() );
576  Eigen::Index pos = Eigen::Index( boundedConstraints.size() ) - 1;
577  while ( !boundedConstraints.empty() ) {
578  constraints.row( pos ) = templateDirectionMatrix.row( boundedConstraints.back() );
579  constants( pos ) = offsets[boundedConstraints.back()].supportValue;
580  boundedConstraints.pop_back();
581  --pos;
582  }
583 
584  mMatrix = constraints;
585  mVector = constants;
586  mTemplateSet = true;
587  }
588  return *this;
589 }
