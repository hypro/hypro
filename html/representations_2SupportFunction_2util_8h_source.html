173  void createNextReduct() const {
174  // the last created reduction pair is at the back, as a std::map is per default sorted ascending
175  assert( parameters.size() > 0 );
176  unsigned powerOfTwo = carl::pow( 2, power );
177  // first compute the new b
178  vector_t<Number> bTrans = ( --parameters.end() )->second.second;
179  matrix_t<Number> aTrans = ( --parameters.end() )->second.first;
180  //std::cout << "Starting from exponent " << (--matrices.end())->first << std::endl;
181  unsigned exponent = ( ( --parameters.end() )->first ) * powerOfTwo;
182  //std::cout << "New exponent: " << exponent << std::endl;
183  // accumulate b
184  for ( std::size_t i = 1; i < powerOfTwo; i++ ) {
185  // Note: aTrans hasn't changed yet -> we can use it for transformation.
186  bTrans = aTrans * bTrans + ( --parameters.end() )->second.second;
187  }
188  // accumulate A
189  for ( std::size_t i = 0; i < power; i++ ) {
190  aTrans = aTrans * aTrans;
191  }
192  parameters[exponent] = std::make_pair( aTrans, bTrans );
193  }
