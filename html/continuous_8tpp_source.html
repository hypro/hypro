114  }

119  }
120 
121  Range<Number> center_of_Xl;
122  Xl.rmConstantTerm( center_of_Xl );
123 
124  TaylorModelVec<Number> tmvCenter( center_of_Xl );
125 
126  if ( !tmvPre.isEmpty() ) {
127  Xl = Xl.substitute( tmv, domain, order );
128  }
129 
130  Range<Number> range_of_Xl;
131  Xl.enclosure( range_of_Xl, domain );
132 
133  // compute the preconditioning factors
134  TaylorModelVec<Number> new_tmvPre;
135 
136  carl::Interval<Number> intStep( 0.0, stepsize );
137

145

151

153

156  }
157 
158  localInitial.assign( iter1->first, intUnit );
159  }
160 
161  result.tmv = Xl;
162

164  x0 = tmvCenter + new_tmvPre;
165 
166  // apply Picard iterations to obtain the Taylor expansion
167  TaylorModelVec<Number> x = x0;
168 
169  for ( unsigned i = 1; i < order; ++i ) {
170  x.Picard_assign( x0, ode, localInitial, t, intStep, i );
171  }
172 
173  for ( auto iter = x.tms.begin(); iter != x.tms.end(); ++iter ) {
174  carl::Interval<Number> intEst;
175  estimation.find_assignment( intEst, iter->first );
176  iter->second.remainder = intEst;
177  }
178 
179  TaylorModelVec<Number> tmvTemp;
180  tmvTemp = x.Picard( x0, ode, localInitial, t, intStep, order );
181

183 
184  auto iter_x = x.tms.begin();
185  auto iter_xp = tmvTemp.tms.begin();
186 
187  for ( ; iter_x != x.tms.end(); ++iter_x, ++iter_xp ) {
188  // std::cout << iter_x->second.remainder << std::endl;
189  // std::cout << iter_xp->second.remainder << std::endl;
190 
191  if ( !( iter_x->second.remainder.contains( iter_xp->second.remainder ) ) ) {
192  bfound = false;
193  break;
194  }
195  }
196 
197  if ( !bfound ) {
198  return 0;
199  } else {
200  iter_x = x.tms.begin();
201  iter_xp = tmvTemp.tms.begin();
202 
203  for ( ; iter_x != x.tms.end(); ++iter_x, ++iter_xp ) {
204  iter_x->second.remainder = iter_xp->second.remainder;
205  }
206  }
207

209  bool bfinish = false;
210 
211  for ( int steps = 0; !bfinish && steps <= Setting::MAX_ITER_NUM; ++steps ) {
212  bfinish = true;
213 
214  tmvTemp = x.Picard( x0, ode, localInitial, t, intStep, order );
215 
216  iter_x = x.tms.begin();
217  iter_xp = tmvTemp.tms.begin();
218 
219  for ( ; iter_x != x.tms.end(); ++iter_x, ++iter_xp ) {
220  Number w1 = iter_x->second.remainder.diameter();
221  Number w2 = iter_xp->second.remainder.diameter();
222 
223  if ( bfinish && w2 / w1 < Number( Setting::STOP_RATIO ) ) {
224  bfinish = false;
225  }
226 
227  iter_x->second.remainder = iter_xp->second.remainder;
228  }
229  }
230 
231  result.tmvPre = x;
232  result.domain = domain;
233 
234  result.domain.assign( t, intStep );
235 
236  return 1;
