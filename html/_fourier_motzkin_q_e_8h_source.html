44 std::pair<matrix_t<Number>, vector_t<Number>> eliminateCol( const matrix_t<Number>& constraints, const vector_t<Number> constants, IdxType col, bool conservative = true ) {
45  std::vector<std::size_t> nobounds;
46  std::vector<std::size_t> upperbounds;
47  std::vector<std::size_t> lowerbounds;
48 
49  // initialize: detect upper and lower bounds
50  for ( Eigen::Index row = 0; row < constraints.rows(); ++row ) {
51  if ( constraints( row, col ) == Number( 0 ) ) {
52  nobounds.push_back( row );
53  } else if ( constraints( row, col ) > Number( 0 ) ) {
54  upperbounds.push_back( row );
55  } else {
56  lowerbounds.push_back( row );
57  }
58  }
59 
60  // initialize result
61  Eigen::Index newConstraintsCount = nobounds.size() + ( upperbounds.size() * lowerbounds.size() );
62  matrix_t<Number> newConstraints = matrix_t<Number>( newConstraintsCount, constraints.cols() );
63  vector_t<Number> newConstants = vector_t<Number>( newConstraintsCount );
64 
65  // compute new constraints
66  std::vector<Eigen::Index> emptyRows;
67  for ( Eigen::Index row = 0; row < newConstraintsCount; ++row ) {
68  if ( std::size_t( row ) < nobounds.size() ) {
69  newConstraints.row( row ) = constraints.row( nobounds[row] );
70  newConstants.row( row ) = constants.row( nobounds[row] );
71  } else {
72  std::size_t combinationIndex = row - nobounds.size();
73  assert( combinationIndex >= 0 and combinationIndex < lowerbounds.size() * upperbounds.size() );
74  std::size_t lbIndex = combinationIndex / upperbounds.size();
75  std::size_t ubIndex = combinationIndex % upperbounds.size();
76  assert( lbIndex < lowerbounds.size() );
77  assert( ubIndex < upperbounds.size() );
78  newConstraints.row( row ) = constraints( upperbounds[ubIndex], col ) * constraints.row( lowerbounds[lbIndex] ) - constraints( lowerbounds[lbIndex], col ) * constraints.row( upperbounds[ubIndex] );
79  newConstants( row ) = constraints( upperbounds[ubIndex], col ) * constants( lowerbounds[lbIndex] ) - constraints( lowerbounds[lbIndex], col ) * constants( upperbounds[ubIndex] );
80  if ( newConstraints.row( row ).isZero() ) {
81  emptyRows.push_back( row );
82  }
83  }
84  }
85 
86  assert( vector_t<Number>( newConstraints.col( col ) ) == vector_t<Number>::Zero( newConstants.rows() ) );
87 
88  // cleanup, if demanded
89  if ( !conservative ) {
90  newConstraints = removeCol( newConstraints, col );
91  }
92 
93  newConstraints = removeRows( newConstraints, emptyRows );
94  newConstants = removeRows( newConstants, emptyRows );
95 
96  auto duplicateRows = getDuplicateRows( newConstraints );
97  newConstraints = removeRows( newConstraints, duplicateRows );
98  newConstants = removeRows( newConstants, duplicateRows );
99 
100  return std::make_pair( newConstraints, newConstants );
101 }
