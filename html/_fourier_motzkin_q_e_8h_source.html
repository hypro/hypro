44 template <typename Number, typename IdxType>
45 std::pair<matrix_t<Number>, vector_t<Number>> eliminateCol( const matrix_t<Number>& constraints, const vector_t<Number> constants, IdxType col, bool conservative = true ) {
46  std::vector<std::size_t> nobounds;
47  std::vector<std::size_t> upperbounds;
48  std::vector<std::size_t> lowerbounds;
49 
50  // initialize: detect upper and lower bounds
51  for ( Eigen::Index row = 0; row < constraints.rows(); ++row ) {
52  if ( constraints( row, col ) == Number( 0 ) ) {
53  nobounds.push_back( row );
54  } else if ( constraints( row, col ) > Number( 0 ) ) {
55  upperbounds.push_back( row );
56  } else {
57  lowerbounds.push_back( row );
58  }
59  }
60 
61  // initialize result
62  Eigen::Index newConstraintsCount = nobounds.size() + ( upperbounds.size() * lowerbounds.size() );
63  matrix_t<Number> newConstraints = matrix_t<Number>( newConstraintsCount, constraints.cols() );
64  vector_t<Number> newConstants = vector_t<Number>( newConstraintsCount );
65 
66  // compute new constraints
67  std::vector<Eigen::Index> emptyRows;
68  for ( Eigen::Index row = 0; row < newConstraintsCount; ++row ) {
69  if ( std::size_t( row ) < nobounds.size() ) {
70  newConstraints.row( row ) = constraints.row( nobounds[row] );
71  newConstants.row( row ) = constants.row( nobounds[row] );
72  } else {
73  std::size_t combinationIndex = row - nobounds.size();
74  assert( combinationIndex >= 0 and combinationIndex < lowerbounds.size() * upperbounds.size() );
75  std::size_t lbIndex = combinationIndex / upperbounds.size();
76  std::size_t ubIndex = combinationIndex % upperbounds.size();
77  assert( lbIndex < lowerbounds.size() );
78  assert( ubIndex < upperbounds.size() );
79  newConstraints.row( row ) = constraints( upperbounds[ubIndex], col ) * constraints.row( lowerbounds[lbIndex] ) - constraints( lowerbounds[lbIndex], col ) * constraints.row( upperbounds[ubIndex] );
80  newConstants( row ) = constraints( upperbounds[ubIndex], col ) * constants( lowerbounds[lbIndex] ) - constraints( lowerbounds[lbIndex], col ) * constants( upperbounds[ubIndex] );
81  if ( newConstraints.row( row ).isZero() ) {
82  emptyRows.push_back( row );
83  }
84  }
85  }
86 
87  assert( vector_t<Number>( newConstraints.col( col ) ) == vector_t<Number>::Zero( newConstants.rows() ) );
88 
89  // cleanup, if demanded
90  if ( !conservative ) {
91  newConstraints = removeCol( newConstraints, col );
92  }
93 
94  newConstraints = removeRows( newConstraints, emptyRows );
95  newConstants = removeRows( newConstants, emptyRows );
96 
97  auto duplicateRows = getDuplicateRows( newConstraints );
98  newConstraints = removeRows( newConstraints, duplicateRows );
99  newConstants = removeRows( newConstants, duplicateRows );
100 
101  return std::make_pair( newConstraints, newConstants );
102 }
