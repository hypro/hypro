64 void ConvexHull<Number>::findOffset() {
65  // compute barycenter
66  mOffset = vector_t<Number>::Zero( mPoints[0].rawCoordinates().size() );
67  for ( const auto& point : mPoints ) {
68  mOffset += point.rawCoordinates();
69  }
70  mOffset = mOffset / mPoints.size();
71 
72  // translate all points to make the barycenter the origin.
73  unsigned index = 0;
74  for ( auto& point : mPoints ) {
75  point.setCoordinates( point.rawCoordinates() - mOffset );
76 
77  // small tuning - remove point that lies exactly on the barycenter, as we know it lies inside
78  // Comment: Not neccesaryly true, what if it is the only point?
79  if ( point.rawCoordinates().nonZeros() == 0 ) {
80  mPoints.erase( mPoints.begin() + index );
81  }
82  ++index;
83  }
84 }

233  aux2[index] = -1 * l[index];
234  }

247  for ( unsigned index = 0; index < hs.normal().size() - 1; ++index ) {
248  aux[index] = hs.normal()[index];

250  vector_t<Number> zero = vector_t<Number>::Zero( hs.normal().size() - 1 );
