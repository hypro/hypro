84  template <typename N = Number, carl::DisableIf<std::is_same<N, double>> = carl::dummy>
85  void makeInteger() {
86  if ( !mIsInteger ) {
87  Number scaling = Number( carl::getDenom( mScalar ) );
88  for ( unsigned i = 0; i < mNormal.rows(); ++i ) {
89  scaling = scaling * Number( carl::getDenom( mNormal( i ) ) );
90  }
91 
92  mScalar = mScalar * scaling;
93  assert( carl::isInteger( mScalar ) );
94 
95  for ( unsigned i = 0; i < mNormal.rows(); ++i ) {
96  mNormal( i ) = mNormal( i ) * scaling;
97  assert( carl::isInteger( mNormal( i ) ) );
98  }
99  mIsInteger = true;
100  }
101  }

418 template <class Number>
419 struct hash<hypro::Halfspace<Number>> {
420  std::size_t operator()( hypro::Halfspace<Number> const& Halfspace ) const {
421  size_t seed = 0;
422  hypro::vector_t<Number> normal = Halfspace.normal();
423  Number scalar = Halfspace.scalar();
424  std::hash<hypro::vector_t<Number>> vectorHasher;
