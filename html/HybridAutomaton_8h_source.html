194  template<typename T>
195  bool equals(const std::set<T*>& lhs, const std::set<T*>& rhs) const {
196 
197  if(lhs.size() != rhs.size()) return false;
198  for(auto lhsIt = lhs.begin(); lhsIt != lhs.end(); ++lhsIt){
199  bool found = false;
200  for(auto rhsIt = rhs.begin(); rhsIt != rhs.end(); ++rhsIt) {
201  //std::cout << "now comparing " << (*(lhsIt))->hash() << " and " << (*(rhsIt))->hash() << std::endl;
202  if((*lhsIt)->hash() == (*rhsIt)->hash()){
203  found = true;
204  break;
205  }
206  }
207  if(!found){
208  return false;
209  }
210  }
211 
212  return true;
213  }

215  /**
216  * @brief Comparison for equality operator.
217  * @param[in] lhs The left hand side.
218  * @param[in] rhs The right hand side.
219  * @return True, if both automata are equal, false otherwise.
220  */
221  friend bool operator==( const HybridAutomaton<Number>& lhs, const HybridAutomaton<Number>& rhs ) {
222  if(!(lhs.equals(lhs.getLocations(),rhs.getLocations()))){
223  TRACE("hypro.datastructures.hybridAutomaton", "no equality of locations.");
224  return false;
225  }
226  if(!(lhs.equals(lhs.getTransitions(),rhs.getTransitions()))){
227  TRACE("hypro.datastructures.hybridAutomaton", "no equality of transitions.");
228  return false;
229  }
230  if(lhs.getInitialStates().size() != rhs.getInitialStates().size()){
231  TRACE("hypro.datastructures.hybridAutomaton", "initial set sizes were different.");
232  return false;
233  }
234  if(lhs.getInitialStates() != rhs.getInitialStates()){
235  TRACE("hypro.datastructures.hybridAutomaton", "no equality of initials.");
236 
237  TRACE("hypro.datastructures.hybridAutomaton", "size of lhs initial set: " << lhs.getInitialStates().size() << " size of rhs initial set: " << rhs.getInitialStates().size());
238  auto rhsIt = rhs.getInitialStates().begin();
239  for(auto lhsIt = lhs.getInitialStates().begin(); lhsIt != lhs.getInitialStates().end(); ++lhsIt){
240  if(*((*lhsIt).first) != *((*rhsIt).first)){
241  TRACE("hypro.datastructures.hybridAutomaton", (*lhsIt).first->getName() << " with hash " << (*lhsIt).first->hash() << " unequal to " << (*rhsIt).first->getName() << " with hash " << (*rhsIt).first->hash());
242  return false;
243  }
244  if((*lhsIt).second != (*rhsIt).second){
245  TRACE("hypro.datastructures.hybridAutomaton", "states were different.");
246  return false;
247  }
248  rhsIt++;
249  }
250  }
251  if(lhs.getLocalBadStates() != rhs.getLocalBadStates()){
252  TRACE("hypro.datastructures.hybridAutomaton", "no equality of local bads.");
253  return false;
254  }
255  if(lhs.getGlobalBadStates() != rhs.getGlobalBadStates()){
256  TRACE("hypro.datastructures.hybridAutomaton", "no equality of global bads.");
257  return false;
258  }
259 
260  return true;
261  }
