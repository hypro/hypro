183  bool equals(const std::set<T*>& lhs, const std::set<T*>& rhs) const {
184 
185  if(lhs.size() != rhs.size()) return false;
186  for(auto lhsIt = lhs.begin(); lhsIt != lhs.end(); ++lhsIt){
187  bool found = false;
188  for(auto rhsIt = rhs.begin(); rhsIt != rhs.end(); ++rhsIt) {
189  //std::cout << "now comparing " << (*(lhsIt))->hash() << " and " << (*(rhsIt))->hash() << std::endl;
190  if((*lhsIt)->hash() == (*rhsIt)->hash()){
191  found = true;
192  break;
193  }
194  }
195  if(!found){
196  return false;
197  }
198  }
199 
200  return true;
201  }
