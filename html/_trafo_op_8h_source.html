80  TrafoOp( const SupportFunctionNewT<Number, Converter, Setting>& origin, const matrix_t<Number>& A, const vector_t<Number>& b )
81  : originCount( 1 )
82  , mChildren( PointerVec( 1, nullptr ) )
83  , mDimension( origin.dimension() )
84  , currentExponent( 1 ) {
85  parameters = std::make_shared<const LinTrafoParameters<Number, Setting>>( A, b );
86 
87  origin.addOperation( this );
88  assert( this->getChildren().size() == 1 );
89 
90  // Determine, if we need to create new parameters or if this matrix and vector pair has already been used (recursive).
91  // in case this transformation has already been performed, parameters will be updated.
92 
93  origin.hasTrafo( parameters, A, b );
94 
95  // best points for reduction are powers of 2 thus we only use these points for possible reduction points
96  if ( Setting::LIN_TRANS_REDUCTION == true ) {
97  bool reduced;
98  do {
99  reduced = false;
100  if ( this->getChildren().at( 0 )->getType() == SFNEW_TYPE::TRAFO ) {
101  auto child = dynamic_cast<TrafoOp<Number, Converter, Setting>*>( this->getChildren().at( 0 ).get() );
102  if ( child->getParameters() == parameters && child->getCurrentExponent() == currentExponent ) {
103  //if(child->getParameters()->matrix().isApprox(parameters->matrix()) && child->getCurrentExponent() == currentExponent){
104  successiveTransformations = child->getSuccessiveTransformations() + 1;
105  } else {
106  successiveTransformations = 0;
107  }
108  } else {
109  successiveTransformations = 0;
110  }
111 
112  if ( successiveTransformations == unsigned( carl::pow( 2, parameters->power ) - 1 ) ) {
113  reduced = true;
114  currentExponent = currentExponent * ( carl::pow( 2, parameters->power ) );
115  std::shared_ptr<RootGrowNode<Number, Converter, Setting>> grandChild = this->getChildren().at( 0 );
116  for ( std::size_t i = 0; i < unsigned( carl::pow( 2, parameters->power ) - 1 ); i++ ) {
117  grandChild = grandChild->getChildren().at( 0 );
118  }
119  this->clearChildren();
120  this->addToChildren( grandChild );
121  assert( this->getChildren().size() == 1 );
122  }
123 
124  } while ( reduced == true );
125  }
126  }
