119  void addConstraint( const vector_t<Number>& normal, Number offset ) {
120  if ( mConstraints.cols() == 0 ) {
121  mConstraints = matrix_t<Number>( normal );
122  mConstants = vector_t<Number>::Zero( 1 );
123  mConstants << offset;
124  } else {
125  assert( mConstraints.cols() == normal.rows() );
126  mConstraints.conservativeResize( mConstraints.rows() + 1, mConstraints.cols() );
127  mConstants.conservativeResize( mConstraints.rows() + 1 );
128  mConstraints.row( mConstraints.rows() - 1 ) = normal;
129  mConstants( mConstants.rows() - 1 ) = offset;
130  }
131  mIsBox = TRIBOOL::NSET;
