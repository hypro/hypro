149  public:
150  /* Problem: Check emptiness of intersection of n convex objects
151  * Solution 1: Template Evaluation
152  * - For each template direction, evaluate every object except one in that direction and the one into the -direction
153  * - If -direction is bigger in absolute value, then intersection not empty
154  * - Needs n*t*LP worst time, since every evaluate needs to solve a LP
155  * - Can stop early if direction directly found -> best time possible time is n*LP
156  * - Scales bad in dimension
157  * Solution 2: Template Evaluation +
158  * - Like Solution 1, but parallel evaluation of 2 directions
159  * - Needs 0.5*n*t*LP time
160  * Solution 3: Center of Gravity (COG)
161  * - For each object, generate a point within the object
162  * - Either on a halfspace (needs an LP, how to choose which halfspace?)
163  * - Or the center of gravity ob the object (needs to compute vertices, then averaging -> expensive as hell)
164  * - Compute the COG of all points and determine the direction as the vector from the COG to the points
165  * - Evaluate all directions -> but how to determine emptiness from that?
166  * - Needs n*LP to generate points and n*LPs for evaluation
167  *
168  * All these solutions were not ideal for dealing with halfspaces. Algorithm now (for only 2 objects):
169  * - If both are not halfspaces: Use template evaluation: 2*#dir*LP
170  * - If one is a halfspace and the other not: Use satisfiesHalfspace: 2*LP
171  * - If none are halfspaces: Check linear dependency of halfspace normals: linear
172  */
173  bool empty( const std::vector<bool>& childrenEmpty ) const override {
174  //Current implementation uses Solution 1: template evaluation.
175 
176  //Quick check: If emptiness already computed, just return computation result
177  if ( RootGrowNode<Number, Converter, Setting>::mEmptyState != SETSTATE::UNKNOWN ) return ( RootGrowNode<Number, Converter, Setting>::mEmptyState == SETSTATE::EMPTY );
178 
179  //Quick check: If not already computed, check if at least one child is empty
180  for ( const auto& child : childrenEmpty ) {
181  if ( child ) {
182  RootGrowNode<Number, Converter, Setting>::mEmptyState = SETSTATE::EMPTY;
183  return true;
184  }
185  }
186 
187  //If no quick check worked: Turn children into SFs
188  std::vector<SupportFunctionNewT<Number, Converter, Setting>> sfChildren;
189  for ( std::size_t i = 0; i < this->getChildren().size(); ++i ) {
190  sfChildren.emplace_back( SupportFunctionNewT<Number, Converter, Setting>( this->getChildren().at( i ) ) );
191  }
192 
193  //Choose one child for which we will always evaluate in the negative direction
194  SupportFunctionNewT<Number, Converter, Setting> chosenOne = sfChildren.back();
195  sfChildren.pop_back();
196  std::vector<vector_t<Number>> directions = computeTemplate<Number>( mDimension, defaultTemplateDirectionCount );
197 
198  for ( const auto& direction : directions ) {
199  //Determine supportValue of direction and reverse direction for chosenOne
200  //Check if the supportValues don't overlap -> there is no intersection between at least two -> empty -> return true
201  //Check in both directions since directions and children do not have to be symmetrical
202  for ( const auto& child : sfChildren ) {
203  EvaluationResult<Number> chosenNegEval = chosenOne.evaluate( -direction, false );
204  EvaluationResult<Number> childPosEval = child.evaluate( direction, false );
205  if ( childPosEval.supportValue < -chosenNegEval.supportValue && childPosEval.errorCode != SOLUTION::INFTY && chosenNegEval.errorCode != SOLUTION::INFTY ) {
206  RootGrowNode<Number, Converter, Setting>::mEmptyState = SETSTATE::EMPTY;
207  return true;
208  }
209  EvaluationResult<Number> chosenPosEval = chosenOne.evaluate( direction, false );
210  EvaluationResult<Number> childNegEval = child.evaluate( -direction, false );
211  if ( -childNegEval.supportValue > chosenPosEval.supportValue && childNegEval.errorCode != SOLUTION::INFTY && chosenPosEval.errorCode != SOLUTION::INFTY ) {
212  RootGrowNode<Number, Converter, Setting>::mEmptyState = SETSTATE::EMPTY;
213  return true;
214  }
215  }
216  }
217  RootGrowNode<Number, Converter, Setting>::mEmptyState = SETSTATE::NONEMPTY;
218  return false;
219  }

240  //Erases all dimensions from a copy of dimensions that are denoted in dims
241  std::vector<std::size_t> intersectDims( const std::vector<std::vector<std::size_t>>& dims ) const override {
242  // we create the intersection of all results. Therefore we iterate over the first vector and check the other
243  // result vectors, if the respective element is contained. Iterating over the first is sufficient, as elements
244  // not in the first vector will not be in the intersection anyways.
245  std::vector<std::size_t> accumulatedResult;
246  for ( unsigned pos = 0; pos < dims.begin()->size(); ++pos ) {
247  std::size_t element = dims.begin()->at( pos );
248  bool elementInAllVectors = true;
249  for ( unsigned resIndex = 1; resIndex < dims.size(); ++resIndex ) {
250  if ( std::find( dims.at( resIndex ).begin(), dims.at( resIndex ).end(), element ) == dims.at( resIndex ).end() ) {
251  elementInAllVectors = false;
252  break;
253  }
254  }
255  if ( elementInAllVectors ) {
256  // if this is a projection, we need to remove non-projected dimensions -> we could not have been in a projection since
257  // since this part of the function was only executed if a node had more than 2 children
258  accumulatedResult.emplace_back( element );
259  }
260  }
261  return accumulatedResult;
262  }
