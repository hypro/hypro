48  //Doing some work here to keep the vertices sorted

53  while(true) {

67  thisInd += 1;
68  }

70  if(!inserted) facet.mVertices[dimension - 1] = visiblePoint;

86  void ExactQuickhull<Number, Euclidian>::FacetSpace::computeNormal(Facet& facet) {
89  matrix_t<Number> matrix(dimension, dimension + 1);
90 
91  if constexpr(Euclidian) {
92  for(size_t i = 0; i < dimension; ++i) {
93  matrix.row(i).head(dimension) = points[facet.mVertices[i]].transpose();
94  matrix.row(i)[dimension] = 1;
95  }
96  } else {
97  for(size_t i = 0; i < dimension; ++i) {
98  matrix.row(i) = points[facet.mVertices[i]].transpose();
99  }
100  }
101 
102  matrix_t<double> m = convert<mpq_class, double>(matrix);
103  TRACE("quickhull", "matrix" << std::endl << m);
104 
105  Eigen::FullPivLU<matrix_t<Number>> lu(matrix);
106 
107  assert(static_cast<size_t>(lu.rank()) == dimension);
108  TRACE("quickhull", "matrix rank " << lu.rank());
109 
110  point_t result = matrix.fullPivLu().kernel().col(0);
111 
112  vector_t<double> r = convert<mpq_class, double>(result);
113  TRACE("quickhull", "result" << std::endl << r);
114 
115  assert(matrix * result == point_t::Zero(dimension));
116 
117  for(size_t i = 0; i < dimension; ++i) {
118  facet.mNormal[i] = result[i];
119  }
120  facet.mOffset = -result[dimension];
121  }

148  void ExactQuickhull<Number, Euclidian>::FacetSpace::establishNeighborhood(facet_ind_t facet_i, facet_ind_t other_i) {
149  Facet& facet = facets[facet_i];
150  Facet& other = facets[other_i];
151 
152  size_t thisNeighborPosition = dimension - 1;
153  size_t otherNeighborPosition = dimension - 1;
154 
155  size_t thisInd = 0;
156  size_t otherInd = 0;
157 
158  bool hammingNonZero = false;
159 
160  while(true) {
161  if(facet.mVertices[thisInd] < other.mVertices[otherInd]) {
162  if(hammingNonZero) return;
163  hammingNonZero = true;
164  thisNeighborPosition = thisInd;
165  thisInd += 1;
166  } else if(facet.mVertices[thisInd] > other.mVertices[otherInd]) {
167  otherNeighborPosition = otherInd;
168  otherInd += 1;
169  } else {
170  thisInd += 1;
171  otherInd += 1;
172  }
173 
174  if(thisInd == dimension) break;
175 
176  if(otherInd == dimension) {
177  if(thisInd + 1 - hammingNonZero < otherInd) return;
178  else break;
179  }
180  }
181 
182  facet.mNeighbors[thisNeighborPosition] = other_i;
183  other.mNeighbors[otherNeighborPosition] = facet_i;
184  }
