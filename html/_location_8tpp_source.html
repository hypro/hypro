505 template <typename Number>
506 void Location<Number>::decompose( const Decomposition& decomposition ) {
507  if ( mLinearFlows.size() > 1 || mInvariant.size() > 1 ) {
508  //already decomposed
509  return;
510  }
511  DEBUG( "hypro.datastructures", "Flow Matrix before: \n " << mLinearFlows.at( 0 ).getFlowMatrix() );
512  auto& vpool = VariablePool::getInstance();
513 
514  // decompose flow
515  matrix_t<Number> oldFlow = mLinearFlows[0].getFlowMatrix();
516  auto oldIntervals = mRectangularFlows[0].getFlowIntervals();
517 
518  std::vector<linearFlow<Number>> newFlows;
519  std::vector<rectangularFlow<Number>> newRectangularFlows;
520  // for each set {i,j,..., k} select the i-th,j-th,...,k-th vector into a new square matrix
521  for ( auto set : decomposition ) {
522 #ifdef HYPRO_LOGGING
523  DEBUG( "hypro.datastructures", "decompose flow for set: {" );
524  for ( auto entry : set ) {
525  DEBUG( "hypro.datastructures", "" << entry << ", " );
526  }
527  DEBUG( "hypro.datastructures", "}" );
528 #endif
529  // +1 row for last-row of affine transformation
530  matrix_t<Number> rowMat = matrix_t<Number>::Zero( set.size() + 1, oldFlow.cols() );
531  // -1 because of last-row
532  for ( Eigen::Index index = 0; index < rowMat.rows() - 1; index++ ) {
533  // select the specific rows into rowMat
534  rowMat.row( index ) = oldFlow.row( set[index] );
535  }
536  //copy last row over
537  rowMat.row( rowMat.rows() - 1 ) = oldFlow.row( oldFlow.rows() - 1 );
538 
539  // +1 for constant column
540  matrix_t<Number> finMat = matrix_t<Number>::Zero( rowMat.rows(), set.size() + 1 );
541  // -1 for constant column
542  for ( Eigen::Index index = 0; index < finMat.cols() - 1; index++ ) {
543  finMat.col( index ) = rowMat.col( set[index] );
544  }
545  finMat.col( finMat.cols() - 1 ) = rowMat.col( rowMat.cols() - 1 );
546  DEBUG( "hypro.datastructures", "Final decomposed Flow: \n"
547  << finMat );
548  newFlows.emplace_back( linearFlow<Number>( finMat ) );
549 
550  rectangularFlow<Number> newRectFlow;
551  for ( auto entry : set ) {
552  auto it = oldIntervals.find( vpool.carlVarByIndex( entry ) );
553  if ( it != oldIntervals.end() ) {
554  newRectFlow.setFlowIntervalForDimension( it->second, it->first );
555  }
556  }
557  newRectangularFlows.emplace_back( newRectFlow );
558  }
559  mLinearFlows = std::move( newFlows );
560  mRectangularFlows = std::move( newRectangularFlows );
561 
562  // decompose invariant
563  mInvariant.decompose( decomposition );
564  // decompose transitions
565  for ( auto& transition : mTransitions ) {
566  transition->decompose( decomposition );
567  }
568  mHash = 0;
569 }
