88  std::vector<EvaluationResult<Number>> aggregate( std::vector<std::vector<EvaluationResult<Number>>>& resultStackBack, const matrix_t<Number>& /*currentParam*/ ) const override {
89  assert( resultStackBack.size() >= 2 );
90  assert( resultStackBack.at( 0 ).size() == resultStackBack.at( 1 ).size() );
91 
92  std::vector<EvaluationResult<Number>> accumulatedResult;
93  for ( unsigned index = 0; index < resultStackBack.front().size(); ++index ) {
94  EvaluationResult<Number> r;
95  r.optimumValue = vector_t<Number>::Zero( resultStackBack.front().front().optimumValue.rows() );
96  for ( const auto& res : resultStackBack ) {
97  if ( res[index].errorCode == SOLUTION::INFEAS ) return res;
98  if ( res[index].errorCode == SOLUTION::INFTY ) {
99  r.errorCode = SOLUTION::INFTY;
100  r.supportValue = 1;
101  break;
102  } else {
103  r.errorCode = SOLUTION::FEAS;
104  r.supportValue += res[index].supportValue;
105  r.optimumValue += res[index].optimumValue;
106  }
107  }
108  accumulatedResult.emplace_back( r );
109  }
110 
111  return accumulatedResult;
112  }

141  //Erases all dimensions from a copy of dimensions that are denoted in dims
142  std::vector<std::size_t> intersectDims( const std::vector<std::vector<std::size_t>>& dims ) const override {
143  // we create the intersection of all results. Therefore we iterate over the first vector and check the other
144  // result vectors, if the respective element is contained. Iterating over the first is sufficient, as elements
145  // not in the first vector will not be in the intersection anyways.
146  std::vector<std::size_t> accumulatedResult;
147  for ( unsigned pos = 0; pos < dims.begin()->size(); ++pos ) {
148  std::size_t element = dims.begin()->at( pos );
149  bool elementInAllVectors = true;
150  for ( unsigned resIndex = 1; resIndex < dims.size(); ++resIndex ) {
151  if ( std::find( dims.at( resIndex ).begin(), dims.at( resIndex ).end(), element ) == dims.at( resIndex ).end() ) {
152  elementInAllVectors = false;
153  break;
154  }
155  }
156  if ( elementInAllVectors ) {
157  // if this is a projection, we need to remove non-projected dimensions -> we could not have been in a projection since
158  // since this part of the function was only executed if a node had more than 2 children
159  accumulatedResult.emplace_back( element );
160  }
161  }
162  return accumulatedResult;
163  }
