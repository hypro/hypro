79  //Given two or more result vecs, sum them coefficientwise
80  std::vector<EvaluationResult<Number>> aggregate( std::vector<std::vector<EvaluationResult<Number>>>& resultStackBack, const matrix_t<Number>& /*currentParam*/ ) const override {
81  assert( resultStackBack.size() >= 2 );
82  assert( resultStackBack.at( 0 ).size() == resultStackBack.at( 1 ).size() );
83 
84  std::vector<EvaluationResult<Number>> accumulatedResult;
85  for ( unsigned index = 0; index < resultStackBack.front().size(); ++index ) {
86  EvaluationResult<Number> r;
87  r.optimumValue = vector_t<Number>::Zero( resultStackBack.front().front().optimumValue.rows() );
88  for ( const auto& res : resultStackBack ) {
89  if ( res[index].errorCode == SOLUTION::INFEAS ) return res;
90  if ( res[index].errorCode == SOLUTION::INFTY ) {
91  r.errorCode = SOLUTION::INFTY;
92  r.supportValue = 1;
93  break;
94  } else {
95  r.errorCode = SOLUTION::FEAS;
96  r.supportValue += res[index].supportValue;
97  r.optimumValue += res[index].optimumValue;
98  }
99  }
100  accumulatedResult.emplace_back( r );
101  }
102 
103  return accumulatedResult;
104  }

133  //Erases all dimensions from a copy of dimensions that are denoted in dims
134  std::vector<std::size_t> intersectDims( const std::vector<std::vector<std::size_t>>& dims ) const override {
135  // we create the intersection of all results. Therefore we iterate over the first vector and check the other
136  // result vectors, if the respective element is contained. Iterating over the first is sufficient, as elements
137  // not in the first vector will not be in the intersection anyways.
138  std::vector<std::size_t> accumulatedResult;
139  for ( unsigned pos = 0; pos < dims.begin()->size(); ++pos ) {
140  std::size_t element = dims.begin()->at( pos );
141  bool elementInAllVectors = true;
142  for ( unsigned resIndex = 1; resIndex < dims.size(); ++resIndex ) {
143  if ( std::find( dims.at( resIndex ).begin(), dims.at( resIndex ).end(), element ) == dims.at( resIndex ).end() ) {
144  elementInAllVectors = false;
145  break;
146  }
147  }
148  if ( elementInAllVectors ) {
149  // if this is a projection, we need to remove non-projected dimensions -> we could not have been in a projection since
150  // since this part of the function was only executed if a node had more than 2 children
151  accumulatedResult.emplace_back( element );
152  }
153  }
154  return accumulatedResult;
