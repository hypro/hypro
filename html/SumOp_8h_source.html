147  // we create the intersection of all results. Therefore we iterate over the first vector and check the other

152  std::size_t element = dims.begin()->at( pos );
153  bool elementInAllVectors = true;
154  for ( unsigned resIndex = 1; resIndex < dims.size(); ++resIndex ) {
155  if ( std::find( dims.at( resIndex ).begin(), dims.at( resIndex ).end(), element ) == dims.at( resIndex ).end() ) {
156  elementInAllVectors = false;

161  // if this is a projection, we need to remove non-projected dimensions -> we could not have been in a projection since
