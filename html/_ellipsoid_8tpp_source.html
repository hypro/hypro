77 EllipsoidT<Number, Converter> EllipsoidT<Number, Converter>::minkowskiSum( const EllipsoidT<Number, Converter>& _rhs, bool _approx ) const {
78  // TODO l needs to be normalized in order for this to work!!
79  vector_t<Number> l;
80  l.setZero( this->mDimension );
81  l( 0 ) = 1;
82  if ( mIsEmpty ) {
83  return EllipsoidT<Number, Converter>( _rhs );
84  }
85  Number lhsScalar, rhsScalar;
86  Number lhsFactor, rhsFactor;
87  lhsScalar = carl::sqrt( l.dot( this->mShapeMatrix * l ) );
88  rhsScalar = carl::sqrt( l.dot( _rhs.mShapeMatrix * l ) );
89  lhsFactor = ( lhsScalar + rhsScalar ) / lhsScalar;
90  rhsFactor = ( lhsScalar + rhsScalar ) / rhsScalar;
91  matrix_t<Number> matrix = lhsFactor * this->mShapeMatrix + rhsFactor * _rhs.mShapeMatrix;
92  if ( _approx ) {
93  return EllipsoidT<Number, Converter>( approxEllipsoidTMatrix( matrix ) );
94  } else {
95  return EllipsoidT<Number, Converter>( matrix );
96  }
97 }
