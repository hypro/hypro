300 void ZonotopeT<Number, Converter, Setting>::reduceOrder( unsigned limit ) {
301  //std::cout << __func__ << ": Current order: " << this->order() << std::endl;
302  while ( this->order() > limit ) {
303  matrix_t<Number> generators = mGenerators;
304  Eigen::Index dim = mGenerators.rows();
305 
306  // create duplicates of generators to sort them
307  std::vector<hypro::vector_t<Number>> sortedGenerators;
308  for ( Eigen::Index i = 0; i < generators.cols(); i++ ) {
309  sortedGenerators.push_back( generators.col( i ) );
310  }
311 
312  // Sort generators according to the difference between 1-norm and infty-norm
313  // (i.e.: How suitable are the generators to
314  // be overapproximated by an interval hull)
315  std::sort( sortedGenerators.begin(), sortedGenerators.end(), ZUtility::compareVectors<Number> );
316 
317  // Row-wise sum of all 2*dim chosen generators (absolute value)
318  vector_t<Number> sumVector = vector_t<Number>::Zero( dim );
319  for ( unsigned i = 0; i < 2 * ( dim ); i++ ) {
320  sumVector += sortedGenerators[i].array().abs().matrix();
321  }
322 
323  Eigen::Index numRemainingGenerators = Eigen::Index( sortedGenerators.size() - ( 2 * dim ) );
324 
325  matrix_t<Number> remainingGenerators = matrix_t<Number>( dim, numRemainingGenerators );
326 
327  // inserts the original remaining vectors
328  for ( Eigen::Index i = 0; i < numRemainingGenerators; i++ ) {
329  remainingGenerators.col( i ) = sortedGenerators[i + ( 2 * dim )];
330  }
331 
332  // calculate interval hull of first 2n generators
333  matrix_t<Number> intervalHull = sumVector.asDiagonal();
334 
335  matrix_t<Number> reducedGenerators = matrix_t<Number>( dim, remainingGenerators.cols() + dim );
336 
337  reducedGenerators << intervalHull, remainingGenerators;
338  mGenerators = reducedGenerators;
339  }
340  //std::cout << __func__ << ": Reduced order: " << this->order() << std::endl;
341 }

506 template <typename Number, typename Converter, typename Setting>
507 Number intersect2d( const ZonotopeT<Number, Converter, Setting>& input, const Halfspace<Number>& hp, int minOrMax ) {
508  assert( input.dimension() == hp.dimension() && input.dimension() == 2 &&
509  "zonotope dimension must be of same dimension (only dim 2 accepted) "
510  "as Halfspace" );
511  Eigen::Index size = Eigen::Index( input.size() );
512  matrix_t<Number> gUpper, gLower, generators = input.generators();
513  matrix_t<Number> s, s1, pmNext, pm = input.center();
514 
515  // ensure that all generators are pointing either upwards or all downwards (depends on minOrMax)
516  // minOrMax == 0 -> start from smallest point (cf. Algorithm Girard HSCC'08)
517  for ( Eigen::Index i = 0; i < size; i++ ) {
518  // point downwards
519  if ( ( generators( 1, i ) > 0 || ( generators( 1, i ) == 0 && generators( 0, i ) > 0 ) ) && minOrMax == 1 ) {
520  generators.col( i ) = -1 * generators.col( i );
521  } // point upwards
522  else if ( ( generators( 1, i ) < 0 || ( generators( 1, i ) == 0 && generators( 0, i ) < 0 ) ) && minOrMax == 0 ) {
523  generators.col( i ) = -1 * generators.col( i );
524  }
525  // set pm successively to be the largest or lowest point starting from the center
526  pm = pm - generators.col( i );
527  }
528 
529  // Determine x-direction to forward - "we should look left or right", delete all generators pointing in the wrong direction.
530  std::vector<unsigned> indexToRemove;
531  if ( pm( 0 ) <= hp.offset() ) {
532  for ( unsigned i = 0; i < generators.cols(); i++ ) {
533  if ( generators.col( i )( 0 ) <= 0 ) indexToRemove.push_back( i );
534  }
535  } else {
536  for ( unsigned i = 0; i < generators.cols(); i++ ) {
537  if ( generators.col( i )( 0 ) >= 0 ) indexToRemove.push_back( i );
538  }
539  }
540  for ( std::vector<unsigned>::reverse_iterator rit = indexToRemove.rbegin(); rit != indexToRemove.rend(); ++rit ) {
541  removeGenerator<Number>( *rit, generators );
542  }
543 
544  s = 2 * generators.rowwise().sum();
545 
546  while ( generators.cols() > 1 ) {
547  std::vector<unsigned> gLowerIndices, gUpperIndices;
548  // splitting the generators according to trigonometric order (angle) in relation to s
549  // TODO: check if atan is available for FLOAT_T types
550  Number sAngle = atan( (Number)s( 1 ) / (Number)s( 0 ) );
551  for ( unsigned i = 0; i < generators.cols(); i++ ) {
552  Number angle = atan( (Number)generators( 1, i ) / (Number)generators( 0, i ) );
553  if ( angle - sAngle <= 0 ) {
554  gLowerIndices.push_back( i );
555  } else if ( angle - sAngle >= 0 ) {
556  gUpperIndices.push_back( i );
557  }
558  }
559 
560  // after splitting, establish two sets gUpper, gLower holding the splitting result.
561  gUpper = generators;
562  gLower = generators;
563 
564  // std::cout << "MinOrMax: " << minOrMax << std::endl;
565  // std::cout << " S is: " << s << std::endl;
566  // std::cout << " PM is: " << pm << std::endl;
567  // std::cout << "Generators: "<< std::endl << generators << std::endl;
568  // std::cout << "Indices gLowerIndices: " << gLowerIndices << " and gUpperIndices: " << gUpperIndices << std::endl;
569 
570  for ( std::vector<unsigned>::reverse_iterator rit = gLowerIndices.rbegin(); rit != gLowerIndices.rend(); ++rit ) {
571  removeGenerator<Number>( *rit, gUpper );
572  }
573  for ( std::vector<unsigned>::reverse_iterator rit = gUpperIndices.rbegin(); rit != gUpperIndices.rend(); ++rit ) {
574  removeGenerator<Number>( *rit, gLower );
575  }
576  assert( gUpper.cols() + gLower.cols() == generators.cols() ); // Verify correct splitting.
577 
578  // Advance from pm to s1 (Line 17). The case distinction is necessary, as the trigonometric ordering is inverted
579  // when advancing towards the negative direction ( pm(0) > hp.offset() ).
580  if ( pm( 0 ) < hp.offset() ) {
581  s1 = ( minOrMax == 1 ) ? 2 * gUpper.rowwise().sum() : 2 * gLower.rowwise().sum();
582  } else {
583  s1 = ( minOrMax == 1 ) ? 2 * gLower.rowwise().sum() : 2 * gUpper.rowwise().sum();
584  }
585 
586  // pmNext is the end of the line segment [p, p+s1]. Verify intersection.
587  pmNext = pm + s1;
588  if ( ( pm( 0 ) <= hp.offset() && pmNext( 0 ) >= hp.offset() ) ||
589  ( pm( 0 ) >= hp.offset() && pmNext( 0 ) <= hp.offset() ) ) {
590  // TODO: Check, if this is still independent from the direction we are advancing to (left/right).
591  generators = ( minOrMax == 1 ) ? gLower : gUpper;
592 
593  // if there is no update, exit the loop -> copes with numeric problems regarding atan and linear dependence
594  // of generators
595  if ( s == s1 ) {
596  break;
597  }
598 
599  s = s1;
600  // This should not happen, otherwise we have an infinite loop.
601  assert( minOrMax != 1 || !gUpperIndices.empty() );
602  assert( minOrMax != 0 || !gLowerIndices.empty() );
603  } else {
604  // TODO: Check, if this is still independent from the direction we are advancing to (left/right).
605  generators = ( minOrMax == 1 ) ? gUpper : gLower;
606  s = s - s1;
607  pm = pmNext;
608 
609  // If there is no progress we can end the algorithm - it is an over-approximation, thus correct.
610  if ( s == s - s1 ) {
611  break;
612  }
613  }
614  }
615  pmNext = pm + s;
616  Number m = pm( 1 ) + ( hp.offset() - pm( 0 ) ) * ( pmNext( 1 ) - pm( 1 ) ) / ( pmNext( 0 ) - pm( 0 ) );
617  return m;
618 }

639 static ZonotopeT<Number, Converter, Setting> intersectZonotopeHalfspace( ZonotopeT<Number, Converter, Setting>& inputZonotope, const Halfspace<Number>& hp,
640  matrix_t<Number>& minMaxOfLine ) {
641  assert( inputZonotope.dimension() == hp.dimension() && "Zonotope and Halfspace have to be of similar dimensions" );
642  std::vector<vector_t<Number>> vertices = inputZonotope.computeZonotopeBoundary();
643  vertices.pop_back();
644 
645  std::size_t numPts = vertices.size();

647 
648  // Compute Halfspace points xhp
649  if ( hp.normal()( 1 ) == 0 ) {
650  Eigen::Matrix<Number, 2, 1> minVecY, maxVecY;
651  minVecY = *std::min_element( vertices.begin(), vertices.end(), ZUtility::compareYVal<Number> );
652  maxVecY = *std::max_element( vertices.begin(), vertices.end(), ZUtility::compareYVal<Number> );
653 
654  xhp.row( 1 ) << ( minVecY( 1 ) - carl::abs( Number( maxVecY( 1 ) - minVecY( 1 ) ) ) / 10 ),
655  ( maxVecY( 1 ) + carl::abs( Number( maxVecY( 1 ) - minVecY( 1 ) ) ) / 10 );
656  Eigen::Matrix<Number, 1, 2> eVec = { hp.offset(), hp.offset() };
657  xhp.row( 0 ) = ( eVec - hp.normal()( 1 ) * xhp.row( 1 ) ) / hp.normal()( 0 );
658  } else {
659  Eigen::Matrix<Number, 2, 1> minVecX, maxVecX;
660  minVecX = *std::min_element( vertices.begin(), vertices.end(), ZUtility::compareXVal<Number> );
661  maxVecX = *std::max_element( vertices.begin(), vertices.end(), ZUtility::compareXVal<Number> );
662 
663  xhp.row( 0 ) << ( minVecX( 0 ) - carl::abs( Number( maxVecX( 0 ) - minVecX( 0 ) ) ) / 10 ),
664  ( maxVecX( 0 ) + carl::abs( Number( maxVecX( 0 ) - minVecX( 0 ) ) ) / 10 );
665  Eigen::Matrix<Number, 1, 2> eVec = { hp.offset(), hp.offset() };
666  xhp.row( 1 ) = ( eVec - hp.normal()( 0 ) * xhp.row( 0 ) ) / hp.normal()( 1 );
667  }
668  // Real algorithm starts here
669 
670  ZUtility::Line_t<Number> lnhp( xhp.col( 0 ), ( xhp.col( 1 ) - xhp.col( 0 ) ) );
671 
672  // Naming variables: dzbme is used for sorting vertices nearest to the line
673  // dzb is used to keep track of d'*zb values
674  std::valarray<Number> dzbme( vertices.size() ), dzb( vertices.size() );
675  unsigned minIdx = 0, i;
676  for ( i = 0; i < vertices.size(); i++ ) {
677  assert( hp.normal().rows() == vertices[i].rows() );
678  matrix_t<Number> res = hp.normal().transpose() * vertices[i];
679  assert( res.rows() == 1 && res.cols() == 1 );
680  dzbme[i] = ( res( 0, 0 ) - hp.offset() );
681  dzb[i] = ( res( 0, 0 ) );
682  }
683 
684  std::vector<unsigned> ltZeroIdxArray, mtZeroIdxArray;
685  for ( i = 0; i < dzbme.size(); i++ ) {
686  if ( carl::abs( dzbme[i] ) < carl::abs( dzbme[minIdx] ) ) {
687  minIdx = i;
688  }
689  if ( dzbme[i] < 0 ) ltZeroIdxArray.push_back( i );
690  if ( dzbme[i] > 0 ) mtZeroIdxArray.push_back( i );
691  }
692 
693  // TODO: circshift vertices vector
694  dzb = dzb.cshift( minIdx );
695  dzbme = dzbme.cshift( minIdx );
696 
697  // Essentially create a temporary val array to do cshift operation
698  // and copy back value into vertices std::vector
699  std::valarray<vector_t<Number>> verticesValArray( vertices.data(), vertices.size() );
700  verticesValArray = verticesValArray.cshift( minIdx );
701  vertices.assign( std::begin( verticesValArray ), std::end( verticesValArray ) );
702 
703  std::size_t nil, il1p2, il2p1, il2p2;
704  if ( dzbme[0] < 0 ) {
705  nil = ltZeroIdxArray.size();
706  if ( dzbme[1] < 0 ) {
707  il1p2 = numPts;
708  il2p1 = nil;
709  } else {
710  il1p2 = 2;
711  il2p1 = numPts - nil + 1;
712  }
713  } else {
714  nil = mtZeroIdxArray.size();
715  if ( dzbme[1] > 0 ) {
716  il1p2 = numPts;
717  il2p1 = nil;
718  } else {
719  il1p2 = 2;
720  il2p1 = numPts - nil + 1;
721  }
722  }
723 
724  if ( nil == 1 )
725  il2p2 = 1;
726  else
727  il2p2 = il2p1 + 1;
728 
729  ZUtility::Line_t<Number> ln( vertices[0], vertices[il1p2 - 1] - vertices[0] );
730  // TODO: compute line intersection
731  Eigen::Matrix<Number, 2, 1> p1 = computeLineIntersection( ln, lnhp );
732 
733  ln.point = vertices[il2p1 - 1];
734  ln.direction = vertices[il2p2 - 1] - vertices[il2p1 - 1];
735  // TODO: compute line intersection
736  Eigen::Matrix<Number, 2, 1> p2 = computeLineIntersection( ln, lnhp );
737 
738  ZonotopeT<Number, Converter, Setting> resZonotope( ( p1 + p2 ) / 2, ( p1 - p2 ) / 2 );
739

741  minMaxOfLine.row( 0 ) = ( comparePoint( p1, p2 ) ) ? p1.transpose() : p2.transpose(); // min [x,y] here
742  minMaxOfLine.row( 1 ) = ( comparePoint( p1, p2 ) ) ? p2.transpose() : p1.transpose(); // max [x,y] here
743 
744  return resZonotope;

754  vector_t<Number> center = inputZonotope.center();

762 
763  identity.resize( inputZonotope.dimension(), inputZonotope.dimension() );
764 
765  identity.setIdentity();
766 
767  new_gen = ( identity - lambda * ( hp.normal().transpose() ) ) * H;
768  zg.setGenerators( new_gen );

775  matrix_t<Number>& minMaxOfLine ) {

777  "Intersect ND: input zonotope and input Halfspace must have same "
778  "dimensionality" );

784  // Using SVD to calculate nullspace (kernel)
785  kernel = svd.matrixV().block( 0, 1, svd.matrixV().rows(), svd.matrixV().cols() - 1 );

789  nd = kernel.cols();

793  matrix_t<Number> resultGenerators;

799  dpQg = dVec.transpose() * convert<Number, double>( inputZonotope.generators() );

802  // construct 2 dimensional Zonotope

804  Eigen::Matrix<double, 2, Eigen::Dynamic> projGenerators;
805

817  // matrix_t<Number> dummyMinMax;

822  p2 = convert<double, Number>( tempResZonotope.center() ) - convert<double, Number>( tempResZonotope.generators() );

826  // space as mentioned earlier

835  hp.offset() * hp.normal() );
836  resultZonotope.setGenerators( resultGenerators );
837 
838  return resultZonotope;

906 template <typename Number, typename Converter, typename Setting>
907 ZonotopeT<Number, Converter, Setting> ZonotopeT<Number, Converter, Setting>::intersectHalfspace( const Halfspace<Number>& rhs ) const {
908  if ( this->empty() ) {
909  return *this;
910  }
911  ZonotopeT<Number, Converter, Setting> result;
912  /* zs holds the 1-norm (Manhattan-Norm) of the direction projected onto the
913  * generators
914  * -> we sum the projections of the direction onto the generators (take only
915  * positive ones to prevent from
916  * canceling)
917  * -> this holds the farest we can go without leaving the zonotope
918  */
919  Number zs = ( rhs.normal().transpose() * this->mGenerators ).array().abs().sum();
920  // projection of normal on center -> offset induced by the center vector.
921  Number dc = rhs.normal().dot( this->mCenter );
922  // qu holds the maximal value one can go into direction of the Halfspace ->
923  // if this is less than the scalar, the
924  // zonotope is fully contained
925  Number qu = dc + zs, qd = dc - zs; // qd holds the minimal value of the zonotope generators
926  // evaluated into the direction of the
927  // Halfspace (with respect to the center)
928  // Qu and Qd are the offsets of the supporting hyperplanes in direction of the normal vector of the half-space
929  if ( qd <= rhs.offset() ) { // the zonotope is below the Halfspace -> there is an
930  // intersection
931  if ( qu <= rhs.offset() ) { // the zonotopes maximal evaluation is also below the
932  // Halfspace -> it is fully contained
933  result = *this;
934  } else { // partly contained
935  // sigma is half the distance between the Halfspace and the "lowest"
936  // point of the zonotope.
937  Number sigma = ( rhs.offset() - qd ) / 2, d = ( qd + rhs.offset() ) / 2; // d is half the distance between the lowest point and the center and the offset
938  matrix_t<Number> HHT = this->mGenerators * this->mGenerators.transpose();
939  vector_t<Number> lambda = HHT * rhs.normal() / ( ( rhs.normal().transpose() * HHT * rhs.normal() )( 0, 0 ) + sigma * sigma );
940  result.setCenter( this->mCenter + lambda * ( d - dc ) );
941  matrix_t<Number> identity = matrix_t<Number>::Identity( mDimension, mDimension );
942  result.setGenerators( ( identity - lambda * rhs.normal().transpose() ) * this->mGenerators );
943  result.addGenerators( sigma * lambda );
944  }
945  }
946  result.reduceOrder();
947  return result;
948 }

954  }
955  assert( mat.rows() == vec.rows() );

959  std::vector<std::pair<unsigned, unsigned>> linePairs;
960  std::set<unsigned> lineConstraints;

990  }
