56  ReachTreeNode<Representation>* currentNode = mWorkQueue.back();
57  mWorkQueue.pop_back();
58 
59  // if node is child of last node in path, collect it in return value
60  if ( currentNode->getDepth() == mPath.elements.size() + 1 ) {

63  }

65  REACHABILITY_RESULT safetyResult;

73  return { Failure{ currentNode } };
74  }
75  }
76 
77  // do not perform discrete jump if jump depth was reached
78  if ( currentNode->getDepth() == mFixedParameters.jumpDepth ) continue;
79 
80  // currentNode is last node in path, create and collect all children
81  if ( currentNode->getDepth() == mPath.elements.size() ) {

83  auto jumpSuccGen = worker.getJumpSuccessors( currentNode->getFlowpipe(), transition.get() );

85 
86  while ( auto* child = childGen.next() ) {
87  pathSuccessors.push_back( child );
88  }
89  }
90  } else {

92

95  for ( auto* child : currentNode->getChildren() ) {

97  matchedOne = true;
98  // check against path

101  }
102  }
103  }
104

108
