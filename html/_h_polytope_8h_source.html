326  template <typename N = Number, carl::DisableIf<std::is_same<N, double>> = carl::dummy>
327  void reduceNumberRepresentation( const std::vector<Point<Number>>& _vertices = std::vector<Point<Number>>(), unsigned limit = fReach_DENOMINATOR ) const {
328  //#ifdef REDUCE_NUMBERS
329  if ( Setting::REDUCE_NUMBERS == true ) {
330  TRACE( "hypro.hPolytope", "Attempt to reduce numbers." );
331  std::vector<Point<Number>> originalVertices;
332  if ( _vertices.empty() ) {
333  TRACE( "hypro.hPolytope", "No passed vertices, computed vertices." );
334  originalVertices = this->vertices();
335  } else {
336  TRACE( "hypro.hPolytope", "Use passed vertices." );
337  originalVertices = _vertices;
338  }
339  TRACE( "hypro.hPolytope", "Vertices empty: " << originalVertices.empty() );
340 
341  if ( !this->empty() ) {
342  // normal reduction
343  for ( unsigned planeIndex = 0; planeIndex < mHPlanes.size(); ++planeIndex ) {
344 #ifdef HPOLY_DEBUG_MSG
345  std::cout << "Original: " << mHPlanes.at( planeIndex ) << std::endl;
346 #endif
347  // find maximal value
348  Number largest = Number( 0 );
349  mHPlanes.at( planeIndex ).makeInteger();
350 #ifdef HPOLY_DEBUG_MSG
351  std::cout << "As Integer: " << mHPlanes.at( planeIndex ) << std::endl;
352 #endif
353  largest = carl::abs( mHPlanes.at( planeIndex ).offset() );
354  for ( unsigned i = 0; i < mDimension; ++i ) {
355  if ( carl::abs( mHPlanes.at( planeIndex ).normal()( i ) ) > largest ) {
356  largest = carl::abs( mHPlanes.at( planeIndex ).normal()( i ) );
357  }
358  }
359 
360  // reduce, if reduction is required
361  if ( largest > ( limit * limit ) ) {
362 #ifdef HPOLY_DEBUG_MSG
363  std::cout << "Actual reduction" << std::endl;
364 #endif
365  vector_t<Number> newNormal( mDimension );
366  for ( unsigned i = 0; i < mDimension; ++i ) {
367  newNormal( i ) = carl::floor( Number( ( mHPlanes.at( planeIndex ).normal()( i ) / largest ) * Number( limit ) ) );
368  assert( carl::abs( Number( mHPlanes.at( planeIndex ).normal()( i ) / largest ) ) <= Number( 1 ) );
369  assert( carl::isInteger( newNormal( i ) ) );
370  assert( newNormal( i ) <= Number( limit ) );
371  }
372  mHPlanes.at( planeIndex ).setNormal( newNormal );
373  Number newOffset = mHPlanes.at( planeIndex ).offset();
374  newOffset = carl::ceil( Number( ( newOffset / largest ) * Number( limit ) ) );
375  for ( const auto& vertex : originalVertices ) {
376  Number tmp = newNormal.dot( vertex.rawCoordinates() );
377  if ( tmp > newOffset ) {
378  newOffset = newOffset + ( tmp - newOffset );
379  assert( newNormal.dot( vertex.rawCoordinates() ) <= newOffset );
380  }
381  assert( Halfspace<Number>( newNormal, newOffset ).contains( vertex ) );
382  }
383  newOffset = carl::ceil( newOffset );
384 #ifdef HPOLY_DEBUG_MSG
385  std::cout << "Reduced to " << convert<Number, double>( newNormal ).transpose() << " <= " << carl::toDouble( newOffset ) << std::endl;
386 #endif
387  mHPlanes.at( planeIndex ).setOffset( newOffset );
388  }
389 #ifdef HPOLY_DEBUG_MSG
390  std::cout << "Reduced: " << mHPlanes.at( planeIndex ) << std::endl;
391 #endif
392  }
393 #ifdef HPOLY_DEBUG_MSG
394 //std::cout << "After Reduction: " << *this << std::endl;
395 #endif
396  }
397  }
398  //#endif
