339  template <typename N = Number, carl::DisableIf<std::is_same<N, double>> = carl::dummy>
340  void reduceNumberRepresentation( const std::vector<Point<Number>>& _vertices = std::vector<Point<Number>>(), unsigned limit = fReach_DENOMINATOR ) const {
341  //#ifdef REDUCE_NUMBERS
342  if ( Setting::REDUCE_NUMBERS == true ) {
343  TRACE( "hypro.hPolytope", "Attempt to reduce numbers." );
344  std::vector<Point<Number>> originalVertices;
345  if ( _vertices.empty() ) {
346  TRACE( "hypro.hPolytope", "No passed vertices, computed vertices." );
347  originalVertices = this->vertices();
348  } else {
349  TRACE( "hypro.hPolytope", "Use passed vertices." );
350  originalVertices = _vertices;
351  }
352  TRACE( "hypro.hPolytope", "Vertices empty: " << originalVertices.empty() );
353 
354  if ( !this->empty() ) {
355  // normal reduction
356  for ( unsigned planeIndex = 0; planeIndex < mHPlanes.size(); ++planeIndex ) {
357 #ifdef HPOLY_DEBUG_MSG
358  std::cout << "Original: " << mHPlanes.at( planeIndex ) << std::endl;
359 #endif
360  // find maximal value
361  Number largest = Number( 0 );
362  mHPlanes.at( planeIndex ).makeInteger();
363 #ifdef HPOLY_DEBUG_MSG
364  std::cout << "As Integer: " << mHPlanes.at( planeIndex ) << std::endl;
365 #endif
366  largest = carl::abs( mHPlanes.at( planeIndex ).offset() );
367  for ( unsigned i = 0; i < mDimension; ++i ) {
368  if ( carl::abs( mHPlanes.at( planeIndex ).normal()( i ) ) > largest ) {
369  largest = carl::abs( mHPlanes.at( planeIndex ).normal()( i ) );
370  }
371  }
372 
373  // reduce, if reduction is required
374  if ( largest > ( limit * limit ) ) {
375 #ifdef HPOLY_DEBUG_MSG
376  std::cout << "Actual reduction" << std::endl;
377 #endif
378  vector_t<Number> newNormal( mDimension );
379  for ( unsigned i = 0; i < mDimension; ++i ) {
380  newNormal( i ) = carl::floor( Number( ( mHPlanes.at( planeIndex ).normal()( i ) / largest ) * Number( limit ) ) );
381  assert( carl::abs( Number( mHPlanes.at( planeIndex ).normal()( i ) / largest ) ) <= Number( 1 ) );
382  assert( carl::isInteger( newNormal( i ) ) );
383  assert( newNormal( i ) <= Number( limit ) );
384  }
385  mHPlanes.at( planeIndex ).setNormal( newNormal );
386  Number newOffset = mHPlanes.at( planeIndex ).offset();
387  newOffset = carl::ceil( Number( ( newOffset / largest ) * Number( limit ) ) );
388  for ( const auto& vertex : originalVertices ) {
389  Number tmp = newNormal.dot( vertex.rawCoordinates() );
390  if ( tmp > newOffset ) {
391  newOffset = newOffset + ( tmp - newOffset );
392  assert( newNormal.dot( vertex.rawCoordinates() ) <= newOffset );
393  }
394  assert( Halfspace<Number>( newNormal, newOffset ).contains( vertex ) );
395  }
396  newOffset = carl::ceil( newOffset );
397 #ifdef HPOLY_DEBUG_MSG
398  std::cout << "Reduced to " << convert<Number, double>( newNormal ).transpose() << " <= " << carl::toDouble( newOffset ) << std::endl;
399 #endif
400  mHPlanes.at( planeIndex ).setOffset( newOffset );
401  }
402 #ifdef HPOLY_DEBUG_MSG
403  std::cout << "Reduced: " << mHPlanes.at( planeIndex ) << std::endl;
404 #endif
405  }
406 #ifdef HPOLY_DEBUG_MSG
407 //std::cout << "After Reduction: " << *this << std::endl;
408 #endif
409  }
410  }
411  //#endif
