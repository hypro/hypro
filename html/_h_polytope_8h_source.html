329  template <typename N = Number, carl::DisableIf<std::is_same<N, double>> = carl::dummy>
330  void reduceNumberRepresentation( const std::vector<Point<Number>>& _vertices = std::vector<Point<Number>>(), unsigned limit = fReach_DENOMINATOR ) const {
331  //#ifdef REDUCE_NUMBERS
332  if ( Setting::REDUCE_NUMBERS == true ) {
333  TRACE( "hypro.hPolytope", "Attempt to reduce numbers." );
334  std::vector<Point<Number>> originalVertices;
335  if ( _vertices.empty() ) {
336  TRACE( "hypro.hPolytope", "No passed vertices, computed vertices." );
337  originalVertices = this->vertices();
338  } else {
339  TRACE( "hypro.hPolytope", "Use passed vertices." );
340  originalVertices = _vertices;
341  }
342  TRACE( "hypro.hPolytope", "Vertices empty: " << originalVertices.empty() );
343 
344  if ( !this->empty() ) {
345  // normal reduction
346  for ( unsigned planeIndex = 0; planeIndex < mHPlanes.size(); ++planeIndex ) {
347 #ifdef HPOLY_DEBUG_MSG
348  std::cout << "Original: " << mHPlanes.at( planeIndex ) << std::endl;
349 #endif
350  // find maximal value
351  Number largest = Number( 0 );
352  mHPlanes.at( planeIndex ).makeInteger();
353 #ifdef HPOLY_DEBUG_MSG
354  std::cout << "As Integer: " << mHPlanes.at( planeIndex ) << std::endl;
355 #endif
356  largest = carl::abs( mHPlanes.at( planeIndex ).offset() );
357  for ( unsigned i = 0; i < mDimension; ++i ) {
358  if ( carl::abs( mHPlanes.at( planeIndex ).normal()( i ) ) > largest ) {
359  largest = carl::abs( mHPlanes.at( planeIndex ).normal()( i ) );
360  }
361  }
362 
363  // reduce, if reduction is required
364  if ( largest > ( limit * limit ) ) {
365 #ifdef HPOLY_DEBUG_MSG
366  std::cout << "Actual reduction" << std::endl;
367 #endif
368  vector_t<Number> newNormal( mDimension );
369  for ( unsigned i = 0; i < mDimension; ++i ) {
370  newNormal( i ) = carl::floor( Number( ( mHPlanes.at( planeIndex ).normal()( i ) / largest ) * Number( limit ) ) );
371  assert( carl::abs( Number( mHPlanes.at( planeIndex ).normal()( i ) / largest ) ) <= Number( 1 ) );
372  assert( carl::isInteger( newNormal( i ) ) );
373  assert( newNormal( i ) <= Number( limit ) );
374  }
375  mHPlanes.at( planeIndex ).setNormal( newNormal );
376  Number newOffset = mHPlanes.at( planeIndex ).offset();
377  newOffset = carl::ceil( Number( ( newOffset / largest ) * Number( limit ) ) );
378  for ( const auto& vertex : originalVertices ) {
379  Number tmp = newNormal.dot( vertex.rawCoordinates() );
380  if ( tmp > newOffset ) {
381  newOffset = newOffset + ( tmp - newOffset );
382  assert( newNormal.dot( vertex.rawCoordinates() ) <= newOffset );
383  }
384  assert( Halfspace<Number>( newNormal, newOffset ).contains( vertex ) );
385  }
386  newOffset = carl::ceil( newOffset );
387 #ifdef HPOLY_DEBUG_MSG
388  std::cout << "Reduced to " << convert<Number, double>( newNormal ).transpose() << " <= " << carl::toDouble( newOffset ) << std::endl;
389 #endif
390  mHPlanes.at( planeIndex ).setOffset( newOffset );
391  }
392 #ifdef HPOLY_DEBUG_MSG
393  std::cout << "Reduced: " << mHPlanes.at( planeIndex ) << std::endl;
394 #endif
395  }
396 #ifdef HPOLY_DEBUG_MSG
397 //std::cout << "After Reduction: " << *this << std::endl;
398 #endif
399  }
400  }
401  //#endif
