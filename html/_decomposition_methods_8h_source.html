81  auto locPtrs = automaton.getLocations();
82  std::for_each( locPtrs.begin(), locPtrs.end(), [&res]( const auto *loc ) {
83  DynamicType cur = getDynamicType( *loc );
84  if ( res == DynamicType::undefined ) {
85  res = cur;
86  } else {
87  res = cur == res ? res : DynamicType::mixed;
88  }
89  } );
90  assert( res != DynamicType::undefined );

96 template <typename Number>
97 DynamicType getDynamicType( const Location<Number> &location ) {
98  // if no flows are defined the automaton is discrete
99  if ( location.getFlows().empty() ) {
100  return DynamicType::discrete;
101  }
102 
103  // first initialization, can either be linear or rectangular
104  DynamicType res{ DynamicType::undefined };
105  // flow
106  for ( std::size_t i = 0; i < location.getFlowTypes().size(); ++i ) {
107  auto curType = std::visit( flowDynamicsTypeVisitor{}, location.getFlows()[i] );
108  if ( res != DynamicType::undefined && res != curType ) {
109  return DynamicType::mixed;
110  }
111  res = curType;
112  }
113  // invariant
114  if ( res != DynamicType::linear && res != DynamicType::rectangular ) {
115  if ( !location.getInvariant().isAxisAligned() ) {
116  res = DynamicType::linear;
117  }
118  }
119  // transitions: guard & reset
120  if ( res != DynamicType::linear && res != DynamicType::rectangular ) {
121  for ( const auto &transition : location.getTransitions() ) {
122  // check guard
123  if ( !transition->getGuard().isAxisAligned() ) {
124  res = DynamicType::linear;
125  }
126  // corner case: resets to intervals
127  if ( std::any_of( transition->getReset().getIntervalResets().begin(), transition->getReset().getIntervalResets().end(), []( const auto &assignment ) { return !assignment.isIdentity(); } ) ) {
128  return DynamicType::mixed;
129  }
130  // check reset
131  if ( res != DynamicType::linear ) {
132  std::for_each( transition->getReset().getAffineResets().begin(), transition->getReset().getAffineResets().end(), [&res]( const auto &reset ) {
133  if ( !reset.isConstant() ) {
134  res = DynamicType::linear;
135  }
136  } );
137  }
138  }
139  }
140  return res;
141 }
