104 bool Grid<Number>::colorAtInduced( const Point<std::size_t> &inducedPoint ) const {
105  // std::cout << __func__ << " " << inducedPoint << std::endl;
106  // the point is not a vertex (vertices are inserted at the beginning) and not
107  // yet calculated.
108  if ( mGridMap.find( inducedPoint ) != mGridMap.end() ) {
109  // std::cout << "known: " << mGridMap.at(inducedPoint) << std::endl;
110  return mGridMap.at( inducedPoint );
111  }
112 
113  // if one coordinate is zero just go along the axes towards origin and count
114  // vertices (origin is always white)
115  bool containsZero = false;
116  std::vector<std::size_t> nonZero;
117  for ( std::size_t d = 0; d < inducedPoint.dimension(); ++d ) {
118  if ( inducedPoint.at( d ) == 0 ) {
119  // std::cout << "Zero at dimension " << d << std::endl;
120  containsZero = true;
121  } else {
122  nonZero.push_back( d );
123  }
124  }
125  if ( containsZero ) { // move along axes to origin, collect predecessors
126  Point<std::size_t> predecessor( iPredecessorInduced( inducedPoint, nonZero.back() ) );
127  std::vector<Point<std::size_t>> predecessors;
128  while ( !nonZero.empty() ) {
129  std::size_t dir = nonZero.back();
130  // std::cout << "Chosen predecessor direction: " << dir << std::endl;
131  while ( mGridMap.find( predecessor ) == mGridMap.end() &&
132  ( iPredecessorInduced( predecessor, dir ) != predecessor ) ) {
133  // std::cout << "Added predecessor " << predecessor << std::endl;
134  predecessors.push_back( predecessor );
135  predecessor = iPredecessorInduced( predecessor, dir );
136  }
137  nonZero.pop_back();
138  }
139  // std::cout << "Found vertex: " << predecessor << ", set color to " <<
140  // mGridMap.at(predecessor) << std::endl;
141  bool color = mGridMap.at( predecessor );
142  mGridMap[inducedPoint] = color;
143  while ( !predecessors.empty() ) {
144  mGridMap[predecessors.back()] = color;
145  predecessors.pop_back();
146  }
147  // std::cout << "Color " << inducedPoint << ": " << color << std::endl;
148  return color;
149  }

151  /* calculate color recursively:
152  * - go through all directions j: There exists j such that for all
153  * j-neighbors(x) (except x itself) it holds:
154  * color(j-pred(jneigh)) == color(j-neigh)
155  * -> color(x) = color(j-pred(j-neigh))
156  */

162  // std::cout << "Evaluate "<< inducedPoint <<" in direction " << j <<
163  // std::endl;
164  setColor = true;
165  std::vector<Point<std::size_t>> jneighs = iNeighborhoodInduced( inducedPoint, j );
166  // for(const auto& p : jneighs) {
167  // std::cout << j << "-Neighbor: " << p << std::endl;
168  //}
169  for ( const auto &x : jneighs ) {
170  if ( x != inducedPoint ) {
171  // std::cout << "Get color at " << x << std::endl;
172  color = colorAtInduced( x );
173 
174  if ( color == colorAtInduced( iPredecessorInduced( x, j ) ) ) {
175  color = colorAtInduced( iPredecessorInduced( inducedPoint, j ) );
176  break;
177  } else {
178  setColor = false;
179  break;
180  }
181  }
182  }
183  if ( setColor ) {
184  mGridMap[inducedPoint] = color;
185  break;
186  }
187  }
188  // std::cout << "Color " << inducedPoint << ": " << color << std::endl;
189  return color;
190 }

351  std::vector<Point<std::size_t>> result;
352  std::size_t d = this->dimension();
353 
354  // calculate direct predecessor point in all dimensions
355  Point<std::size_t> directPredecessor( _inducedPoint );
356  for ( std::size_t dim = 0; dim < d; ++dim ) {
357  directPredecessor[dim] = _inducedPoint.at( dim ) > 0 ? _inducedPoint.at( dim ) - 1 : 0;
358  }
359 
360  // get all 2^d i-neighbors by combination of the point and its direct
361  // predecessor
362  for ( std::size_t i = 0; i < std::size_t( std::pow( 2, d ) ); ++i ) {
363  std::bitset<MAX_DIMENSION_LIMIT> map( i );
364  std::size_t pos = 0;
365  Point<std::size_t> neighbor( directPredecessor );
366  while ( pos < d ) {
367  if ( map.test( pos ) ) neighbor[pos] = _inducedPoint.at( pos );
368  ++pos;
369  }

371  }

409  // std::cout << "Point is not on grid coordinate for dim " << i <<
410  // std::endl;
411  return false;
412  }
413

415 
416  // std::cout << "Is on " << i << "-facet " << _point << " (" << inducedPoint

420  if ( inducedPoint == Point<std::size_t>::Zero( inducedPoint.dimension() ) ) {

580 template <typename Number>
581 std::pair<Point<std::size_t>, bool> Grid<Number>::calculateInduced( const Point<Number> &point ) const {
582  vector_t<std::size_t> coordinates( mInducedGridPoints.size() );
583  bool precise = true;
584  // std::cout << *this << std::endl;
585  for ( std::size_t dim = 0; dim < mInducedGridPoints.size(); ++dim ) {
586  std::vector<Number> inducedGridPoints = mInducedGridPoints.at( dim );
587 
588  // get the position of the first element greater than the coordinate + 1
589  std::size_t pos = 0;
590  while ( pos < inducedGridPoints.size() - 1 && inducedGridPoints[pos] < point.at( dim ) ) ++pos;
591 
592  // insert the index of the element one before the element found above
593  if ( inducedGridPoints[pos] <= point.at( dim ) ) { // here the points coordinate is either equal or we reached
594  // the last element and it is larger
595  coordinates[Eigen::Index( dim )] = pos;
596  } else {
597  // std::cout << "dim: " << dim << ", " << inducedGridPoints[pos] << " > "
598  // << point.at(dim) << std::endl;
599  coordinates[Eigen::Index( dim )] = pos - 1;
600  precise = false;
