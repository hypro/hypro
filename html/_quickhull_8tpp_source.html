144 bool ExactQuickhull<Number, Euclidian>::constructInitialFacet() {
145  Facet& facet = fSpace.insertNew();
146 
147  Eigen::FullPivLU<matrix_t<Number>> lu( dimension, dimension + 1 );
148  matrix_t<Number> matrix = matrix_t<Number>::Zero( dimension, dimension + 1 );
149 
150  //Do first iteration
151  facet.mVertices[0] = 0;
152 
153  //Set the first row
154  if constexpr ( Euclidian ) {
155  matrix.row( 0 ).head( dimension ) = points[0].transpose();
156  matrix( 0, dimension ) = 1;
157  } else {
158  matrix.row( 0 ) = points[0].transpose();
159  }
160 
161  //Compute normal and offset
162  lu.compute( matrix );
163  vector_t<Number> result = lu.kernel().col( 0 );
164 
165  facet.mNormal = result.head( dimension );
166  facet.mOffset = -result[dimension];
167 
168  for ( size_t d = 1; d < dimension; ++d ) {
169  auto [onPlane, furthestPoint_i] = findFurthestPoint( facet );
170 
171  if ( onPlane ) {
172  constructLowerDimensional();
173  return true;
174  }
175 
176  facet.mVertices[d] = furthestPoint_i;
177  //Set the (d+1)-th row
178  if constexpr ( Euclidian ) {
179  matrix.row( d ).head( dimension ) = points[furthestPoint_i].transpose();
180  matrix( d, dimension ) = 1;
181  } else {
182  matrix.row( d ) = points[furthestPoint_i].transpose();
183  }
184 
185  //Compute normal and offset
186  lu.compute( matrix );
187  result = lu.kernel().col( 0 );
188 
189  facet.mNormal = result.head( dimension );
190  facet.mOffset = -result[dimension];
191  }
192  std::sort( facet.mVertices.begin(), facet.mVertices.end() );
193 
194  return false;
195 }

243 template <typename Number, bool Euclidian>
244 void ExactQuickhull<Number, Euclidian>::buildCone( facet_ind_t currentFacet_i, point_ind_t visiblePoint_i, bitset_t& visited ) {
245  fSpace.deleteFacet( currentFacet_i );
246 
247  point_t& visiblePoint = points[visiblePoint_i];
248 
249  for ( size_t neighbor_pos = 0; neighbor_pos < dimension; ++neighbor_pos ) {
250  facet_ind_t neighbor_i = fSpace.facets[currentFacet_i].mNeighbors[neighbor_pos];
251 
252  if ( !visited[neighbor_i] ) {
253  //neighbor is "inside" horizon
254  if ( fSpace.facets[neighbor_i].visible( visiblePoint ) ) {
255  visited.set( neighbor_i );
256  buildCone( neighbor_i, visiblePoint_i, visited );
257  } else {
258  //this ridge to the neighbor is part of the horizon
259 
260  //Find the index of the ridge from the facet outside the horizon (neighbor_i) to the facet inside the horizon (currentFacet_i).
261  size_t ridgeIndex_outer_inner = fSpace.facets[neighbor_i].findNeighborIndex( currentFacet_i );
262 
263  size_t differentiatingPosition = fSpace.insertConePart( neighbor_i, visiblePoint_i, ridgeIndex_outer_inner );
264 
265 #ifndef NDEBUG
266  fSpace.containsAllPoints( fSpace.facets.back() );
267 #endif
268  //Set the outside facet as neighbor of the new facet at the index where the differentiating point was inserted.
269  fSpace.facets.back().mNeighbors[differentiatingPosition] = neighbor_i;
270  //Replace the facet inside the horizon (currentFacet_i) as neighbor of neighbor_i with the newly created facet.
271  fSpace.facets[neighbor_i].mNeighbors[ridgeIndex_outer_inner] = fSpace.facets.size() - 1 /*The index of the new facet*/;
272 
273  findConeNeighbors( fSpace.facets.size() - 1 );
274  }
275  }
