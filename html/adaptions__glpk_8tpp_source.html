39 template <typename Number>
40 vector_t<Number> refineSolution( glpk_context& context, const matrix_t<Number>& constraints, const vector_t<Number>& constants ) {
41  matrix_t<Number> exactSolutionMatrix = matrix_t<Number>::Zero( constraints.cols(), constraints.cols() );
42  vector_t<Number> exactSolutionVector = vector_t<Number>::Zero( constraints.cols() );
43  unsigned pos = 0;
44  if ( glp_get_obj_dir( context.lp ) == GLP_MAX ) {
45  for ( unsigned i = 1; i <= constraints.rows(); ++i ) {
46  // we search for d non-basic variables at their upper bound, which define the optimal point.
47  if ( glp_get_row_stat( context.lp, i ) == GLP_NU ) {
48  exactSolutionMatrix.row( pos ) = constraints.row( i - 1 );
49  exactSolutionVector( pos ) = constants( i - 1 );
50  ++pos;
51  }
52  }
53  } else {
54  assert( glp_get_obj_dir( context.lp ) == GLP_MIN );
55  for ( unsigned i = 1; i <= constraints.rows(); ++i ) {
56  // we search for d non-basic variables at their lower bound, which define the optimal point.
57  if ( glp_get_row_stat( context.lp, i ) == GLP_NL ) {
58  exactSolutionMatrix.row( pos ) = constraints.row( i - 1 );
59  exactSolutionVector( pos ) = constants( i - 1 );
60  ++pos;
61  }
62  }
63  }
64  // solve and return
65  return Eigen::FullPivLU<matrix_t<Number>>( exactSolutionMatrix ).solve( exactSolutionVector );
66 }

76  // setup glpk

78  glp_set_col_bnds( context.lp, i + 1, GLP_FR, 0.0, 0.0 );
79  glp_set_obj_coef( context.lp, i + 1, carl::toDouble( _direction( i ) ) );
80  }
81  /* solve problem */
82  if ( useExact ) {
83  glp_exact( context.lp, &context.parm );
84  } else {
85  glp_simplex( context.lp, &context.parm );
86  }
87 
88  vector_t<Number> exactSolution;
89  switch ( glp_get_status( context.lp ) ) {
90  case GLP_OPT:
91  case GLP_FEAS: {
92  // if satisfiable, derive exact solution by intersecting all constraints, which are at their upper respectively lower bounds .
93  exactSolution = refineSolution( context, constraints, constants );
94  return EvaluationResult<Number>( _direction.dot( exactSolution ), exactSolution, SOLUTION::FEAS );
95  break;
96  }
97  case GLP_UNBND: {
98  vector_t<Number> glpkModel( constraints.cols() );
99  for ( unsigned i = 1; i <= constraints.cols(); ++i ) {
100  glpkModel( i - 1 ) = carl::rationalize<Number>( glp_get_col_prim( context.lp, i ) );
101  }
102  return EvaluationResult<Number>( 1, glpkModel, SOLUTION::INFTY );
103  break;
104  }
105  default:
106  return EvaluationResult<Number>( 0, vector_t<Number>::Zero( 1 ), SOLUTION::INFEAS );

145  for ( std::size_t constraintIndex = std::size_t( constraints.rows() - 1 );; --constraintIndex ) {

148  //assert(actualRes.supportValue <= constants(constraintIndex));

155  res.push_back( constraintIndex );

163  }
