157  int numberOfConstraints = int( constraints.rows() );
158  if ( numberOfConstraints > 0 ) {
159  // convert constraint constants
160  glp_add_rows( lp, numberOfConstraints );
161  for ( int i = 0; i < numberOfConstraints; i++ ) {
162  // Set relation symbols correctly
163  switch ( relations[i] ) {
164  case carl::Relation::LEQ:
165  // set upper bounds, lb-values (here 0.0) are ignored.
166  glp_set_row_bnds( lp, i + 1, GLP_UP, 0.0, carl::toDouble( constants( i ) ) );
167  break;
168  case carl::Relation::GEQ:
169  // if it is an equality, the value is read from the lb-value, ub.values (here 0.0) are ignored.
170  glp_set_row_bnds( lp, i + 1, GLP_LO, carl::toDouble( constants( i ) ), 0.0 );
171  break;
172  case carl::Relation::EQ:
173  // if it is an equality, the value is read from the lb-value, ub.values (here 0.0) are ignored.
174  glp_set_row_bnds( lp, i + 1, GLP_FX, carl::toDouble( constants( i ) ), 0.0 );
175  break;
176  default:
177  // glpk cannot handle strict inequalities.
178  assert( false );
179  std::cout << "This should not happen." << std::endl;
180  }
181  }
182  // add cols here
183  int cols = int( constraints.cols() );
184  glp_add_cols( lp, cols );
185  createArrays( unsigned( numberOfConstraints * cols ) );
186 
187  // convert constraint matrix
188  ia[0] = 0;
189  ja[0] = 0;
190  ar[0] = 0;
191  assert( constraints.size() == numberOfConstraints * cols );
