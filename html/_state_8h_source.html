552  friend bool operator==( const State<Number, Representation, Rargs...>& lhs, const State<Number, Representation, Rargs...>& rhs ) {
553  // quick checks first
554  if ( lhs.getNumberSets() != rhs.getNumberSets() || lhs.mTimestamp != rhs.mTimestamp ) {
555  return false;
556  }
557 
558  // light-weight check on mIsEmpty
559  for ( const auto b1 : lhs.getEmptyStates() ) {
560  for ( const auto b2 : rhs.getEmptyStates() ) {
561  if ( ( b1 == TRIBOOL::TRUE && b2 == TRIBOOL::FALSE ) || ( b2 == TRIBOOL::TRUE && b1 == TRIBOOL::FALSE ) ) {
562  return false;
563  }
564  }
565  }
566 
567  // location-based checks
568  if ( lhs.mLoc != nullptr ) {
569  if ( rhs.mLoc != nullptr ) {
570  if ( *( lhs.mLoc ) != *( rhs.mLoc ) ) {
571  return false;
572  }
573  } else {
574  return false;
575  }
576  } else {
577  if ( rhs.mLoc != nullptr ) {
578  return false;
579  }
580  }
581 
582  // set-based checks
583  for ( std::size_t i = 0; i < lhs.getNumberSets(); ++i ) {
584  if ( lhs.getSetType( i ) != rhs.getSetType( i ) ) {
585  return false;
586  }
587  if ( !std::visit( genericCompareVisitor(), lhs.getSet( i ), rhs.getSet( i ) ) ) {
588  return false;
589  }
590  }
591  return true;
592  }
