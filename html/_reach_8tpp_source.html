43 typename Reach<Number, ReacherSettings, State>::flowpipe_t Reach<Number, ReacherSettings, State>::computeForwardTimeClosure( ReachTreeNode<State>* currentTreeNode ) {
44  assert( !currentTreeNode->getState().getTimestamp().isUnbounded() );
45 #ifdef REACH_DEBUG
46  INFO( "hypro.reacher", "Location: " << currentTreeNode->getState().getLocation()->hash() );
47  INFO( "hypro.reacher", "Location printed : " << *( currentTreeNode->getState().getLocation() ) );
48  INFO( "hypro.reacher", "Time step size: " << mSettings.timeStep );
49 #endif
50  // new empty Flowpipe
51  typename Reach<Number, ReacherSettings, State>::flowpipe_t flowpipe;
52  std::vector<std::tuple<Transition<Number>*, State>> nextInitialSets;
53 
54  std::tuple<hypro::CONTAINMENT, State, matrix_t<Number>, vector_t<Number>, Box<Number>> initialSetup = computeFirstSegment<Number, State>( currentTreeNode->getState(), mSettings.timeStep );
55 #ifdef REACH_DEBUG
56  INFO( "hypro.reacher", "Valuation fulfills Invariant?: " << std::get<0>( initialSetup ) );
57 #endif
58  if ( std::get<0>( initialSetup ) != CONTAINMENT::NO ) { // see convenienceOperators for details
59  assert( !std::get<1>( initialSetup ).getTimestamp().isUnbounded() );
60  bool noFlow = false;
61 
62  // if the location does not have dynamic behaviour, check guards and exit loop.
63  if ( std::get<2>( initialSetup ) == matrix_t<Number>::Identity( std::get<2>( initialSetup ).rows(), std::get<2>( initialSetup ).cols() ) &&
64  std::get<3>( initialSetup ) == vector_t<Number>::Zero( std::get<3>( initialSetup ).rows() ) ) {
65  noFlow = true;
66  // Collect potential new initial states from discrete behaviour.
67  if ( int( mCurrentLevel ) <= mSettings.jumpDepth || mSettings.jumpDepth < 0 ) {
68  INFO( "hypro.reacher", "-- Checking Transitions from initial!" );
69  checkTransitions( currentTreeNode->getState(), carl::Interval<tNumber>( tNumber( 0 ), mSettings.timeBound ), nextInitialSets );
70  }
71  }
72 
73  // insert first Segment into the empty flowpipe
74  State currentSegment;
75  if ( noFlow ) {
76  currentSegment = currentTreeNode->getState();
77  } else {
78  currentSegment = std::get<1>( initialSetup );
79  }
80  flowpipe.push_back( currentSegment );
81 
82  // Check for bad states intersection. The first segment is validated against the invariant, already.
83  if ( intersectBadStates( currentSegment ) ) {
84  // clear queue to stop whole algorithm
85  while ( !mWorkingQueue.isEmpty() ) {
86  mWorkingQueue.dequeueFront();
87  }
88  return flowpipe;
89  }
90 
91  // Set after linear transformation
92  State nextSegment;
93 #ifdef REACH_DEBUG
94  if ( !noFlow ) {
95  INFO( "hypro.reacher", "--- Loop entered ---" );
96  }
97 #endif
98  // the first segment covers one time step already
99  tNumber currentLocalTime = mSettings.timeStep;
100  bool forward = mSettings.timeStep > 0;
101  INFO( "hypro.reacher", "Compute forwards time closure: " << forward );
102  // intersection of bad states and violation of invariant is handled inside the loop
103  while ( !noFlow && ( ( forward && currentLocalTime <= mSettings.timeBound ) || ( !forward && currentLocalTime >= -mSettings.timeBound ) ) ) {
104  INFO( "hypro.reacher", "Time: " << std::setprecision( 4 ) << std::setw( 8 ) << std::fixed << carl::toDouble( currentLocalTime ) );
105  // Verify transitions on the current set.
106  if ( int( mCurrentLevel ) < mSettings.jumpDepth || mSettings.jumpDepth < 0 ) {
107  //State currentState = _state;
108  State currentState = currentSegment;
109  INFO( "hypro.reacher", "--- Checking Transitions!" );
110  checkTransitions( currentState, currentState.getTimestamp(), nextInitialSets );
111  }
112 
113  // perform linear transformation on the last segment of the flowpipe
114 
115  nextSegment = currentSegment.partiallyApplyTimeStep( ConstraintSet<Number>( std::get<2>( initialSetup ), std::get<3>( initialSetup ) ), mSettings.timeStep, 0 );
116 
117  // extend flowpipe (only if still within Invariant of location)
118  std::pair<hypro::CONTAINMENT, State> newSegment = nextSegment.satisfies( currentTreeNode->getState().getLocation()->getInvariant() );
119 
120 #ifdef REACH_DEBUG
121  INFO( "hypro.reacher", "still within Invariant?: " << newSegment.first );
122 #endif
123  if ( newSegment.first != CONTAINMENT::NO ) {
124  flowpipe.push_back( newSegment.second );
125  if ( intersectBadStates( newSegment.second ) ) {
126  // clear queue to stop whole algorithm
127  while ( !mWorkingQueue.isEmpty() ) {
128  mWorkingQueue.dequeueFront();
129  }
130  return flowpipe;
131  }
132  // update currentSegment
133  currentSegment = newSegment.second;
134  } else {
135  // the flowpipe does not longer satisfy the invariant -> quit loop
136  break;
137  }
138  currentLocalTime += mSettings.timeStep;
139  }
140 #ifdef REACH_DEBUG
141  if ( !noFlow ) {
142  INFO( "hypro.reacher", "--- Loop left ---" );
143  }
144  INFO( "hypro.reacher", "flowpipe: " << flowpipe.size() << " Segments computed." );
145  INFO( "hypro.reacher", "Process " << nextInitialSets.size() << " new initial sets." );
146  INFO( "hypro.reacher", "current level: " << mCurrentLevel );
147 
148 #endif
149  // The loop terminated correctly (i.e. no bad states were hit), process discrete behavior.
150  assert( nextInitialSets.empty() || ( int( mCurrentLevel ) < mSettings.jumpDepth || mSettings.jumpDepth < 0 ) );
151  if ( int( mCurrentLevel ) < mSettings.jumpDepth || mSettings.jumpDepth < 0 ) {
152  processDiscreteBehaviour( nextInitialSets, currentTreeNode );
153  }
154  return flowpipe;
155  } else {
156  // return an empty flowpipe
157  return flowpipe;
158  }
159 }
