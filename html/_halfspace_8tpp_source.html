337 template <typename Number>
338 vector_t<Number> Halfspace<Number>::computePlaneNormal( const std::vector<vector_t<Number>>& pointSet ) {
339  assert( !pointSet.empty() );
340  // case: dimension-many vertices in the plane
341  if ( pointSet.size() <= unsigned( pointSet.begin()->rows() ) ) {
342  // method avoiding glpk and using Eigen instead (higher precision)
343  Eigen::Index dim = pointSet.begin()->rows();
344  matrix_t<Number> constraints( pointSet.size(), dim + 1 );
345  for ( unsigned pos = 0; pos < pointSet.size(); ++pos ) {
346  constraints.row( pos ).head( dim ) = pointSet.at( pos ).transpose();
347  constraints.row( pos )( dim ) = Number( 1 );
348  }
349 
350  TRACE( "hypro.datastructures", "computing kernel of " << constraints );
351  TRACE( "hypro.datastructures", "rows: " << constraints.rows() << ", cols: " << constraints.cols() );
352 
353  matrix_t<Number> kernel = constraints.fullPivLu().kernel();
354 
355  vector_t<Number> normal = kernel.col( 0 ).head( dim );
356 
357  TRACE( "hypro.datastructures", "Computed kernel: " << kernel );
358 
359  return normal;
360  } else {
361  /*
362  * Setup LP with GLPK
363  */
364 
365  // TODO: Re-think this: apparently this only works, if all points lie exactly on the plane, in which case you
366  // could have selected only the dim-first ones.
367 
368  glp_prob* normal;
369  normal = glp_create_prob();
370  glp_set_obj_dir( normal, GLP_MAX );
371 
372  // we have one row for each edge in our set
373  glp_add_rows( normal, int( pointSet.size() ) );
374 
375  // constraints of auxiliary variables (bounds for rows)
376  for ( int i = 1; i <= int( pointSet.size() ); ++i ) {
377  glp_set_row_bnds( normal, i, GLP_FX, 0.0, 0.0 );
378  }
379 
380  // each column corresponds to one dimension of a vector in our edgeSet
381  // TODO consider p1 & p2 of different dimensions? (-> two edge sets)
382  glp_add_cols( normal, int( pointSet.at( 0 ).rows() ) );
383 
384  // coefficients of objective function:
385  for ( int i = 1; i <= pointSet.at( 0 ).rows(); ++i ) {
386  glp_set_obj_coef( normal, i, 1.0 );
387  }
388 
389  // constraints for structural variables
390  for ( int i = 1; i <= pointSet.at( 0 ).rows(); ++i ) {
391  glp_set_col_bnds( normal, i, GLP_DB, -1.0, 1.0 );
392  }
393 
394  // setup matrix coefficients
395  std::size_t elements = ( pointSet.size() ) * ( std::size_t( pointSet.at( 0 ).rows() ) );
396  int* ia = new int[elements + 1];
397  int* ja = new int[elements + 1];
398  double* ar = new double[elements + 1];
399  int pos = 1;
400 
401  // to prevent bugs
402  ia[0] = 0;
403  ja[0] = 0;
404  ar[0] = 0;
405 
406  for ( int i = 1; i <= int( pointSet.size() ); ++i ) {
407  for ( int j = 1; j <= pointSet.at( 0 ).rows(); ++j ) {
408  ia[pos] = i;
409  ja[pos] = j;
410  vector_t<Number> tmpVec = pointSet.at( i - 1 );
411  ar[pos] = carl::toDouble( tmpVec( j - 1 ) );
412  ++pos;
413  }
414  }
415  assert( pos - 1 <= int( elements ) );
416 
417  glp_load_matrix( normal, int( elements ), ia, ja, ar );
418  glp_simplex( normal, NULL );
419  glp_exact( normal, NULL );
420 
421  vector_t<Number> result = vector_t<Number>( pointSet.at( 0 ).rows(), 1 );
422 
423  // fill the result vector based on the optimal solution returned by the LP
424  for ( unsigned i = 1; i <= pointSet.at( 0 ).rows(); ++i ) {
425  result( i - 1 ) = carl::rationalize<Number>( glp_get_col_prim( normal, i ) );
426  }
427 
428  glp_delete_prob( normal );
429  delete[] ja;
430  delete[] ia;
431  delete[] ar;
432 
433  return result;
434  }
