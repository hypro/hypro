88  //Given two or more result vecs, get maximum
89  std::vector<EvaluationResult<Number>> aggregate( std::vector<std::vector<EvaluationResult<Number>>>& resultStackBack, const matrix_t<Number>& ) const override {
90  TRACE( "hypro.representations.supportFunction", ": UNITE, accumulate results." )
91  assert( resultStackBack.size() >= 2 );
92  std::vector<EvaluationResult<Number>> accumulatedResult = resultStackBack.front();
93  for ( auto resIt = resultStackBack.begin(); resIt != resultStackBack.end(); ++resIt ) {
94  for ( unsigned resultId = 0; resultId < accumulatedResult.size(); ++resultId ) {
95  if ( ( *resIt )[resultId].errorCode == SOLUTION::INFEAS ) {
96  accumulatedResult[resultId] = ( *resIt )[resultId];
97  break;
98  } else if ( ( *resIt )[resultId].errorCode == SOLUTION::INFTY ) {
99  ( *resIt )[resultId].supportValue = 1;
100  accumulatedResult[resultId] = ( *resIt )[resultId];
101  //std::cout << "UnionOp::aggregate, entry was infty" << std::endl;
102  } else {
103  assert( ( *resIt )[resultId].errorCode == SOLUTION::FEAS );
104  accumulatedResult[resultId] = ( *resIt )[resultId] > accumulatedResult[resultId] ? ( *resIt )[resultId] : accumulatedResult[resultId];
105  //std::cout << "UnionOp::aggregate, after choosing mas between " << accumulatedResult[resultId] << " and " << ( *resIt )[resultId] << " r is:" << accumulatedResult[resultId] << std::endl;
106  }
107  }
108  }
109  return accumulatedResult;
110  }

139  //Erases all dimensions from a copy of dimensions that are denoted in dims
140  std::vector<std::size_t> intersectDims( const std::vector<std::vector<std::size_t>>& dims ) const override {
141  // we create the intersection of all results. Therefore we iterate over the first vector and check the other
142  // result vectors, if the respective element is contained. Iterating over the first is sufficient, as elements
143  // not in the first vector will not be in the intersection anyways.
144  std::vector<std::size_t> accumulatedResult;
145  for ( unsigned pos = 0; pos < dims.begin()->size(); ++pos ) {
146  std::size_t element = dims.begin()->at( pos );
147  bool elementInAllVectors = true;
148  for ( unsigned resIndex = 1; resIndex < dims.size(); ++resIndex ) {
149  if ( std::find( dims.at( resIndex ).begin(), dims.at( resIndex ).end(), element ) == dims.at( resIndex ).end() ) {
150  elementInAllVectors = false;
151  break;
152  }
153  }
154  if ( elementInAllVectors ) {
155  // if this is a projection, we need to remove non-projected dimensions -> we could not have been in a projection since
156  // since this part of the function was only executed if a node had more than 2 children
157  accumulatedResult.emplace_back( element );
158  }
159  }
160  return accumulatedResult;
161  }
