45 template <typename Number, class Setting>
46 PolytopeSupportFunction<Number, Setting>::PolytopeSupportFunction( const std::vector<Point<Number>> &_points ) {
47  TRACE( "hypro.representations.supportFunction", "" );
48  //std::cout << __func__ << std::endl;
49  if ( !_points.empty() ) {
50  //std::cout << "Points not empty" << std::endl;
51  mDimension = _points.begin()->dimension();
52  //std::cout << "Space dimension is " << mDimension << std::endl;
53  // check affine independence - verify object dimension.
54  std::vector<vector_t<Number>> coordinates;
55  for ( const auto &vertex : _points ) {
56  coordinates.emplace_back( vertex.rawCoordinates() );
57  }
58  int effectiveDim = effectiveDimension( coordinates );
59  //std::cout << "Effective dimension of the pointset: " << effectiveDim << std::endl;
60  assert( effectiveDim >= 0 );
61 
62  if ( unsigned( effectiveDim ) < mDimension ) {
63  //std::cout << "Compute oriented box." << std::endl;
64  std::vector<Halfspace<Number>> boxConstraints = computeOrientedBox( _points );
65 
66  mConstraints = matrix_t<Number>( boxConstraints.size(), mDimension );
67  mConstraintConstants = vector_t<Number>( boxConstraints.size() );
68 
69  unsigned rowCnt = 0;
70  for ( const auto &constraint : boxConstraints ) {
71  mConstraints.row( rowCnt ) = constraint.normal();
72  mConstraintConstants( rowCnt ) = constraint.offset();
73  ++rowCnt;
74  }
75  } else {
76  //std::cout << "Use convex hull algorithm." << std::endl;
77  std::vector<std::shared_ptr<Facet<Number>>> facets = convexHull( _points ).first;
78 
79  mConstraints = matrix_t<Number>( facets.size(), mDimension );
80  mConstraintConstants = vector_t<Number>( facets.size() );
81 
82  unsigned rowCnt = 0;
83  for ( auto &facet : facets ) {
84  assert( facet->halfspace().contains( _points ) );
85  mConstraints.row( rowCnt ) = facet->halfspace().normal();
86  mConstraintConstants( rowCnt ) = facet->halfspace().offset();
87  ++rowCnt;
88  }
89  facets.clear();
90  }
91  }
92  mOpt = Optimizer<Number>( mConstraints, mConstraintConstants );
93 }

140 template <typename Number, class Setting>
141 std::vector<Point<Number>> PolytopeSupportFunction<Number, Setting>::vertices() const {
142  typename std::vector<Point<Number>> vertices;
143  if ( mConstraints.rows() != 0 ) {
144  unsigned dim = this->dimension();
145 
146  Permutator permutator( mConstraints.rows(), dim );
147  std::vector<std::size_t> permutation;
148  while ( !permutator.end() ) {
149  permutation = permutator();
150 
151  matrix_t<Number> A( dim, dim );
152  vector_t<Number> b( dim );
153 
154  // set up linear problem
155  unsigned pos = 0;
156  for ( auto planeIt = permutation.begin(); planeIt != permutation.end(); ++planeIt ) {
157  A.row( pos ) = mConstraints.row( *planeIt );
158  b( pos ) = mConstraintConstants( *planeIt );
159  ++pos;
160  }
161 
162  // solve
163  Eigen::FullPivLU<matrix_t<Number>> lu_decomp( A );
164  if ( lu_decomp.rank() < A.rows() ) {
165  continue;
166  }
167 
168  vector_t<Number> res = lu_decomp.solve( b );
169 
170  // Check if the computed vertex is a real vertex
171  bool outside = false;
172  for ( unsigned planePos = 0; planePos < mConstraints.rows(); ++planePos ) {
173  bool skip = false;
174  for ( unsigned permPos = 0; permPos < permutation.size(); ++permPos ) {
175  // naturally the computed vertex satisfies the planes we used for its computation -> skip the plane
176  if ( planePos == permutation.at( permPos ) ) {
177  skip = true;
178  break;
179  }
180  }
181 
182  if ( !skip ) {
183  // verify if the vertex is outside the current plane
184  if ( mConstraintConstants( planePos ) - mConstraints.row( planePos ).dot( res ) < 0 ) {
185  outside = true;
186  break;
187  }
188  }
189  }
190  if ( !outside ) {
191  // insert, if no duplicate
192  Point<Number> tmp( res );
193  if ( std::find( vertices.begin(), vertices.end(), tmp ) == vertices.end() ) {
194  vertices.push_back( tmp );
195  }
196  }
197  }
198  }
199  return vertices;
200 }

202 template <typename Number, class Setting>
203 Point<Number> PolytopeSupportFunction<Number, Setting>::supremumPoint() const {
204  assert( !this->empty() );
205  EvaluationResult<Number> sup;
206  sup.errorCode = SOLUTION::UNKNOWN;
207 
208  for ( unsigned d = 0; d < mDimension; ++d ) {
209  // evaluate in each main direction, positive and negative
210  vector_t<Number> posDir = vector_t<Number>::Zero( mDimension );
211  posDir( d ) = 1;
212  vector_t<Number> negDir = vector_t<Number>::Zero( mDimension );
213  negDir( d ) = -1;
214 
215  EvaluationResult<Number> positive = this->evaluate( posDir, false );
216  EvaluationResult<Number> negative = this->evaluate( negDir, false );
217  assert( positive.errorCode != SOLUTION::UNKNOWN );
218  assert( negative.errorCode != SOLUTION::UNKNOWN );
219 
220  // empty object
221  if ( positive.errorCode == SOLUTION::INFEAS ) {
222  return Point<Number>::Zero( mDimension );
223  }
224 
225  // init once
226  if ( sup.errorCode == SOLUTION::UNKNOWN ) {
227  sup = positive;
228  }
229 
230  if ( positive.supportValue > negative.supportValue ) {
231  sup = sup.supportValue > positive.supportValue ? sup : positive;
232  } else {
233  sup = sup.supportValue > negative.supportValue ? sup : negative;
234  }
235  }
236  return Point<Number>( sup.optimumValue );
237 }

239 template <typename Number, class Setting>
240 EvaluationResult<Number> PolytopeSupportFunction<Number, Setting>::evaluate( const vector_t<Number> &l, bool useExact ) const {
241  TRACE( "hypro.representations.supportFunction", "" );
242  // catch half-space
243  if ( mConstraints.rows() == 1 ) {
244  COUNT( "Single constraint evaluate." );
245  //std::cout << "only one constraint! -> we evaluate against a plane!" << std::endl;
246  // TODO: extend check to linear dependence. Here temporarily sufficient as we will initialize and evaluate with the plane normals, which should be the same vectors.
247  if ( l == vector_t<Number>( mConstraints.row( 0 ) ) ) {
248  //std::cout << "Managed to optimize into correct direction! which is: " << convert<Number,double>(mConstraints.row(0)).transpose() << std::endl;
249  // The vectors are EXACTLY the same -> to find point on plane and to avoid squareroot computations, return vector
250  // which contains zeroes except of the position with the first non-zero coeff, which is set to the stored distance.
251  vector_t<Number> pointOnPlane = vector_t<Number>::Zero( l.rows() );
252  unsigned i = 0;
253  while ( i < l.rows() && l( i ) == 0 ) {
254  ++i;
255  }
256  if ( l( i ) == 0 ) {
257  return EvaluationResult<Number>(); // TODO: Check if this is correct -> we evaluated something weird with a weird direction (both zero vertors).
258  }
259  pointOnPlane( i ) = mConstraintConstants( 0 );
260  return EvaluationResult<Number>( mConstraintConstants( 0 ), pointOnPlane, SOLUTION::FEAS );
261  } else {
262  //std::cout << "Eval to infty." << std::endl;
263  return EvaluationResult<Number>( 0, SOLUTION::INFTY );
264  }
265  }
266 
267  COUNT( "Polytope evaluate." );
268 
269  EvaluationResult<Number> res( mOpt.evaluate( l, useExact ) );
270  if ( Setting::PPOLYTOPESUPPORTFUNCTION_VERBOSE ) {
271  std::cout << __func__ << ": " << *this << " evaluated in direction " << convert<Number, double>( l ) << " results in " << res << std::endl;
272  }
273 #if defined( HYPRO_USE_SMTRAT ) || defined( HYPRO_USE_Z3 ) || defined( HYPRO_USE_SOPLEX )
274  assert( res.errorCode != SOLUTION::FEAS || this->contains( res.optimumValue ) );
275 #endif
276  //assert( std::size_t(l.rows()) == mDimension );
277  //assert( l.rows() == mDimension );
278  return res;
279 }

282 std::vector<EvaluationResult<Number>> PolytopeSupportFunction<Number, Setting>::multiEvaluate( const matrix_t<Number> &_A, bool useExact, bool ) const {
283  assert( std::size_t( _A.cols() ) == mDimension );
284  std::vector<EvaluationResult<Number>> res;
285  TRACE( "hypro.representations.supportFunction", "Evaluate in directions " << matrix_t<double>( convert<Number, double>( _A ) ) << std::endl
286  << "POLY SF IS " << *this );
287 
288  // catch half-space
289  if ( mConstraints.rows() == 1 ) {
290  vector_t<Number> pointOnPlane = vector_t<Number>::Zero( mDimension );
291  assert( mConstraints.row( 0 ).nonZeros() != 0 );
292  unsigned nonZeroPos = 0;
293  while ( mConstraints( 0, nonZeroPos ) == 0 ) {
294  ++nonZeroPos;
295  }
296  pointOnPlane( nonZeroPos ) = mConstraintConstants( 0 );
297  for ( unsigned index = 0; index < _A.rows(); ++index ) {
298  std::pair<bool, Number> dependent = linearDependent( vector_t<Number>( mConstraints.row( 0 ) ), vector_t<Number>( _A.row( index ) ) );
299  if ( dependent.first ) {
300  if ( dependent.second > 0 ) {
301  //std::cout << "Vectors " << convert<Number,double>(vector_t<Number>(mConstraints.row(0))).transpose() << " and " << convert<Number,double>(vector_t<Number>(_A.row(index))).transpose() << " are dependent!" << std::endl;
302  //std::cout << "Factor: " << dependent.second << std::endl;
303  res.emplace_back( mConstraintConstants( 0 ) * dependent.second, pointOnPlane, SOLUTION::FEAS );
304  } else {
305  res.emplace_back( 1, pointOnPlane, SOLUTION::INFTY );
306  }
307  } else {
308  res.emplace_back( 1, pointOnPlane, SOLUTION::INFTY );
309  }
310  }
311  return res;
312  }
313 
314  for ( unsigned index = 0; index < _A.rows(); ++index ) {
315  res.emplace_back( mOpt.evaluate( _A.row( index ), useExact ) );
316  }
317  assert( res.size() == std::size_t( _A.rows() ) );
318  return res;
319 }

379 template <typename Number, class Setting>
380 void PolytopeSupportFunction<Number, Setting>::removeRedundancy() {
381  TRACE( "hypro.representations.supportFunction", "" );
382  if ( mConstraints.rows() > 1 ) {
383  std::vector<std::size_t> redundant = mOpt.redundantConstraints();
384  //std::cout << __func__ << ": found " << redundant.size() << " redundant constraints." << std::endl;
385 
386  if ( !redundant.empty() ) {
387  matrix_t<Number> newConstraints = matrix_t<Number>( mConstraints.rows() - redundant.size(), mConstraints.cols() );
388  vector_t<Number> newConstants = vector_t<Number>( mConstraints.rows() - redundant.size() );
389  int insertionIndex = newConstants.rows() - 1;
390  for ( int rowIndex = mConstraints.rows() - 1; rowIndex >= 0; --rowIndex ) {
391  if ( redundant.empty() || unsigned( rowIndex ) != redundant.back() ) {
392  newConstraints.row( insertionIndex ) = mConstraints.row( rowIndex );
393  newConstants( insertionIndex ) = mConstraintConstants( rowIndex );
394  --insertionIndex;
395  } else {
396  redundant.pop_back();
397  }
398  }
399  assert( insertionIndex == -1 );
400  mConstraints = newConstraints;
401  mConstraintConstants = newConstants;
402  mOpt.cleanContexts();
403  mOpt = Optimizer<Number>( mConstraints, mConstraintConstants );
404  }
405  assert( redundant.empty() );
406  }
