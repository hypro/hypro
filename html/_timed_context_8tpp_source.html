106 template <typename State>
107 void TimedContext<State>::checkTransition() {
108  // Collect potential new initial states from discrete behaviour.
109  if ( !( this->mSettings.jumpDepth < 0 || int( this->mTask->treeNode->getDepth() ) <= this->mSettings.jumpDepth ) ) {
110  return;
111  }
112 
113  bool urgentTransitionEnabled = false;
114 
115  for ( auto it = this->mTransitionHandlerMap.begin(); it != this->mTransitionHandlerMap.end(); ++it ) {
116 #ifdef SINGLE_THREAD_FIXED_POINT_TEST
117  // if the considered transition is disabled (for one iteration), skip the test by starting the next loop iteration.
118  auto disabledTransitionIt = std::find( this->mDisabledTransitions.begin(), this->mDisabledTransitions.end(), it->first );
119  if ( disabledTransitionIt != this->mDisabledTransitions.end() ) {
120  this->mDisabledTransitions.erase( disabledTransitionIt );
121  continue;
122  }
123 #endif
124 
125  if ( !it->first->isUrgent() && LTIContext<State>::omitTransition( it->first ) ) {
126  // store that transition was not enabled for this time interval
127  this->mTask->treeNode->rGetRefinements().at( this->mTask->btInfo.btLevel ).mTimings.insertTransition( it->first, this->mCurrentGlobalTimeInterval, hypro::CONTAINMENT::NO );
128  continue;
129  }
130 
131  TRACE( "hydra.worker", "Initializing " << it->second.size() << " guard handlers for transition " << it->first->getSource()->hash() << " -> " << it->first->getTarget()->hash() );
132 
133  State* guardStatePtr = new State( this->mComputationState );
134  std::shared_ptr<State> shGuardPtr = std::shared_ptr<State>( guardStatePtr );
135 
136  for ( std::size_t i = 0; i < it->second.size(); i++ ) {
137  it->second.at( i )->setState( shGuardPtr );
138  }
139 
140  // apply handlers
141  bool guardSatisfied = true;
142 
143  if ( it->second.size() > 0 ) {
144  //START_BENCHMARK_OPERATION(FULL_CUT);
145  for ( auto guardIt = it->second.begin(); guardIt != it->second.end(); ++guardIt ) {
146  //START_BENCHMARK_OPERATION(SUBSPACE_CUT);
147  ( *guardIt )->reinitialize();
148  ( *guardIt )->handle();
149  guardSatisfied &= ( *guardIt )->satisfiesGuard();
150  //EVALUATE_BENCHMARK_RESULT(SUBSPACE_CUT);
151 
152  if ( !guardSatisfied ) {
153  break;
154  }
155  }
156  }
157 
158  // handle time-triggered transitions
159  if ( it->first->isUrgent() ) {
160  TRACE( "hydra.worker.discrete", "Checking urgent transition " << it->first->getSource()->hash() << " -> " << it->first->getTarget()->hash() );
161  if ( guardSatisfied ) {
162  this->mDiscreteSuccessorBuffer.push_back( std::tuple<hypro::Transition<Number>*, State>( it->first, *guardStatePtr ) );
163  urgentTransitionEnabled = true;
164  }
165  } // handle normal transitions, but only if they cannot be omitted (they can if the timing intervals do not intersect)
166  else if ( guardSatisfied ) {
167  // <FULLTIME-SPECIFIC>
168 
169  carl::Interval<tNumber> timeStamp = handshakeTimestamps( this->mComputationState, *( guardStatePtr ) );
170  if ( timeStamp.isEmpty() ) {
171  this->mTask->treeNode->rGetRefinements().at( this->mTask->btInfo.btLevel ).mTimings.insertTransition( it->first, this->mCurrentGlobalTimeInterval, hypro::CONTAINMENT::NO );
172  } else {
173  guardStatePtr->setTimestamp( timeStamp );
174  TRACE( "hydra.worker.discrete", "hybrid transition enabled with timestamp " << guardStatePtr->getTimestamp() );
175  TRACE( "hydra.worker.discrete", "Enqueued state: " << *( guardStatePtr ) );
176  this->mDiscreteSuccessorBuffer.push_back( std::tuple<hypro::Transition<Number>*, State>( it->first, *( guardStatePtr ) ) );
177  if ( SettingsProvider<State>::getInstance().useLocalTiming() ) {
178  carl::Interval<tNumber> entryTimeStamp = this->mTask->treeNode->getStateAtLevel( this->mTask->btInfo.btLevel ).getTimestamp();
179  carl::Interval<tNumber> guardSatTimeStamp = entryTimeStamp + guardStatePtr->getTimestamp();
180  this->mTask->treeNode->rGetRefinements().at( this->mTask->btInfo.btLevel ).mTimings.insertTransition( it->first, guardSatTimeStamp, hypro::CONTAINMENT::YES );
181 
182  } else {
183  this->mTask->treeNode->rGetRefinements().at( this->mTask->btInfo.btLevel ).mTimings.insertTransition( it->first, guardStatePtr->getTimestamp(), hypro::CONTAINMENT::YES );
184  }
185  }
186  // </FULLTIME-SPECIFIC>
187  } else {
188  // store that transition was not enabled for this time interval
189  this->mTask->treeNode->rGetRefinements().at( this->mTask->btInfo.btLevel ).mTimings.insertTransition( it->first, this->mCurrentGlobalTimeInterval, hypro::CONTAINMENT::NO );
190  }
191  }
192 
193  if ( urgentTransitionEnabled ) {
194  // quit loop after firing time triggered transition -> time triggered transitions are handled as urgent.
195  TRACE( "hydra.worker.discrete", "Urgent transition enabled, quit flowpipe computation immediately." << std::endl );
196  throw FinishWithDiscreteProcessingException( "Urgent transition enabled, quit flowpipe computation immediately." );
197  }

200 template <typename State>
201 carl::Interval<tNumber> TimedContext<State>::computeShortestElapsedTimeIntervalForSubspace( State& in, size_t index ) {
202  carl::Interval<tNumber> minInterval = carl::Interval<tNumber>::emptyInterval();
203  tNumber minDiameter = -1.0;
204  // this should always be a dbm. If not, we deliberately throw the bad get exception
205  hypro::DifferenceBounds<Number> dbm = std::get<hypro::DifferenceBounds<Number>>( in.getSet( index ) );
206  if ( dbm.dimension() > 0 ) {
207  size_t numclocks = dbm.dimension();
208  for ( size_t i = 0; i < numclocks; i++ ) {
209  // for each clock project the DBM to its axis
210  std::vector<size_t> v;
211  v.push_back( i );
212  hypro::DifferenceBounds<Number> projectedDBM = dbm.projectOn( v );
213  // compute vertices ie endpoints of interval
214  std::vector<hypro::Point<Number>> vert = projectedDBM.vertices();
215  // if less than 2, this is a point interval and therefore minimal (diameter 0)
216  if ( vert.size() == 0 ) {
217  //can happen due to rounding issues on eg: [-2.22507e-308, 0] -> return point
218  return carl::Interval<tNumber>( tNumber( 0.0 ) );
219  } else if ( vert.size() < 2 ) {
220  hypro::Point<Number> origin = hypro::Point<Number>( vector_t<Number>::Zero( 1 ) );
221  Number timePoint = origin.distance( vert[0] );
222  if ( abs( timePoint ) > 0 && abs( timePoint ) < 0.000000000001 ) {
223  timePoint = Number( 0 );
224  }
225  return carl::Interval<tNumber>( carl::convert<Number, tNumber>( timePoint ) );
226  }
227  // as we projected to axis and the entries were not the same, that size should be 2
228  assert( vert.size() == 2 );
229 
230  // compute diameter of interval
231  hypro::Point<Number> origin = hypro::Point<Number>( vector_t<Number>::Zero( 1 ) );
232  Number timePoint1 = origin.distance( vert[0] );
233  if ( abs( timePoint1 ) > 0 && abs( timePoint1 ) < 0.000000000001 ) {
234  timePoint1 = Number( 0 );
235  }
236  Number timePoint2 = origin.distance( vert[1] );
237  if ( abs( timePoint2 ) > 0 && abs( timePoint2 ) < 0.000000000001 ) {
238  timePoint2 = Number( 0 );
239  }
240  Number diameter = vert[0].distance( vert[1] );
241  if ( abs( diameter ) > 0 && abs( diameter ) < 0.000000000001 ) {
242  diameter = Number( 0 );
243  }
244 
245  // if interval is minimal
246  if ( diameter < minDiameter || minDiameter < 0.0 ) { // on first iteration this might be -1.0 so update
247  minDiameter = carl::convert<Number, tNumber>( diameter );
248  minInterval = timePoint1 < timePoint2 ? carl::Interval<tNumber>( carl::convert<Number, tNumber>( timePoint1 ), carl::convert<Number, tNumber>( timePoint2 ) ) : carl::Interval<tNumber>( carl::convert<Number, tNumber>( timePoint2 ), carl::convert<Number, tNumber>( timePoint1 ) );
249  }
250  }
251  }
252  return minInterval;
253 }

255 template <typename State>
256 carl::Interval<tNumber> TimedContext<State>::handshakeTimestamps( State& preOp, State& postOp ) {
257  assert( preOp.getNumberSets() == postOp.getNumberSets() );
258  auto map = SettingsProvider<State>::getInstance().getLocationSubspaceTypeMap();
259  std::vector<hypro::SUBSPACETYPE> types = *( map.find( preOp.getLocation() )->second );
260 
261  // for each timed subspace compute its minimal elapsed time interval before the operation happened
262  std::map<size_t, carl::Interval<tNumber>> preOpTimeIntervals;
263  for ( size_t index = 0; index < preOp.getNumberSets(); index++ ) {
264  if ( types.at( index ) != hypro::SUBSPACETYPE::TIMED ) {
265  // only compute for timed subspaces
266  continue;
267  }
268  preOpTimeIntervals.insert( std::make_pair( index, computeShortestElapsedTimeIntervalForSubspace( preOp, index ) ) );
269  }
270 
271  // for each timed subspace compute its minimal elapsed time interval before the operation happened
272  std::map<size_t, carl::Interval<tNumber>> postOpTimeIntervals;
273  for ( size_t index = 0; index < postOp.getNumberSets(); index++ ) {
274  if ( types.at( index ) != hypro::SUBSPACETYPE::TIMED ) {
275  // only compute for timed subspaces
276  continue;
277  }
278  // build normalized vectors by substracting the lower bounds of the preOpTimeIntervals
279  // e.g. lets assume we are between [7,19] before. We perform the guard cut and obtain [16,17]
280  // as the time interval that satisfies the cut. [16,18] - [7,7] will give [9,11] i.e. the time in
281  // which the cut happened after entering with the preop timestamp. Intersecting these timestamps
282  // will either yield an empty interval (guards were satisfied at non intersecting timestamps)
283  // or all timestamp will agree upon a time interval, eg. [9,10] in the above case.
284  carl::Interval<tNumber> timestampPost = computeShortestElapsedTimeIntervalForSubspace( postOp, index );
285  carl::Interval<tNumber> timestampPre = preOpTimeIntervals.find( index )->second;
286 
287  carl::Interval<tNumber> normalized = carl::Interval<tNumber>( timestampPost.lower() - timestampPre.lower(), timestampPost.upper() - timestampPre.lower() );
288  postOpTimeIntervals.insert( std::make_pair( index, normalized ) );
289  }
290 
291  carl::Interval<tNumber> agreedTimeStamp = carl::Interval<tNumber>::emptyInterval();
292  bool first = true;
293  for ( auto it = postOpTimeIntervals.begin(); it != postOpTimeIntervals.end(); ++it ) {
294  if ( first ) {
295  // just copy over first timestamp
296  agreedTimeStamp = it->second;
297  first = false;
298  } else {
299  // intersect timestamps
300  agreedTimeStamp = carl::set_intersection( agreedTimeStamp, it->second );
301  }
302  }
303 
304  if ( agreedTimeStamp.isEmpty() ) {
305  // there was no agreement on the jump timestamp
306  return carl::Interval<tNumber>::emptyInterval();
307  }
308  for ( size_t index = 0; index < postOp.getNumberSets(); index++ ) {
309  if ( types.at( index ) != hypro::SUBSPACETYPE::TIMED ) {
310  // only compute for timed subspaces
311  continue;
312  }
313 
314  hypro::DifferenceBounds<Number> preOpDBM = std::get<hypro::DifferenceBounds<Number>>( preOp.getSet( index ) );
315  hypro::DifferenceBounds<Number> postOpDBM = std::get<hypro::DifferenceBounds<Number>>( postOp.getSet( index ) );
316  for ( int i = 1; i < postOpDBM.getDBM().rows(); i++ ) {
317  Number lowerBound = preOpDBM.getDBM()( 0, i ).first;
318 
319  // eg clock[5,10] , agreed jumptimestep [2,4] <=> [5+2,5+4] = [7,9]
320  Number newLowerBound = lowerBound - carl::convert<tNumber, Number>( agreedTimeStamp.lower() );
321  Number newUpperBound = abs( lowerBound ) + carl::convert<tNumber, Number>( agreedTimeStamp.upper() );
322  // cut with x-0 <= c
323  postOpDBM = postOpDBM.intersectConstraint( i, 0, typename hypro::DifferenceBounds<Number>::DBMEntry( newUpperBound, hypro::DifferenceBounds<Number>::BOUND_TYPE::SMALLER_EQ ) );
324  // cut with 0-x <= c
325  postOpDBM = postOpDBM.intersectConstraint( 0, i, typename hypro::DifferenceBounds<Number>::DBMEntry( newLowerBound, hypro::DifferenceBounds<Number>::BOUND_TYPE::SMALLER_EQ ) );
326  }
327  postOp.setSet( postOpDBM, index );
328  }
329 
330  return agreedTimeStamp;
331 }
