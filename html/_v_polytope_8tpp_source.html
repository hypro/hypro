55 template <typename Number, typename Converter, typename S>
56 VPolytopeT<Number, Converter, S>::VPolytopeT( const matrix_t<Number>& _constraints, const vector_t<Number> _constants ) {
57  if constexpr ( is_exact<Number> ) {
58  auto dimension = _constraints.cols();
59  typename QuickIntersection<Number>::pointVector_t inputHalfspaces;
60 
61  for ( int i = 0; i < _constraints.rows(); ++i ) {
62  inputHalfspaces.emplace_back( dimension + 1 );
63  inputHalfspaces.back().head( dimension ) = _constraints.row( i );
64  inputHalfspaces.back()[dimension] = -_constants[i];
65  }
66 
67  QuickIntersection<Number> qInt{ inputHalfspaces, (size_t)dimension };
68  qInt.compute();
69 
70  for ( auto& facet : qInt.getFacets() ) {
71  facet.mNormal /= -facet.mOffset;
72  mVertices.emplace_back( Point( std::move( facet.mNormal ) ) );
73  }
74  } else {
75  // calculate all possible Halfspace intersections
76  TRACE( "hypro.representations.vpolytope", "Construct from " << _constraints << " <= " << _constants );
77  //std::cout << __func__ << ": matrix: " << _constraints << " and vector: " << _constants << std::endl;
78  assert( _constraints.rows() == _constants.rows() );
79  Permutator permutator = Permutator( _constraints.rows(), _constraints.cols() );
80  matrix_t<Number> intersection = matrix_t<Number>( _constraints.cols(), _constraints.cols() );
81  vector_t<Number> intersectionConstants = vector_t<Number>( _constraints.cols() );
82  std::set<vector_t<Number>> possibleVertices;
83  std::vector<std::size_t> permutation;
84  while ( !permutator.end() ) {
85  permutation = permutator();
86  unsigned rowCount = 0;
87  // std::cout << "Intersect :" << std::endl;
88  for ( const auto& rowIndex : permutation ) {
89  // std::cout << _constraints.row(rowIndex) << " <= " <<
90  // _constants(rowIndex) << std::endl;
91  assert( rowCount < _constraints.cols() );
92  intersection.row( rowCount ) = _constraints.row( rowIndex );
93  intersectionConstants( rowCount ) = _constants( rowIndex );
94  ++rowCount;
95  }
96  // check if rank is full
97  if ( intersection.fullPivLu().rank() == intersection.cols() ) {
98  vector_t<Number> vertex = intersection.fullPivLu().solve( intersectionConstants );
99  assert( vertex.rows() == _constraints.cols() );
100  //assert(insidePlanes(vertex, intersection, intersectionConstants));
101  // avoid duplicate entries
102  if ( std::find( possibleVertices.begin(), possibleVertices.end(), vertex ) == possibleVertices.end() ) {
103  possibleVertices.emplace( std::move( vertex ) );
104  }
105  }
106  }
107  TRACE( "hypro.representations.vpolytope", ": Computed " << possibleVertices.size() << " possible vertices." );
108 
109  // check if vertices are true vertices (i.e. they fulfill all constraints)
110  for ( auto vertex = possibleVertices.begin(); vertex != possibleVertices.end(); ) {
111  // std::cout<<__func__ << " : " <<__LINE__ << " current position : " << i <<
112  // std::endl;
113  // std::cout<<__func__ << " : " <<__LINE__ << "number of vertices : " <<
114  // possibleVertices.size() << std::endl;
115  bool deleted = false;
116  for ( unsigned rowIndex = 0; rowIndex < _constraints.rows(); ++rowIndex ) {
117  Number res = vertex->dot( _constraints.row( rowIndex ) );
118  if ( !carl::AlmostEqual2sComplement( res, _constants( rowIndex ), default_double_comparison_ulps ) && res > _constants( rowIndex ) ) {
119  vertex = possibleVertices.erase( vertex );
120  deleted = true;
121  break;
122  }
123  }
124  if ( !deleted ) {
125  ++vertex;
126  }
127  }
128 
129  TRACE( "hypro.representations.vpolytope", "Deleted vertices. Remaining " << possibleVertices.size() << " vertices." );
130 
131  // std::cout<<__func__ << " : " <<__LINE__ <<std::endl;
132  // finish initialization
133  for ( const auto& point : possibleVertices ) {
134  mVertices.emplace_back( point );
135  mNeighbors.push_back( std::set<unsigned>() );
136  // std::cout << "Real vertex " << point.transpose() << std::endl;
137  }
138  // std::cout<<__func__ << " : " <<__LINE__ <<std::endl;
139  mReduced = false;
140 
141  //reduceNumberRepresentation();
142  }

217 template <typename Number, typename Converter, typename S>
218 VPolytopeT<Number, Converter, S> VPolytopeT<Number, Converter, S>::intersect( const VPolytopeT<Number, Converter, S>& rhs ) const {
219  // create a set of possible points via combination of all coordinates
220  if ( rhs.size() == 0 ) {
221  return VPolytopeT<Number, Converter, S>();
222  } else {
223  pointVector possibleVertices;
224  for ( const auto& lhsVertex : mVertices ) {
225  possibleVertices.push_back( lhsVertex );
226  for ( unsigned coordIndex = 0; coordIndex < lhsVertex.rawCoordinates().rows(); ++coordIndex ) {
227  for ( const auto& rhsVertex : rhs.mVertices ) {
228  vector_t<Number> newVertex = rhsVertex.rawCoordinates();
229  newVertex( coordIndex ) = lhsVertex.at( coordIndex );
230  possibleVertices.push_back( Point<Number>( newVertex ) );
231  possibleVertices.push_back( Point<Number>( rhsVertex.rawCoordinates() ) );
232  }
233  }
234  }
235  // remove all vertices, which are only contained in one of the polytopes
236  for ( auto vertexIt = possibleVertices.begin(); vertexIt != possibleVertices.end(); ) {
237  if ( !( this->contains( *vertexIt ) && rhs.contains( *vertexIt ) ) ) {
238  vertexIt = possibleVertices.erase( vertexIt );
239  } else {
240  ++vertexIt;
241  }
242  }
243  return VPolytopeT<Number, Converter, S>( std::move( possibleVertices ) );
244  }

265 template <typename Number, typename Converter, typename S>
266 std::pair<CONTAINMENT, VPolytopeT<Number, Converter, S>> VPolytopeT<Number, Converter, S>::satisfiesHalfspace( const Halfspace<Number>& rhs ) const {
267  // if we want to have fast pre-checks, this is the time to do them
268  if ( S::checkVerticesBeforeConversion ) {
269  // we count the number of vertices satisfying the halfspace to get a first impression about the result type.
270  std::size_t numberVerticesInRhs = numberOfPointsInHalfspace( mVertices, rhs );
271  // no vertex is contained - the result will be empty.
272  if ( numberVerticesInRhs == 0 ) {
273  COUNT( "VPolyAvoidConversion" );
274  return std::make_pair( CONTAINMENT::NO, VPolytopeT<Number, Converter, S>::Empty() );
275  } else if ( numberVerticesInRhs == mVertices.size() ) {
276  // all vertices are contained - the full set is the result.
277  COUNT( "VPolyAvoidConversion" );
278  return std::make_pair( CONTAINMENT::FULL, *this );
279  }
280  }
281 
282  auto intermediate = Converter::toHPolytope( *this );
283  auto resultPair = intermediate.satisfiesHalfspace( rhs );
284  if ( resultPair.first != CONTAINMENT::NO ) {
285  //resultPair.second.removeRedundancy();
286  VPolytopeT<Number, Converter, S> res( Converter::toVPolytope( resultPair.second ) );
287  return std::make_pair( resultPair.first, std::move( res ) );
288  }
289  return std::make_pair( resultPair.first, std::move( VPolytopeT<Number, Converter, S>::Empty() ) );
290 }

292 template <typename Number, typename Converter, typename S>
293 std::pair<CONTAINMENT, VPolytopeT<Number, Converter, S>> VPolytopeT<Number, Converter, S>::satisfiesHalfspaces( const matrix_t<Number>& _mat, const vector_t<Number>& _vec ) const {
294  // this stores results from pre-checks to reduce the effort of intersections later, if applicable.
295  std::vector<Eigen::Index> halfspacesToAvoid;
296  if ( S::checkVerticesBeforeConversion ) {
297  // we check for each halfspace individually to allow for premature exiting and reductions.
298  for ( Eigen::Index rowIndex = 0; rowIndex < _mat.rows(); ++rowIndex ) {
299  // we count the number of vertices satisfying the halfspace to get a first impression about the result type.
300  std::size_t numberVerticesInRhs = numberOfPointsInHalfspace( mVertices, Halfspace<Number>( _mat.row( rowIndex ), _vec( rowIndex ) ) );
301  // no vertex is contained - the result will be empty.
302  if ( numberVerticesInRhs == 0 ) {
303  COUNT( "VPolyAvoidConversion" );
304  return std::make_pair( CONTAINMENT::NO, VPolytopeT<Number, Converter, S>::Empty() );
305  } else if ( numberVerticesInRhs == mVertices.size() ) {
306  // all vertices are contained - the full set is the result for this halfspace, we can avoid it.
307  halfspacesToAvoid.push_back( rowIndex );
308  }
309  }
310  // none of the halfspaces cuts the set - return full set.
311  if ( halfspacesToAvoid.size() == std::size_t( _mat.rows() ) ) {
312  COUNT( "VPolyAvoidConversion" );
313  return std::make_pair( CONTAINMENT::FULL, *this );
314  }
315  }
316 
317  auto intermediate = Converter::toHPolytope( *this );
318  intermediate.reduceNumberRepresentation( mVertices );
319  //std::cout << typeid(*this).name() << "::" << __func__ << ": Intermediate hpoly: " << intermediate << std::endl;
320  auto resultPair = intermediate.satisfiesHalfspaces( removeRows( _mat, halfspacesToAvoid ), removeRows( _vec, halfspacesToAvoid ) );
321  if ( resultPair.first != CONTAINMENT::NO ) {
322  //resultPair.second.removeRedundancy();
323  assert( !resultPair.second.empty() );
324  TRACE( "hypro.representations.vpolytope", ": Intermediate hpoly convert back: " << resultPair.second );
325  VPolytopeT<Number, Converter, S> res( Converter::toVPolytope( resultPair.second ) );
326  TRACE( "hypro.representations.vpolytope", "Re-Converted v-poly: " << res );
327  assert( !res.empty() );
328  return std::make_pair( resultPair.first, std::move( res ) );
329  }
330  return std::make_pair( resultPair.first, std::move( VPolytopeT<Number, Converter, S>::Empty() ) );
331 }

338 template <typename Number, typename Converter, typename S>
339 bool VPolytopeT<Number, Converter, S>::contains( const vector_t<Number>& vec ) const {
340  if ( this->empty() ) {
341  return false;
342  }
343  if ( S::useLpForPointContainment ) {
344  // create LP Ax = b to solve whether vec can be represented as a convex combination of the vertices.
345  // 1 create matrix from vertices.
346  // 2 use vec as b
347  // 3 set bounds for cols to [0,1]
348  // 4 add constraint stating that all col-values need to sum up to 1.
349 
350  // 1 create matrix
351  matrix_t<Number> A = matrix_t<Number>( this->dimension(), mVertices.size() );
352  for ( Eigen::Index i = 0; i < A.cols(); ++i ) {
353  A.col( i ) = mVertices[i].rawCoordinates();
354  }
355 
356  Optimizer<Number> opt( A, vec );
357 
358  // 3 bound cols
359  for ( Eigen::Index i = 0; i < A.cols(); ++i ) {
360  vector_t<Number> constraint = vector_t<Number>::Zero( A.cols() );
361  constraint( i ) = 1;
362  opt.addConstraint( constraint, 1, carl::Relation::LEQ );
363  opt.addConstraint( constraint, 0, carl::Relation::GEQ );
364  }
365 
366  // 4 add constraint that all coefficients add up to 1
367  vector_t<Number> constraint = vector_t<Number>::Ones( A.cols() );
368  opt.addConstraint( constraint, 1, carl::Relation::EQ );
369 
370  return opt.checkConsistency();
371 
372  } else {
373  // check via conversion
374  auto tmpHPoly = Converter::toHPolytope( *this );
375  return tmpHPoly.contains( vec );
376  }
377 }

442 template <typename Number, typename Converter, typename S>
443 VPolytopeT<Number, Converter, S> VPolytopeT<Number, Converter, S>::unite( const std::vector<VPolytopeT>& rhs ) {
444  if ( rhs.size() == 0 ) {
445  return VPolytopeT<Number, Converter, S>::Empty();
446  } else {
447  VPolytopeT<Number, Converter, S> result;
448  //std::cout << __func__ << " : of " << *this << " and " << rhs << std::endl;
449  VPolytopeT<Number, Converter, S>::pointVector points;
450 
451  std::set<Point<Number>> pointSet;
452  for ( const auto& poly : rhs ) {
453  pointSet.insert( rhs.mVertices.begin(), rhs.mVertices.end() );
454  }
455 
456  if ( pointSet.empty() ) {
457  return VPolytopeT<Number, Converter, S>();
458  }
459 
460  unsigned effDim = unsigned( effectiveDimension( pointSet ) );
461  points.insert( points.end(), pointSet.begin(), pointSet.end() );
462  assert( !points.empty() );
463 
464  if ( effDim < points.begin()->dimension() ) {
465  pointSet.clear();
466  std::vector<Halfspace<Number>> orientedHalfspaces = computeOrientedBox( points );
467  //std::cout << "Box has " << orientedHalfspaces.size() << " halfspaces in dimension " << points.begin()->dimension() << std::endl;
468  //assert(orientedHalfspaces.size() == points.begin()->dimension());
469 
470  // vertex computation of the oriented box
471  unsigned dim = points.begin()->dimension();
472  Permutator permutator( orientedHalfspaces.size(), dim );
473  std::vector<std::size_t> permutation;
474  while ( !permutator.end() ) {
475  permutation = permutator();
476 
477  matrix_t<Number> A( dim, dim );
478  vector_t<Number> b( dim );
479  unsigned pos = 0;
480  for ( auto planeIt = permutation.begin(); planeIt != permutation.end(); ++planeIt ) {
481  A.row( pos ) = orientedHalfspaces.at( *planeIt ).normal().transpose();
482  b( pos ) = orientedHalfspaces.at( *planeIt ).offset();
483  ++pos;
484  }
485  Eigen::FullPivLU<matrix_t<Number>> lu_decomp( A );
486  if ( lu_decomp.rank() < A.rows() ) {
487  continue;
488  }
489 
490  vector_t<Number> res = lu_decomp.solve( b );
491 
492  // Check if the computed vertex is a real vertex
493  bool outside = false;
494  for ( unsigned planePos = 0; planePos < orientedHalfspaces.size(); ++planePos ) {
495  bool skip = false;
496  for ( unsigned permPos = 0; permPos < permutation.size(); ++permPos ) {
497  if ( planePos == permutation.at( permPos ) ) {
498  skip = true;
499  break;
500  }
501  }
502 
503  if ( !skip ) {
504  if ( orientedHalfspaces.at( planePos ).offset() - orientedHalfspaces.at( planePos ).normal().dot( res ) < 0 ) {
505  //std::cout << "Drop vertex: " << res << " because of plane " << planePos << std::endl;
506  outside = true;
507  break;
508  }
509  }
510  }
511  if ( !outside ) {
512  pointSet.emplace( res );
513  }
514  }
515  for ( const auto& point : pointSet ) {
516  result.insert( point );
517  }
518  } else if ( points.size() > points.begin()->dimension() ) {
519  std::vector<std::shared_ptr<Facet<Number>>> facets = convexHull( points ).first;
520  std::set<Point<Number>> preresult;
521  for ( unsigned i = 0; i < facets.size(); i++ ) {
522  for ( unsigned j = 0; j < facets[i]->vertices().size(); j++ ) {

545 template <typename Number, typename Converter, typename S>
546 EvaluationResult<Number> VPolytopeT<Number, Converter, S>::evaluate( const vector_t<Number>& direction ) const {
547  if ( direction == vector_t<Number>::Zero( direction.rows() ) )
548  return EvaluationResult<Number>();
549  Number maxDist = Number( -1e20 );
550  std::size_t vIndex = 0;
551  EvaluationResult<Number> res;
552  for ( std::size_t i = 0; i < mVertices.size(); ++i ) {
553  //Point with the furthest distance to the direction vector is the optimal point and is always a corner
554  Number dist = direction.dot( mVertices[i].rawCoordinates() );
555  if ( dist > maxDist ) {
556  maxDist = dist;
557  vIndex = i;
558  }
559  }
560  res.optimumValue = mVertices[vIndex].rawCoordinates();
561  res.supportValue = Number( maxDist );
562  res.errorCode = SOLUTION::FEAS;
563  return res;
564 }

588 template <typename Number, typename Converter, typename S>
589 void VPolytopeT<Number, Converter, S>::removeRedundancy() {
590  if ( !mReduced ) {
591 #ifdef USE_SMTRAT
592  std::set<Point<Number>> toDelete;
593  smtrat::SimplexSolver simplex;
594  simplex.push();
595 
596  // create mapping of variables (lambdas') to vertices.
597  std::map<Point<Number>, carl::Variable> lambdas;
598  for ( const auto& vertex : mVertices ) {
599  carl::Variable lambda = carl::freshRealVariable();
600  lambdas.insert( std::make_pair( vertex, lambda ) );
601  //std::cout << "Assigned " << lambdas.find(vertex)->second << " to " << lambdas.find(vertex)->first << std::endl;
602  }
603 
604  for ( auto currVertex = mVertices.begin(); currVertex != mVertices.end(); ) {
605  //std::cout << currVertex << ": " << std::endl;
606  // create constraint for Sum(lambdas) == 1
607  carl::MultivariatePolynomial<smtrat::Rational> sumBound;
608  for ( const auto& vertex : mVertices ) {
609  if ( vertex != *currVertex ) {
610  carl::Variable lambda = lambdas.find( vertex )->second;
611  sumBound += lambda;
612  smtrat::FormulaT constr = smtrat::FormulaT( carl::MultivariatePolynomial<smtrat::Rational>( lambda ), carl::Relation::GEQ );
613  simplex.inform( constr );
614  simplex.add( constr );
615  }
616  }
617  sumBound -= smtrat::Rational( 1 );
618  smtrat::FormulaT constr = smtrat::FormulaT( sumBound, carl::Relation::EQ );
619  simplex.inform( constr );
620  simplex.add( constr );
621 
622  for ( unsigned dim = 0; dim < this->dimension(); ++dim ) {
623  carl::MultivariatePolynomial<smtrat::Rational> row;
624  for ( const auto& vertex : mVertices ) {
625  if ( *currVertex != vertex ) {
626  carl::Variable tmp = lambdas.find( vertex )->second;
627  row += tmp * carl::convert<Number, smtrat::Rational>( vertex.at( dim ) );
628  } else {
629  row -= carl::convert<Number, smtrat::Rational>( vertex.at( dim ) );
630  }
631  }
632  smtrat::FormulaT constr = smtrat::FormulaT( row, carl::Relation::EQ );
633  //std::cout << constr << std::endl;
634  simplex.inform( constr );
635  simplex.add( constr );
636  }
637 
638  //std::cout << "Checking: " << std::endl << simplex.formula().toString() << std::endl;
639 
640  smtrat::Answer res = simplex.check();
641 
642  if ( res == smtrat::Answer::SAT ) {
643  currVertex = mVertices.erase( currVertex );
644  } else {
645  ++currVertex;
646  }
647 
648  //std::cout << "Is extreme point: " << (res == smtrat::Answer::UNSAT) << std::endl << std::endl << std::endl << std::endl;
649  simplex.pop();
650  simplex.push();
651  }
652 
653  mReduced = true;
654 #else
655  updateNeighbors();
656 #endif
657  }
658 }
