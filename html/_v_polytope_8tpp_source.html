55 template <typename Number, typename Converter, typename S>
56 VPolytopeT<Number, Converter, S>::VPolytopeT( const matrix_t<Number> &_constraints, const vector_t<Number> _constants ) {
57  if constexpr ( is_exact<Number> ) {
58  auto dimension = _constraints.cols();
59  typename QuickIntersection<Number>::pointVector_t inputHalfspaces;
60 
61  for ( int i = 0; i < _constraints.rows(); ++i ) {
62  inputHalfspaces.emplace_back( dimension + 1 );
63  inputHalfspaces.back().head( dimension ) = _constraints.row( i );
64  inputHalfspaces.back()[dimension] = -_constants[i];
65  }
66 
67  QuickIntersection<Number> qInt{inputHalfspaces, (size_t)dimension};
68  qInt.compute();
69 
70  for ( auto &facet : qInt.getFacets() ) {
71  facet.mNormal /= -facet.mOffset;
72  mVertices.emplace_back( Point( std::move( facet.mNormal ) ) );
73  }
74  } else {
75  // calculate all possible Halfspace intersections
76  TRACE( "hypro.representations.vpolytope", "Construct from " << _constraints << " <= " << _constants );
77  //std::cout << __func__ << ": matrix: " << _constraints << " and vector: " << _constants << std::endl;
78  assert( _constraints.rows() == _constants.rows() );
79  Permutator permutator = Permutator( _constraints.rows(), _constraints.cols() );
80  matrix_t<Number> intersection = matrix_t<Number>( _constraints.cols(), _constraints.cols() );
81  vector_t<Number> intersectionConstants = vector_t<Number>( _constraints.cols() );
82  std::set<vector_t<Number>> possibleVertices;
83  std::vector<std::size_t> permutation;
84  while ( !permutator.end() ) {
85  permutation = permutator();
86  unsigned rowCount = 0;
87  // std::cout << "Intersect :" << std::endl;
88  for ( const auto &rowIndex : permutation ) {
89  // std::cout << _constraints.row(rowIndex) << " <= " <<
90  // _constants(rowIndex) << std::endl;
91  assert( rowCount < _constraints.cols() );
92  intersection.row( rowCount ) = _constraints.row( rowIndex );
93  intersectionConstants( rowCount ) = _constants( rowIndex );
94  ++rowCount;
95  }
96  // check if rank is full
97  if ( intersection.fullPivLu().rank() == intersection.cols() ) {
98  vector_t<Number> vertex = intersection.fullPivLu().solve( intersectionConstants );
99  assert( vertex.rows() == _constraints.cols() );
100  //assert(insidePlanes(vertex, intersection, intersectionConstants));
101  // avoid duplicate entries
102  if ( std::find( possibleVertices.begin(), possibleVertices.end(), vertex ) == possibleVertices.end() ) {
103  possibleVertices.emplace( std::move( vertex ) );
104  }
105  }
106  }
107  TRACE( "hypro.representations.vpolytope", ": Computed " << possibleVertices.size() << " possible vertices." );
108 
109  // check if vertices are true vertices (i.e. they fulfill all constraints)
110  for ( auto vertex = possibleVertices.begin(); vertex != possibleVertices.end(); ) {
111  // std::cout<<__func__ << " : " <<__LINE__ << " current position : " << i <<
112  // std::endl;
113  // std::cout<<__func__ << " : " <<__LINE__ << "number of vertices : " <<
114  // possibleVertices.size() << std::endl;
115  bool deleted = false;
116  for ( unsigned rowIndex = 0; rowIndex < _constraints.rows(); ++rowIndex ) {
117  Number res = vertex->dot( _constraints.row( rowIndex ) );
118  if ( !carl::AlmostEqual2sComplement( res, _constants( rowIndex ), default_double_comparison_ulps ) && res > _constants( rowIndex ) ) {
119  vertex = possibleVertices.erase( vertex );
120  deleted = true;
121  break;
122  }
123  }
124  if ( !deleted ) {
125  ++vertex;
126  }
127  }
128 
129  TRACE( "hypro.representations.vpolytope", "Deleted vertices. Remaining " << possibleVertices.size() << " vertices." );
130 
131  // std::cout<<__func__ << " : " <<__LINE__ <<std::endl;
132  // finish initialization
133  for ( const auto &point : possibleVertices ) {
134  mVertices.emplace_back( point );
135  mNeighbors.push_back( std::set<unsigned>() );
136  // std::cout << "Real vertex " << point.transpose() << std::endl;
137  }
138  // std::cout<<__func__ << " : " <<__LINE__ <<std::endl;
139  mReduced = false;
140 
141  //reduceNumberRepresentation();
142  }

206 template <typename Number, typename Converter, typename S>
207 VPolytopeT<Number, Converter, S> VPolytopeT<Number, Converter, S>::intersect( const VPolytopeT<Number, Converter, S> &rhs ) const {
208  // create a set of possible points via combination of all coordinates
209  if ( rhs.size() == 0 ) {
210  return VPolytopeT<Number, Converter, S>();
211  } else {
212  pointVector possibleVertices;
213  for ( const auto &lhsVertex : mVertices ) {
214  possibleVertices.push_back( lhsVertex );
215  for ( unsigned coordIndex = 0; coordIndex < lhsVertex.rawCoordinates().rows(); ++coordIndex ) {
216  for ( const auto &rhsVertex : rhs.mVertices ) {
217  vector_t<Number> newVertex = rhsVertex.rawCoordinates();
218  newVertex( coordIndex ) = lhsVertex.at( coordIndex );
219  possibleVertices.push_back( Point<Number>( newVertex ) );
220  possibleVertices.push_back( Point<Number>( rhsVertex.rawCoordinates() ) );
221  }
222  }
223  }
224  // remove all vertices, which are only contained in one of the polytopes
225  for ( auto vertexIt = possibleVertices.begin(); vertexIt != possibleVertices.end(); ) {
226  if ( !( this->contains( *vertexIt ) && rhs.contains( *vertexIt ) ) ) {
227  vertexIt = possibleVertices.erase( vertexIt );
228  } else {
229  ++vertexIt;
230  }
231  }
232  return VPolytopeT<Number, Converter, S>( std::move( possibleVertices ) );
233  }

254 template <typename Number, typename Converter, typename S>
255 std::pair<CONTAINMENT, VPolytopeT<Number, Converter, S>> VPolytopeT<Number, Converter, S>::satisfiesHalfspace( const Halfspace<Number> &rhs ) const {
256  // if we want to have fast pre-checks, this is the time to do them
257  if ( S::checkVerticesBeforeConversion ) {
258  // we count the number of vertices satisfying the halfspace to get a first impression about the result type.
259  std::size_t numberVerticesInRhs = numberOfPointsInHalfspace( mVertices, rhs );
260  // no vertex is contained - the result will be empty.
261  if ( numberVerticesInRhs == 0 ) {
262  COUNT( "VPolyAvoidConversion" );
263  return std::make_pair( CONTAINMENT::NO, VPolytopeT<Number, Converter, S>::Empty() );
264  } else if ( numberVerticesInRhs == mVertices.size() ) {
265  // all vertices are contained - the full set is the result.
266  COUNT( "VPolyAvoidConversion" );
267  return std::make_pair( CONTAINMENT::FULL, *this );
268  }
269  }
270 
271  auto intermediate = Converter::toHPolytope( *this );
272  auto resultPair = intermediate.satisfiesHalfspace( rhs );
273  if ( resultPair.first != CONTAINMENT::NO ) {
274  //resultPair.second.removeRedundancy();
275  VPolytopeT<Number, Converter, S> res( Converter::toVPolytope( resultPair.second ) );
276  return std::make_pair( resultPair.first, std::move( res ) );
277  }
278  return std::make_pair( resultPair.first, std::move( VPolytopeT<Number, Converter, S>::Empty() ) );
279 }

281 template <typename Number, typename Converter, typename S>
282 std::pair<CONTAINMENT, VPolytopeT<Number, Converter, S>> VPolytopeT<Number, Converter, S>::satisfiesHalfspaces( const matrix_t<Number> &_mat, const vector_t<Number> &_vec ) const {
283  // this stores results from pre-checks to reduce the effort of intersections later, if applicable.
284  std::vector<Eigen::Index> halfspacesToAvoid;
285  if ( S::checkVerticesBeforeConversion ) {
286  // we check for each halfspace individually to allow for premature exiting and reductions.
287  for ( Eigen::Index rowIndex = 0; rowIndex < _mat.rows(); ++rowIndex ) {
288  // we count the number of vertices satisfying the halfspace to get a first impression about the result type.
289  std::size_t numberVerticesInRhs = numberOfPointsInHalfspace( mVertices, Halfspace<Number>( _mat.row( rowIndex ), _vec( rowIndex ) ) );
290  // no vertex is contained - the result will be empty.
291  if ( numberVerticesInRhs == 0 ) {
292  COUNT( "VPolyAvoidConversion" );
293  return std::make_pair( CONTAINMENT::NO, VPolytopeT<Number, Converter, S>::Empty() );
294  } else if ( numberVerticesInRhs == mVertices.size() ) {
295  // all vertices are contained - the full set is the result for this halfspace, we can avoid it.
296  halfspacesToAvoid.push_back( rowIndex );
297  }
298  }
299  // none of the halfspaces cuts the set - return full set.
300  if ( halfspacesToAvoid.size() == std::size_t( _mat.rows() ) ) {
301  COUNT( "VPolyAvoidConversion" );
302  return std::make_pair( CONTAINMENT::FULL, *this );
303  }
304  }
305 
306  auto intermediate = Converter::toHPolytope( *this );
307  intermediate.reduceNumberRepresentation( mVertices );
308  //std::cout << typeid(*this).name() << "::" << __func__ << ": Intermediate hpoly: " << intermediate << std::endl;
309  auto resultPair = intermediate.satisfiesHalfspaces( removeRows( _mat, halfspacesToAvoid ), removeRows( _vec, halfspacesToAvoid ) );
310  if ( resultPair.first != CONTAINMENT::NO ) {
311  //resultPair.second.removeRedundancy();
312  assert( !resultPair.second.empty() );
313  TRACE( "hypro.representations.vpolytope", ": Intermediate hpoly convert back: " << resultPair.second );
314  VPolytopeT<Number, Converter, S> res( Converter::toVPolytope( resultPair.second ) );
315  TRACE( "hypro.representations.vpolytope", "Re-Converted v-poly: " << res );
316  assert( !res.empty() );
317  return std::make_pair( resultPair.first, std::move( res ) );
318  }
319  return std::make_pair( resultPair.first, std::move( VPolytopeT<Number, Converter, S>::Empty() ) );
320 }

327 template <typename Number, typename Converter, typename S>
328 bool VPolytopeT<Number, Converter, S>::contains( const vector_t<Number> &vec ) const {
329  if ( this->empty() ) {
330  return false;
331  }
332  if ( S::useLpForPointContainment ) {
333  // create LP Ax = b to solve whether vec can be represented as a convex combination of the vertices.
334  // 1 create matrix from vertices.
335  // 2 use vec as b
336  // 3 set bounds for cols to [0,1]
337  // 4 add constraint stating that all col-values need to sum up to 1.
338 
339  // 1 create matrix
340  matrix_t<Number> A = matrix_t<Number>( this->dimension(), mVertices.size() );
341  for ( Eigen::Index i = 0; i < A.cols(); ++i ) {
342  A.col( i ) = mVertices[i].rawCoordinates();
343  }
344 
345  Optimizer<Number> opt( A, vec );
346 
347  // 3 bound cols
348  for ( Eigen::Index i = 0; i < A.cols(); ++i ) {
349  vector_t<Number> constraint = vector_t<Number>::Zero( A.cols() );
350  constraint( i ) = 1;
351  opt.addConstraint( constraint, 1, carl::Relation::LEQ );
352  opt.addConstraint( constraint, 0, carl::Relation::GEQ );
353  }
354 
355  // 4 add constraint that all coefficients add up to 1
356  vector_t<Number> constraint = vector_t<Number>::Ones( A.cols() );
357  opt.addConstraint( constraint, 1, carl::Relation::EQ );
358 
359  return opt.checkConsistency();
360 
361  } else {
362  // check via conversion
363  auto tmpHPoly = Converter::toHPolytope( *this );
364  return tmpHPoly.contains( vec );
365  }
366 }

431 template <typename Number, typename Converter, typename S>
432 VPolytopeT<Number, Converter, S> VPolytopeT<Number, Converter, S>::unite( const std::vector<VPolytopeT> &rhs ) {
433  if ( rhs.size() == 0 ) {
434  return VPolytopeT<Number, Converter, S>::Empty();
435  } else {
436  VPolytopeT<Number, Converter, S> result;
437  //std::cout << __func__ << " : of " << *this << " and " << rhs << std::endl;
438  VPolytopeT<Number, Converter, S>::pointVector points;
439 
440  std::set<Point<Number>> pointSet;
441  for ( const auto &poly : rhs ) {
442  pointSet.insert( rhs.mVertices.begin(), rhs.mVertices.end() );
443  }
444 
445  if ( pointSet.empty() ) {
446  return VPolytopeT<Number, Converter, S>();
447  }
448 
449  unsigned effDim = unsigned( effectiveDimension( pointSet ) );
450  points.insert( points.end(), pointSet.begin(), pointSet.end() );
451  assert( !points.empty() );
452 
453  if ( effDim < points.begin()->dimension() ) {
454  pointSet.clear();
455  std::vector<Halfspace<Number>> orientedHalfspaces = computeOrientedBox( points );
456  //std::cout << "Box has " << orientedHalfspaces.size() << " halfspaces in dimension " << points.begin()->dimension() << std::endl;
457  //assert(orientedHalfspaces.size() == points.begin()->dimension());
458 
459  // vertex computation of the oriented box
460  unsigned dim = points.begin()->dimension();
461  Permutator permutator( orientedHalfspaces.size(), dim );
462  std::vector<std::size_t> permutation;
463  while ( !permutator.end() ) {
464  permutation = permutator();
465 
466  matrix_t<Number> A( dim, dim );
467  vector_t<Number> b( dim );
468  unsigned pos = 0;
469  for ( auto planeIt = permutation.begin(); planeIt != permutation.end(); ++planeIt ) {
470  A.row( pos ) = orientedHalfspaces.at( *planeIt ).normal().transpose();
471  b( pos ) = orientedHalfspaces.at( *planeIt ).offset();
472  ++pos;
473  }
474  Eigen::FullPivLU<matrix_t<Number>> lu_decomp( A );
475  if ( lu_decomp.rank() < A.rows() ) {
476  continue;
477  }
478 
479  vector_t<Number> res = lu_decomp.solve( b );
480 
481  // Check if the computed vertex is a real vertex
482  bool outside = false;
483  for ( unsigned planePos = 0; planePos < orientedHalfspaces.size(); ++planePos ) {
484  bool skip = false;
485  for ( unsigned permPos = 0; permPos < permutation.size(); ++permPos ) {
486  if ( planePos == permutation.at( permPos ) ) {
487  skip = true;
488  break;
489  }
490  }
491 
492  if ( !skip ) {
493  if ( orientedHalfspaces.at( planePos ).offset() - orientedHalfspaces.at( planePos ).normal().dot( res ) < 0 ) {
494  //std::cout << "Drop vertex: " << res << " because of plane " << planePos << std::endl;
495  outside = true;
496  break;
497  }
498  }
499  }
500  if ( !outside ) {
501  pointSet.emplace( res );
502  }
503  }
504  for ( const auto &point : pointSet ) {
505  result.insert( point );
506  }
507  } else if ( points.size() > points.begin()->dimension() ) {
508  std::vector<std::shared_ptr<Facet<Number>>> facets = convexHull( points ).first;
509  std::set<Point<Number>> preresult;
510  for ( unsigned i = 0; i < facets.size(); i++ ) {
511  for ( unsigned j = 0; j < facets[i]->vertices().size(); j++ ) {
512  preresult.insert( facets[i]->vertices().at( j ) );
513  }
514  }
515  VPolytopeT<Number, Converter, S>::pointVector res;
516  for ( const auto &point : preresult ) {
517  res.push_back( point );

534 template <typename Number, typename Converter, typename S>
535 EvaluationResult<Number> VPolytopeT<Number, Converter, S>::evaluate( const vector_t<Number> &direction ) const {
536  if ( direction == vector_t<Number>::Zero( direction.rows() ) )
537  return EvaluationResult<Number>();
538  Number maxDist = Number( -1e20 );
539  std::size_t vIndex = 0;
540  EvaluationResult<Number> res;
541  for ( std::size_t i = 0; i < mVertices.size(); ++i ) {
542  //Point with the furthest distance to the direction vector is the optimal point and is always a corner
543  Number dist = direction.dot( mVertices[i].rawCoordinates() );
544  if ( dist > maxDist ) {
545  maxDist = dist;
546  vIndex = i;
547  }
548  }
549  res.optimumValue = mVertices[vIndex].rawCoordinates();
550  res.supportValue = Number( maxDist );
551  res.errorCode = SOLUTION::FEAS;
552  return res;
553 }

577 template <typename Number, typename Converter, typename S>
578 void VPolytopeT<Number, Converter, S>::removeRedundancy() {
579  if ( !mReduced ) {
580 #ifdef USE_SMTRAT
581  std::set<Point<Number>> toDelete;
582  smtrat::SimplexSolver simplex;
583  simplex.push();
584 
585  // create mapping of variables (lambdas') to vertices.
586  std::map<Point<Number>, carl::Variable> lambdas;
587  for ( const auto &vertex : mVertices ) {
588  carl::Variable lambda = carl::freshRealVariable();
589  lambdas.insert( std::make_pair( vertex, lambda ) );
590  //std::cout << "Assigned " << lambdas.find(vertex)->second << " to " << lambdas.find(vertex)->first << std::endl;
591  }
592 
593  for ( auto currVertex = mVertices.begin(); currVertex != mVertices.end(); ) {
594  //std::cout << currVertex << ": " << std::endl;
595  // create constraint for Sum(lambdas) == 1
596  carl::MultivariatePolynomial<smtrat::Rational> sumBound;
597  for ( const auto &vertex : mVertices ) {
598  if ( vertex != *currVertex ) {
599  carl::Variable lambda = lambdas.find( vertex )->second;
600  sumBound += lambda;
601  smtrat::FormulaT constr = smtrat::FormulaT( carl::MultivariatePolynomial<smtrat::Rational>( lambda ), carl::Relation::GEQ );
602  simplex.inform( constr );
603  simplex.add( constr );
604  }
605  }
606  sumBound -= smtrat::Rational( 1 );
607  smtrat::FormulaT constr = smtrat::FormulaT( sumBound, carl::Relation::EQ );
608  simplex.inform( constr );
609  simplex.add( constr );
610 
611  for ( unsigned dim = 0; dim < this->dimension(); ++dim ) {
612  carl::MultivariatePolynomial<smtrat::Rational> row;
613  for ( const auto &vertex : mVertices ) {
614  if ( *currVertex != vertex ) {
615  carl::Variable tmp = lambdas.find( vertex )->second;
616  row += tmp * carl::convert<Number, smtrat::Rational>( vertex.at( dim ) );
617  } else {
618  row -= carl::convert<Number, smtrat::Rational>( vertex.at( dim ) );
619  }
620  }
621  smtrat::FormulaT constr = smtrat::FormulaT( row, carl::Relation::EQ );
622  //std::cout << constr << std::endl;
623  simplex.inform( constr );
624  simplex.add( constr );
625  }
626 
627  //std::cout << "Checking: " << std::endl << simplex.formula().toString() << std::endl;
628 
629  smtrat::Answer res = simplex.check();
630 
631  if ( res == smtrat::Answer::SAT ) {
632  currVertex = mVertices.erase( currVertex );
633  } else {
634  ++currVertex;
635  }
636 
637  //std::cout << "Is extreme point: " << (res == smtrat::Answer::UNSAT) << std::endl << std::endl << std::endl << std::endl;
638  simplex.pop();
639  simplex.push();
640  }
641 
642  mReduced = true;
643 #else
644  updateNeighbors();
645 #endif
646  }
647 }
