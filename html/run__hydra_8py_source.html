1 #!/usr/bin/env python3
2 """
3 Run multiple benchmarks consecutively.
4 Required arguments (positional):
5  tool: The path to the tool (hydra) executable
6  settings: Path to a text file that contains the command line
7  arguments for the benchmarks to run. For example, if hydra
8  should be called with
9  `./bin/hydra -m bball.model -r support_function --setting BoxLinearOptimizationOn`
10  then the settings file should contain a line
11  `-m bball.model -r support_function --setting BoxLinearOptimizationOn`
12  Paths to the model (-m) should be given relative to the settings file.
13  Empty lines and lines that start with '#' are ignored.
14  output: Path to a file to write the output to. The output will
15  be given in csv-format with the measured time, safety result and
16  error messages if an error occured.
17 Optional arguments:
18  -t <seconds> A time limit in seconds. If the execution of a benchmark
19  model takes longer than the given time it will be terminated.
20  This is indicated in the output.
21  -m <megabyte> A memory limit in megabyte. If the execution of a benchmark
22  model uses more space than allowed it will be terminated.
23  This is NOT indicated in the output but will cause an error
24  in the tool execution and the error message has to be interpreted.
25  --verbose Verbose execution: The output of the tool will be printed.
26  --benchmark Faster execution: The output is written to a file and analyzed
27  after execution is complete. This lowers runtime but disables
28  the timeout and verbose option as well as skipping functionality.
29  (These arguments are ignored.)
30 Example calls:
31  (1) `python3 run_hydra.py ../bin/hydra input/settings results.csv -t 1200 -m 16000 --verbose`
32  Call for testing models with a timeout of 20min and memout of 16gb with verbose mode.
33  (2) `python3 run_hydra.py ../bin/hydra input/settings results.csv --benchmark`
34  Call for benchmarking and faster execution. Adding the timeout or verbose option
35  will have no effect.
36 """

132  def _run_test(self, result, tool, timeLimit, memLimit, verbose):
133  params = self.get_process_params(tool)
134  process = subprocess.Popen(
135  params, stdout=subprocess.PIPE, stderr=subprocess.PIPE,
136  universal_newlines=True, preexec_fn=lambda: limit_mem(memLimit))
137  # Make stdout non-blocking to enable reading during execution (to avoid
138  # filling the pipe and losing output), while also checking for timeouts
139  # and skips.
140  pStdoutFlags = fcntl.fcntl(process.stdout, fcntl.F_GETFL)
141  fcntl.fcntl(process.stdout, fcntl.F_SETFL, pStdoutFlags | os.O_NONBLOCK)
142 
143  while process.poll() is None:
144  for line in process.stdout.readlines():
145  self._parse_line(result, line, verbose)
146  if self._check_skip(result):
147  self._stop_run(process)
148  print('[+] Skipped %s'%self.name)
149  if self._check_timeout(result, timeLimit):
150  print('[-] Timeout %s'%self.name)
151  self._stop_run(process)
152  # Read errors after process is done because stderr usually fits in
153  # the pipe.
154  errors = process.stderr.readlines()
155  if errors:
156  print('[-] Errors during execution: ')
157  result.set_error(errors[0])
158  for line in errors:
159  print(line, end='')
160  return result
