33  smtrat::ModelValue valuation = solver.optimum(objective);

35  assert(!valuation.isBool());

41  if(valuation.isMinusInfinity() || valuation.isPlusInfinity() ){
42  // std::cout << __func__ << ": INFTY" << std::endl;
43  res = EvaluationResult<Number>( 1, INFTY );
44  } else {
45  assert(valuation.isRational());

47  smtrat::Model assignment = solver.model();
48  vector_t<Number> point = vector_t<Number>(dimension);
49  for(unsigned d = 0; d < dimension; ++d){
50  // if the variable did not occur in the current call, set to 0.
51  if(assignment.find(hypro::VariablePool::getInstance().carlVarByIndex(d)) == assignment.end()){
52  point(d) = 0;
53  } else {
54  assert(assignment.find(hypro::VariablePool::getInstance().carlVarByIndex(d)) != assignment.end());
55  assert(assignment.find(hypro::VariablePool::getInstance().carlVarByIndex(d))->second.isRational());
56  point(d) = carl::convert<smtrat::Rational, Number>(assignment.find(hypro::VariablePool::getInstance().carlVarByIndex(d))->second.asRational());
57  }
58  }
59  res.optimumValue = point;
60  #ifdef VERIFY_RESULT
61  solver.push();
62  smtrat::FormulaT inversedObjective(objective-valuation.asRational(), carl::Relation::GREATER);
63  solver.inform(inversedObjective);
64  solver.add(inversedObjective);
65  if(solver.check() != smtrat::Answer::UNSAT) {

67  }
68  solver.pop();
69  #endif
