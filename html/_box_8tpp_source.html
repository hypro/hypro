295 EvaluationResult<Number> BoxT<Number, Converter, Setting>::evaluate( const vector_t<Number>& _direction, bool ) const {
296  DEBUG( "hypro.representations.box", "In evaluate. direction: " << std::endl
297  << _direction );
298  assert( _direction.rows() == Eigen::Index( this->dimension() ) || this->empty() );
299  if ( this->empty() ) {
300  return EvaluationResult<Number>(); // defaults to infeasible, i.e. empty.
301  }
302  // find the point, which represents the maximum towards the direction - compare signs.
303  vector_t<Number> furthestPoint = vector_t<Number>::Zero( this->dimension() );
304  for ( Eigen::Index i = 0; i < furthestPoint.rows(); ++i ) {
305  if ( _direction( i ) >= 0 ) {
306  //Unboundedness check
307  if ( mLimits[i].upperBoundType() == carl::BoundType::INFTY ) {
308  return EvaluationResult<Number>( 0, furthestPoint, SOLUTION::INFTY );
309  }
310  furthestPoint( i ) = mLimits[i].upper();
311  } else {
312  //Unboundedness check
313  if ( mLimits[i].lowerBoundType() == carl::BoundType::INFTY ) {
314  return EvaluationResult<Number>( 0, furthestPoint, SOLUTION::INFTY );
315  }
316  furthestPoint( i ) = mLimits[i].lower();
317  }
318  }
319  return EvaluationResult<Number>( furthestPoint.dot( _direction ), furthestPoint, SOLUTION::FEAS );
320 }

376 std::pair<CONTAINMENT, BoxT<Number, Converter, Setting>> BoxT<Number, Converter, Setting>::satisfiesHalfspaces( const matrix_t<Number>& _mat, const vector_t<Number>& _vec ) const {
377  if ( _mat.rows() == 0 ) {
378  return std::make_pair( CONTAINMENT::FULL, *this );
379  }
380 
381  if ( this->empty() ) {
382  //std::cout << __func__ << " Box is empty." << std::endl;
383  return std::make_pair( CONTAINMENT::NO, *this );
384  }
385 
386  // std::cout << __func__ << " This: " << convert<Number,double>(*this) << std::endl;
387  // std::cout << __func__ << ": input matrix: " << convert<Number,double>(_mat) << std::endl << "input vector: " << convert<Number,double>(_vec) << std::endl;
388  // std::cout << __func__ << ": This->dimension() = " << this->dimension() << std::endl;
389  assert( this->dimension() == unsigned( _mat.cols() ) );
390  std::vector<unsigned> limitingPlanes;
391 
392  for ( unsigned rowIndex = 0; rowIndex < _mat.rows(); ++rowIndex ) {
393  carl::Interval<Number> evaluatedBox = carl::Interval<Number>( 0 );
394  for ( unsigned d = 0; d < _mat.cols(); ++d ) {
395  evaluatedBox += mLimits[d] * Number( _mat( rowIndex, d ) );
396  }
397 
398  //std::cout << __func__ << " Row: " << convert<Number,double>(_mat.row(rowIndex)) << std::endl;
399  //std::cout << __func__ << " Evaluated box: " << evaluatedBox << std::endl;
400  //std::cout << __func__ << " Distance: " << carl::convert<Number,double>(_vec(rowIndex)) << std::endl;
401 
402  if ( evaluatedBox.lower() > _vec( rowIndex ) ) {
403  return std::make_pair( CONTAINMENT::NO, Empty() );
404  }
405 
406  if ( evaluatedBox.upper() > _vec( rowIndex ) ) {
407  limitingPlanes.push_back( rowIndex );
408  }
409  }
410 
411  // at this point the box cannot be empty.
412  if ( limitingPlanes.empty() ) {
413  return std::make_pair( CONTAINMENT::FULL, *this );
414  }
415 
416  // at this point the box will be limited but not empty.
417  matrix_t<Number> newPlanes = matrix_t<Number>( limitingPlanes.size(), _mat.cols() );
418  vector_t<Number> newDistances = vector_t<Number>( limitingPlanes.size() );
419  Eigen::Index rowPos = newPlanes.rows() - 1;
420  while ( !limitingPlanes.empty() ) {
421  assert( rowPos >= 0 );
422  newPlanes.row( rowPos ) = _mat.row( limitingPlanes.back() );
423  newDistances( rowPos ) = _vec( limitingPlanes.back() );
424  --rowPos;
425  limitingPlanes.pop_back();
426  }
427  assert( newPlanes.rows() == newDistances.rows() );
428  auto tmp = this->intersectHalfspaces( newPlanes, newDistances );
429  bool empty = tmp.empty();
430  if ( empty ) {
431  return std::make_pair( CONTAINMENT::NO, tmp );
432  } else {
433  return std::make_pair( CONTAINMENT::PARTIAL, tmp );
434  }

564 BoxT<Number, Converter, Setting> BoxT<Number, Converter, Setting>::intersect( const BoxT<Number, Converter, Setting>& rhs ) const {
565  if ( this->empty() || rhs.empty() ) {
566  return Empty();
567  }
568  std::vector<carl::Interval<Number>> newIntervals;
569  std::size_t dim = this->dimension();
570  std::size_t rdim = rhs.dimension();
571  newIntervals.reserve( std::max( dim, rdim ) );
572  for ( std::size_t d = 0; d < this->dimension(); ++d ) {
573  // intersection if both agree on the dimension

576  else // if this->dimension() > rdim use projection
577  newIntervals.emplace_back( mLimits[d] );
578  }
579  // if the other box has a larger dimension use projection as well.
580  if ( rdim > dim ) {
581  for ( std::size_t d = dim; d < rdim; ++d ) {
582  newIntervals.emplace_back( rhs.interval( d ) );
583  }
584  }
585  return BoxT<Number, Converter, Setting>( newIntervals );

588 template <typename Number, typename Converter, class Setting>
589 BoxT<Number, Converter, Setting> BoxT<Number, Converter, Setting>::intersectHalfspace( const Halfspace<Number>& hspace ) const {
590  //std::cout << __func__ << " of " << *this << " and " << hspace << std::endl;
591  if ( !this->empty() ) {
592  if ( Setting::USE_INTERVAL_ARITHMETIC ) {
593  std::vector<carl::Interval<Number>> intervals = this->intervals();
594  bool empty = icpIntersectHalfspace( intervals, hspace );
595  if ( empty ) {
596  return Empty();
597  }
598  return BoxT<Number, Converter, Setting>( intervals );
599  } else { // Do not use interval arithmetic.
600  BoxT<Number, Converter, Setting> boxcopy( *this );
601  // Preprocessing: If any two points opposite to each other are contained, the box stays the same - test limit points
602  bool holdsMin = hspace.contains( boxcopy.min() );
603  bool holdsMax = hspace.contains( boxcopy.max() );
604  if ( holdsMin && holdsMax ) {
605  //std::cout << __func__ << " Min and Max are below the halfspace." << std::endl;
606  return *this;
607  }
608 
609  // another special case: if the hspace normal is axis-aligned, i.e. it has only one non-zero entry, we simply can use interval-
610  // style intersection.
611  if ( hspace.normal().nonZeros() == 1 ) {
612  // from previous checks we know that either the lowest or the highest point is not contained. If both are not
613  // contained and the normal is axis-aligned, the set is empty.
614  if ( !holdsMin && !holdsMax ) {
615  return Empty();
616  }
617 
618  // find the one, non-zero component
619  unsigned nonZeroDim = 0;
620  while ( hspace.normal()( nonZeroDim ) == 0 ) ++nonZeroDim;
621 
622  if ( hspace.normal()( nonZeroDim ) > 0 ) {
623  boxcopy.rInterval( nonZeroDim ).setUpper( hspace.offset() / hspace.normal()( nonZeroDim ) );
624  } else {
625  boxcopy.rInterval( nonZeroDim ).setLower( hspace.offset() / hspace.normal()( nonZeroDim ) );
626  }
627  return boxcopy;
628  }
629 
630  //std::cout << __func__ << " Min below: " << holdsMin << ", Max below: " << holdsMax << std::endl;
631  std::size_t dim = this->dimension();
632 
633  // Phase 1: Find starting point (point outside) for phase 2 by depth-first search or use limit points, if applicable
634  Point<Number> farestPointOutside = boxcopy.min();
635  Point<Number> farestPointInside = boxcopy.min();
636  std::size_t usedDimension = 0;
637  // determine walk direction by using plane normal and variable order
638  for ( ; usedDimension < dim; ++usedDimension ) {
639  if ( hspace.normal()( usedDimension ) > 0 ) {
640  if ( farestPointOutside.at( usedDimension ) != boxcopy.intervals()[usedDimension].upper() ) {
641  farestPointOutside[usedDimension] = boxcopy.intervals()[usedDimension].upper();
642  }
643  if ( farestPointInside.at( usedDimension ) != boxcopy.intervals()[usedDimension].lower() ) {
644  farestPointInside[usedDimension] = boxcopy.intervals()[usedDimension].lower();
645  }
646  } else if ( hspace.normal()( usedDimension ) < 0 ) {
647  if ( farestPointOutside.at( usedDimension ) != boxcopy.intervals()[usedDimension].lower() ) {
648  farestPointOutside[usedDimension] = boxcopy.intervals()[usedDimension].lower();
649  }
650  if ( farestPointInside.at( usedDimension ) != boxcopy.intervals()[usedDimension].upper() ) {
651  farestPointInside[usedDimension] = boxcopy.intervals()[usedDimension].upper();
652  }
653  }
654  }
655  // farestPointOutside is the point farest point in direction of the plane normal - if it is contained in the halfspace, there is no intersection.
656  if ( hspace.contains( farestPointOutside.rawCoordinates() ) ) {
657  //std::cout << __func__ << " Farest point outside is contained - return full box." << std::endl;
658  return *this;
659  }
660  if ( !hspace.contains( farestPointInside.rawCoordinates() ) ) {
661  //std::cout << __func__ << " Farest point inside is NOT contained - return EMPTY box." << std::endl;
662  return BoxT<Number, Converter, Setting>::Empty();
663  }
664 
665  //std::cout << __func__ << " Farest point outside: " << convert<Number,double>(farestPointOutside.rawCoordinates()).transpose() << std::endl;
666 
667  // at this point farestPointOutside is outside and farestPointInside is inside - the plane intersects the box somehow.
668  std::vector<Point<Number>> discoveredPoints;
669  std::vector<Point<Number>> intersectionPoints;
670  std::queue<Point<Number>> workingQueue;
671  workingQueue.push( farestPointOutside );
672 
673  // BFS search of points outside and intersection points.
674  //std::cout << __func__ << " Start BFS search." << std::endl;
675  while ( !workingQueue.empty() ) {
676  //std::cout << "Queue size: " << workingQueue.size() << std::endl;
677  Point<Number> current = workingQueue.front();
678  workingQueue.pop();
679  //std::cout << "Current Point: " << convert<Number,double>(current.rawCoordinates()).transpose() << std::endl;
680  // create and test neighbors
681  for ( unsigned d = 0; d < dim; ++d ) {
682  Point<Number> tmp = current;
683  if ( hspace.normal()( d ) < 0 && current.at( d ) == boxcopy.intervals()[d].lower() ) {
684  tmp[d] = boxcopy.intervals()[d].upper();
685  } else if ( hspace.normal()( d ) > 0 && current.at( d ) == boxcopy.intervals()[d].upper() ) {
686  tmp[d] = boxcopy.intervals()[d].lower();
687  } else if ( hspace.normal()( d ) == 0 ) {
688  tmp[d] = tmp.at( d ) == boxcopy.intervals()[d].lower() ? boxcopy.intervals()[d].upper() : boxcopy.intervals()[d].lower();
689  } else {
690  // UNSINN!?
691  //std::cout << "Could create point " << tmp << ", but is the same as " << current << std::endl;
692  assert( tmp == current );
693  continue;
694  }
695 
696  //std::cout << "Created search-point: " << convert<Number,double>(tmp.rawCoordinates()).transpose() << std::endl;
697  // if neighbor is new, test is, otherwise skip.
698  if ( std::find( discoveredPoints.begin(), discoveredPoints.end(), tmp ) == discoveredPoints.end() ) {
699  if ( !hspace.contains( tmp.rawCoordinates() ) ) {
700  //std::cout << "is also outside, enqueue." << std::endl;
701  workingQueue.push( tmp );
702  } else {
703  Number dCoord = 0;
704  for ( unsigned i = 0; i < dim; ++i ) {
705  if ( i != d ) {
706  dCoord += hspace.normal()( i ) * current.at( i );
707  }
708  }
709  dCoord -= hspace.offset();
710  dCoord /= -hspace.normal()( d );
711  Point<Number> intersectionPoint = tmp;
712  intersectionPoint[d] = dCoord;
713  //std::cout << "is inside, intersection point is " << convert<Number,double>(intersectionPoint.rawCoordinates()).transpose() << std::endl;
714  intersectionPoints.push_back( intersectionPoint );
715  }
716  } else {
717  //std::cout << "Already discovered - skip." << std::endl;
718  }
719  }
720  discoveredPoints.push_back( current );
721  }
722  //std::cout << __func__ << " BFS search finished." << std::endl;
723 
724  // at this point we know that either min or max or both are outside but not both inside.
725  if ( !holdsMin && !holdsMax ) {
726  return BoxT<Number, Converter, Setting>( intersectionPoints );
727  } else {
728  if ( holdsMin ) {
729  intersectionPoints.push_back( boxcopy.min() );
730  } else {
731  intersectionPoints.push_back( boxcopy.max() );
732  }
733  return BoxT<Number, Converter, Setting>( intersectionPoints );
734  }
735  }
736  }
737  return Empty();
738 }

740 template <typename Number, typename Converter, class Setting>
741 BoxT<Number, Converter, Setting> BoxT<Number, Converter, Setting>::intersectHalfspaces( const matrix_t<Number>& _mat, const vector_t<Number>& _vec ) const {
742  assert( _mat.rows() == _vec.rows() );
743  assert( _mat.cols() == Eigen::Index( this->dimension() ) );
744  if ( _mat.rows() == 0 ) {
745  return *this;
746  }
747  if ( this->empty() ) {
748  return Empty();
749  }
750 
751  if ( Setting::USE_INTERVAL_ARITHMETIC ) {
752  std::vector<carl::Interval<Number>> intervals = this->intervals();
753  // Todo: This is a first draft using the function for single halfspaces - maybe we can check more than one plane at the same time.
754  for ( unsigned planeIndex = 0; planeIndex < _mat.rows(); ++planeIndex ) {
755  bool empty = icpIntersectHalfspace( intervals, Halfspace<Number>( _mat.row( planeIndex ), _vec( planeIndex ) ) );
756  if ( empty ) {
757  return Empty();
758  }
759  }
760  return BoxT<Number, Converter, Setting>( intervals );
761  } else if ( Setting::HYPRO_BOX_AVOID_LINEAR_OPTIMIZATION ) {
762  if ( _mat.rows() == 0 ) {
763  return *this;
764  }
765  if ( !this->empty() ) {
766  BoxT<Number, Converter, Setting> result = *this;
767  // Todo: This is a first draft using the function for single halfspaces - maybe we can check more than one plane at the same time.
768  for ( unsigned planeIndex = 0; planeIndex < _mat.rows(); ++planeIndex ) {
769  result = result.intersectHalfspace( Halfspace<Number>( _mat.row( planeIndex ), _vec( planeIndex ) ) );
770  if ( result.empty() ) {
771  break;
772  }
773  }
774  return result;
775  }
776  return Empty();
777  } else {
778  // convert box to a set of constraints, add other halfspaces and evaluate in box main directions to get new intervals.
779  std::vector<vector_t<Number>> tpl = computeTemplate<Number>( this->dimension(), 4 );
780  matrix_t<Number> boxDirections = this->matrix();
781  vector_t<Number> boxDistances = this->vector();
782 
783  // At this point the constraints for the box are created as a matrix-vector pair.
784  // Now add the halfspace constraints in a fresh matrix (we re-use the box template later).
785  matrix_t<Number> constraints = matrix_t<Number>( boxDirections.rows() + _mat.rows(), _mat.cols() );
786  vector_t<Number> constants = vector_t<Number>( boxDistances.rows() + _mat.rows() );
787  constraints.block( 0, 0, boxDirections.rows(), boxDirections.cols() ) = boxDirections;
788  constraints.block( boxDirections.rows(), 0, _mat.rows(), _mat.cols() ) = _mat;
789  constants.block( 0, 0, boxDirections.rows(), 1 ) = boxDistances;
790  constants.block( boxDirections.rows(), 0, _vec.rows(), 1 ) = _vec;
791 
792  // evaluate in box directions.
793  Optimizer<Number> opt( constraints, constants );
794  std::vector<EvaluationResult<Number>> results;
795  for ( Eigen::Index rowIndex = 0; rowIndex < boxDirections.rows(); ++rowIndex ) {
796  results.emplace_back( opt.evaluate( boxDirections.row( rowIndex ), false ) );
797  }
798  opt.cleanContexts();
799  assert( Eigen::Index( results.size() ) == boxDirections.rows() );
800 
801  // re-construct box from results.
802  std::pair<Point<Number>, Point<Number>> newLimits = std::make_pair( Point<Number>( vector_t<Number>::Zero( this->dimension() ) ), Point<Number>( vector_t<Number>::Zero( this->dimension() ) ) );
803  for ( Eigen::Index rowIndex = 0; rowIndex < boxDirections.rows(); ++rowIndex ) {
804  for ( Eigen::Index colIndex = 0; colIndex < boxDirections.cols(); ++colIndex ) {
805  if ( boxDirections( rowIndex, colIndex ) > 0 ) {
806  newLimits.second[colIndex] = results[rowIndex].supportValue;
807  } else if ( boxDirections( rowIndex, colIndex ) < 0 ) {
808  newLimits.first[colIndex] = -results[rowIndex].supportValue;
809  }
810  }
811  }
812  BoxT<Number, Converter, Setting> res( newLimits );
813  res.reduceNumberRepresentation();
814  return res;
815  }
816 }
