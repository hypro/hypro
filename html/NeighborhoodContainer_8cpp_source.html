51 void NeighborhoodContainer::calculateNeighborhood( const Point<int>& p ) {
52  std::vector<Point<int> > neighbors;
53  unsigned dim = p.dimension();
54 
55  // the number of neighbors is 2^(dimension) - 1
56  int nrofNeighbors = ( int( pow( 2, dim ) ) - 1 );
57 
58  vector_t<int> coordinates = p.rawCoordinates();
59  vector_t<int> pointCoordinates = p.rawCoordinates();
60  assert( coordinates.rows() == dim );
61 
62  // iterate through all neighbors
63  for ( int neighborNr = 1; neighborNr <= nrofNeighbors; neighborNr++ ) {
64  // then iterate through all dimensions
65  int i = 0;
66  for ( unsigned d = 0; d < dim; ++d ) {
67  // look if the bit of the current coordinate is set
68  // thus the first point will have 1 less in every second dimension,
69  // the second point will have 1 less in every fourth dimension etc.
70  if ( ( neighborNr >> i++ ) & 1 ) {
71  coordinates( d ) = pointCoordinates( d ) - 1;
72  } else {
73  coordinates( d ) = pointCoordinates( d );
74  }
75  }

78  }
