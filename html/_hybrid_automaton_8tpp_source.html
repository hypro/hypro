233 template <typename Number>
234 bool HybridAutomaton<Number>::isComposedOf( const HybridAutomaton<Number>& rhs ) const {
235  // trivial case.
236  if ( *this == rhs ) return true;
237 
238  // check variable sets
239  for ( const auto& v : rhs.getVariables() ) {
240  if ( std::find( mVariables.begin(), mVariables.end(), v ) == mVariables.end() ) {
241  //std::cout << "Variable " << v << " not contained in this, return false" << std::endl;
242  return false;
243  }
244  }
245 
246  // check locations:
247  // try to find *exactly* one location, which matches - matching is defined by name, flow and invariant.
248  for ( auto& locPtr : this->mLocations ) {
249  bool foundOne = false;
250  for ( auto& rhsLocPtr : rhs.getLocations() ) {
251  if ( locPtr->isComposedOf( *rhsLocPtr, rhs.getVariables(), this->getVariables() ) ) {
252  if ( foundOne ) {
253  return false;
254  }
255  foundOne = true;
256  }
257  }
258  if ( !foundOne ) {
259  return false;
260  }
261  }
262 
263  return true;
264 }
