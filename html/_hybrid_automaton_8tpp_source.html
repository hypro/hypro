242 template <typename Number>
243 bool HybridAutomaton<Number>::isComposedOf( const HybridAutomaton<Number>& rhs ) const {
244  // trivial case.
245  if ( *this == rhs ) return true;
246 
247  // check variable sets
248  for ( const auto& v : rhs.getVariables() ) {
249  if ( std::find( mVariables.begin(), mVariables.end(), v ) == mVariables.end() ) {
250  //std::cout << "Variable " << v << " not contained in this, return false" << std::endl;
251  return false;
252  }
253  }
254 
255  // check locations:
256  // try to find *exactly* one location, which matches - matching is defined by name, flow and invariant.
257  for ( auto& locPtr : this->mLocations ) {
258  bool foundOne = false;
259  for ( auto& rhsLocPtr : rhs.getLocations() ) {
260  if ( locPtr->isComposedOf( *rhsLocPtr, rhs.getVariables(), this->getVariables() ) ) {
261  if ( foundOne ) {
262  return false;
263  }
264  foundOne = true;
265  }
266  }
267  if ( !foundOne ) {
268  return false;
269  }
270  }
271 
272  return true;
273 }
