109  std::vector<unsigned> N, unsigned& i, unsigned& j, unsigned f, unsigned g ) {
110  // step 1
111  unsigned r = 0;

114  bool secondSet = false;
115  for ( ; s < N.size(); ++s ) {
116  assert( N.at( s ) >= 0 && N.at( s ) <= A.cols() );
117  if ( A( f, N.at( s ) ) > 0 ) {
118  for ( ; r < B.size(); ++r ) {
119  if ( B.at( r ) != f && A( B.at( r ), N.at( s ) ) < 0 &&
120  -( A( B.at( r ), g ) / A( B.at( r ), N.at( s ) ) ) < min ) {
121  secondSet = true;
122  break;
123  }
124  }
125  }
126  if ( secondSet ) break;
127  }
128  i = r;
129  j = s;
