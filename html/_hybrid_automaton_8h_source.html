183  bool equals( const std::vector<T*>& lhs, const std::vector<T*>& rhs ) const {
184  if ( lhs.size() != rhs.size() ) return false;
185  for ( auto lhsIt = lhs.begin(); lhsIt != lhs.end(); ++lhsIt ) {
186  bool found = false;
187  for ( auto rhsIt = rhs.begin(); rhsIt != rhs.end(); ++rhsIt ) {
188  //std::cout << "now comparing " << (*(lhsIt))->hash() << " and " << (*(rhsIt))->hash() << std::endl;
189  if ( ( *lhsIt )->hash() == ( *rhsIt )->hash() ) {
190  found = true;
191  break;
192  }
193  }
194  if ( !found ) {
195  return false;
196  }
197  }
198 
199  return true;
200  }

208  friend bool operator==( const HybridAutomaton<Number>& lhs, const HybridAutomaton<Number>& rhs ) {
209  if ( !( lhs.equals( lhs.getLocations(), rhs.getLocations() ) ) ) {
210  TRACE( "hypro.datastructures.hybridAutomaton", "no equality of locations." );
211  return false;
212  }
213  //if(!(lhs.equals(lhs.getTransitions(),rhs.getTransitions()))){
214  // TRACE("hypro.datastructures.hybridAutomaton", "no equality of transitions.");
215  // return false;
216  //}
217  if ( lhs.getInitialStates().size() != rhs.getInitialStates().size() ) {
218  TRACE( "hypro.datastructures.hybridAutomaton", "initial set sizes were different." );
219  return false;
220  }
221  if ( lhs.getInitialStates() != rhs.getInitialStates() ) {
222  TRACE( "hypro.datastructures.hybridAutomaton", "no equality of initials." );
223 
224  TRACE( "hypro.datastructures.hybridAutomaton", "size of lhs initial set: " << lhs.getInitialStates().size() << " size of rhs initial set: " << rhs.getInitialStates().size() );
225  auto rhsIt = rhs.getInitialStates().begin();
226  for ( auto lhsIt = lhs.getInitialStates().begin(); lhsIt != lhs.getInitialStates().end(); ++lhsIt ) {
227  if ( *( ( *lhsIt ).first ) != *( ( *rhsIt ).first ) ) {
228  TRACE( "hypro.datastructures.hybridAutomaton", ( *lhsIt ).first->getName() << " with hash " << ( *lhsIt ).first->hash() << " unequal to " << ( *rhsIt ).first->getName() << " with hash " << ( *rhsIt ).first->hash() );
229  return false;
230  }
231  if ( ( *lhsIt ).second != ( *rhsIt ).second ) {
232  TRACE( "hypro.datastructures.hybridAutomaton", "states were different." );
233  return false;
234  }
235  rhsIt++;
236  }
237  }
238  if ( lhs.getLocalBadStates() != rhs.getLocalBadStates() ) {
239  TRACE( "hypro.datastructures.hybridAutomaton", "no equality of local bads." );
240  return false;
241  }
242  if ( lhs.getGlobalBadStates() != rhs.getGlobalBadStates() ) {
243  TRACE( "hypro.datastructures.hybridAutomaton", "no equality of global bads." );
244  return false;
245  }
246 
247  return true;
248  }
