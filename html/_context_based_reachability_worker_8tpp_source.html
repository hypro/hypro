144 template <typename State>
145 bool ContextBasedReachabilityWorker<State>::isValidTask( const TaskType& t, WorkQueue<TaskType>* localCEXQueue ) {
146  if ( t->treeNode->getRefinements().at( t->btInfo.btLevel ).fullyComputed ) {
147  // This node has already been computed for the required level. This can only happen during concurrent BT-runs.
148  DEBUG( "hydra.worker", std::this_thread::get_id() << ": Got node which is already on the desired level." );
149  assert( t->btInfo.btLevel > 0 );
150  // Follow bt-path by creating new tasks
151  // 1: Find correct transition.
152  // 2: Fine correct potential child. Note: At least the children (their order and links) are updated already, so we can
153  // safely read those. Nonetheless they might be locked for computation.
154  std::pair<hypro::Transition<Number>*, carl::Interval<tNumber>> transitionTimingPair = t->btInfo.btPath.getTransitionToJumpDepth( t->treeNode->getDepth() );
155  // Transition is only nullptr, if the jump depth is zero (which cannot be here) or the queried jump depth is larger than
156  // the path's length (which cannot be either, as there would have been a bt-run, which already computed the whole path).
157  if ( transitionTimingPair.first == nullptr ) {
158  // the whole path has been computed already.
159  TRACE( "hydra.worker", "Do not continue refinement as the path has been computed already to full extend." );
160  t->treeNode->getMutex().unlock();
161  return false;
162  }
163  // iterate over all children to match the timings
164  DEBUG( "hydra.worker", std::this_thread::get_id() << ": Search for next nodes on btPath: " << t->btInfo.btPath );
165  DEBUG( "hydra.worker", std::this_thread::get_id() << ": Consider child nodes for transition " << *transitionTimingPair.first );
166  for ( auto child : t->treeNode->getChildrenForTransition( transitionTimingPair.first ) ) {
167  DEBUG( "hydra.worker", std::this_thread::get_id() << ": Consider child " << child );
168  if ( child->getLatestBTLevel() >= t->btInfo.btLevel && carl::set_have_intersection( child->getTimestamp( t->btInfo.btLevel ), transitionTimingPair.second ) ) {
169  DEBUG( "hydra.worker", std::this_thread::get_id() << ": Timestamp matches, create follow-up task." );
170  TaskType newTask( new Task<State>( child ) );
171  newTask->btInfo = t->btInfo;
172  newTask->btInfo.currentBTPosition += 2;
173  DEBUG( "hydra.worker", "Enqueue follow up child with tree node " << child << " to local CEX queue." );
174  if ( !SettingsProvider<State>::getInstance().useGlobalQueuesOnly() ) {
175  localCEXQueue->nonLockingEnqueue( std::move( newTask ) );
176  } else {
177  localCEXQueue->enqueue( std::move( newTask ) );
178  }
179  }
180  }
181  DEBUG( "hydra.worker", std::this_thread::get_id() << ": Local CEX-Queue size: " << localCEXQueue->nonLockingSize() << "localCEXQueue is now:\n"
182  << localCEXQueue );
183  t->treeNode->getMutex().unlock();
184  return false;
185  }
186 
187  // If a node is set to empty, skip this node. This can happen, if the number of child-nodes has changed upon refinement - in this case
188  // we do not erase subtrees but rather set the particular level to empty.
189  if ( t->treeNode->getRefinements().at( t->btInfo.btLevel ).isEmpty ) {
190  t->treeNode->getMutex().unlock();
191  return false;
192  }
193 
194  return true;
195 }
