1 /**
2  * \example example_bouncingBall.cpp
3  * An example file showing how to construct the automaton for a bouncing ball system and afterwards determine the set of reachable states
4  * for this system using H-polytopes as a state set representation.
5  */

7 #include "config.h"

11 #include "datastructures/Point.h"
12 #include "representations/GeometricObject.h"

16 int main()
17 {
18  using namespace hypro;
19  using namespace carl;
20 
21  // typedefs for simplification.
22  typedef mpq_class Number;
23  typedef hypro::HPolytope<Number> Representation;
24 
25  // LocationManager holds all created locations and allows to create new locations.
26  LocationManager<Number>& lManager = LocationManager<Number>::getInstance();
27 
28  // create the discrete structure of the automaton and the automaton itself.
29  std::unique_ptr<Location<Number>> loc1 = std::make_unique<Location<Number>>();
30  std::unique_ptr<Transition<Number>> trans = std::make_unique<Transition<Number>>();
31  HybridAutomaton<Number> bBallAutomaton = HybridAutomaton<Number>();
32 
33  // matrix defining the flow (note: 3rd dimension for constant parts).
34  matrix_t<Number> flowMatrix = matrix_t<Number>(3,3);
35 
36  // location parameter setting.
37  // creation of the invariant.
38  matrix_t<Number> invariantMat = matrix_t<Number>(1,2);
39  vector_t<Number> invariantVec = vector_t<Number>(1);
40 
41  invariantVec(0) = Number(0);
42 
43  invariantMat(0,0) = Number(-1);
44  invariantMat(0,1) = Number(0);
45 
46  Condition<Number> inv(invariantMat,invariantVec);
47 
48  loc1->setInvariant(inv);
49 
50  // setup flow matrix (3x3, we add an artificial dimension to cope with constants).
51  flowMatrix(0,0) = Number(0);
52  flowMatrix(0,1) = Number(1);
53  flowMatrix(0,2) = Number(0);
54  flowMatrix(1,0) = Number(0);
55  flowMatrix(1,1) = Number(0);
56  flowMatrix(1,2) = Number(carl::rationalize<Number>(-9.81));
57  flowMatrix(2,0) = Number(0);
58  flowMatrix(2,1) = Number(0);
59  flowMatrix(2,2) = Number(0);
60 
61  loc1->setFlow(flowMatrix);
62 
63  // setup of the transition.
64  // guard
65  Condition<Number> guard;
66  matrix_t<Number> guardMat = matrix_t<Number>(3,2);
67  vector_t<Number> guardVec = vector_t<Number>(3);
68 
69  guardVec(0) = Number(0);
70  guardVec(1) = Number(0);
71  guardVec(2) = Number(0);
72 
73  guardMat(0,0) = Number(1);
74  guardMat(0,1) = Number(0);
75  guardMat(1,0) = Number(-1);
76  guardMat(1,1) = Number(0);
77  guardMat(2,0) = Number(0);
78  guardMat(2,1) = Number(1);
79 
80  guard.setMatrix(guardMat);
81  guard.setVector(guardVec);
82 
83  // reset function
84  Reset<Number> reset;
85  vector_t<Number> constantReset = vector_t<Number>(2,1);
86  matrix_t<Number> linearReset = matrix_t<Number>(2,2);
87 
88  constantReset(0) = Number(0);
89  constantReset(1) = Number(0);
90 
91  linearReset(0,0) = Number(1);
92  linearReset(0,1) = Number(0);
93  linearReset(1,0) = Number(0);
94  linearReset(1,1) = Number(carl::rationalize<Number>(-0.9));
95 
96  reset.setVector(constantReset);
97  reset.setMatrix(linearReset);
98 
99  // setup transition
100  trans->setAggregation(Aggregation::parallelotopeAgg);
101  trans->setGuard(guard);
102  trans->setSource(loc1);
103  trans->setTarget(loc1);
104  trans->setReset(reset);
105 
106 
107  // add defined location and transition to the automaton.
108  loc1->addTransition(trans.get());
109 
110  // create Box holding the initial set.
111  matrix_t<Number> boxMat = matrix_t<Number>(4,2);
112  vector_t<Number> boxVec = vector_t<Number>(4);
113 
114  boxVec(0) = Number(carl::rationalize<Number>(10.2));
115  boxVec(1) = Number(-10);
116  boxVec(2) = Number(0);
117  boxVec(3) = Number(0);
118 
119  boxMat(0,0) = Number(1);
120  boxMat(0,1) = Number(0);
121  boxMat(1,0) = Number(-1);
122  boxMat(1,1) = Number(0);
123  boxMat(2,0) = Number(0);
124  boxMat(2,1) = Number(1);
125  boxMat(3,0) = Number(0);
126  boxMat(3,1) = Number(-1);
127 
128  // create initial state.
129  hypro::State_t<Number> initialState;
130  initialState.setLocation(loc1.get());
131  initialState.setSet(ConstraintSet<Number>(boxMat, boxVec));
132  bBallAutomaton.addInitialState(initialState);
133 
134  bBallAutomaton.addLocation(std::move(loc1));
135  bBallAutomaton.addTransition(std::move(trans));
136 
137  // vector of sets to collect flowpipes (which are again a vector of sets).
138  std::vector<std::vector<Representation>> flowpipes;
139 
140  // instanciate reachability analysis algorithm.
141  hypro::reachability::Reach<Number> reacher(bBallAutomaton);
142  hypro::ReachabilitySettings settings = reacher.settings();
143  settings.timeStep = carl::convert<double,Number>(0.01);
144  settings.timeBound = Number(3);
145  settings.jumpDepth = 3;
146  reacher.setSettings(settings);
147  reacher.setRepresentationType(Representation::type());
148 
149  // perform reachability analysis.
150  std::vector<std::pair<unsigned, reachability::flowpipe_t>> flowpipeIndices = reacher.computeForwardReachability();
151 
152  // plot flowpipes.
153  Plotter<Number>& plotter = Plotter<Number>::getInstance();
154  plotter.setFilename("bouncingBall");
155  for(auto& indexPair : flowpipeIndices){
156  std::vector<hypro::State_t<Number>> flowpipe = indexPair.second;
157  // Plot single flowpipe
158  for(auto& set : flowpipe) {
159  std::vector<Point<Number>> points = set.vertices();
160  if(!points.empty() && points.size() > 2) {
161  for(auto& point : points) {
162  point.reduceDimension(2);
163  }
164  plotter.addObject(points);
165  points.clear();
166  }
167  }
168  }
169 
170  // write output.
171  plotter.plot2d();
172 
173  return 0;
174 }
