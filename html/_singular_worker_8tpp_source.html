62  // for each transition intersect each computed time successor set with the guard. If the intersection is non-empty, store for post-processing.
63 
64  JumpPredecessors transitionEnablingStateSets;

67  auto [containment, set] = intersect( stateSet, transitionPtr->getGuard() );
