626  return 0;
627  }
628  long maxDim = vertices.begin()->rawCoordinates().rows();
629  matrix_t<Number> matr = matrix_t<Number>( vertices.size() - 1, maxDim );
630  // use first vertex as origin, start at second vertex
631  long rowIndex = 0;
632  for ( auto vertexIt = ++vertices.begin(); vertexIt != vertices.end(); ++vertexIt, ++rowIndex ) {
633  matr.row( rowIndex ) = ( vertexIt->rawCoordinates() - vertices.begin()->rawCoordinates() ).transpose();
634  }

636  }
