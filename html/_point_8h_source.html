630 int effectiveDimension( const std::vector<Point<Number>>& vertices ) {
631  if ( !vertices.empty() ) {
632  if ( vertices.size() == 1 ) {
633  return 0;
634  }
635  long maxDim = vertices.begin()->rawCoordinates().rows();
636  matrix_t<Number> matr = matrix_t<Number>( vertices.size() - 1, maxDim );
637  // use first vertex as origin, start at second vertex
638  long rowIndex = 0;
639  for ( auto vertexIt = ++vertices.begin(); vertexIt != vertices.end(); ++vertexIt, ++rowIndex ) {
640  matr.row( rowIndex ) = ( vertexIt->rawCoordinates() - vertices.begin()->rawCoordinates() ).transpose();
641  }
642  return int( matr.fullPivLu().rank() );
643  }
644  return -1;
645 }
