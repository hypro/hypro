121  void addConstraint( const vector_t<Number>& normal, Number offset ) {
122  if ( mConstraints.cols() == 0 ) {
123  mConstraints = matrix_t<Number>( normal );
124  mConstants = vector_t<Number>::Zero( 1 );
125  mConstants << offset;
126  } else {
127  assert( mConstraints.cols() == normal.rows() );
128  mConstraints.conservativeResize( mConstraints.rows() + 1, mConstraints.cols() );
129  mConstants.conservativeResize( mConstraints.rows() + 1 );
130 
131  mConstraints.row( mConstraints.rows() - 1 ) = normal;
132  mConstants( mConstants.rows() - 1 ) = offset;
133  }
