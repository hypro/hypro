126  return std::make_pair( CONTAINMENT::FULL, *this );

146

154  break;
155  } else if ( resultPair.first == CONTAINMENT::PARTIAL ) {

159  } else if ( resultPair.first == CONTAINMENT::YES ) {
160  strictestContainment = CONTAINMENT::YES;
161  res.rGetEmptyStates()[i] = TRIBOOL::FALSE;
162  } else {
163  res.rGetEmptyStates()[i] = TRIBOOL::FALSE;
164  }
165  }
166  DEBUG( "hypro.datastructures", "State::satisfies: End of loop" );
167  return std::make_pair( strictestContainment, res );

607 template <typename Number, typename Representation, typename... Rargs>
608 void State<Number, Representation, Rargs...>::decompose( const Decomposition& decomposition ) {
609  if ( decomposition.subspaces.size() == 1 || mSets.size() != 1 ) {
610  // no decomposition/already decomposed
611  }
612  // initial set is a constraint set
613  matrix_t<Number> constraintsOld( std::get<hypro::ConstraintSet<Number>>( mSets.at( 0 ) ).matrix() );
614  vector_t<Number> constantsOld( std::get<hypro::ConstraintSet<Number>>( mSets.at( 0 ) ).vector() );
615  int i = 0;
616  for ( auto decomp : decomposition.subspaces ) {
617 #ifdef HYPRO_LOGGING
618  DEBUG( "hypro.datastructures", "Trying to project set: \n " << mSets.at( 0 ) << "\n to dimensions: " );
619  DEBUG( "hypro.datastructures", "{" );
620  for ( auto entry : decomp ) {
621  DEBUG( "hypro.datastructures", "" << entry << ", " );
622  }
623  DEBUG( "hypro.datastructures", "}" );
624 #endif
625 
626  // for each row of the constraints check if it contains an entry for one of the variables of the set
627  // and add the corresponding rows to a list of indices that are later added to the result matrix
628  std::vector<int> indicesToAdd;
629  for ( int i = 0; i < constraintsOld.rows(); i++ ) {
630  vector_t<Number> row = constraintsOld.row( i );
631  bool containsVar = false;
632  for ( int j = 0; j < row.rows(); j++ ) {
633  if ( row( j, 0 ) != 0 ) {
634  if ( std::find( decomp.begin(), decomp.end(), j ) != decomp.end() ) {
635  //set contains variable j, which is also contained in this constraint
636  containsVar = true;
637  break;
638  }
639  }
640  }
641  if ( containsVar ) {
642  // this row contains information for one of the variables of this decomposition
643  indicesToAdd.push_back( i );
644  }
645  }
646 
647  // we found information for our decomposition
648  if ( indicesToAdd.size() > 0 ) {
649  // create a row matrix with numIndicesToAdd many rows
650  matrix_t<Number> rowMat = matrix_t<Number>::Zero( indicesToAdd.size(), constraintsOld.cols() );
651  for ( Eigen::Index index = 0; index < rowMat.rows(); index++ ) {
652  // copy over preselected rows
653  rowMat.row( index ) = constraintsOld.row( indicesToAdd[index] );
654  }
655  // create final matrix that does not contain columns not in this set
656  matrix_t<Number> finMat = matrix_t<Number>::Zero( rowMat.rows(), decomp.size() );
657  // -1 for constant column
658  for ( Eigen::Index index = 0; index < finMat.cols(); index++ ) {
659  finMat.col( index ) = rowMat.col( decomp[index] );
660  }
661  // create final constant vector
662  vector_t<Number> finVec = vector_t<Number>::Zero( indicesToAdd.size() );
663  for ( Eigen::Index index = 0; index < finVec.rows(); index++ ) {
664  finVec( index ) = constantsOld( indicesToAdd[index] );
665  }
666 
667  ConstraintSet<Number> res( finMat, finVec );
668  DEBUG( "hypro.datastructures", "Final decomposed ConstraintSet: \n"
669  << res );
670  setSetDirect( hypro::Converter<Number>::toConstraintSet( res ), i );
671  setSetType( hypro::representation_name::constraint_set, i );
672  } else {
673  DEBUG( "hypro.datastructures", "No constraints for set found." );
674  ConstraintSet<Number> res = ConstraintSet<Number>();
675  setSetDirect( hypro::Converter<Number>::toConstraintSet( res ), i );
676  setSetType( hypro::representation_name::constraint_set, i );
677  }
678  i++;
679  }
680  mIsEmpty = std::vector<TRIBOOL>{ mSets.size(), TRIBOOL::NSET };
681  DEBUG( "hypro.datastructures", "State after decomposition: " << *this );
682 }
