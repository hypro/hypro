132 template <typename Number>
133 void VertexEnumeration<Number>::enumerateVerticesNaive() {
134  if ( !mHsv.empty() ) {
135  unsigned dim = mHsv.begin()->dimension();
136  // permutate over all combinations of d half-spaces
137  Permutator permutator( mHsv.size(), dim );
138  std::vector<std::size_t> permutation;
139  while ( !permutator.end() ) {
140  permutation = permutator();
141  // create equation system.
142  matrix_t<Number> A( dim, dim );
143  vector_t<Number> b( dim );
144  unsigned pos = 0;
145  // set coordinates.
146  for ( auto planeIt = permutation.begin(); planeIt != permutation.end(); ++planeIt ) {
147  A.row( pos ) = mHsv.at( *planeIt ).normal().transpose();
148  b( pos ) = mHsv.at( *planeIt ).offset();
149  ++pos;
150  }
151  // solve equation system with LU-decomposition.
152  Eigen::FullPivLU<matrix_t<Number>> lu_decomp( A );
153  if ( lu_decomp.rank() < A.rows() ) { // if the problem is not fully determined, continue - the vertex is not described properly.
154  continue;
155  }
156  // get solution
157  vector_t<Number> res = lu_decomp.solve( b );
158 
159  // Check if the computed vertex is a real vertex (lies below all hyperplanes)
160  bool outside = false;
161  for ( unsigned planePos = 0; planePos < mHsv.size(); ++planePos ) {
162  bool skip = false;
163  for ( unsigned permPos = 0; permPos < permutation.size(); ++permPos ) {
164  if ( planePos == permutation.at( permPos ) ) {
165  skip = true;
166  break;
167  }
168  }
169  // account for rounding errors.
170  if ( !skip && !carl::AlmostEqual2sComplement( mHsv.at( planePos ).offset(), mHsv.at( planePos ).normal().dot( res ), default_double_comparison_ulps ) && mHsv.at( planePos ).offset() - mHsv.at( planePos ).normal().dot( res ) < 0 ) {
171  outside = true;
172  break;
173  }
174  }
175  if ( !outside ) {
176  // insert, if no duplicate
177  Point<Number> tmp( res );
178  if ( std::find( mPoints.begin(), mPoints.end(), tmp ) == mPoints.end() ) {
179  mPoints.push_back( tmp );
180  }
181  }
182  }
183  }

321 template <typename Number>
322 Dictionary<Number> VertexEnumeration<Number>::findFirstVertex() {
323  Eigen::Index d = Eigen::Index( mHsv[0].dimension() ); // get dimension
324  Eigen::Index n0 = Eigen::Index( mHsv.size() ); // holds the original size (note: lineality adds half-spaces later).
325  Dictionary<Number> dictionary = Dictionary<Number>( mHsv ); // Calls dictionary constructor.
326  DEBUG( "hypro.vertexEnumeration", "first dictionary" );
327  DEBUG( "hypro.vertexEnumeration", dictionary );
328  DEBUG( "hypro.vertexEnumeration", dictionary.constrainSet() );
329  while ( dictionary.fixOutOfBounds() ) {
330  } // Pivot and update assignments until a point in the satisfyable region is reached. Throws exception if unsat.
331  DEBUG( "hypro.vertexEnumeration", "fixOutOfBounds" );
332  DEBUG( "hypro.vertexEnumeration", dictionary );
333  DEBUG( "hypro.vertexEnumeration", dictionary.constrainSet() );
334  dictionary.nonSlackToBase( mLinealtySpace ); // finds linealty
335  DEBUG( "hypro.vertexEnumeration", "nonSlackToBase" );
336  DEBUG( "hypro.vertexEnumeration", dictionary );
337  DEBUG( "hypro.vertexEnumeration", dictionary.constrainSet() );
338  DEBUG( "hypro.vertexEnumeration", "lineality space size: " << mLinealtySpace.size() );
339  addLinealtyConstrains(); // adds new half-spaces describing the lineality space.
340 
341  // At this point we need to start over again, as we added artificial constraints for the lineality space.
342  matrix_t<Number> dictio = matrix_t<Number>::Zero( mHsv.size() + 1, d + 1 );
343  for ( Eigen::Index colIndex = 0; colIndex <= d; ++colIndex ) { //copy
344  for ( Eigen::Index rowIndex = 0; rowIndex < n0; ++rowIndex ) {
345  dictio( rowIndex, colIndex ) = dictionary.get( rowIndex, colIndex ); // copy old values.
346  }
347  dictio( mHsv.size(), colIndex ) = dictionary.get( n0, colIndex ); // fresh last row contains column indices/var indices.
348  }
349  for ( Eigen::Index rowIndex = 0; rowIndex < n0; ++rowIndex ) { //build the linealty constrains, the part with the var in the basis
350  if ( std::size_t( dictionary.basis()[rowIndex] ) >= dictionary.basis().size() ) {
351  for ( Eigen::Index colIndex = 0; colIndex < d; ++colIndex ) {
352  for ( std::size_t rowIndexLinealty = 0; rowIndexLinealty < mLinealtySpace.size(); ++rowIndexLinealty ) {
353  dictio( n0 + 2 * rowIndexLinealty, colIndex ) -=
354  dictionary.get( rowIndex, colIndex ) * mLinealtySpace[rowIndexLinealty][dictionary.basis()[rowIndex] - dictionary.basis().size()];
355  }
356  }
357  }
358  }
359  for ( Eigen::Index colIndex = 0; colIndex < d; ++colIndex ) { //build the linealty constrains, the part with the var in the cobasis
360  for ( std::size_t rowIndexLinealty = 0; rowIndexLinealty < mLinealtySpace.size(); ++rowIndexLinealty ) {
361  if ( std::size_t( dictionary.cobasis()[colIndex] ) >= dictionary.basis().size() ) {
362  dictio( n0 + 2 * rowIndexLinealty, colIndex ) -= mLinealtySpace[rowIndexLinealty][colIndex];
363  }
364  dictio( n0 + 2 * rowIndexLinealty + 1, colIndex ) = -dictio( n0 + 2 * rowIndexLinealty, colIndex );
365  }
366  }
367 
368  // update variable indices.
369  std::vector<Eigen::Index> basis = dictionary.basis();
370  std::vector<Eigen::Index> cobasis = dictionary.cobasis();
371  for ( auto& index : basis ) {
372  if ( index > n0 ) {
373  index += Eigen::Index( 2 * mLinealtySpace.size() );
374  }
375  }
376  for ( auto& index : cobasis ) {
377  if ( index > n0 ) {
378  index += Eigen::Index( 2 * mLinealtySpace.size() );
379  }
380  }
381  Eigen::Index back = basis.back();
382  basis.pop_back();
383 
384  for ( std::size_t index = std::size_t( n0 ) + 1; index <= mHsv.size(); ++index ) {
385  basis.push_back( Eigen::Index( index ) );
386  }
387  basis.push_back( back );
388  ConstrainSet<Number> constrains;
389  for ( Eigen::Index index = 0; index < n0; ++index ) {
390  constrains.add( dictionary.constrainSet().get( index ) );
391  }
392  for ( std::size_t index = 0; index < 2 * mLinealtySpace.size(); ++index ) {
393  constrains.add( std::tuple<std::pair<bool, Number>, std::pair<bool, Number>, Number>(
394  std::pair<bool, Number>( false, Number( 0 ) ), std::pair<bool, Number>( true, Number( 0 ) ), Number( 0 ) ) ); //fix
395  }
396  for ( Eigen::Index index = n0; index < n0 + d; ++index ) {
397  constrains.add( dictionary.constrainSet().get( index ) );
398  }
399  Dictionary<Number> newDictionary = Dictionary<Number>( dictio, basis, cobasis, constrains ); // creation of a new dictionary with the linealty constrains
400  while ( newDictionary.fixOutOfBounds() ) {
401  }
402  DEBUG( "hypro.vertexEnumeration", "the new dico" );
403  DEBUG( "hypro.vertexEnumeration", dictionary );
404  DEBUG( "hypro.vertexEnumeration", dictionary.constrainSet() );
405  newDictionary.nonSlackToBase();
406  std::set<Eigen::Index> hyperplanes;
407  for ( std::size_t index = 0; index < basis.size(); ++index ) { // search for already saturated constrains
408  if ( newDictionary.constrainSet().isSaturated( index ) ) {
409  hyperplanes.insert( Eigen::Index( index ) );
410  }
411  }
412  std::set<Eigen::Index> frozenCols = newDictionary.toCobase( hyperplanes );
413  DEBUG( "hypro.vertexEnumeration", "frozen cols:" );
414  DEBUG( "hypro.vertexEnumeration", frozenCols );
415  for ( Eigen::Index colIndex = 0; colIndex < Eigen::Index( d ); ++colIndex ) {
416  if ( frozenCols.end() == frozenCols.find( colIndex ) ) {
417  newDictionary.pushToBounds( colIndex );
418  }
419  }
420  return newDictionary;
421 }

424 int VertexEnumeration<Number>::linearIndependance( std::map<int, vector_t<Number>> collection, vector_t<Number>& candidateRef ) const {
425  for ( typename std::map<int, vector_t<Number>>::iterator it = collection.begin(); it != collection.end(); ++it ) {
426  Number ratio = ( candidateRef[it->first] ) / ( it->second )[it->first];
427  for ( int i = 0; i < candidateRef.size(); ++i ) {
428  candidateRef[i] -= ratio * it->second[i];
429  }
430  }
431  int nonNulIndex = -1;
432  int i = 0;
433  while ( nonNulIndex < 0 && i < candidateRef.size() ) {
434  if ( candidateRef[i] != 0 ) {
435  return i;
436  }
437  ++i;
438  }
439  return nonNulIndex;
440 }

477 template <typename Number>
478 bool VertexEnumeration<Number>::findPositiveConstrains() {
479  try {
480  // if there is no separating half-space defined, return whole R^d (-> set lineality space).
481  if ( mHsv.size() == 0 ) {
482  vector_t<Number> one = vector_t<Number>( 1 );
483  one[0] = 1;
484  mLinealtySpace.push_back( one );
485  throw std::string( "\n WARNING: no constrains. \n" );
486  }
487  Dictionary<Number> dictionary( findFirstVertex() );
488  DEBUG( "hypro.vertexEnumeration", "findFirstVertex" );
489  DEBUG( "hypro.vertexEnumeration", dictionary );
490  DEBUG( "hypro.vertexEnumeration", dictionary.constrainSet() );
491  DEBUG( "hypro.vertexEnumeration", "Is optimal: " << dictionary.isOptimal() );
492  std::size_t dimension = dictionary.cobasis().size() - 1;
493  Eigen::Index constraintsCount = dictionary.basis().size() - 1;
494  DEBUG( "hypro.vertexEnumeration", "Dimension: " << dimension << ", constraintsCount: " << constraintsCount );
495  matrix_t<Number> A1 = matrix_t<Number>( dimension, dimension );
496  matrix_t<Number> A2 = matrix_t<Number>( mHsv.size() - dimension, dimension );
497  vector_t<Number> b1 = vector_t<Number>( dimension );
498  std::vector<Eigen::Index> mB;
499  std::vector<Eigen::Index> mN;
500  // write the first d normals to A1 (in the order they appear in the dictionary), set b1 to contain the corresponding constants.
501  for ( std::size_t rowIndex = 0; rowIndex < dimension; ++rowIndex ) {
502  for ( std::size_t colIndex = 0; colIndex < dimension; ++colIndex ) {
503  A1( rowIndex, colIndex ) = ( mHsv.at( dictionary.cobasis()[rowIndex] - 1 ) ).normal()[colIndex];
504  }
505  b1[rowIndex] = ( mHsv.at( dictionary.cobasis()[rowIndex] - 1 ) ).offset();
506  }
507 
508  // A1*y <= b1 describes the first vertex y? -> during find first vertex, non-original variables are guaranteed to be in the co-basis (nonSlackToBase).
509 
510  matrix_t<Number> invA1 = A1.inverse();
511  std::size_t skipped = 0;
512  for ( Eigen::Index rowIndex = 0; rowIndex < constraintsCount; ++rowIndex ) {
513  if ( dictionary.basis()[rowIndex] - 1 < constraintsCount ) { // pick indices, which correspond to an original constraint in mHsv and skip original variables.
514  for ( std::size_t colIndex = 0; colIndex < dimension; ++colIndex ) {
515  A2( rowIndex - skipped, colIndex ) = ( mHsv.at( dictionary.basis()[rowIndex] - 1 ) ).normal()[colIndex];
516  }
517  } else {
518  ++skipped;
519  }
520  }
521 
522  // Original A is now split into A1 and A2, where A1 defines the initial vertex and A2 contains the rest of constraints.
523 
524  DEBUG( "hypro.vertexEnumeration", "A1: " << A1 );
525  DEBUG( "hypro.vertexEnumeration", "b1: " << b1 );
526  DEBUG( "hypro.vertexEnumeration", "A2: " << A2 );
527 
528  matrix_t<Number> newDictionary = matrix_t<Number>::Zero( constraintsCount - dimension + 1, dimension + 1 ); // create new dictionary
529  skipped = 0;
530  for ( Eigen::Index rowIndex = 0; rowIndex < constraintsCount; ++rowIndex ) {
531  if ( dictionary.basis()[rowIndex] - 1 < constraintsCount ) { // again only pick those indices, which correspond to original constraints.
532  for ( std::size_t colIndex = 0; colIndex < dimension; ++colIndex ) {
533  //newDictionary(rowIndex-skipped,colIndex)=0; // Set all zero? -> removed by initiating all to zero in the creation of the matrix.
534  for ( std::size_t index = 0; index < dimension; ++index ) {
535  newDictionary( rowIndex - skipped, colIndex ) += A2( rowIndex - skipped, index ) * invA1( index, colIndex );
536  }
537  }

591 void VertexEnumeration<Number>::findLinealtySpace() {
592  std::size_t dim = mHsv[0].dimension();
593  std::size_t count = 0;
594  std::size_t index = 0;
595  std::vector<Number> norms;
596  std::vector<vector_t<Number>> collection;
597  while ( count < dim && index < mHsv.size() ) {
598  vector_t<Number> candidate = mHsv[index].normal();
599  for ( std::size_t vectorIndex = 0; vectorIndex < count; ++vectorIndex ) {
600  candidate = candidate - collection[vectorIndex] * ( collection[vectorIndex].dot( candidate ) / norms[vectorIndex] );
601  }
602  int i = 0;
603  while ( i < candidate.size() ) {
604  if ( candidate[i] != 0 ) {
605  break;
606  }
607  ++i;
608  }
609  if ( i < candidate.size() ) {
610  collection.push_back( candidate );
611  norms.push_back( candidate.dot( candidate ) );
612  ++count;
613  }
614  ++index;
615  }
616  index = 0;
617  std::size_t linealtySpaceSize = dim - count;
618  while ( count < dim ) {
619  vector_t<Number> baseVector = vector_t<Number>::Zero( dim );
620  baseVector[index] = Number( -1 );
621  for ( std::size_t vectorIndex = 0; vectorIndex < count; ++vectorIndex ) {
622  baseVector = baseVector - collection[vectorIndex] * ( collection[vectorIndex].dot( baseVector ) / norms[vectorIndex] );
623  }
624  int i = 0;
625  while ( i < baseVector.size() ) {
626  if ( baseVector[i] != 0 ) {
627  break;
628  }
629  ++i;
630  }
631  if ( i < baseVector.size() ) {
632  collection.push_back( baseVector );
633  norms.push_back( baseVector.dot( baseVector ) );
634  ++count;
635  }
636  ++index;
637  }
638  for ( index = 0; index < linealtySpaceSize; ++index ) {
639  mLinealtySpace.push_back( collection[dim - index - 1] );
