88  matrix_t<Number> tmpMatrix = matrix_t<Number>::Zero( lhsRows + rhsRows, haVar.size() );
89 
90  size_t col = 0;

92  while ( colLhs < lhsMatrix.cols() ) {
93  if ( haVar[col] == lhsVar[colLhs] ) {
94  tmpMatrix.block( 0, col, lhsRows, 1 ) = lhsMatrix.block( 0, colLhs, lhsRows, 1 );
95  col++;
96  colLhs++;
97  continue;
98  }
99  if ( haVar[col] < lhsVar[colLhs] ) {
100  col++;
101  continue;
102  }
103  }
104 
105  col = 0;

107  while ( colRhs < rhsMatrix.cols() ) {

109  tmpMatrix.block( lhsRows, col, rhsRows, 1 ) = rhsMatrix.block( 0, colRhs, rhsRows, 1 );
110  col++;
111  colRhs++;
112  continue;
113  }

115  col++;
116  continue;
117  }
