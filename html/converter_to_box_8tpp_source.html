293  //first column describes upper bounds, first row describes lower bounds, cast these to point
294  std::vector<Number> upper;

296  for ( int i = 1; i < _source.getDBM().rows(); i++ ) {
297  if ( _source.getDBM()( i, 0 ).second == DifferenceBoundsT<Number, Converter<Number>, inSetting>::BOUND_TYPE::INFTY ) {
298  // no infinite boxes
299  return BoxT<Number, Converter, BoxSetting>();
300  }
301  upper.push_back( _source.getDBM()( i, 0 ).first );
302  }
303 
304  std::vector<Number> lower;
305  // i=1 -- do not consider 0 clock
306  for ( int i = 1; i < _source.getDBM().cols(); i++ ) {
307  if ( _source.getDBM()( 0, i ).second == DifferenceBoundsT<Number, Converter<Number>, inSetting>::BOUND_TYPE::INFTY ) {
308  // no infinite boxes
309  return BoxT<Number, Converter, BoxSetting>();
310  }
311  // note that the entries are always </<= so a lower bound is given by a negative number
312  // eg. (-3,<=) <=> 0-x <= -3 <=> -x <= -3 <=> x >= 3
313  // eg. (3, <=) <=> 0-x <= 3 <=> -x <= 3 <=> x >= -3
314  Number value = _source.getDBM()( 0, i ).first;
315  if ( value != Number( 0 ) ) {
316  value = -1.0 * value;
317  }
318  lower.push_back( value );
319  }
320 
321  Point<Number> minimum( lower );
322  Point<Number> maximum( upper );
323  return BoxT<Number, Converter, BoxSetting>( std::make_pair( minimum, maximum ) );
