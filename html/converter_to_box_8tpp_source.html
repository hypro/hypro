295  if ( _source.getDBM()( i, 0 ).second == DifferenceBoundsT<Number, Converter<Number>, inSetting>::BOUND_TYPE::INFTY ) {
296  // no infinite boxes
297  return BoxT<Number, Converter, BoxSetting>();
298  }
299  upper.push_back( _source.getDBM()( i, 0 ).first );
300  }
301 
302  std::vector<Number> lower;
303  // i=1 -- do not consider 0 clock
304  for ( int i = 1; i < _source.getDBM().cols(); i++ ) {
305  if ( _source.getDBM()( 0, i ).second == DifferenceBoundsT<Number, Converter<Number>, inSetting>::BOUND_TYPE::INFTY ) {
306  // no infinite boxes
307  return BoxT<Number, Converter, BoxSetting>();
308  }
309  // note that the entries are always </<= so a lower bound is given by a negative number
310  // eg. (-3,<=) <=> 0-x <= -3 <=> -x <= -3 <=> x >= 3
311  // eg. (3, <=) <=> 0-x <= 3 <=> -x <= 3 <=> x >= -3
312  Number value = _source.getDBM()( 0, i ).first;
313  if ( value != Number( 0 ) ) {
314  value = -1.0 * value;
315  }
316  lower.push_back( value );
317  }
318 
319  Point<Number> minimum( lower );
320  Point<Number> maximum( upper );
321  return BoxT<Number, Converter, BoxSetting>( std::make_pair( minimum, maximum ) );
