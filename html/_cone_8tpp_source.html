32 vectorSet<Number> Cone<Number>::vectors() const {
33  // create all possible vectors
34  vectorSet<Number> intersectionVectors;
35  for ( const auto &lhsPlane : mPlanes ) {
36  for ( const auto &rhsPlane : mPlanes ) {
37  if ( lhsPlane != rhsPlane ) {
38  intersectionVectors.insert( lhsPlane.get()->intersectionVector( *rhsPlane.get() ) );
39  }
40  }
41  }
42  // adjust vectors to the right direction
43 
44  // std::cout<<__func__ << " : " <<__LINE__ << " : "<<mPlanes.size() <<
45  // std::endl;
46  vectorSet<Number> correctIntersectionVectors;
47  while ( !intersectionVectors.empty() ) {
48  bool changed = false;
49  // std::cout<<__func__ << " : " <<__LINE__ << *intersectionVectors.begin()
50  // << std::endl;
51  for ( const auto &plane : mPlanes ) {
52  // std::cout<<__func__ << " : " <<__LINE__ << " plane : " << plane <<
53  // std::endl;
54  if ( !plane->contains( *intersectionVectors.begin() ) ) {
55  // std::cout<<__func__ << " : " <<__LINE__ <<std::endl;
56  vector_t<Number> tmp = ( *intersectionVectors.begin() ) * -1;
57  intersectionVectors.erase( intersectionVectors.begin() );
58  correctIntersectionVectors.insert( tmp );
59  changed = true;
60  // std::cout << tmp << " is contained in " << *plane << " : " <<
61  // plane->contains(tmp) << std::endl;
62  assert( plane->contains( tmp ) );
63  break;
64  }
65  }
66  if ( !changed ) {
67  vector_t<Number> tmp = ( *intersectionVectors.begin() );
68  intersectionVectors.erase( intersectionVectors.begin() );
69  correctIntersectionVectors.insert( tmp );
70  }

73  intersectionVectors = correctIntersectionVectors;

77  // std::cout << vec << ", " << std::endl;
78  //}

80  // for each vector, determine, if it is inside the original cone

82  if ( !this->contains( vector ) ) intersectionVectors.erase( vector );

105 template <typename Number>
106 void Cone<Number>::add( vector_t<Number> _vector ) {
107  // collect all vectors
108  vectorSet<Number> lhsVectors = this->vectors();
109 
110  // check if vector is already inside cone_l -> drop
111  planeVector insidePlanes;
112  std::set<std::shared_ptr<Halfspace<Number>>> outsidePlanes;
113  if ( !this->contains( _vector, insidePlanes, outsidePlanes ) ) { // not inside
114  // std::cout << "vector is not inside." << std::endl;
115  assert( !insidePlanes.empty() );
116 
117  // get horizon vectors, create new Halfspaces, delete inside planes.
118  std::vector<vector_t<Number>> horizonVectors;
119  for ( const auto &horizonCandidate : lhsVectors ) {
120  bool found = false;
121  for ( const auto &insidePlane : insidePlanes ) {
122  // std::cout << "Inside plane: " << *insidePlane.get() << std::endl;
123  if ( insidePlane->contains( horizonCandidate ) ) {
124  // std::cout << "is contained (normal*horizonCandidate=" <<
125  // insidePlane.get()->normal().dot(horizonCandidate) << ")" <<
126  // std::endl;
127  // check all other planes, which are not inside, if the candidate is
128  // part of one -> horizon vector
129  for ( const auto &outsidePlane : outsidePlanes ) {
130  // std::cout << "Consider vector " << horizonCandidate << " to be
131  // inside " << *insidePlane.get()
132  // << " and " << *outsidePlane.get() << std::endl;
133  if ( outsidePlane->contains( horizonCandidate ) ) {
134  horizonVectors.push_back( horizonCandidate );
135  // std::cout << "horizon vector: " << horizonCandidate <<
136  // std::endl;
137  found = true;
138  break;
139  }
140  }
141  }
142  if ( found ) break;
143  }
144  found = false;
145  }
146 
147  // std::cout << "Create new Halfspaces." << std::endl;
148  // create new Halfspaces
149  std::vector<vector_t<Number>> HalfspaceBasis;
150  HalfspaceBasis.push_back( _vector );
151  for ( const auto &horizonVector : horizonVectors ) {
152  // Todo: check if origin zero is okay.
153  vector_t<Number> origin = vector_t<Number>::Zero( _vector.rows() );
154 
155  // std::cout << "Create Halfspace from " << _vector << " and " <<
156  // horizonVector << std::endl;
157 
158  HalfspaceBasis.push_back( horizonVector );
159  outsidePlanes.insert(
160  std::shared_ptr<Halfspace<Number>>( new Halfspace<Number>( origin, HalfspaceBasis ) ) );
161  // std::cout << "new plane: " << *outsidePlanes.back().get() << std::endl;
162  HalfspaceBasis.pop_back();
163  }
164 
165  // assign new planes
166  mPlanes.clear();
167  for ( const auto &plane : outsidePlanes ) mPlanes.push_back( plane );
168  }
169 }
