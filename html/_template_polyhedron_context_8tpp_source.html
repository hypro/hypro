63  assert( initialOffsets.rows() == invTPoly.vector().rows() );
64  for(int i = 0; i < initialOffsets.rows(); ++i){
65  assert(initialOffsets(i) <= invTPoly.vector()(i));
66  }
67  #endif

69  unsigned invRows = invTPoly.matrix().rows();
70  unsigned invCols = invTPoly.matrix().cols();
71  assert( invCols == this->mComputationState.getDimension() );
72 
73  //Used as storage for solution of last iteration, called a(j) in paper
74  vector_t<Number> lastStrengthenedInv = vector_t<Number>::Zero( invTPoly.vector().rows() );
75  //Contains the newest relaxed invariant, called a(j+1) in paper
76  vector_t<Number> nextStrengthenedInv = invTPoly.vector();
77  //Contains the relaxed invariant with the smallest sum of coefficients
78  vector_t<Number> smallestInv = invTPoly.vector();
79  //Contains all strengthened invariant computed, needed for cycle detection
80  std::vector<vector_t<Number>> allInvs;
81  //Counts the number of iterations to stop when we reached 10 iterations
82  int counter = 0;
83 
85 
86  //1. Compute pi(j) (certificate of relaxed invariance for a(j+1)) by solving Dj : max evalDir*x subject to Ax <= b && lambda >= 0
87  //Construct A as: first transpose inv matrix, then insert lambda >= 0 constraints
88  //NOTE: Since A is the same in every iteration, we construct A once the loop
89  //NOTE: We add one row and col for the lie derivate offset later, where we force lambda_{m+1} = 1
90  //NOTE: Thus A is of size d+m+1 x m+1
91  matrix_t<Number> A = matrix_t<Number>::Zero( invCols + invRows + 1, invRows + 1 );
92  A.block( 0, 0, invCols, invRows ) = invTPoly.matrix().transpose();
93  A.block( invCols, 0, invRows, invRows ) = -1 * matrix_t<Number>::Identity( invRows, invRows );
94  A( invCols + invRows, invRows ) = Number( 1 );
95 
96  //std::cout << "TemplatePolyhedronContext::LIS, A is: \n" << A << std::endl;
97 
98  //Set equality relations for the first d equations
99  std::vector<carl::Relation> equalRelations( invCols + invRows + 1, carl::Relation::LEQ );
100  for ( unsigned i = 0; i < invCols; ++i ) {
101  equalRelations.at( i ) = carl::Relation::EQ;
102  }
103  //The last for lamdba_{m+1} = 1
104  equalRelations[invCols + invRows] = carl::Relation::EQ;
105 
106  //while a(j) != a(j+1)
107  while ( lastStrengthenedInv != nextStrengthenedInv ) {
108 
109  counter++;
110 
111  //Certificate pi(j): Saves the solutions for each Dj
112  matrix_t<Number> certificate = matrix_t<Number>::Zero( invRows, invRows );
113 
114  //Save lie derivate offsets
115  vector_t<Number> derivativeOffsets = vector_t<Number>::Zero( invRows );
116 
117  //For each (important) row solve Dj
118  for ( unsigned rowI = 0; rowI < invRows; ++rowI ) {
119 
120  //2.Construct b = (mScaling * current template row + lie derivative of current row)^T = (mü(H_j) + H_j')^T,
121  //extended with zeros to make sure that the optimumValue only contains non negative entries.
122  //NOTE: b is of dimension d+m+1, since there are m lambdas and one extra row for being able to calculate with the lie derivate offset
123  vector_t<Number> b = vector_t<Number>::Zero( invRows + invCols + 1 );
124  assert( b.rows() == A.rows() );
125  auto deriv = lieDerivative( invTPoly.matrix().row( rowI ) );
126  b.block( 0, 0, invCols, 1 ) = mScaling * vector_t<Number>( invTPoly.matrix().row( rowI ) ) + deriv.first;
127  b( invRows + invCols ) = Number( 1 );
128  derivativeOffsets( rowI ) = deriv.second;
129 
130  //std::cout << "TemplatePolyhedronContext::LIS, for rowI: " << rowI << " b is: \n" << b << std::endl;
131 
132  //3.Set A and b as matrix and vector for mOptimizer
133  mOptimizer.setMatrix( A );
134  mOptimizer.setVector( b );
135  mOptimizer.setMaximize( false );
136  mOptimizer.setRelations( equalRelations );
137 
138  //4.Minimize into direction nextStrengthenedInv - Solution is the optimumValue of minimizeA, but only the first invRows coefficients from that.
139  //The rest was only needed to make sure that the optimumValue only contains positive entries.
140  vector_t<Number> evalDir = vector_t<Number>::Zero( invRows + 1 );
141  evalDir.block( 0, 0, invRows, 1 ) = nextStrengthenedInv;
142  evalDir( invRows ) = -mScaling * evalDir( rowI ) + deriv.second;
143  auto minimizeA = mOptimizer.evaluate( evalDir, std::is_same_v<Number, mpq_class> );
144 
145 #ifndef NDEBUG
146  assert(minimizeA.errorCode == SOLUTION::FEAS);
147  assert( minimizeA.optimumValue.rows() == invRows + 1 );
148  for ( int i = 0; i < minimizeA.optimumValue.rows(); ++i ) {
149  //make sure all values are greater equal than 0
150  assert( minimizeA.optimumValue( i ) >= 0 );
151  }
152 #endif
153 
154  //std::cout << "TemplatePolyhedronContext::LIS, evaldir: " << evalDir << std::endl;
155  //std::cout << "TemplatePolyhedronContext::LIS, minimizeA: " << minimizeA << std::endl;
156 
157  //6.Save into certificate, if support value <= 0, since then row is non-frozen and invariant therefore changeable
158  //NOTE: -mScaling * evalDir(rowI) + deriv.second is already a part of evalDir
159  if ( minimizeA.supportValue <= 0 ) {
160  vector_t<Number> res = minimizeA.optimumValue.block( 0, 0, invRows, 1 );
161  certificate.row( rowI ) = res.transpose();
162  }
163 
164  //Numerical corrections
165  if( TemplatePolyhedron<Number>::Settings::USE_NUMERICAL_CORRECTIONS ){
166  if(certificate.row(rowI) == vector_t<Number>::Zero(certificate.cols()).transpose()){
167  //Usually when the condition above holds, it also holds nextStrengthenedInv(rowI) == invTPoly.vector()(rowI),
168  //but because of numerical issues it can be unequal. In that case correct to the upper case.
169  if(nextStrengthenedInv(rowI) < invTPoly.vector()(rowI)){
170  //numeric error correction
171  COUNT("Correction Insert Certificate Row");
172  vector_t<Number> correction = minimizeA.optimumValue.block( 0, 0, invRows, 1 );
173  certificate.row( rowI ) = correction.transpose();
174  }
175  } else {
176  if(nextStrengthenedInv.dot(certificate.row(rowI)) - mScaling*nextStrengthenedInv(rowI) + deriv.second > 0){
177  //numeric error correction - as the value is actually zero or less, we need to set back to row to a zero row.
178  COUNT("Correction Remove Certificate Row");
179  vector_t<Number> correction = vector_t<Number>::Zero(invRows);
180  certificate.row(rowI) = correction.transpose();
181  }
182  }
183  }
184 
185  //NOTE: this would usually hold, but because of numerical issues this is not always the case
186  //#ifndef NDEBUG
187  //if(certificate.row(rowI) == vector_t<Number>::Zero(certificate.cols()).transpose()){
188  // assert(nextStrengthenedInv(rowI) == invTPoly.vector()(rowI));
189  //} else {
190  // assert(nextStrengthenedInv.dot(certificate.row(rowI)) - mScaling*nextStrengthenedInv(rowI) + deriv.second <= 0);
191  //}
192  //#endif
193  }
194 
195  //std::cout << "TemplatePolyhedronContext::LIS, certificate is now: \n" << certificate << std::endl;
196 
198 
199  //Build the system L*y = c with which we get better invariant bounds
200  assert( initialOffsets.rows() == invTPoly.vector().rows() );
201  matrix_t<Number> L = matrix_t<Number>::Zero( 3 * invRows, invRows );
202  vector_t<Number> c = vector_t<Number>::Zero( 3 * invRows );
203 
204  //6.Build L_pi(j): Add constraints: init <= y
205  L.block( 0, 0, invRows, invRows ) = std::move( -1 * matrix_t<Number>::Identity( invRows, invRows ) );
206  c.block( 0, 0, invRows, 1 ) = -1 * initialOffsets;
207  //std::cout << "TemplatePolyhedronContext::LIS, added init <= y to L and c: L: \n" << L << "c: \n" << c << std::endl;
208 
209  //7.Build L_pi(j): Add constraints: y <= inv_j
210  L.block( invRows, 0, invRows, invRows ) = std::move( matrix_t<Number>::Identity( invRows, invRows ) );
211  c.block( invRows, 0, invRows, 1 ) = invTPoly.vector();
212  //std::cout << "TemplatePolyhedronContext::LIS, added y <= inv_j to L and c: L: \n" << L << "c: \n" << c << std::endl;
213 
214  //8.Build L_pi(j): Add constraints. Row frozen if value >= inv
215  //If row j frozen, then y = inv_j
216  //If row j not frozen, then Delta_j^T*y - mü*y_j + h_j <= 0
217  assert( nextStrengthenedInv.rows() == invRows );
218  for ( unsigned i = 0; i < nextStrengthenedInv.rows(); ++i ) {
219  if ( certificate.row( i ) == vector_t<Number>::Zero( invRows ).transpose() ) {
220  //frozen row - normally, insert y_j = inv_j, but since y_j <= inv_j is already in L, insert y_j >= inv_j aka -y_j <= -inv_j
221  vector_t<Number> rowToInsert = vector_t<Number>::Zero( invRows );
222  rowToInsert( i ) = -1;
223  L.row( 2 * invRows + i ) = std::move( rowToInsert.transpose() );
224  c( 2 * invRows + i ) = -invTPoly.vector()( i );
225  } else {
226  //non frozen row
227  vector_t<Number> rowToInsert = certificate.row( i );
228  rowToInsert( i ) -= mScaling;
229  L.row( 2 * invRows + i ) = std::move( rowToInsert.transpose() );
230  c( 2 * invRows + i ) = - derivativeOffsets( i );
231  }
232  }
233  //std::cout << "TemplatePolyhedronContext::LIS, added (non-)frozen rows to L and c: L: \n" << L << "c: \n" << c << std::endl;
234 
235  //9.Minimize L to get nextStrengthenedInv
236  mOptimizer.setMatrix( L );
237  mOptimizer.setVector( c );
238  mOptimizer.setMaximize( false );
239  mOptimizer.setRelations( std::vector<carl::Relation>( 3 * invRows, carl::Relation::LEQ ) );
240  auto minimizeL = mOptimizer.evaluate( vector_t<Number>::Ones( invRows ), std::is_same_v<Number, mpq_class> );
241  if(minimizeL.errorCode == SOLUTION::INFEAS){
242  //If next solution infeasible, return currently saved solution
243  //TODO: find out why infeas return even occurs
244  COUNT("Infeas return");
245  return nextStrengthenedInv;
246  }
247  assert( minimizeL.errorCode == SOLUTION::FEAS );
248 
249  //NOTE: normally, this would hold, too, but sadly there are numerical effects again
250  //#ifndef NDEBUG
251  //for ( unsigned i = 0; i < invRows; ++i ) {
252  // std::cout << i << std::endl;
253  // assert( initialOffsets( i ) <= minimizeL.optimumValue( i ) );
254  // assert( minimizeL.optimumValue( i ) <= invTPoly.vector()( i ) );
255  //}
256  //#endif
257 
258  //std::cout << "TemplatePolyhedronContext::LIS, minimizeL optimumValue is: \n" << minimizeL.optimumValue << std::endl;
259 
260  //10.Set as next strenghtened inv
261  lastStrengthenedInv = nextStrengthenedInv;
262  nextStrengthenedInv = minimizeL.optimumValue;
263 
264  //Wrong fixed point escape: save inv with smallest coefficient
265  if(nextStrengthenedInv.sum() < smallestInv.sum()){
266  smallestInv = nextStrengthenedInv;
267  }
268 
269  //std::cout << "TemplatePolyhedronContext::LIS, lastStrengthenedInv is now: \n" << lastStrengthenedInv << "nextStrengthenedInv is now: \n" << nextStrengthenedInv << std::endl;
270  //std::cout << "TemplatePolyhedronContext::LIS, nextStrengthenedInv: \n" << nextStrengthenedInv << std::endl;
271  //std::cout << "TemplatePolyhedronContext::LIS, counter: " << counter << std::endl;
272 
273  //Cycle Detection
274  bool cycleDetected = false;
275  for(const auto& inv : allInvs){
276  if(nextStrengthenedInv.isApprox(inv)){
277  COUNT("Cycle Detection Return");
278  cycleDetected = true;
279  break;
280  }
281  }
282  if(cycleDetected){
283  break;
284  } else {
285  allInvs.push_back(nextStrengthenedInv);
286  }
287 
288  //Stop after 10 iterations to stop cycles of size greater 10
289  if(counter >= 10){
290  COUNT("Counter max return");
291  break;
292  }
293  }
294 
295  //Return strengthened invariant with smallest sum
296  if(nextStrengthenedInv.sum() < smallestInv.sum()){
297  return nextStrengthenedInv;
298  } else {
299  return smallestInv;
300  }
301 }

369 bool TemplatePolyhedronContext<State>::isPositiveInvariant( const TemplatePolyhedron<Number>& tpoly, const vector_t<Number>& invVector ) {
370 #ifndef NDEBUG
371  //All offset values must be below or equal to invariant offset values
372  assert( tpoly.matrix().rows() == invVector.rows() );
373  assert( tpoly.vector().rows() == invVector.rows() );
374  for ( int i = 0; i < invVector.rows(); ++i ) {
375  assert( tpoly.vector()( i ) <= invVector( i ) );
376  }
377 #endif
378 
379  //Template matrix and vector extended by two constraints to test if current constraint is a positive invariant.
380  matrix_t<Number> extendedMatrix = matrix_t<Number>::Zero( tpoly.matrix().rows() + 2, tpoly.matrix().cols() );
381  extendedMatrix.block( 0, 0, tpoly.matrix().rows(), tpoly.matrix().cols() ) = tpoly.matrix();
382  vector_t<Number> extendedVector = vector_t<Number>::Zero( tpoly.vector().rows() + 2 );
383  extendedVector.block( 0, 0, tpoly.vector().rows(), 1 ) = tpoly.vector();
384 
385  for ( int i = 0; i < tpoly.matrix().rows(); ++i ) {
386  //std::cout << "TemplatePolyhedronContext::isPositiveInvariant, tpoly.vector(i) = " << tpoly.vector()( i ) << " != " << invVector( i ) << std::endl;
387 
388  //Avoid test if value is equal to invariant value
389  if ( tpoly.vector()( i ) != invVector( i ) ) {
390  //Add constraint "row i >= vector_i" to get equality "row_i == vector_i" since "row_i <= vector_i" is already in template
391  extendedMatrix.row( tpoly.matrix().rows() ) = -tpoly.matrix().row( i );
392  extendedVector( tpoly.vector().rows() ) = -tpoly.vector()( i );
393 
394  //Add constraint "lie derivative of row_i > 0"
395  //NOTE: this is equivalent to "-(lie derivative) < 0 " <=> "-(lie derivative only terms with variables) < lie derivate offset"
396  auto deriv = lieDerivative( tpoly.matrix().row( i ) );
397  extendedMatrix.row( tpoly.matrix().rows() + 1 ) = -deriv.first.transpose();
398  extendedVector( tpoly.vector().rows() + 1 ) = deriv.second;
399  //extendedMatrix.row( tpoly.matrix().rows() + 1 ) = -lieDerivative( tpoly.matrix().row( i ) ).first.transpose();
400 
401  //If unfeasible, then not a positive invariant
402  mOptimizer.setMatrix( extendedMatrix );
403  mOptimizer.setVector( extendedVector );
404  mOptimizer.setMaximize( true );
405  //std::cout << "TemplatePolyhedronContext::isPositiveInvariant, extendedMatrix: \n" << extendedMatrix << " extendedVector: \n" << extendedVector << std::endl;
406  if ( !mOptimizer.checkConsistency() ) {
407  return false;
408  }
409  }
410  }
411 
412  return true;
413 }

427  matrix_t<Number> extendedMatrix = matrix_t<Number>::Zero( tpoly.matrix().rows() + 1, tpoly.matrix().cols() );
428  extendedMatrix.block( 0, 0, tpoly.matrix().rows(), tpoly.matrix().cols() ) = tpoly.matrix();
429  vector_t<Number> extendedVector = vector_t<Number>::Zero( tpoly.vector().rows() + 1 );
430  extendedVector.block( 0, 0, tpoly.vector().rows(), 1 ) = tpoly.vector();
431 
432  for ( int i = 0; i < tpoly.matrix().rows(); ++i ) {
433  //std::cout << "TemplatePolyhedronContext::isRelaxedInvariant, tpoly.vector(i) = " << tpoly.vector()( i ) << " != " << invVector( i ) << std::endl;
434 
435  //Avoid test if value is equal to invariant value
436  if ( tpoly.vector()( i ) < invVector( i ) ) {
437 
438  //Compute Lie(H_i*x - d_i)
439  vector_t<Number> tmp = vector_t<Number>::Zero(tpoly.matrix().cols() + 1);
440  tmp.block( 0, 0, tpoly.matrix().cols(), 1) = tpoly.matrix().row(i).transpose();
441  tmp(tpoly.matrix().cols()) = -tpoly.vector()(i);
442  auto deriv = lieDerivativeExtended(tmp);
443 
444  extendedMatrix.row( tpoly.matrix().rows() ) = deriv.first + mScaling * tpoly.matrix().row(i).transpose();
445  extendedVector( tpoly.vector().rows() ) = -deriv.second + mScaling * tpoly.vector()(i);
446 
447  //If unfeasible, then not a positive invariant
448  mOptimizer.setMatrix( extendedMatrix );
449  mOptimizer.setVector( extendedVector );
450  mOptimizer.setMaximize( true );
451  //std::cout << "TemplatePolyhedronContext::isRelaxedInvariant, extendedMatrix: \n" << extendedMatrix << " extendedVector: \n" << extendedVector << std::endl;
452  if ( !mOptimizer.checkConsistency() ) {
453  return false;
454  }
455  }
456  }
457 
458  return true;
