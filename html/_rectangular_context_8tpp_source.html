548 void RectangularContext<State>::checkTransition() {
549  DEBUG( "hypro.worker", "State before guard intersection: " << mCurrentSegment );
550  // Collect potential new initial states from discrete behaviour.
551  if ( !( mSettings.jumpDepth < 0 || int( mTask->treeNode->getDepth() ) <= mSettings.jumpDepth ) ) {
552  return;
553  }
554 
555  std::vector<typename std::map<Transition<Number>*, std::vector<IGuardHandler<State>*>>::iterator> alwaysDisabledTransitions;
556  std::vector<typename std::map<Transition<Number>*, std::vector<IGuardHandler<State>*>>::iterator> sortIndexChanged;
557  bool urgentTransitionEnabled = false;
558 
559  for ( auto it = mTransitionHandlerMap.begin(); it != mTransitionHandlerMap.end(); ++it ) {
560 #ifdef SINGLE_THREAD_FIXED_POINT_TEST
561  // if the considered transition is disabled (for one iteration), skip the test by starting the next loop iteration.
562  auto disabledTransitionIt = std::find( mDisabledTransitions.begin(), mDisabledTransitions.end(), it->first );
563  if ( disabledTransitionIt != mDisabledTransitions.end() ) {
564  mDisabledTransitions.erase( disabledTransitionIt );

566  }
567 #endif
568 
569  if ( !it->first->isUrgent() && omitTransition( it->first ) ) {
570  // store that transition was not enabled for this time interval
571  //mTask->treeNode->rGetRefinements().at(mTask->btInfo.btLevel).mTimings.insertTransition(it->first, mCurrentSegment.getTimestamp(), CONTAINMENT::NO);
572  if ( mSettings.useGuardTimingInformation ) {
573  mLocalTimings.insertTransition( it->first, mCurrentSegment.getTimestamp(), CONTAINMENT::NO );
574  }
575  continue;
576  }
577 
578  TRACE( "hypro.worker", "Initializing " << it->second.size() << " guard handlers for transition " << it->first->getSource()->hash() << " -> " << it->first->getTarget()->hash() );

580  State* guardStatePtr = new State( mCurrentSegment );
581  std::shared_ptr<State> shGuardPtr = std::shared_ptr<State>( guardStatePtr );
582 
583  for ( std::size_t i = 0; i < it->second.size(); i++ ) {
584  it->second.at( i )->setState( shGuardPtr );
585  }
586 
587  // apply handlers
588  bool guardSatisfied = true;
589 
590  // if there are handlers, apply them
591  if ( it->second.size() > 0 ) {
592  bool deleteRequested = false;
593  //START_BENCHMARK_OPERATION(FULL_CUT);
594  int i = 0;
595  // iterate over handlers
596  COUNT( "GuardCheck" );
597  for ( auto guardIt = it->second.begin(); guardIt != it->second.end(); ++guardIt ) {
598  //START_BENCHMARK_OPERATION(SUBSPACE_CUT);
599  ( *guardIt )->reinitialize();
600  ( *guardIt )->handle();
601  guardSatisfied &= ( *guardIt )->satisfiesGuard();
602  //EVALUATE_BENCHMARK_RESULT(SUBSPACE_CUT);
603 
604  // TODO: strcmp is quite expensive, get rid of.
605  if ( !guardSatisfied && ( strcmp( ( *guardIt )->handlerName(), "noFlowGuardHandler" ) == 0 ) ) {
606  //this transition will never be enabled, remove it from the map
607  TRACE( "hypro.worker.discrete", "noFlowGuardHandler unsatisfied for transition " << it->first->getSource()->hash() << " -> " << it->first->getTarget()->hash() << ". Transition will be removed from further checks." );
608  //std::cout << "Guard not satisfied and no Flow handler. Removing Transition." << std::endl;
609  alwaysDisabledTransitions.push_back( it );
610  deleteRequested = false;
611  break;
612  }
613  // if this guardHandler is not satisfied, it is likely not satisfied next time, so adjust its sortindex to move it to the front
614  else if ( !guardSatisfied ) {
615  if ( i != 0 ) { //lassen wir einen handler an position 0 mal nicht in den sonnenuntergang reiten
616  ( *guardIt )->setSortIndex( it->second.at( 0 )->getSortIndex() - 1 );
617  sortIndexChanged.push_back( it );
618  }
619  break;
620  } else if ( guardSatisfied && ( strcmp( ( *guardIt )->handlerName(), "noFlowGuardHandler" ) == 0 ) ) {
621  // do not recheck again
622  ( *guardIt )->setMarkedForDelete( true );
623  deleteRequested = true;
624  }
625  i++;
626  }
627  //EVALUATE_BENCHMARK_RESULT(FULL_CUT);
628  if ( deleteRequested ) {
629  for ( auto handler = it->second.begin(); handler != it->second.end(); ) {
630  if ( ( *handler )->getMarkedForDelete() ) {
631  delete *handler;
632  handler = it->second.erase( handler );
633  } else {
634  ++handler;
635  }
636  }
637  }
638  }
639  // handle time-triggered transitions
640  if ( it->first->isUrgent() ) {
641  TRACE( "hypro.worker.discrete", "Checking urgent transition " << it->first->getSource()->hash() << " -> " << it->first->getTarget()->hash() );
642  if ( guardSatisfied ) {
643  mDiscreteSuccessorBuffer.push_back( std::tuple<Transition<Number>*, State>( it->first, *guardStatePtr ) );
644  urgentTransitionEnabled = true;
645  }
646  } // handle normal transitions, but only if they cannot be omitted (they can if the timing intervals do not intersect)
647  else if ( guardSatisfied ) {
648  TRACE( "hypro.worker.discrete", "hybrid transition enabled with timestamp " << guardStatePtr->getTimestamp() );
649 
650  DEBUG( "hypro.worker", "Guard satisfying state: " << *guardStatePtr );
651  COUNT( "GuardEnabled" );
652  mDiscreteSuccessorBuffer.push_back( std::tuple<Transition<Number>*, State>( it->first, *( guardStatePtr ) ) );
653  //mTask->treeNode->rGetRefinements().at(mTask->btInfo.btLevel).mTimings.insertTransition(it->first, mCurrentSegment.getTimestamp(), CONTAINMENT::YES);
654  if ( mSettings.useGuardTimingInformation ) {
655  mLocalTimings.insertTransition( it->first, mCurrentSegment.getTimestamp(), CONTAINMENT::YES );
656  }
657  } else {
658  // store that transition was not enabled for this time interval
659  //mTask->treeNode->rGetRefinements().at(mTask->btInfo.btLevel).mTimings.insertTransition(it->first, mCurrentSegment.getTimestamp(), CONTAINMENT::NO);
660  if ( mSettings.useGuardTimingInformation ) {
661  mLocalTimings.insertTransition( it->first, mCurrentSegment.getTimestamp(), CONTAINMENT::NO );
662  }
663  }
664  }
665  for ( auto disabledTransition : alwaysDisabledTransitions ) {
666  mTransitionHandlerMap.erase( disabledTransition );
667  }
668  for ( auto sortIterator : sortIndexChanged ) {
669  //order the handlers according to their sort index
670  std::sort( sortIterator->second.begin(), sortIterator->second.end(), guardHandlerSorter() );

672  if ( urgentTransitionEnabled ) {

718 template <typename State>
719 bool RectangularContext<State>::omitTransition( Transition<Number>* transition ) {
720  TRACE( "hypro.worker.discrete", "Check if transition " << transition << " can be omitted." );
721  // do not use timing information -> return false
722  if ( !mSettings.useGuardTimingInformation ) {
723  COUNT( "CannotOmitTransition" );
724  return false;
725  }
726  // the timings for a transition are either empty because this node is not fully computed or because this transition cannot
727  // be enabled although the node has been visited before - the first case requires to check the transition while the later
728  // allows to omit it.
729 
730  // temporary for dbg-output
731 #ifdef HYPRO_LOGGING
732  auto tmp = convert<tNumber, double>( mCurrentSegment.getTimestamp() );
733 #endif
734 
735  // if the transition is irrelevant for the backtracking, we still want to find potential successors to avoid re-computation.
736  if ( mTask->btInfo.currentBTPosition < mTask->btInfo.btPath.size() && mTask->btInfo.btPath.at( mTask->btInfo.currentBTPosition + 1 ).transition != transition ) {
737  DEBUG( "hypro.worker.discrete", "Cannot Omit transition " << transition->getSource()->getName() << " -> " << transition->getTarget()->getName() << " for time interval " << tmp );
738  COUNT( "CannotOmitTransition" );
739  return false;
740  }
741 
742  // If there is no recorded event and the node has never been fully computed, we cannot omit the transition. However,
743  // if the node has been fully computed but there is no event there never will be a non-spurious event ever and we can savely
744  // omit the transition.
745  TRACE( "hypro.worker.discrete", "Has Transition: " << mLocalTimings.hasTransitionEvent( transition ) );
746  TRACE( "hypro.worker.discrete", "Is fully computed: " << mTask->treeNode->isFullyComputedOnSomeLevel() );
747  TRACE( "hypro.worker.discrete", "Have transition information for the time interval: " << mLocalTimings.hasTransitionInformation( mCurrentSegment.getTimestamp(), transition ) );
748  if ( !mLocalTimings.hasTransitionInformation( mCurrentSegment.getTimestamp(), transition ) ) {
749  DEBUG( "hypro.worker.discrete", "Cannot Omit transition " << transition->getSource()->getName() << " -> " << transition->getTarget()->getName() << " for time interval " << tmp );
750  COUNT( "CannotOmitTransition" );
751  return false;
752  } else {
753  // has transition information for that specific event, now check whether the transition was enabled
754  if ( !mLocalTimings.hasTransitionEvent( mCurrentSegment.getTimestamp(), transition ) ) {
755  DEBUG( "hypro.worker.discrete", "Omit transition " << transition->getSource()->getName() << " -> " << transition->getTarget()->getName() << " for time interval " << tmp << " as we know it was not enabled." );
756  COUNT( "OmitTransition" );
757  return true;
758  } else {
759  DEBUG( "hypro.worker.discrete", "Cannot Omit transition " << transition->getSource()->getName() << " -> " << transition->getTarget()->getName() << " for time interval " << tmp << " as there is no information available." );
760  COUNT( "CannotOmitTransition" );
761  return false;
762  }
763  }
764 
765  // this should not be reachable.
766  assert( false );
767  std::cout << __func__ << ": This should not be reachable." << std::endl;
768  COUNT( "CannotOmitTransition" );
769  return false;
770 }
