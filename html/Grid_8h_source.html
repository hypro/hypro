26  pointIt( const std::vector<std::vector<Number>>& _v ) : v( _v ) {
27  for ( unsigned i = 0; i < v.size(); ++i ) it.push_back( v.at( i ).begin() );
28 
29  end = false;
30  }
31 
32  // prefix
33  Point<Number> operator++() {
34  if ( end ) return std::move( current() );
35  ++it.back();
36  unsigned i = it.size() - 1;
37  while ( it.at( i ) == v.at( i ).end() ) {
38  it[i] = v.at( i ).begin();
39  if ( i == 0 ) {
40  end = true;
41  return std::move( current() );
42  }
43  --i;
44  it[i]++;
45  }
46  return std::move( current() );
47  }
48 
49  // postfix
50  Point<Number> operator++( int ) { return ++( *this ); }
51 
52  Point<Number> current() {
53  assert( v.size() == it.size() );
54  Point<Number> res;
55  for ( unsigned i = 0; i < v.size(); ++i ) res[i] = *it.at( i );
56 
57  return std::move( res );
58  }
