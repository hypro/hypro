19  * @brief Struct holding a point iterator allowing to iterate over a vector of points represented by their coordinates.
20  *
21  * @tparam Number The used number type.
22  */
23 template <typename Number>
24 struct pointIt {
25  typename std::vector<std::vector<Number>> v;
26  std::vector<typename std::vector<Number>::const_iterator> it;
27  bool end = false;
28 
29  pointIt( const std::vector<std::vector<Number>>& _v ) : v( _v ) {
30  for ( std::size_t i = 0; i < v.size(); ++i ) it.push_back( v.at( i ).begin() );
31 
32  end = false;
33  }
34 
35  // prefix
36  Point<Number> operator++() {
37  if ( end ) return std::move( current() );
38  ++it.back();
39  std::size_t i = it.size() - 1;
40  while ( it.at( i ) == v.at( i ).end() ) {
41  it[i] = v.at( i ).begin();
42  if ( i == 0 ) {
43  end = true;
44  return std::move( current() );
45  }
46  --i;
47  it[i]++;
48  }
49  return current();
50  }
51 
52  // postfix
53  Point<Number> operator++( int ) { return ++( *this ); }
54 
55  Point<Number> current() {
56  assert( v.size() == it.size() );
57  Point<Number> res(vector_t<Number>::Zero(v.size()));
58  for ( std::size_t i = 0; i < v.size(); ++i ) res[i] = *it.at( i );
59 
60  return res;
61  }
62 };
