58 template <typename Number>
59 EvaluationResult<Number> BallSupportFunction<Number>::evaluate( const vector_t<Number> &l ) const {
60  EvaluationResult<Number> result;
61  // the ball is empty.
62  if ( mRadius < 0 ) {
63  return result;
64  }
65  // there is no cost function but the ball is not empty.
66  if ( l.rows() == 0 ) {
67  return EvaluationResult<Number>( SOLUTION::FEAS );
68  }
69  // there is a cost function, but it is zero (we know its dimension nonetheless) and the ball is not empty.
70  if ( l.nonZeros() == 0 ) {
71  return EvaluationResult<Number>( l, SOLUTION::FEAS );
72  }
73 
74  // there is a non-zero cost function and the ball is not empty.
75  switch ( mType ) {
76  case SF_TYPE::INFTY_BALL: {
77  unsigned max = 0;
78  for ( unsigned i = 1; i < l.rows(); ++i ) {
79  max = abs( l( i ) ) > abs( l( max ) ) ? i : max;
80  }
81  result.supportValue = ( mRadius / abs( l( max ) ) );
82  result.optimumValue = result.supportValue * normalize( l );
83  result.errorCode = SOLUTION::FEAS;
84  break;
85  }
86  case SF_TYPE::TWO_BALL: {
87  Number length = norm( l, true );
88  if ( length == 0 ) {
89  return EvaluationResult<Number>( l, SOLUTION::FEAS );
90  }
91  DEBUG( "hypro.representations.BallSupportFunction", "l: " << l << ", length: " << length );
92  result.supportValue = ( mRadius / length );
93  result.optimumValue = result.supportValue * normalize( l );
94  result.errorCode = SOLUTION::FEAS;
95  break;
96  }
97  default:
98  assert( false );
99  }
100 
101  return result;
102 }
