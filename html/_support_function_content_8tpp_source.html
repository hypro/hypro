469 template <typename Number, typename Setting>
470 std::vector<EvaluationResult<Number>> SupportFunctionContent<Number, Setting>::multiEvaluate( const matrix_t<Number>& _directions, bool useExact ) const {
471  assert( checkTreeValidity() );
472 
473  using Node = std::shared_ptr<SupportFunctionContent<Number, Setting>>;
474  using Param = matrix_t<Number>;
475  using Res = std::vector<EvaluationResult<Number>>;
476 
477  std::vector<Node> callStack;
478  std::vector<Param> paramStack;
479  std::vector<std::pair<int, std::vector<Res>>> resultStack; // The first value is an iterator to the calling frame
480 
481  callStack.push_back( getThis() );
482  paramStack.push_back( _directions );
483  resultStack.push_back( std::make_pair( -1, std::vector<Res>() ) );
484 
485  while ( !callStack.empty() ) {
486  Node cur = callStack.back();
487  Param currentParam = paramStack.back();
488 
489  if ( cur->originCount() == 0 ) {
490  // Do computation and write results in case recursion ends.
491 
492  std::pair<int, std::vector<Res>> currentResult = resultStack.back();
493 
494  // update result
495  // special case: When the node is a leaf, we directly return the result.
496  if ( currentResult.first == -1 ) {
497  // we reached the top, exit (or return result or whatever)
498  //TRACE("hypro.representations.supportFunction","Is leaf. Evaluate and return.");
499  switch ( cur->type() ) {
500  case SF_TYPE::ELLIPSOID:
501  return ellipsoid()->multiEvaluate( currentParam );
502  case SF_TYPE::INFTY_BALL:
503  [[fallthrough]];
504  case SF_TYPE::TWO_BALL:
505  return ball()->multiEvaluate( currentParam );
506  case SF_TYPE::POLY:
507  return polytope()->multiEvaluate( currentParam, useExact );
508  case SF_TYPE::BOX:
509  return box()->multiEvaluate( currentParam, useExact );
510  default:
511  assert( false );
512  FATAL( "hypro.representations.supportFunction", "Wrong type." );
513  }
514  } else {
515  //TRACE("hypro.representations.supportFunction","Evaluate and add to results stack.");
516  switch ( cur->type() ) {
517  case SF_TYPE::ELLIPSOID:
518  resultStack.at( currentResult.first ).second.emplace_back( cur->ellipsoid()->multiEvaluate( currentParam ) );
519  break;
520  case SF_TYPE::INFTY_BALL:
521  [[fallthrough]];
522  case SF_TYPE::TWO_BALL:
523  resultStack.at( currentResult.first ).second.emplace_back( cur->ball()->multiEvaluate( currentParam ) );
524  break;
525  case SF_TYPE::POLY:
526  resultStack.at( currentResult.first ).second.emplace_back( cur->polytope()->multiEvaluate( currentParam, useExact ) );
527  break;
528  case SF_TYPE::BOX:
529  resultStack.at( currentResult.first ).second.emplace_back( cur->box()->multiEvaluate( currentParam, useExact ) );
530  break;
531  default:
532  assert( false );
533  FATAL( "hypro.representations.supportFunction", "Wrong type." );
534  }
535  }
536  //TRACE("hypro.representations.supportFunction","Leave recursive call.");
537 
538  // leave recursive call.
539  callStack.pop_back();
540  paramStack.pop_back();
541  resultStack.pop_back();
542 
543  } else {
544  // Detect, if this call is finished or new.
545  if ( resultStack.back().second.size() == cur->originCount() ) {
546  // the call is finished, perform accumulating operations and forward result.
547  //TRACE("hypro.representations.supportFunction", ": Accumulate results.");
548  // accumulate results - in this case sum.
549  Res accumulatedResult;
550  switch ( cur->type() ) {
551  case SF_TYPE::LINTRAFO: {
552  //TRACE("hypro.representations.supportFunction", ": LINTRAFO, accumulate results.")
553  assert( resultStack.back().second.size() == 1 );
554  const std::pair<matrix_t<Number>, vector_t<Number>>& parameterPair = cur->linearTrafoParameters()->parameters->getParameterSet( cur->linearTrafoParameters()->currentExponent );
555  //TRACE("hypro.representations.supportFunction", "Matrix: " << parameterPair.first);
556  //TRACE("hypro.representations.supportFunction", "Vector: " << parameterPair.second);
557  if ( resultStack.back().second.front().begin()->errorCode != SOLUTION::INFEAS ) {
558  unsigned directionCnt = 0;
559  for ( auto& entry : resultStack.back().second.front() ) {
560  vector_t<Number> currentDir( currentParam.row( directionCnt ) );
561  if ( entry.errorCode == SOLUTION::INFTY ) {
562  entry.supportValue = 1;
563  } else {
564  //TRACE("hypro.representations.supportFunction", ": Entry val before trafo: " << entry.optimumValue);
565  entry.optimumValue = parameterPair.first * entry.optimumValue + parameterPair.second;
566  // As we know, that the optimal vertex lies on the supporting Halfspace, we can obtain the distance by dot product.
567  entry.supportValue = entry.optimumValue.dot( currentDir );
568  }
569  //auto t = convert<Number,double>(currentParam.row(directionCnt));
571  ++directionCnt;
572  }
573  }
574  //TRACE("hypro.representations.supportFunction",": LINTRAFO, accumulate results done.");
575  accumulatedResult = resultStack.back().second.front();
576  assert( !accumulatedResult.empty() );
577  break;
578  }
579  case SF_TYPE::PROJECTION: {
580  //TRACE("hypro.representations.supportFunction", ": PROJECTION, accumulate results.")
581  assert( resultStack.back().second.size() == 1 );
582  // simply forward the results
583  accumulatedResult = resultStack.back().second.front();
584  assert( !accumulatedResult.empty() );
585  break;
586  }
587  case SF_TYPE::SCALE: {
588  //TRACE("hypro.representations.supportFunction", ": SCALE, accumulate results.")
589  assert( resultStack.back().second.size() == 1 );
590  // if one result is infeasible, the others will be too -> do not process.
591  if ( resultStack.back().second.front().begin()->errorCode != SOLUTION::INFEAS ) {
592  for ( auto& singleRes : resultStack.back().second.front() ) {
593  assert( singleRes.errorCode != SOLUTION::INFEAS );
594  if ( singleRes.errorCode == SOLUTION::FEAS ) {
595  singleRes.supportValue *= cur->scaleParameters()->factor;
596  singleRes.optimumValue *= cur->scaleParameters()->factor;
597  }
598  }
599  }
600  accumulatedResult = resultStack.back().second.front();
601  assert( !accumulatedResult.empty() );
602  break;
603  }
604  case SF_TYPE::SUM: {
605  //TRACE("hypro.representations.supportFunction", ": SUM, accumulate results.")
606  assert( resultStack.back().second.size() == 2 );
607  assert( resultStack.back().second.at( 0 ).size() == std::size_t( currentParam.rows() ) );
608  assert( resultStack.back().second.at( 0 ).size() == resultStack.back().second.at( 1 ).size() );
609  Res& resA = resultStack.back().second.at( 0 );
610  Res& resB = resultStack.back().second.at( 1 );
611 
612  // only process if both are feasible. If one result is infeasible, the others will be too, so stop processing.
613  if ( resA.begin()->errorCode != SOLUTION::INFEAS && resB.begin()->errorCode != SOLUTION::INFEAS ) {
614  for ( unsigned index = 0; index < resA.size(); ++index ) {
615  assert( resA.at( index ).errorCode != SOLUTION::INFEAS && resB.at( index ).errorCode != SOLUTION::INFEAS );
616  EvaluationResult<Number> r;
617  if ( resA[index].errorCode == SOLUTION::INFTY || resB[index].errorCode == SOLUTION::INFTY ) {
618  r.errorCode = SOLUTION::INFTY;
619  r.supportValue = 1;
620  } else {
621  r.errorCode = SOLUTION::FEAS;
622  r.supportValue = resA[index].supportValue + resB[index].supportValue;
623  r.optimumValue = resA[index].optimumValue + resB[index].optimumValue;
624  }
625  accumulatedResult.emplace_back( r );
626  }
627  assert( accumulatedResult.size() == std::size_t( currentParam.rows() ) );
628  assert( !accumulatedResult.empty() );
629  } else {
630  accumulatedResult.emplace_back( EvaluationResult<Number>{} );
631  }
632  assert( !accumulatedResult.empty() );
633  break;
634  }
635  case SF_TYPE::UNITE: {
636  //TRACE("hypro.representations.supportFunction", ": UNITE, accumulate results.")
637  assert( resultStack.back().second.size() > 0 );
638  assert( resultStack.back().second.size() == cur->unionParameters()->items.size() );
639  accumulatedResult = resultStack.back().second.front();
640  for ( auto resIt = resultStack.back().second.begin(); resIt != resultStack.back().second.end(); ++resIt ) {
641  for ( unsigned resultId = 0; resultId < accumulatedResult.size(); ++resultId ) {
642  if ( ( *resIt )[resultId].errorCode == SOLUTION::INFEAS ) {
643  accumulatedResult[resultId] = ( *resIt )[resultId];
644  break;
645  } else if ( ( *resIt )[resultId].errorCode == SOLUTION::INFTY ) {
646  ( *resIt )[resultId].supportValue = 1;
647  accumulatedResult[resultId] = ( *resIt )[resultId];
648  } else {
649  assert( ( *resIt )[resultId].errorCode == SOLUTION::FEAS );
650  accumulatedResult[resultId] = ( *resIt )[resultId] > accumulatedResult[resultId] ? ( *resIt )[resultId] : accumulatedResult[resultId];
651  //assert(accumulatedResult[resultId].errorCode == SOLUTION::FEAS);
652  }
653  }
654  }
655  assert( !accumulatedResult.empty() );
656  break;
657  }
658  case SF_TYPE::INTERSECT: {
659  //TRACE("hypro.representations.supportFunction", ": INTERSECT, accumulate results.")
660  assert( resultStack.back().second.size() == 2 );
661  Res& resA = resultStack.back().second[0];
662  Res& resB = resultStack.back().second[1];
663  assert( resA.size() == resB.size() );
664  // in case one of the results is infeasible (the set is empty), return this result.
665  if ( resA.begin()->errorCode == SOLUTION::INFEAS ) {
666  accumulatedResult = std::move( resA );
667  assert( !accumulatedResult.empty() );
668  break;
669  }
670  if ( resB.begin()->errorCode == SOLUTION::INFEAS ) {
671  accumulatedResult = std::move( resB );
672  assert( !accumulatedResult.empty() );
673  break;
674  }
675  for ( unsigned i = 0; i < resA.size(); ++i ) {
676  EvaluationResult<Number> res;
677  // intersection with the empty set is the empty set -> skip further computation
678  if ( resA[i].errorCode != SOLUTION::INFEAS && resB[i].errorCode != SOLUTION::INFEAS ) {
679  if ( resA[i].errorCode == SOLUTION::INFTY ) {
680  //TRACE("hypro.representations.supportFunction","resA infinite");
681  res.errorCode = resB[i].errorCode;
682  res.supportValue = resB[i].supportValue;
683  res.optimumValue = resB[i].optimumValue;
684  } else if ( resB[i].errorCode == SOLUTION::INFTY ) {
685  //TRACE("hypro.representations.supportFunction","resB infinite");
686  assert( resA[i].errorCode == SOLUTION::FEAS );
687  res.errorCode = resA[i].errorCode;
688  res.supportValue = resA[i].supportValue;
689  res.optimumValue = resA[i].optimumValue;
690  } else {
691  assert( resA[i].errorCode == SOLUTION::FEAS && resB[i].errorCode == SOLUTION::FEAS );
692  //TRACE("hypro.representations.supportFunction","Both finite: A " << resA[i].supportValue << " vs B " << resB[i].supportValue);
693  res.errorCode = SOLUTION::FEAS;
694  if ( resA[i].supportValue < resB[i].supportValue ) {
695  res.supportValue = resA[i].supportValue;
696  res.optimumValue = resA[i].optimumValue;
697  } else {
698  res.supportValue = resB[i].supportValue;
699  res.optimumValue = resB[i].optimumValue;
700  }
701  }
702  }
703 
704  //auto t = convert<Number,double>(res.optimumValue);
706  accumulatedResult.emplace_back( std::move( res ) );
707  }
708  assert( accumulatedResult.size() == std::size_t( currentParam.rows() ) );
709  assert( !accumulatedResult.empty() );
710  break;
711  }
712  case SF_TYPE::NONE: {
713  std::cout << __func__ << ": SF Type not properly initialized!" << std::endl;
714  assert( false );
715  break;
716  }
717  default: {
718  assert( false );
719  return std::vector<EvaluationResult<Number>>();
720  }
721  }
722 
723  if ( resultStack.back().first == -1 ) {
724  // we reached the top, exit
725  //TRACE("hypro.representations.supportFunction","Return accumulated result.");
726  assert( !accumulatedResult.empty() );
727  return accumulatedResult;
728  }
729 
730  // forward result.
732  //resultStack.at(resultStack.back().first).second.emplace_back(accumulatedResult);
733  assert( !accumulatedResult.empty() );
734  resultStack.at( resultStack.back().first ).second.emplace_back( std::move( accumulatedResult ) );
735 
736  // delete result frame and close recursive call
737  callStack.pop_back();
738  paramStack.pop_back();
739  resultStack.pop_back();
740 
741  } else {
742  // this is the branch for calling recursively
743  //TRACE("hypro.representations.supportFunction","Invoke recursive call.");
744 
745  // here we create the new stack levels.
746  std::size_t callingFrame = callStack.size() - 1;
747 
748  // Do some parameter transformation, i.e. create passed parameters
749  switch ( cur->type() ) {
750  case SF_TYPE::ELLIPSOID:
751  [[fallthrough]];
752  case SF_TYPE::INFTY_BALL:
753  [[fallthrough]];
754  case SF_TYPE::POLY:
755  [[fallthrough]];
756  case SF_TYPE::BOX:
757  [[fallthrough]];
758  case SF_TYPE::TWO_BALL:
759  assert( false );
760  FATAL( "hypro.representations.supportFunction", "Leaf node cannot be an intermediate case." );
761  break;
762  case SF_TYPE::LINTRAFO: {
763  callStack.push_back( cur->linearTrafoParameters()->origin );
764 #ifndef HYPRO_USE_VECTOR_CACHING
765  paramStack.push_back( currentParam * cur->linearTrafoParameters()->parameters->getParameterSet( cur->linearTrafoParameters()->currentExponent ).first );
766 #else
767  paramStack.push_back( cur->linearTrafoParameters()->parameters->getTransformedDirections( currentParam, cur->linearTrafoParameters()->currentExponent ) );
768 #endif
769  resultStack.emplace_back( std::make_pair( callingFrame, std::vector<Res>() ) );
770  break;
771  }
772  case SF_TYPE::PROJECTION: {
773  matrix_t<Number> projectedParameters = matrix_t<Number>::Zero( _directions.rows(), _directions.cols() );
774  Eigen::Index entryIndex = 0;
775  for ( const auto& entry : cur->projectionParameters()->dimensions ) {
776  //TRACE("hypro.representations.supportFunction","Entry: " << entry)
777  if ( entry < mDimension ) {
778  projectedParameters.col( entry ) = currentParam.col( entry );
779  ++entryIndex;
780  }
781  }
782  assert( entryIndex == Eigen::Index( cur->projectionParameters()->dimensions.size() ) );
783  callStack.push_back( cur->projectionParameters()->origin );
784  paramStack.emplace_back( projectedParameters );
785  resultStack.emplace_back( std::make_pair( callingFrame, std::vector<Res>() ) );
786  break;
787  }
788  case SF_TYPE::SCALE: {
789  // Do nothing for scaling -> processing is done afterwards.
790  callStack.push_back( cur->scaleParameters()->origin );
791  paramStack.emplace_back( currentParam );
792  resultStack.emplace_back( std::make_pair( callingFrame, std::vector<Res>() ) );
793  break;
794  }
795  case SF_TYPE::SUM: {
796  // Do nothing for sum -> processing is done afterwards.
797  callStack.push_back( cur->summands()->rhs );
798  callStack.push_back( cur->summands()->lhs );
799  paramStack.emplace_back( currentParam );
800  paramStack.emplace_back( currentParam );
801  resultStack.emplace_back( std::make_pair( callingFrame, std::vector<Res>() ) );
802  resultStack.emplace_back( std::make_pair( callingFrame, std::vector<Res>() ) );
803  break;
804  }
805  case SF_TYPE::UNITE: {
806  // Do nothing for union -> processing is done afterwards.
807  for ( unsigned i = 0; i < cur->unionParameters()->items.size(); ++i ) {
808  callStack.push_back( cur->unionParameters()->items.at( i ) );
809  paramStack.emplace_back( currentParam );
810  resultStack.emplace_back( std::make_pair( callingFrame, std::vector<Res>() ) );
811  }
812  break;
813  }
814  case SF_TYPE::INTERSECT: {
815  // Do nothing for intersection -> processing is done afterwards.
816  callStack.push_back( cur->intersectionParameters()->rhs );
817  callStack.push_back( cur->intersectionParameters()->lhs );
818  paramStack.emplace_back( currentParam );
819  paramStack.emplace_back( currentParam );
820  resultStack.emplace_back( std::make_pair( callingFrame, std::vector<Res>() ) );
821  resultStack.emplace_back( std::make_pair( callingFrame, std::vector<Res>() ) );
822  break;
823  }
824  case SF_TYPE::NONE: {
825  std::cout << __func__ << ": SF Type not properly initialized!" << std::endl;
826  assert( false );
827  break;
828  }
829  default: {
830  assert( false );
831  return std::vector<EvaluationResult<Number>>();
832  }
833  }
834  }
835  }
836  }
837  assert( false );
838  std::cout << "THIS SHOULD NOT HAPPEN." << std::endl;
839  return std::vector<EvaluationResult<Number>>();
840 }

867  using Node = std::shared_ptr<SupportFunctionContent<Number, Setting>>;
868  using Res = long int;
869  std::vector<Node> callStack;
870  std::vector<std::pair<std::size_t, std::vector<Res>>> resultStack; // The first value is an iterator to the calling frame
871 
872  callStack.push_back( getThis() );
873  resultStack.push_back( std::make_pair( -1, std::vector<Res>() ) );
874 
875  while ( !callStack.empty() ) {
876  Node cur = callStack.back();
877 
878  if ( cur->originCount() == 0 ) {
879  // Do computation and write results in case recursion ends.
880 
881  std::pair<std::size_t, std::vector<Res>> currentResult = resultStack.back();
882 
883  // update result
884  // special case: When the node is a leaf, we directly return the result.
885  if ( currentResult.first == 1 ) {
886  // we reached the top
887  assert( currentResult.second.size() == 1 );
888  return currentResult.second.front();
889  }
890 
891  switch ( cur->type() ) {
892  case SF_TYPE::INFTY_BALL:
893  [[fallthrough]];
894  case SF_TYPE::TWO_BALL:
895  [[fallthrough]];
896  case SF_TYPE::BOX:
897  [[fallthrough]];
898  case SF_TYPE::ELLIPSOID:
899  resultStack.at( currentResult.first ).second.push_back( 1 );
900  break;
901  case SF_TYPE::POLY: {
902  unsigned maxValue = std::max( polytope()->constraints().rows(), polytope()->constraints().cols() );
903  resultStack.at( currentResult.first ).second.push_back( carl::pow( maxValue, 2 ) * polytope()->constraints().cols() );
904  break;
905  }
906  default:
907  // TODO: Extend for other bottom cases.
908  assert( false );
909  }
910 
911  // leave recursive call.
912  callStack.pop_back();
913  resultStack.pop_back();
914 
915  } else {
916  // Detect, if this call is finished or new.
917  if ( resultStack.back().second.size() == cur->originCount() ) {
918  // the call is finished, perform accumulating operations and forward result.
919 
920  // accumulate results
921  Res accumulatedResult = 0;
922  switch ( cur->type() ) {
923  case SF_TYPE::INTERSECT:
924  [[fallthrough]];
925  case SF_TYPE::PROJECTION:
926  [[fallthrough]];
927  case SF_TYPE::LINTRAFO:
928  accumulatedResult += 1;
929  break;
930  case SF_TYPE::UNITE:
931  for ( auto val : resultStack.back().second ) {
932  accumulatedResult += val;
933  }
934  break;
935  case SF_TYPE::NONE: {
936  std::cout << __func__ << ": SF Type not properly initialized!" << std::endl;
937  assert( false );
938  break;
939  }
940  default:
941  std::cout << __func__ << ": SF Type unexpected!" << std::endl;
942  assert( false );
943  }
944 
945  if ( resultStack.back().first == -1 ) {
946  // we reached the top
947  return accumulatedResult;
948  }
949 
950  resultStack.at( resultStack.back().first ).second.push_back( accumulatedResult );
951 
952  // delete result frame and close recursive call
953  callStack.pop_back();
954  resultStack.pop_back();
955 
956  } else {
957  // this is the branch for calling recursively
958 
959  // here we create the new stack levels. As the parameters are all the same, we do not care for their order (could be extended).
960  std::size_t callingFrame = callStack.size() - 1;
961 
962  switch ( cur->type() ) {
963  case SF_TYPE::SUM:
964  callStack.push_back( cur->summands()->rhs );
965  callStack.push_back( cur->summands()->lhs );
966  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
967  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
968  break;
969  case SF_TYPE::INTERSECT:
970  callStack.push_back( cur->intersectionParameters()->rhs );
971  callStack.push_back( cur->intersectionParameters()->lhs );
972  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
973  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
974  break;
975  case SF_TYPE::LINTRAFO:
976  callStack.push_back( cur->linearTrafoParameters()->origin );
977  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
978  break;
979  case SF_TYPE::SCALE:
980  callStack.push_back( cur->scaleParameters()->origin );
981  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
982  break;
983  case SF_TYPE::UNITE:
984  for ( unsigned i = 0; i < cur->unionParameters()->items.size(); ++i ) {
985  callStack.push_back( cur->unionParameters()->items.at( i ) );
986  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
987  }
988  break;
989  case SF_TYPE::POLY:
990  [[fallthrough]];
991  case SF_TYPE::INFTY_BALL:
992  [[fallthrough]];
993  case SF_TYPE::TWO_BALL:
994  [[fallthrough]];
995  case SF_TYPE::ELLIPSOID:
996  [[fallthrough]];

998  [[fallthrough]];
999  case SF_TYPE::ZONOTOPE:
1000  assert( false );
1001  break;
1002  case SF_TYPE::PROJECTION:
1003  callStack.push_back( cur->projectionParameters()->origin );
1004  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
1005  break;
1006  case SF_TYPE::NONE:
1007  default:
1008  assert( false );
1009  }
1010  }
1011  }
1012  }
1013  std::cout << __func__ << ": This should not be reachable." << std::endl;
1014  return 0;

1086  assert( checkTreeValidity() );
1087  switch ( mType ) {
1088  case SF_TYPE::INFTY_BALL:
1089  [[fallthrough]];
1090  case SF_TYPE::TWO_BALL:
1091  return ball()->supremumPoint();
1092  case SF_TYPE::LINTRAFO: {
1093  Point<Number> supPoint = linearTrafoParameters()->origin->supremumPoint();
1094  if ( supPoint.dimension() == 0 ) {
1095  return supPoint;
1096  }

1098  return supPoint.affineTransformation( parameterPair.first, parameterPair.second );
1099  }
1100  case SF_TYPE::POLY:
1101  return polytope()->supremumPoint();
1102  case SF_TYPE::BOX:
1103  return box()->supremumPoint();
1104  case SF_TYPE::PROJECTION: {
1105  Point<Number> tmpRes = projectionParameters()->origin->supremumPoint();
1106  vector_t<Number> tmp = vector_t<Number>::Zero( mDimension );
1107  for ( const auto& entry : projectionParameters()->dimensions ) {
1108  if ( entry < mDimension ) {
1109  tmp( entry ) = tmpRes.at( entry );
1110  }
1111  }
1112  return Point<Number>( tmp );
1113  }
1114  case SF_TYPE::SCALE: {
1115  if ( scaleParameters()->factor == 0 ) {
1116  return Point<Number>::Zero( mDimension );
1117  } else {
1118  Point<Number> supPoint = scaleParameters()->origin->supremumPoint();
1119  if ( supPoint.dimension() == 0 ) {
1120  return supPoint;
1121  }
1122  return supPoint * scaleParameters()->factor;
1123  }
1124  }
1125  case SF_TYPE::SUM: {
1126  Point<Number> lhsPoint = summands()->lhs->supremumPoint();
1127  Point<Number> rhsPoint = summands()->rhs->supremumPoint();
1128  if ( lhsPoint.dimension() == 0 ) {
1129  return rhsPoint;
1130  }
1131  if ( rhsPoint.dimension() == 0 ) {
1132  return lhsPoint;
1133  }
1134  return lhsPoint + rhsPoint;
1135  }
1136  case SF_TYPE::UNITE: {

1138  if ( resPoint.dimension() == 0 ) {
1139  return resPoint;
1140  }
1141  auto itemIt = unionParameters()->items.begin();
1142  ++itemIt;
1143  for ( ; itemIt != unionParameters()->items.end(); ++itemIt ) {
1144  Point<Number> tmpPoint = ( *itemIt )->supremumPoint();
1145  resPoint = Point<Number>::inftyNorm( resPoint ) > Point<Number>::inftyNorm( tmpPoint ) ? resPoint : tmpPoint;
1146  }
1147  return resPoint;
1148  }
1149  case SF_TYPE::INTERSECT: {
1150  Point<Number> lhsPoint = intersectionParameters()->lhs->supremumPoint();

1159  return lhsPoint;
1160  }
1161  return rhsPoint;
1162  }
1163  case SF_TYPE::NONE: {

1165  assert( false );
1166  return Point<Number>();

1168  default:
1169  assert( false );
