119 template <typename Number>
120 void Condition<Number>::decompose( const Decomposition& decomposition ) {
121  if ( mConstraints.size() > 1 ) {
122  //already decomposed/empty constraints
123  return;
124  } else if ( mConstraints.size() == 0 && decomposition.subspaces.size() > 0 ) {
125  //fill mConstaints with empty constraint sets
126  std::vector<ConstraintSetT<Number>> newCset;
127  for ( std::size_t i = 0; i < decomposition.subspaces.size(); i++ ) {
128  ConstraintSetT<Number> res = ConstraintSetT<Number>();
129  newCset.push_back( res );
130  }
131  mConstraints = newCset;
132  mConditionIsBox = std::vector<TRIBOOL>{ mConstraints.size(), TRIBOOL::NSET };
133  mHash = 0;
134  return;
135  }
136 
137  ConstraintSetT<Number> cset = mConstraints.at( 0 );
138  DEBUG( "hypro.datastructures", "Constraint Set before: \n " << cset );
139 
140  matrix_t<Number> constraintsOld( cset.matrix() );
141  vector_t<Number> constantsOld( cset.vector() );
142 
143  std::vector<ConstraintSetT<Number>> newCset;
144  // for each set {i,j,..., k} select each constraint that defines over {i,j,k etc.}
145  for ( auto set : decomposition.subspaces ) {
146 #ifdef HYPRO_LOGGING
147  DEBUG( "hypro.datastructures", "decompose constraint for set: {" );
148  for ( auto entry : set ) {
149  DEBUG( "hypro.datastructures", "" << entry << ", " );
150  }
151  DEBUG( "hypro.datastructures", "}" );
152 #endif
153 
154  // for each row of the constraints check if it contains an entry for one of the variables of the set
155  // and add the corresponding rows to a list of indices that are later added to a matrix
156  std::vector<Eigen::Index> indicesToAdd;
157  for ( Eigen::Index i = 0; i < constraintsOld.rows(); i++ ) {
158  bool containsVar = false;
159  for ( Eigen::Index j = 0; j < constraintsOld.cols(); j++ ) {
160  if ( constraintsOld( i, j ) != 0 && std::find( set.begin(), set.end(), j ) != set.end() ) {
161  //set contains variable j, which is also contained in this constraint
162  containsVar = true;
163  break;
164  }
165  }
166  if ( containsVar ) {
167  TRACE( "hypro.datastructures", "Row " << i << " is used in constraint " << constraintsOld.row( i ) << " for the current set." );
168  indicesToAdd.push_back( i );
169  }
170  }
171 
172  if ( indicesToAdd.size() > 0 ) {
173  // create a row matrix with numIndicesToAdd many rows
174  matrix_t<Number> newMatrix = selectRows( constraintsOld, indicesToAdd );
175  newMatrix = selectCols( newMatrix, set );
176 
177  // create final constant vector
178  vector_t<Number> newVec = selectRows( constantsOld, indicesToAdd );
179 
180  ConstraintSetT<Number> res( newMatrix, newVec );
181  DEBUG( "hypro.datastructures", "Final decomposed ConstraintSet: \n"
182  << res );
183  newCset.push_back( res );
184  } else {
185  DEBUG( "hypro.datastructures", "No constraints for set found." );
186  ConstraintSetT<Number> res = ConstraintSetT<Number>();
187  newCset.push_back( res );
188  }
189  }
190 
191  mConstraints = newCset;
192  mConditionIsBox = std::vector<TRIBOOL>{ mConstraints.size(), TRIBOOL::NSET };
193  mHash = 0;
194  DEBUG( "hypro.datastructures", "Decomposed Condition, created " << mConstraints.size() << " new constraint sets." );
195 }
