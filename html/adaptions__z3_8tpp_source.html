168 std::vector<std::size_t> z3RedundantConstraints( const matrix_t<Number>& constraints, const vector_t<Number>& constants ) {
169  std::vector<std::size_t> res;
170  z3Context c;
171  z3::solver z3Solver( c );
172  z3Solver.push();
173 
174  // TODO: ATTENTION: This relies upon that Z3 maintains the order of the constraints!
175  z3::expr_vector formulas = createFormula( constraints, constants, c );
176 
177  if ( formulas.size() == 1 ) {
178  return res;
179  }
180 
181  assert( unsigned( formulas.size() + res.size() ) == constraints.rows() );
182  for ( unsigned i = 0; i < formulas.size(); ++i ) {
183  z3Solver.add( formulas[i] );
184  }
185 
186  // first call to check satisfiability
187  z3::check_result firstCheck = z3Solver.check();
188 #ifdef DEBUG_MSG
189  //std::cout << __func__ << " Original problem solution: " << firstCheck << std::endl;
190 #endif
191  switch ( firstCheck ) {
192  case z3::check_result::unsat: {
193  return res;
194  break;
195  }
196  case z3::check_result::sat: {
197  break;
198  }
199  default: {
200  assert( false );
201  break;
202  }
203  }
204 
205  z3Solver.pop();
206  for ( unsigned constraintIndex = 0; constraintIndex < formulas.size(); ++constraintIndex ) {
207  z3::expr originalConstraint = formulas[constraintIndex];
208 #ifdef DEBUG_MSG
209  //std::cout << __func__ << " Original constraint: " << originalConstraint << std::endl;
210 #endif
211  z3::expr negatedConstraint = !originalConstraint;
212 
213  z3Solver.push();
214  for ( unsigned i = 0; i < formulas.size(); ++i ) {
215  //If a constraint has already been marked as redundant, we do not need to
216  //to check satisfiability with that constraint in this current iteration.
217  bool ignore = false;
218  for ( unsigned alreadyRedundant = 0; alreadyRedundant < res.size(); ++alreadyRedundant ) {
219  if ( i == res[alreadyRedundant] ) {
220  ignore = true;
221  }
222  }
223  if ( ignore ) break;
224 
225  //Add constraints
226  if ( i == constraintIndex ) {
227  z3Solver.add( negatedConstraint );
228  } else {
229  z3Solver.add( formulas[i] );
230  }
231  }
232 
233  z3::check_result isRedundant = z3Solver.check();
234  assert( isRedundant != z3::check_result::unknown );
235  if ( isRedundant == z3::check_result::unsat ) {
236 #ifdef DEBUG_MSG
237  //std::cout << __func__ << " is redundant." << std::endl;
238 #endif
239  res.push_back( constraintIndex );
240  }
241 
242  z3Solver.pop();
243  }
244  return res;
245 }
