42 void singularJumpHandler<Representation>::applyReset( Representation& stateSet, Transition<Number>* transitionPtr ) const {
43  if ( !transitionPtr->getReset().empty() ) {
44  assert( transitionPtr->getReset().getAffineReset().isIdentity() && "Singular automata do not support linear/affine resets." );
45 
46  IntervalAssignment<Number> intervalReset = transitionPtr->getReset().getIntervalReset();
47  if ( !intervalReset.isIdentity() ) {
48  std::vector<std::size_t> projectOutDimensions;
49  HPolytope<Number> projectedSet = Converter<Number>::toHPolytope( stateSet );
50  std::vector<Halfspace<Number>> newConstraints;
51  for ( std::size_t i = 0; i < intervalReset.size(); ++i ) {
52  if ( !intervalReset.mIntervals[i].isEmpty() ) {
53  // non-empty intervals represent some reset different from identity -> project out dimension, memorize new interval bounds
54  projectOutDimensions.push_back( i );
55  // create and store new interval bounds
56  vector_t<Number> normal = vector_t<Number>::Zero( stateSet.dimension() );
57  normal( i ) = Number( 1 );
58  newConstraints.emplace_back( normal, intervalReset.mIntervals[i].upper() );
59  normal = -normal;
60  newConstraints.emplace_back( normal, intervalReset.mIntervals[i].lower() );
61  }
62  }
63  // add interval bounds as new constraints
64  projectedSet = projectedSet.projectOutConservative( projectOutDimensions );
65  std::cout << "projectedSet dimension: " << projectedSet.dimension() << ", newConstraints: " << newConstraints << std::endl;
66  projectedSet.insert( newConstraints.begin(), newConstraints.end() );
67 
68  // convert back and assign to original representation type
69  convert( projectedSet, stateSet );
70  }
71  }
72 }
