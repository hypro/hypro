174  Point<Number> result = Point<Number>( _point );
175  std::vector<carl::Variable> variables = _point.variables();

193  }
194  }

204 template <typename Number, class DebugSetting>
205 bool adjOracle( Point<Number>& result, Point<Number>& _vertex, std::pair<int, int>& _counter ) {
206  // retrieve the edge that is defined by the counter (j,i)
207  // first get both source & target vertex (dependent on the counter param.)
208  if ( DebugSetting::fukuda_DEBUG ) {
209  std::cout << "-------------------------" << std::endl;
210  std::cout << "AdjOracle for vertex: " << _vertex << std::endl;
211  std::cout << "-------------------------" << std::endl;
212  }
213 
214  std::vector<Point<Number>> vertexComposition = _vertex.composedOf();
215  Point<Number> sourceVertex;
216  Point<Number> targetVertex;
217 
218  if ( _counter.first == 1 || _counter.first == 2 ) {
219  sourceVertex = vertexComposition[_counter.first - 1];
220  }
221  std::vector<Point<Number>> neighbors = sourceVertex.neighbors();
222  if ( neighbors.size() < _counter.second ) {
223  // this neighbor does not exist for this vertex
224  if ( DebugSetting::fukuda_DEBUG ) {
225  std::cout << "-------------------------" << std::endl;
226  std::cout << "AdjOracle result: no neighbor in this direction" << std::endl;
227  std::cout << "-------------------------" << std::endl;
228  }
229  return false;
230  } else {
231  targetVertex = neighbors[_counter.second - 1];
232  }
233 
234  // the edge is represented by a vector
235  vector_t<Number> edge = computeEdge( sourceVertex, targetVertex );
236 
237  // check if there is a parallel edge in the other polytope that points in the same direction (not the origin of the
238  // sourceVertex)
239  // add all non-parallel edges to a set (needed for constraints in LP)
240  vector_t<Number> parallelEdge;
241  // indicates whether a parallel edge has been identified
242  bool parallelFlag = false;
243  std::vector<vector_t<Number>> nonParallelEdges;
244  Point<Number> otherSource;
245 
246  if ( _counter.first == 1 ) {
247  otherSource = vertexComposition[1];
248  } else {
249  otherSource = vertexComposition[0];
250  }
251 
252  std::vector<Point<Number>> otherNeighbors = otherSource.neighbors();
253 
254  vector_t<Number> tempEdge;
255  Number dotProduct;
256  Number normFactor;
257  for ( typename std::vector<Point<Number>>::iterator it = otherNeighbors.begin(); it != otherNeighbors.end();
258  ++it ) {
259  tempEdge = computeEdge( otherSource, *it );
260 
261  // check if edges are parallel (considering direction too)
262  // for this the following has to hold: (x dot_product y) / (|x|*|y|) = 1
263 
264  dotProduct = tempEdge.dot( edge );
265  normFactor = tempEdge.norm() * edge.norm();
266 
267  // has to be done... to avoid rounding problems
268  dotProduct = std::round( dotProduct.toDouble() * 1000000 );
269  normFactor = std::round( normFactor.toDouble() * 1000000 );
270 
271  if ( DebugSetting::fukuda_DEBUG ) {
272  std::cout << "Dot Product: " << dotProduct << std::endl;
273  std::cout << "Norm Factor: " << normFactor << std::endl;
274  std::cout << "Parallelism Factor: " << dotProduct / normFactor << std::endl;
275  std::cout << "Value of the if condition: " << ( dotProduct / normFactor == 1 ) << std::endl;
276  }
277 
278  if ( ( dotProduct / normFactor == 1 + EPSILON ) || ( dotProduct / normFactor == 1 - EPSILON ) ||
279  ( dotProduct / normFactor == -1 + EPSILON ) || ( dotProduct / normFactor == -1 - EPSILON ) ||
280  ( dotProduct / normFactor == -1 ) || ( dotProduct / normFactor == 1 ) ) {
281  if ( DebugSetting::fukuda_DEBUG ) {
282  std::cout << "Parallel Edge detected" << std::endl;
283  }
284  parallelEdge = tempEdge;
285  parallelFlag = true;
286  } else {
287  nonParallelEdges.push_back( tempEdge );
288  }
289  }
290 
291  // add all edges incident of the original source vertex to the Set of non-parallel edges as well
292  for ( typename std::vector<Point<Number>>::iterator it = neighbors.begin(); it != neighbors.end(); ++it ) {
293  // dont add the original edge to the set
294  if ( *it != neighbors[_counter.second - 1] ) {
295  vector_t<Number> tempEdge2 = computeEdge( sourceVertex, *it );
296  nonParallelEdges.push_back( tempEdge2 );
297  }
298  }
299 
300  /*
301  * Setup Linear Feasibility Problem (in form of a min. LP) with GLPK
302  */
303  glp_prob* feasibility;
304  feasibility = glp_create_prob();
305  glp_set_obj_dir( feasibility, GLP_MIN );
306 
307  // each row corresponds to one non-parallel edge (except for the first row, which refers to the edge we examine)
308  glp_add_rows( feasibility, nonParallelEdges.size() + 1 );
309 
310  // set bound of first row
311  // since < 0 is not possible, we have <= -EPSILON
312  glp_set_row_bnds( feasibility, 1, GLP_UP, 0.0, -( EPSILON ) );
313 
314  // constraints of auxiliary variables
315  for ( int i = 1; i <= nonParallelEdges.size(); ++i ) {
316  // start from second row
317  glp_set_row_bnds( feasibility, i + 1, GLP_LO, 0.0, 0.0 );
318  }
319 
320  // each column corresponds to one dimension of the vector
321  glp_add_cols( feasibility, edge.rows() );
322 
323  // coefficients of objective function
324  for ( int i = 1; i <= edge.rows(); ++i ) {
325  glp_set_obj_coef( feasibility, i, 1.0 );
326  }
327 
328  // constrains for structural variables: x>=0, y>=0 for Lambda
329  // not specified in paper, but else there is no feasible primal solution
330  for ( unsigned i = 1; i <= edge.rows(); ++i ) {
331  glp_set_col_bnds( feasibility, i, GLP_DB, -1.0, 1.0 );
332  }
333 
334  // setup the matrix coefficients
335  unsigned elements = ( nonParallelEdges.size() + 1 ) * ( edge.rows() );
336  if ( DebugSetting::fukuda_DEBUG ) {
337  std::cout << "source Vertex: " << sourceVertex << std::endl;
338  std::cout << "target Vertex: " << targetVertex << std::endl;
339  std::cout << "other source Vertex: " << otherSource << std::endl;
340  std::cout << "considered Edge: " << edge << std::endl;
341  }
342  int* ia = new int[1 + elements];
343  int* ja = new int[1 + elements];
344  double* ar = new double[1 + elements];
345  unsigned pos = 1;
346 
347  // to prevent bugs
348  ia[0] = 0;
349  ja[0] = 0;
350  ar[0] = 0;
351 
352  // setup the first row coefficients individually
353  for ( unsigned j = 1; j <= edge.rows(); ++j ) {
354  ia[pos] = 1;
355  ja[pos] = j;
356  ar[pos] = edge( j - 1 ).toDouble();
357  if ( DebugSetting::fukuda_DEBUG ) {
358  std::cout << "Coeff. at (1," << j << "): " << ar[pos] << std::endl;
359  }
360  ++pos;
361  }
362 
363  // then the rest (for the set of non-parallel edges)
364  for ( unsigned i = 2; i <= nonParallelEdges.size() + 1; ++i ) {
365  for ( unsigned j = 1; j <= edge.rows(); ++j ) {
366  ia[pos] = i;
367  ja[pos] = j;
368  vector_t<Number> tmpVec = nonParallelEdges.at( i - 2 );
369  ar[pos] = tmpVec( j - 1 ).toDouble();
370  if ( DebugSetting::fukuda_DEBUG ) {
371  std::cout << "Coeff. at (" << i << "," << j << "): " << ar[pos] << std::endl;
372  }
373  ++pos;
374  }
375  }
376  assert( pos - 1 <= elements );
377 
378  glp_load_matrix( feasibility, elements, ia, ja, ar );
379  glp_simplex( feasibility, NULL );
380 
381  if ( DebugSetting::fukuda_DEBUG ) {
382  std::cout << "Parallel Flag: " << parallelFlag << std::endl;
383  }
384 
385  // check if a feasible solution exists
386  if ( glp_get_status( feasibility ) == GLP_NOFEAS ) {
387  if ( DebugSetting::fukuda_DEBUG ) {
388  std::cout << "-------------------------" << std::endl;
389  std::cout << "AdjOracle result: no feasible solution" << std::endl;
390  std::cout << "-------------------------" << std::endl;
391  }
392  return false;
393  } else {
394  // since there is a feasible solution, our edge determines an edge direction of P=P1+P2
395  // compute the new adjacent vertex in P using this information
396  if ( !parallelFlag ) {
397  // if there was no parallel edge: v_new = a1(v1,i1) + v2
398  // a1(v1,i1) := i1-th vertex adjacent to v1 (ref. Fukuda)
399  result = targetVertex.extAdd( otherSource );
400 
401  // set the composition of the new Vertex accordingly
402  result.addToComposition( targetVertex );
403  result.addToComposition( otherSource );
404  } else {
405  // if there was a parallel edge: v_new = a1(v1,i1) + a2(v2,i2)
406  Point<Number> otherTargetVertex = computePoint( otherSource, parallelEdge, true );
407  if ( DebugSetting::fukuda_DEBUG ) {
408  std::cout << "parallel Edge: " << parallelEdge << std::endl;
409  std::cout << "Other Target Vertex: " << otherTargetVertex << std::endl;
410  }
411  result = targetVertex.extAdd( otherTargetVertex );
412 
413  result.addToComposition( targetVertex );
414  result.addToComposition( otherTargetVertex );
415  }
416  }
417 
418  glp_delete_prob( feasibility );
419 
420  if ( DebugSetting::fukuda_DEBUG ) {
421  std::cout << "-------------------------" << std::endl;
422  std::cout << "AdjOracle result: " << result << std::endl;
423  std::cout << "-------------------------" << std::endl;
424  }
425 
426  return true;
427 }

432 template <typename Number, class DebugSetting>
433 vector_t<Number> computeMaximizerVector( Point<Number>& _targetVertex, Point<Number>& _vertex ) {
434  // to prepare the LP, compute all incident edges of v1 & v2 for v=v1+v2
435  std::vector<Point<Number>> vertexComposition = _vertex.composedOf();
436  Point<Number> sourceVertex1 = vertexComposition[0];
437  Point<Number> sourceVertex2 = vertexComposition[1];
438 
439  if ( DebugSetting::fukuda_DEBUG ) {
440  std::cout << "Decomposition: " << sourceVertex1 << ", " << sourceVertex2 << std::endl;
441  }
442 
443  std::vector<Point<Number>> neighbors1 = sourceVertex1.neighbors();
444  std::vector<Point<Number>> neighbors2 = sourceVertex2.neighbors();
445 
446  std::vector<vector_t<Number>> edges;
447  vector_t<Number> tmpEdge;
448 
449  // traverse neighbors of v1
450  for ( auto neighbor : neighbors1 ) {
451  tmpEdge = computeEdge( sourceVertex1, neighbor );
452  edges.push_back( tmpEdge );
453  }
454 
455  // traverse neighbors of v2
456  for ( auto neighbor : neighbors2 ) {
457  tmpEdge = computeEdge( sourceVertex2, neighbor );
458  edges.push_back( tmpEdge );
459  }
460 
461  // check for degeneracy of the summand polytopes
462  std::vector<bool> degeneracyCheck( tmpEdge.rows(), true );
463  for ( auto& edge : edges ) {
464  for ( unsigned i = 0; i < edge.rows(); ++i ) {
465  if ( edge( i ) != 0 ) {
466  degeneracyCheck.at( i ) = false;
467  }
468  }
469  }
470 
471  if ( DebugSetting::fukuda_DEBUG ) {
472  std::cout << "Bool Vector: " << degeneracyCheck << std::endl;
473  }
474 
475  /*
476  * Setup LP with GLPK
477  */
478  glp_prob* maximizer;
479  maximizer = glp_create_prob();
480  glp_set_obj_dir( maximizer, GLP_MAX );
481 
482  // we have one row for each edge in our set
483  glp_add_rows( maximizer, edges.size() );
484 
485  // constraints of auxiliary variables (bounds for rows)
486  for ( int i = 1; i <= edges.size(); ++i ) {
487  glp_set_row_bnds( maximizer, i, GLP_UP, 0.0, 0.0 );
488  }
489 
490  // each column corresponds to one dimension of the vector + one column for Lambda0
491  glp_add_cols( maximizer, tmpEdge.rows() + 1 );
492 
493  // coefficients of objective function: only Lambda0 has a non-zero coefficient (last column)
494  for ( int i = 1; i <= tmpEdge.rows(); ++i ) {
495  glp_set_obj_coef( maximizer, i, 0.0 );
496  }
497  glp_set_obj_coef( maximizer, tmpEdge.rows() + 1, 1.0 );
498 
499  // constraints for structural variables
500  for ( int i = 1; i <= tmpEdge.rows(); ++i ) {
501  glp_set_col_bnds( maximizer, i, GLP_DB, -1.0, 1.0 );
502  }
503  glp_set_col_bnds( maximizer, tmpEdge.rows() + 1, GLP_UP, 0.0, POS_CONSTANT );
504 
505  // setup matrix coefficients
506  unsigned elements = ( edges.size() ) * ( tmpEdge.rows() + 1 );
507  int* ia = new int[1 + elements];
508  int* ja = new int[1 + elements];
509  double* ar = new double[1 + elements];
510  unsigned pos = 1;
511 
512  // to prevent bugs
513  ia[0] = 0;
514  ja[0] = 0;
515  ar[0] = 0;
516 
517  for ( unsigned i = 1; i <= edges.size(); ++i ) {
518  // the rest depends on the current edge
519  for ( unsigned j = 1; j <= tmpEdge.rows(); ++j ) {
520  ia[pos] = i;
521  ja[pos] = j;
522  vector_t<Number> tmpVec = edges.at( i - 1 );
523  ar[pos] = tmpVec( j - 1 ).toDouble();
524  if ( DebugSetting::fukuda_DEBUG ) {
525  std::cout << "Coeff. at (" << i << "," << j << "): " << ar[pos] << std::endl;
526  }
527  ++pos;
528  }
529 
530  // for each row the last column is handled individually (Lambda0)
531  ia[pos] = i;
532  ja[pos] = tmpEdge.rows() + 1;
533  ar[pos] = 1;
534  if ( DebugSetting::fukuda_DEBUG ) {
535  std::cout << "Coeff. at (" << i << "," << tmpEdge.rows() + 1 << "): " << ar[pos] << std::endl;
536  }
537  ++pos;
538  }
539 
540  assert( pos - 1 <= elements );
541 
542  glp_load_matrix( maximizer, elements, ia, ja, ar );
543  glp_simplex( maximizer, NULL );
544 
545  vector_t<Number> result = vector_t<Number>( tmpEdge.rows(), 1 );
546 
547  // fill the result vector based on the optimal solution returned by the LP
548  for ( unsigned i = 1; i <= tmpEdge.rows(); ++i ) {
549  result( i - 1 ) = glp_get_col_prim( maximizer, i );
550  }
551 
552  // compute the target of this new edge
553  _targetVertex = computePoint( _vertex, result, false );
554 
555  glp_delete_prob( maximizer );
556 
557  if ( DebugSetting::fukuda_DEBUG ) {
558  std::cout << "-------------------------" << std::endl;
559  std::cout << "Computed MaximizerVector: " << result << std::endl;
560  std::cout << "-------------------------" << std::endl;
561  }
562 
563  return result;
564 }

574  */
575  glp_prob* coneVector;
576  coneVector = glp_create_prob();
577  glp_set_obj_dir( coneVector, GLP_MAX );
578

585  }
586

588  glp_add_cols( coneVector, _edgeSet.at( 0 ).rows() );
589 
590  // coefficients of objective function:
591  // maximize the scalar product between the computed Maximizer Vector of the vertex in P and the vector of the cone
592  // that we look for
593  //=> this secures that our returned vector is oriented "the right way"
594  for ( unsigned i = 1; i <= _edgeSet.at( 0 ).rows(); ++i ) {
595  glp_set_obj_coef( coneVector, i, _maximizerVector( i - 1 ).toDouble() );
596  }

600  glp_set_col_bnds( coneVector, i, GLP_DB, -1.0, 1.0 );
601  }

609 
610  // to prevent bugs
611  ia[0] = 0;
612  ja[0] = 0;
613  ar[0] = 0;
614 
615  for ( unsigned i = 1; i <= _edgeSet.size(); ++i ) {
616  for ( unsigned j = 1; j <= _edgeSet.at( 0 ).rows(); ++j ) {
617  ia[pos] = i;
618  ja[pos] = j;
619  vector_t<Number> tmpVec = _edgeSet.at( i - 1 );
620  ar[pos] = tmpVec( j - 1 ).toDouble();
621  if ( DebugSetting::fukuda_DEBUG ) {
622  std::cout << "Coeff. at (" << i << "," << j << "): " << ar[pos] << std::endl;
623  }
624  ++pos;
625  }
626  }
627  assert( pos - 1 <= elements );
628 
629  glp_load_matrix( coneVector, elements, ia, ja, ar );
630  glp_simplex( coneVector, NULL );
631 
632  vector_t<Number> result = vector_t<Number>( _edgeSet.at( 0 ).rows(), 1 );
633

635  for ( unsigned i = 1; i <= _edgeSet.at( 0 ).rows(); ++i ) {
636  result( i - 1 ) = glp_get_col_prim( coneVector, i );
637  }
638 
639  glp_delete_prob( coneVector );
640 
641  return result;
642 }
