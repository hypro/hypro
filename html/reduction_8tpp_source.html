152 template <typename Number, typename Converter, class Setting>
153 vector_t<Number> HPolytopeT<Number, Converter, Setting>::computeNormal( std::vector<vector_t<Number>> vertices, vector_t<Number> check ) const {
154  unsigned dimension = vertices.at( 0 ).size();
155  std::vector<vector_t<Number>> baseVectors;
156  for ( unsigned i = 1; i < vertices.size(); i++ ) {
157  baseVectors.push_back( vertices.at( i ) - vertices.at( 0 ) );
158  }
159 
160  if ( vertices.size() != dimension ) { // check if we have the exact number of vertices for the calculation
161  std::cout << __func__ << " : " << __LINE__ << "\n Error - the number of vertices does not fit for the calculation.\nActual Number: " << vertices.size() << "\nNeeded Number: " << dimension - 1 << std::endl;
162  return check;
163  }
164 
165  vector_t<Number> res = vector_t<Number>::Zero( dimension );
166 
167  for ( unsigned i = 0; i < dimension; i++ ) { // iterate through each coordinate of the normal vector
168  // create the matrix and get determinant for each matrix
169  matrix_t<Number> m( dimension - 1, dimension - 1 );
170  for ( unsigned r = 0; r < dimension; r++ ) {
171  if ( r != i ) {
172  unsigned rCorrect = r;
173  if ( r > i ) rCorrect--;
174  for ( unsigned v = 0; v < baseVectors.size(); v++ ) {
175  m( rCorrect, v ) = baseVectors[v][r];
176  }
177  }
178  }
179  Number determinant = Number( m.determinant() );
180  if ( ( i % 2 ) == 0 ) {
181  res[i] = determinant;
182  } else {
183  res[i] = ( -1 ) * determinant;
184  }
185  }
186 
187  if ( check.dot( res ) <= 0 ) { // check direction ...should not be 0?
188  return ( -1 ) * res;
189  }
190 
191  return res;
192 }

454 /*
455  * Reduction-Function
456  * @Input unsigned strat for the strategy, unsigned a for the facet (Drop, drop_smooth) and first facet for (unite, unite_...), unsigned b for the seconde facet
457  * @return the reduced polytope or if the result would be unbounded the inital polytope
458  */
459 template <typename Number, typename Converter, class Setting>
460 HPolytopeT<Number, Converter, Setting> HPolytopeT<Number, Converter, Setting>::reduce( unsigned a, unsigned b, REDUCTION_STRATEGY strat ) const { // REDUCTION_STRATEGY
461 
462  // init
463  HPolytopeT<Number, Converter, Setting> res = *this;
464 
465  std::vector<vector_t<Number>> evaluations;
466  std::vector<Point<Number>> vertices;
467  std::vector<std::vector<unsigned>> membersOfVertices;
468  std::vector<unsigned> neighborsOf_a, neighborsOf_b;
469 
470  if ( strat > 0 ) {
471  vertices = res.vertices();
472  membersOfVertices = getMembersOfVertices( vertices );
473 
474  neighborsOf_a = getNeighborsOfIndex( a, membersOfVertices );
475  neighborsOf_b = getNeighborsOfIndex( b, membersOfVertices );
476  }
477 
478  // neighbor test for unite
479  if ( strat > 1 && strat < 6 && std::find( neighborsOf_a.begin(), neighborsOf_a.end(), b ) == neighborsOf_a.end() ) {
480  std::cout << "Error - second facet is no neighbor of first facet" << std::endl;
481  return res;
482  }
483 
484  // (try to) reduce
485  switch ( strat ) {
486  case REDUCTION_STRATEGY::DROP: {
487  evaluations.push_back( res.constraints().at( a ).normal() );
488  res.erase( a ); // delete facet
489  break;
490  }
491 
492  case REDUCTION_STRATEGY::DROP_SMOOTH: {
493  std::vector<vector_t<Number>> smoothVectors;
494  std::vector<Number> smoothVectors_offset;
495  std::vector<unsigned> smoothVectors_oldIndices;
496 
497  for ( unsigned neighbor : neighborsOf_a ) {
498  smoothVectors.push_back( res.constraints().at( neighbor ).normal() + res.constraints().at( a ).normal() );
499  smoothVectors_offset.push_back( res.constraints().at( neighbor ).offset() + res.constraints().at( a ).offset() );
500  smoothVectors_oldIndices.push_back( neighbor );
501  }
502 
503  neighborsOf_a.push_back( a );
504  std::sort( neighborsOf_a.begin(), neighborsOf_a.end() );
505  std::reverse( neighborsOf_a.begin(), neighborsOf_a.end() );
506 
507  for ( unsigned i = 0; i < smoothVectors.size(); i++ ) {
508  res.insert( Halfspace<Number>( smoothVectors[i], smoothVectors_offset[i] ) ); // update neighbor facets
509  }
510 
511  for ( unsigned neighbor : neighborsOf_a ) {
512  evaluations.push_back( res.constraints().at( neighbor ).normal() );
513  res.erase( neighbor ); // delete not smoothed facet
514  }
515 
516  break;
517  }
518 
519  case REDUCTION_STRATEGY::UNITE: {
520  vector_t<Number> uniteVector = ( res.constraints().at( a ).normal() + res.constraints().at( b ).normal() );
521  Number uniteVector_offset = res.constraints().at( a ).offset() + res.constraints().at( b ).offset();
522 
523  evaluations.push_back( res.constraints().at( a ).normal() );
