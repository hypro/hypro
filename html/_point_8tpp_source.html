171 template <typename Number>
172 std::vector<Number> Point<Number>::polarCoordinates( const Point<Number> &_origin, bool _radians ) const {
173  Point<Number> base = *this - _origin;
174  vector_t<double> transformedCoordinates = vector_t<double>( base.dimension() );
175  for ( unsigned d = 0; d < base.dimension(); ++d ) {
176  transformedCoordinates( d ) = carl::convert<Number, double>( base.at( d ) );
177  }
178 
179  std::vector<Number> result;
180 
181  // 1st component of the result is the radial part, the following components
182  // are the angles.
183  double radialCoordinate = 0;
184  for ( unsigned i = 0; i < base.dimension(); ++i ) {
185  double square;
186  square = carl::pow( transformedCoordinates( i ), 2 );
187  radialCoordinate += square;
188  }
189  radialCoordinate = carl::sqrt( radialCoordinate );
190  result.insert( result.begin(), carl::convert<double, Number>( radialCoordinate ) );
191 
192  // compute polar angles
193  for ( unsigned dimension = 0; dimension < base.dimension() - 1; ++dimension ) {
194  double angle( 0 );
195  for ( auto dimension2 = dimension; dimension2 < base.dimension(); ++dimension2 ) {
196  angle += carl::pow( transformedCoordinates( dimension2 ), 2 );
197  }
198  angle = carl::sqrt( angle );
199  angle = ( transformedCoordinates( dimension ) / angle );
200  angle = std::acos( angle );
201 
202  if ( !_radians ) {
203  angle /= 2 * PI_DN;
204  angle *= 360;
205  }
206  result.emplace_back( std::move( carl::convert<double, Number>( angle ) ) );
207  }
208  if ( ( base.mCoordinates( base.dimension() - 1 ) ) < Number( 0 ) ) {
209  Number tmp = result.back();
210  result.pop_back();
211  if ( !_radians ) {
212  tmp = Number( 360 ) - tmp;
213  } else {
214  tmp = -tmp;
215  }
216  result.push_back( tmp );
217  }
218  assert( result.size() == this->dimension() );
219  return result;
220 }
