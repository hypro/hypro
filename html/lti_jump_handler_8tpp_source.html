6 auto ltiJumpHandler<State>::applyJump( const TransitionStateMap& states, Transition<Number>* transition, const AnalysisParameters& strategy ) -> TransitionStateMap {
7  // holds a mapping of transitions to states which need to be aggregated
8  TransitionStateMap toAggregate;
9  // holds a mapping of transitions to states which are ready to apply the reset function and the intersection with the invariant
10  TransitionStateMap toProcess;
11  // holds a mapping of transitions to already processed (i.e. aggregated, resetted and reduced) states
12  TransitionStateMap processedStates;
13  for ( const auto& [transitionPtr, statesVec] : states ) {
14  // only handle sets related to the passed transition, in case any is passed.
15  if ( transition == nullptr || transitionPtr == transition ) {
16  // check aggregation settings
17 
18  if ( ( strategy.aggregation == AGG_SETTING::NO_AGG && strategy.clustering == -1 ) ||
19  ( strategy.aggregation == AGG_SETTING::MODEL && transitionPtr->getAggregation() == Aggregation::none ) ) {
20  // just copy the states to the toProcess map.
21 
22  auto& targetVec = toProcess[transitionPtr];
23  targetVec.insert( targetVec.end(), statesVec.begin(), statesVec.end() );
24  } else { // store for aggregation
25  auto& targetVec = toAggregate[transitionPtr];
26  targetVec.insert( targetVec.end(), statesVec.begin(), statesVec.end() );
27  }
28  }
29  }
30 
31  // aggregate all sets marked for aggregation
32  aggregate( toProcess, toAggregate, strategy );
33 
34  DEBUG( "hydra.worker", "Apply jump on " << toProcess.size() << " transitions." );
35 
36  for ( const auto& [transitionPtr, statesVec] : toProcess ) {
37  DEBUG( "hydra.worker", "Apply jump on " << statesVec.size() << " states." );
38  for ( const auto& state : statesVec ) {
39  // copy state - as there is no aggregation, the containing set and timestamp is already valid
40  // TODO: Why copy?
41  assert( !state.getTimestamp().isEmpty() );
42  State newState( state );
43 
44  // apply reset function
45  applyReset( newState, transitionPtr );
46 
47  // set target location in state set
48  newState.setLocation( transitionPtr->getTarget() );
49 
50  // check invariant in new location
51  auto [containment, stateSet] = ltiIntersectInvariant( newState );
52  if ( containment == CONTAINMENT::NO ) {
53  continue;
54  }
55 
56  // reduce if possible (Currently only for support functions)
57  applyReduction( stateSet );
58 
59  DEBUG( "hydra.worker.discrete", "State after reduction: " << stateSet );
60 
61  // Note: Here we misuse the state's timestamp to carry the transition timing to the next stage -> just don't reset
62  // the timestamp in case no aggregation happens.
63 
64  if ( processedStates.find( transitionPtr ) == processedStates.end() ) {
65  processedStates[transitionPtr] = std::vector<State>();
66  }
67  processedStates[transitionPtr].emplace_back( stateSet );
68  }
69  }
70  return processedStates;
71 }

74 void ltiJumpHandler<State>::applyReset( State& state, Transition<Number>* transitionPtr ) const {
75  if ( !transitionPtr->getReset().empty() ) {
76  for ( size_t i = 0; i < state.getNumberSets(); i++ ) {
77  if ( state.getSetType( i ) == representation_name::carl_polytope ) {
78  IntervalAssignment<Number> intervalReset = transitionPtr->getReset().getIntervalReset( i );

80  } else {

83  }

89 void ltiJumpHandler<State>::applyReduction( State& state ) const {
90  for ( size_t i = 0; i < state.getNumberSets(); i++ ) {
91  if ( state.getSetType( i ) == representation_name::support_function ) {
92  state.partiallyReduceRepresentation( i );
93  }
94  if ( state.getSetType( i ) == representation_name::SFN ) {
95  //Cut off the subtrees from the root of the supportfunction by overapproximating the representation with a hpolytope (or possibly a box)
96  //and set it as the leaf of a new tree
97  auto tmpSFN = std::visit( genericConvertAndGetVisitor<SupportFunctionNew<typename State::NumberType>>(), state.getSet( i ) );
98  if ( tmpSFN.getSettings().DETECT_BOX ) {
99  //if(!tmpSFN.empty()){
100  tmpSFN.reduceRepresentation();
101  auto isHPolyBox = isBox( tmpSFN.matrix(), tmpSFN.vector() );
102  if ( std::get<0>( isHPolyBox ) ) {
103  Box<Number> tmpBox( std::get<1>( isHPolyBox ) );
104  tmpSFN = SupportFunctionNew<Number>( tmpBox );
105  } else {
106  HPolytopeT<Number, hypro::Converter<Number>, HPolytopeOptimizerCaching> tmpHPoly( tmpSFN.matrix(), tmpSFN.vector() );
107  tmpSFN = SupportFunctionNew<Number>( tmpHPoly );
108  }
109  state.setSet( std::visit( genericInternalConversionVisitor<typename State::repVariant, SupportFunctionNew<Number>>( tmpSFN ), state.getSet( i ) ), i );
110  //}
111  }
112  }

116 template <typename State>
117 void ltiJumpHandler<State>::aggregate( TransitionStateMap& processedStates, const TransitionStateMap& toAggregate, const AnalysisParameters& strategy ) const {
118  // Aggregation
119  DEBUG( "hydra.worker.discrete", "Number of transitions to aggregate: " << toAggregate.size() << std::endl );
120  for ( const auto& [transition, stateSets] : toAggregate ) {
121  assert( !stateSets.empty() );
122  TRACE( "hydra.worker.discrete", stateSets.size() << " sets to aggregate for transition " << transition->getSource()->getName() << " -> " << transition->getTarget()->getName() );
123  std::vector<State> aggregatedStates;
124  // Aggregate sets by using sequential unite operations (TODO Implement and make use of multi-unite).
125  State aggregatedState( *stateSets.begin() );
126  // counts the number of processed sets - just for debugging.
127  unsigned setCnt = 1;
128  // counts the number of sets in the current cluster.
129  std::size_t clusterCnt = 1;
130  // determines whether there are still sets to be clustered together.
131  bool leftovers = true;
132 
133  TRACE( "hydra.worker.discrete", "Find subspace mapping for loc " << aggregatedState.getLocation()->getName() << " (" << aggregatedState.getLocation() << ")" );
134 
135  // perform union directly on the current set vector to avoid an extreme amount of consistency checks
136  std::vector<typename State::repVariant> currentSets = aggregatedState.getSets();
137  // contains the aggregated timestamp, initialized with the first timestamp
138  carl::Interval<tNumber> aggregatedTimestamp = ( *stateSets.begin() ).getTimestamp();
139  //START_BENCHMARK_OPERATION(AGGREGATE);
140  for ( auto stateIt = ++stateSets.begin(); stateIt != stateSets.end(); ++stateIt ) {
141  TRACE( "hydra.worker.discrete", "Agg. aggState and set " << setCnt );
142  // actual aggregation.
143  for ( std::size_t i = 0; i < aggregatedState.getNumberSets(); i++ ) {
144  //START_BENCHMARK_OPERATION(UNION);
145  currentSets[i] = std::visit( genericUniteVisitor<typename State::repVariant>(), currentSets[i], stateIt->getSet( i ) );
146  //EVALUATE_BENCHMARK_RESULT(UNION);
147  }
148 
149  // timestamp handling
150  assert( !( *stateIt ).getTimestamp().isEmpty() );
151  aggregatedTimestamp = aggregatedTimestamp.convexHull( ( *stateIt ).getTimestamp() );
152 
153  leftovers = true;
154  ++setCnt;
155  ++clusterCnt;
156 #ifdef CLUSTERING_NUMBER_LIMITS_SUCESSORS
157  if ( strategy.clustering > 0 && clusterCnt == std::ceil( double( stateSets.size() ) / double( strategy.clustering ) ) ) {
158  TRACE( "hydra.worker.discrete", "Clustering." );
159  aggregatedStates.emplace_back( aggregatedState );
160  leftovers = false;
161  if ( stateIt + 1 != stateSets.end() ) {
162  aggregatedState = *( ++stateIt );
163  aggregatedTimestamp = aggregatedState.getTimestamp();
164  clusterCnt = 1;
165  } else {
166  break;
167  }
168  }
169 #else
170  if ( strategy.clustering > 0 && clusterCnt == std::size_t( strategy.clustering ) ) {
171  TRACE( "hydra.worker.discrete", "Clustering." );
172  aggregatedStates.emplace_back( aggregatedState );
173  leftovers = false;
174  if ( stateIt + 1 != stateSets.end() ) {
175  aggregatedState = *( ++stateIt );
176  aggregatedTimestamp = aggregatedState.getTimestamp();
177  clusterCnt = 1;
178  } else {
179  break;
180  }
181  }
182 #endif
183  }
184  aggregatedState.setSets( currentSets );
185  //EVALUATE_BENCHMARK_RESULT(AGGREGATE);
186 
187  // set timestamps accordingly
188  aggregatedState.setTimestamp( aggregatedTimestamp );
189 
190  if ( strategy.clustering < 1 || leftovers ) {
191  TRACE( "hydra.worker.discrete", "No clustering." );
192  aggregatedStates.emplace_back( aggregatedState );
193  }
194 
195  // add to final mapping.
196  for ( auto& state : aggregatedStates ) {
197  processedStates[transition].emplace_back( state );
198  }
199  }
200 }
