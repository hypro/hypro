180 template <typename State>
181 auto ltiJumpHandler<State>::applyJump( const TransitionStateMap& states, Transition<Number>* transition, const AnalysisParameters& strategy ) -> TransitionStateMap {
182  // holds a mapping of transitions to states which need to be aggregated
183  TransitionStateMap toAggregate;
184  // holds a mapping of transitions to states which are ready to apply the reset function and the intersection with the invariant
185  TransitionStateMap toProcess;
186  // holds a mapping of transitions to already processed (i.e. aggregated, resetted and reduced) states
187  TransitionStateMap processedStates;
188  for ( const auto& [transitionPtr, statesVec] : states ) {
189  // only handle sets related to the passed transition, in case any is passed.
190  if ( transition == nullptr || transitionPtr == transition ) {
191  // check aggregation settings
192 
193  if ( ( strategy.aggregation == AGG_SETTING::NO_AGG && strategy.clustering == -1 ) ||
194  ( strategy.aggregation == AGG_SETTING::MODEL && transitionPtr->getAggregation() == Aggregation::none ) ) {
195  // just copy the states to the toProcess map.
196 
197  auto& targetVec = toProcess[transitionPtr];
198  targetVec.insert( targetVec.end(), statesVec.begin(), statesVec.end() );
199  } else { // store for aggregation
200  auto& targetVec = toAggregate[transitionPtr];
201  targetVec.insert( targetVec.end(), statesVec.begin(), statesVec.end() );
202  }
203  }
204  }
205 
206  // aggregate all sets marked for aggregation
207  aggregate( toProcess, toAggregate, strategy );
208 
209  DEBUG( "hydra.worker", "Apply jump on " << toProcess.size() << " transitions." );
210 
211  for ( const auto& [transitionPtr, statesVec] : toProcess ) {
212  DEBUG( "hydra.worker", "Apply jump on " << statesVec.size() << " states." );
213  for ( const auto& state : statesVec ) {
214  // copy state - as there is no aggregation, the containing set and timestamp is already valid
215  // TODO: Why copy?
216  assert( !state.getTimestamp().isEmpty() );
217  State newState( state );
218 
219  // apply reset function
220  applyReset( newState, transitionPtr );
221 
222  // set target location in state set
223  newState.setLocation( transitionPtr->getTarget() );
224 
225  // check invariant in new location
226  auto containmentStateSetPair = ltiIntersectInvariant( newState );
227  if ( containmentStateSetPair.first == CONTAINMENT::NO ) {
228  continue;
229  }
230 
231  // reduce if possible (Currently only for support functions)
232  applyReduction( containmentStateSetPair.second );
233 
234  DEBUG( "hydra.worker.discrete", "State after reduction: " << containmentStateSetPair.second );
235 
236  // Note: Here we misuse the state's timestamp to carry the transition timing to the next stage -> just don't reset
237  // the timestamp in case no aggregation happens.
238 
239  if ( processedStates.find( transitionPtr ) == processedStates.end() ) {
240  processedStates[transitionPtr] = std::vector<State>();
241  }
242  processedStates[transitionPtr].emplace_back( containmentStateSetPair.second );
243  }
244  }
245  return processedStates;

292 void ltiJumpHandler<State>::aggregate( TransitionStateMap& processedStates, const TransitionStateMap& toAggregate, const AnalysisParameters& strategy ) const {
293  // Aggregation
294  DEBUG( "hydra.worker.discrete", "Number of transitions to aggregate: " << toAggregate.size() << std::endl );
295  for ( const auto& [transition, stateSets] : toAggregate ) {
296  assert( !stateSets.empty() );
297  TRACE( "hydra.worker.discrete", stateSets.size() << " sets to aggregate for transition " << transitionStatePair.first->getSource()->getName() << " -> " << transition->getTarget()->getName() );
298  std::vector<State> aggregatedStates;
299  // Aggregate sets by using sequential unite operations (TODO Implement and make use of multi-unite).
300  State aggregatedState( *stateSets.begin() );
301  // counts the number of processed sets - just for debugging.
302  unsigned setCnt = 1;
303  // counts the number of sets in the current cluster.
304  std::size_t clusterCnt = 1;
305  // determines whether there are still sets to be clustered together.
306  bool leftovers = true;
307 
308  TRACE( "hydra.worker.discrete", "Find subspace mapping for loc " << aggregatedState.getLocation()->getName() << " (" << aggregatedState.getLocation() << ")" );
309 
310  // perform union directly on the current set vector to avoid an extreme amount of consistency checks
311  std::vector<typename State::repVariant> currentSets = aggregatedState.getSets();
312  // contains the aggregated timestamp, initialized with the first timestamp
313  carl::Interval<tNumber> aggregatedTimestamp = ( *stateSets.begin() ).getTimestamp();
314  //START_BENCHMARK_OPERATION(AGGREGATE);
315  for ( auto stateIt = ++stateSets.begin(); stateIt != stateSets.end(); ++stateIt ) {
316  TRACE( "hydra.worker.discrete", "Agg. aggState and set " << setCnt );
317  // actual aggregation.
318  for ( std::size_t i = 0; i < aggregatedState.getNumberSets(); i++ ) {
319  //START_BENCHMARK_OPERATION(UNION);
320  currentSets[i] = std::visit( genericUniteVisitor<typename State::repVariant>(), currentSets[i], stateIt->getSet( i ) );
321  //EVALUATE_BENCHMARK_RESULT(UNION);
322  }
323 
324  // timestamp handling
325  assert( !( *stateIt ).getTimestamp().isEmpty() );
326  aggregatedTimestamp = aggregatedTimestamp.convexHull( ( *stateIt ).getTimestamp() );
327 
328  leftovers = true;
329  ++setCnt;
330  ++clusterCnt;
331 #ifdef CLUSTERING_NUMBER_LIMITS_SUCESSORS
332  if ( strategy.clustering > 0 && clusterCnt == std::ceil( double( stateSets.size() ) / double( strategy.clustering ) ) ) {
333  TRACE( "hydra.worker.discrete", "Clustering." );
334  aggregatedStates.emplace_back( aggregatedState );
335  leftovers = false;
336  if ( stateIt + 1 != stateSets.end() ) {
337  aggregatedState = *( ++stateIt );
338  aggregatedTimestamp = aggregatedState.getTimestamp();
339  clusterCnt = 1;
340  } else {
341  break;
342  }
343  }
344 #else
345  if ( strategy.clustering > 0 && clusterCnt == std::size_t( strategy.clustering ) ) {
346  TRACE( "hydra.worker.discrete", "Clustering." );
347  aggregatedStates.emplace_back( aggregatedState );
348  leftovers = false;
349  if ( stateIt + 1 != stateSets.end() ) {
350  aggregatedState = *( ++stateIt );
351  aggregatedTimestamp = aggregatedState.getTimestamp();
352  clusterCnt = 1;
353  } else {
354  break;
355  }
356  }
357 #endif
358  }
359  aggregatedState.setSets( currentSets );
360  //EVALUATE_BENCHMARK_RESULT(AGGREGATE);
361 
362  // set timestamps accordingly
363  aggregatedState.setTimestamp( aggregatedTimestamp );
364 
365  if ( strategy.clustering < 1 || leftovers ) {
366  TRACE( "hydra.worker.discrete", "No clustering." );
367  aggregatedStates.emplace_back( aggregatedState );
368  }
369 
370  // add to final mapping.
371  for ( auto& state : aggregatedStates ) {
372  if ( processedStates.find( transition ) == processedStates.end() ) {
373  processedStates[transition] = std::vector<State>();
374  }
375  processedStates[transition].emplace_back( state );
376  }
377  }
