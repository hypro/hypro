180 template <typename State>
181 auto ltiJumpHandler<State>::applyJump( const TransitionStateMap& states, Transition<Number>* transition, const AnalysisParameters& strategy ) -> TransitionStateMap {
182  // holds a mapping of transitions to states which need to be aggregated
183  TransitionStateMap toAggregate;
184  // holds a mapping of transitions to states which are ready to apply the reset function and the intersection with the invariant
185  TransitionStateMap toProcess;
186  // holds a mapping of transitions to already processed (i.e. aggregated, resetted and reduced) states
187  TransitionStateMap processedStates;
188  for ( const auto& [transitionPtr, statesVec] : states ) {
189  // only handle sets related to the passed transition, in case any is passed.
190  if ( transition == nullptr || transitionPtr == transition ) {
191  // check aggregation settings
192 
193  if ( ( strategy.aggregation == AGG_SETTING::NO_AGG && strategy.clustering == -1 ) ||
194  ( strategy.aggregation == AGG_SETTING::MODEL && transitionPtr->getAggregation() == Aggregation::none ) ) {
195  // just copy the states to the toProcess map.
196 
197  auto& targetVec = toProcess[transitionPtr];
198  targetVec.insert( targetVec.end(), statesVec.begin(), statesVec.end() );
199  } else { // store for aggregation
200  auto& targetVec = toAggregate[transitionPtr];
201  targetVec.insert( targetVec.end(), statesVec.begin(), statesVec.end() );
202  }
203  }
204  }
205 
206  // aggregate all sets marked for aggregation
207  aggregate( toProcess, toAggregate, strategy );
208 
209  DEBUG( "hydra.worker", "Apply jump on " << toProcess.size() << " transitions." );
210 
211  for ( const auto& [transitionPtr, statesVec] : toProcess ) {
212  DEBUG( "hydra.worker", "Apply jump on " << statesVec.size() << " states." );
213  for ( const auto& state : statesVec ) {
214  // copy state - as there is no aggregation, the containing set and timestamp is already valid
215  // TODO: Why copy?
216  assert( !state.getTimestamp().isEmpty() );
217  State newState( state );
218 
219  // apply reset function
220  applyReset( newState, transitionPtr );
221 
222  // set target location in state set
223  newState.setLocation( transitionPtr->getTarget() );
224 
225  // check invariant in new location
226  auto containmentStateSetPair = ltiIntersectInvariant( newState );
227  if ( containmentStateSetPair.first == CONTAINMENT::NO ) {
228  continue;
229  }
230 
231  // reduce if possible (Currently only for support functions)
232  applyReduction( containmentStateSetPair.second );
233 
234  DEBUG( "hydra.worker.discrete", "State after reduction: " << containmentStateSetPair.second );
235 
236  // Note: Here we misuse the state's timestamp to carry the transition timing to the next stage -> just don't reset
237  // the timestamp in case no aggregation happens.
238 
239  if ( processedStates.find( transitionPtr ) == processedStates.end() ) {
240  processedStates[transitionPtr] = std::vector<State>();
241  }
242  processedStates[transitionPtr].emplace_back( containmentStateSetPair.second );
243  }
244  }
245  return processedStates;

293 void ltiJumpHandler<State>::aggregate( TransitionStateMap& processedStates, const TransitionStateMap& toAggregate, const AnalysisParameters& strategy ) const {
294  // Aggregation
295  DEBUG( "hydra.worker.discrete", "Number of transitions to aggregate: " << toAggregate.size() << std::endl );
296  for ( const auto& transitionStatePair : toAggregate ) {
297  assert( !transitionStatePair.second.empty() );
298  TRACE( "hydra.worker.discrete", transitionStatePair.second.size() << " sets to aggregate for transition " << transitionStatePair.first->getSource()->getName() << " -> " << transitionStatePair.first->getTarget()->getName() );
299  std::vector<State> aggregatedStates;
300  // Aggregate sets by using sequential unite operations (TODO: Implement and make use of multi-unite).
301  State aggregatedState( *transitionStatePair.second.begin() );
302  // counts the number of processed sets - just for debugging.
303  unsigned setCnt = 1;
304  // counts the number of sets in the current cluster.
305  std::size_t clusterCnt = 1;
306  // determines whether there are still sets to be clustered together.
307  bool leftovers = true;
308 
309  TRACE( "hydra.worker.discrete", "Find subspace mapping for loc " << aggregatedState.getLocation()->getName() << " (" << aggregatedState.getLocation() << ")" );
310 
311  // perform union directly on the current set vector to avoid an extreme amount of consistency checks
312  std::vector<typename State::repVariant> currentSets = aggregatedState.getSets();
313  // contains the aggregated timestamp, initialized with the first timestamp
314  carl::Interval<tNumber> aggregatedTimestamp = ( *transitionStatePair.second.begin() ).getTimestamp();
315  //START_BENCHMARK_OPERATION(AGGREGATE);
316  for ( auto stateIt = ++transitionStatePair.second.begin(); stateIt != transitionStatePair.second.end(); ++stateIt ) {
317  TRACE( "hydra.worker.discrete", "Agg. aggState and set " << setCnt );
318  // actual aggregation.
319  for ( std::size_t i = 0; i < aggregatedState.getNumberSets(); i++ ) {
320  //START_BENCHMARK_OPERATION(UNION);
321  currentSets[i] = std::visit( genericUniteVisitor<typename State::repVariant>(), currentSets[i], stateIt->getSet( i ) );
322  //EVALUATE_BENCHMARK_RESULT(UNION);
323  }
324 
325  // timestamp handling
326  assert( !( *stateIt ).getTimestamp().isEmpty() );
327  aggregatedTimestamp = aggregatedTimestamp.convexHull( ( *stateIt ).getTimestamp() );
328 
329  leftovers = true;
330  ++setCnt;
331  ++clusterCnt;
332 #ifdef CLUSTERING_NUMBER_LIMITS_SUCESSORS
333  if ( strategy.clustering > 0 && clusterCnt == std::ceil( double( transitionStatePair.second.size() ) / double( strategy.clustering ) ) ) {
334  TRACE( "hydra.worker.discrete", "Clustering." );
335  aggregatedStates.emplace_back( aggregatedState );
336  leftovers = false;
337  if ( stateIt + 1 != transitionStatePair.second.end() ) {
338  aggregatedState = *( ++stateIt );
339  aggregatedTimestamp = aggregatedState.getTimestamp();
340  clusterCnt = 1;
341  } else {
342  break;
343  }
344  }
345 #else
346  if ( strategy.clustering > 0 && clusterCnt == std::size_t( strategy.clustering ) ) {
347  TRACE( "hydra.worker.discrete", "Clustering." );
348  aggregatedStates.emplace_back( aggregatedState );
349  leftovers = false;
350  if ( stateIt + 1 != transitionStatePair.second.end() ) {
351  aggregatedState = *( ++stateIt );
352  aggregatedTimestamp = aggregatedState.getTimestamp();
353  clusterCnt = 1;
354  } else {
355  break;
356  }
357  }
358 #endif
359  }
360  aggregatedState.setSets( currentSets );
361  //EVALUATE_BENCHMARK_RESULT(AGGREGATE);
362 
363  // set timestamps accordingly
364  aggregatedState.setTimestamp( aggregatedTimestamp );
365 
366  if ( strategy.clustering < 1 || leftovers ) {
367  TRACE( "hydra.worker.discrete", "No clustering." );
368  aggregatedStates.emplace_back( aggregatedState );
369  }
370 
371  // add to final mapping.
372  for ( auto& state : aggregatedStates ) {
373  if ( processedStates.find( transitionStatePair.first ) == processedStates.end() ) {
374  processedStates[transitionStatePair.first] = std::vector<State>();
375  }
376  processedStates[transitionStatePair.first].emplace_back( state );
377  }
378  }
