40  for ( std::size_t j = i + 1; j < _source.dimension(); j++ ) {

52  constraints.push_back( -evaluation );

57  constraints.push_back( -evaluation );

59  l( i ) = 0;
60  evaluation = _source.evaluate( l );
61  constraints.push_back( evaluation );
62  constraints.push_back( -evaluation );

64  }

233 template <typename Number>
234 template <typename HPolySetting, typename inSetting>
235 HPolytopeT<Number, Converter<Number>, HPolySetting> Converter<Number>::toHPolytope( const DifferenceBoundsT<Number, Converter<Number>, inSetting>& _source, const CONV_MODE ) {
236  assert( _source.getDBM().rows() == _source.getDBM().cols() );
237  int numclocks = _source.getDBM().cols() - 1;
238  int numconstraints = 0; //all entries of the DBM (except the diagonal and inifinities) define a constraint
239  for ( int i = 0; i < _source.getDBM().rows(); i++ ) {
240  for ( int j = 0; j < _source.getDBM().rows(); j++ ) {
241  if ( i != j && !( _source.getDBM()( i, j ).second == DifferenceBoundsT<Number, Converter<Number>, DifferenceBoundsSetting>::BOUND_TYPE::INFTY ) ) {
242  numconstraints++;
243  }
244  }
245  }
246  //constraints of the polytope
247  hypro::matrix_t<Number> HPolyConstraints = hypro::matrix_t<Number>::Zero( numconstraints, numclocks );
248  hypro::vector_t<Number> HPolyConstants = hypro::vector_t<Number>::Zero( numconstraints );
249 
250  int counter = 0; // counter for indexing constraints
251  for ( int i = 0; i < _source.getDBM().rows(); i++ ) {
252  for ( int j = 0; j < _source.getDBM().cols(); j++ ) {
253  // do not consider diagonals
254  if ( i != j && !( _source.getDBM()( i, j ).second == DifferenceBoundsT<Number, Converter<Number>, DifferenceBoundsSetting>::BOUND_TYPE::INFTY ) ) {
255  // the constraint to add
256  matrix_t<Number> constraintVars = matrix_t<Number>::Zero( 1, numclocks );
257  if ( i == 0 ) {
258  // constraint of the form 0-x_j <= c_ij
259  constraintVars( 0, j - 1 ) = -1.0; // j-1 because we don't consider static clock 0
260  } else if ( j == 0 ) {
261  // constraint of the form x_i - 0 <= c_ij
262  constraintVars( 0, i - 1 ) = 1.0; // i-1 because we don't consider static clock 0
263  } else {
264  // constraint of the form x_i - x_j <= c_ij
265  constraintVars( 0, i - 1 ) = 1.0; // i-1 because we don't consider static clock 0
266  constraintVars( 0, j - 1 ) = -1.0; // j-1 because we don't consider static clock 0
267  }
268  HPolyConstraints.row( counter ) = constraintVars;
269  HPolyConstants( counter, 0 ) = _source.getDBM()( i, j ).first;
270  counter++;
271  }
272  }
273  }
274  return HPolytope( HPolyConstraints, HPolyConstants );
275 }
