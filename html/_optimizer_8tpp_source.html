168 template <typename Number>
169 EvaluationResult<Number> Optimizer<Number>::evaluate( const vector_t<Number>& _direction, bool useExactGlpk ) const {
170  assert( isSane() );
171  updateConstraints();
172  assert( _direction.rows() == mConstraintMatrix.cols() );
173 
174  if ( mConstraintMatrix.rows() == 0 ) {
175  return EvaluationResult<Number>( Number( 0 ), vector_t<Number>::Zero( 1 ), SOLUTION::INFTY );
176  }
177 
178  // there is no non-zero cost function -> check for emptiness.
179  if ( _direction.rows() == 0 || _direction.nonZeros() == 0 ) {
180  if ( this->checkConsistency() ) {
181  return EvaluationResult<Number>( Number( 0 ), vector_t<Number>::Zero( 1 ), SOLUTION::FEAS ); // origin, feasible
182  }
183  return EvaluationResult<Number>(); // defaults to infeasible.
184  }
185 
186  // return value
187  EvaluationResult<Number> res;
188 
189  // call to first solver
190 #if HYPRO_PRIMARY_SOLVER == SOLVER_GLPK
191  res = glpkOptimizeLinear( mGlpkContexts[std::this_thread::get_id()], _direction, mConstraintMatrix, mConstraintVector, useExactGlpk );
192 #elif HYPRO_PRIMARY_SOLVER == SOLVER_CLP
193  res = clpOptimizeLinear( mClpContexts[std::this_thread::get_id()], _direction, mConstraintMatrix, mConstraintVector, useExactGlpk );
194 #endif
195 
196 // call to secondary solver
197 #if HYPRO_SECONDARY_SOLVER == SOLVER_SMTRAT
198 
199 #elif HYPRO_SECONDARY_SOLVER == SOLVER_ZTHREE
200 #elif HYPRO_SECONDARY_SOLVER == SOLVER_SOPLEX
201 
202 #else
203  return res;
204 #endif
205 #if defined( HYPRO_USE_SMTRAT ) || defined( HYPRO_USE_Z3 ) || defined( HYPRO_USE_SOPLEX )
206 
207  // At this point we can check, whether the glpk result is already exact and optimal.
208  // We do this by inserting the solution into the constraints. The solution is exact,
209  // whenever it lies at least on one hyperplane (the respective constraint is saturated). Moreover
210  // the solution should always satisfy all constraints and if the direction is linear independent from
211  // all normals, at least d (d is the dimension) constraints are saturated.
212  unsigned saturatedPositiveConstraints = 0;
213  int closestLinearDependentConstraint = -1;
214  vector_t<Number> closestPointOnLinearDependentPlane;
215  bool invalidResult = false;
216  for ( unsigned constraintIndex = 0; constraintIndex < mConstraintMatrix.rows(); ++constraintIndex ) {
217  Number optVal = mConstraintMatrix.row( constraintIndex ).dot( res.optimumValue );
218  if ( optVal > mConstraintVector( constraintIndex ) ) {
219  invalidResult = true;
220  break;
221  }
222  // check for saturation. Only when not linear dependent and the normal points in the same direction.
223  if ( closestLinearDependentConstraint >= 0 &&
224  mConstraintMatrix.row( constraintIndex ).dot( _direction ) > 0 &&
225  optVal == mConstraintVector( constraintIndex ) ) {
226  ++saturatedPositiveConstraints;
227  }
228  if ( linearDependent( vector_t<Number>( mConstraintMatrix.row( constraintIndex ) ), _direction ).first ) {
229  // obtain point on the plane: scale the normal with the factor of the difference of the
230  // length of the normal and the target distance.
231  Number factor = mConstraintVector( constraintIndex ) / ( mConstraintMatrix.row( constraintIndex ).dot( mConstraintMatrix.row( constraintIndex ) ) );
232  vector_t<Number> pointOnPlane = mConstraintMatrix.row( constraintIndex ) * factor;
233  // update closest point if required (compare first nonZero component).
234  unsigned nonZeroPos = 0;
235  while ( _direction( nonZeroPos ) == Number( 0 ) ) {
236  ++nonZeroPos;
237  }
238  // positive pointing direction - take the point with the smallest component in this dimension.
239  if ( _direction( nonZeroPos ) > 0 ) {
240  if ( closestLinearDependentConstraint == -1 || closestPointOnLinearDependentPlane( nonZeroPos ) > pointOnPlane( nonZeroPos ) ) {
241  closestPointOnLinearDependentPlane = pointOnPlane;
242  closestLinearDependentConstraint = constraintIndex;
243  }
244  } else {
245  if ( closestLinearDependentConstraint == -1 || closestPointOnLinearDependentPlane( nonZeroPos ) < pointOnPlane( nonZeroPos ) ) {
246  closestPointOnLinearDependentPlane = pointOnPlane;
247  closestLinearDependentConstraint = constraintIndex;
248  }
249  }
250  }
251  }
252 
253  // re-solve when: Either the result invalidates one constraint, or the direction is linear independent from all
254  // constraint-normals and there are not at least d constraints saturated or if the direction is linear
255  // dependent and the closest linear dependent constraint is not satisfied.
256  if ( invalidResult ||
257  ( closestLinearDependentConstraint == -1 && saturatedPositiveConstraints < mConstraintMatrix.cols() ) ||
258  ( mConstraintMatrix.row( closestLinearDependentConstraint ).dot( res.optimumValue ) != mConstraintVector( closestLinearDependentConstraint ) ) ) {
259  // TODO: just for statistics, delete in productive code!
260  if ( invalidResult ) {
261  COUNT( "invalid result" )
262  };
263 
264  if ( ( closestLinearDependentConstraint == -1 && saturatedPositiveConstraints < mConstraintMatrix.cols() ) ) {
265  COUNT( "saturation failure" )
266  };
267 
268  if ( ( closestLinearDependentConstraint != -1 && mConstraintMatrix.row( closestLinearDependentConstraint ).dot( res.optimumValue ) != mConstraintVector( closestLinearDependentConstraint ) ) ) {
269  COUNT( "linear dependence failure" )
270  };
271 
272 #ifdef HYPRO_USE_Z3
273  COUNT( "z3" );
274  res = z3OptimizeLinear( maximize, _direction, mConstraintMatrix, mConstraintVector, res );
275 #elif defined( HYPRO_USE_SMTRAT ) // elif HYPRO_USE_SMTRAT
276  COUNT( "smtrat" );
277  res = smtratOptimizeLinear( _direction, mConstraintMatrix, mConstraintVector, mRelationSymbols, res );
278 #elif defined( HYPRO_USE_SOPLEX )
279  COUNT( "soplex" );
280  res = soplexOptimizeLinear( _direction, mConstraintMatrix, mConstraintVector, res );
281 #endif
282  }
283 
284  // if there is a valid solution (FEAS), it implies the optimumValue is set.
285  //assert(res.errorCode !=SOLUTION::FEAS || (res.optimumValue.rows() > 1 || (res.optimumValue != vector_t<Number>::Zero(0) && res.supportValue > 0 )));
286  //std::cout << "Point: " << res.optimumValue << " contained: " << checkPoint(Point<Number>(res.optimumValue)) << ", Solution is feasible: " << (res.errorCode==SOLUTION::FEAS) << std::endl;
287  //assert(res.errorCode !=SOLUTION::FEAS || checkPoint(Point<Number>(res.optimumValue)));
288  return res;
289 #endif
290 }

293 bool Optimizer<Number>::checkConsistency() const {
294  //assert((mConsistencyChecked && mLastConsistencyAnswer != SOLUTION::UNKNOWN) || (!mConsistencyChecked && mLastConsistencyAnswer == SOLUTION::UNKNOWN));
295  assert( isSane() );
296  updateConstraints();

301  }

303 #ifdef HYPRO_USE_SMTRAT
304  //TRACE("hypro.optimizer","Use smtrat for consistency check.");
305  mLastConsistencyAnswer = smtratCheckConsistency( mConstraintMatrix, mConstraintVector, mRelationSymbols ) == true ? SOLUTION::FEAS : SOLUTION::INFEAS;
306  mConsistencyChecked = true;
307 #elif defined( HYPRO_USE_Z3 )
308  mLastConsistencyAnswer = z3CheckConsistency( mConstraintMatrix, mConstraintVector ) == true ? SOLUTION::FEAS : SOLUTION::INFEAS;
309  mConsistencyChecked = true;
310 #elif defined( HYPRO_USE_SOPLEX )
311  mLastConsistencyAnswer = soplexCheckConsistency( mConstraintMatrix, mConstraintVector ) == true ? SOLUTION::FEAS : SOLUTION::INFEAS;
312  mConsistencyChecked = true;
313 #endif
314 
315  if ( !mConsistencyChecked ) {
316 #if HYPRO_PRIMARY_SOLVER == SOLVER_GLPK
317  glp_simplex( mGlpkContexts[std::this_thread::get_id()].lp, &mGlpkContexts[std::this_thread::get_id()].parm );
318  glp_exact( mGlpkContexts[std::this_thread::get_id()].lp, &mGlpkContexts[std::this_thread::get_id()].parm );
319  mLastConsistencyAnswer = glp_get_status( mGlpkContexts[std::this_thread::get_id()].lp ) == GLP_NOFEAS ? SOLUTION::INFEAS : SOLUTION::FEAS;
320  mConsistencyChecked = true;
321 #elif HYPRO_PRIMARY_SOLVER == SOLVER_CLP
322  mLastConsistencyAnswer = clpCheckConsistency( mClpContexts[std::this_thread::get_id()] );
323  mConsistencyChecked = true;
324 #endif
325  }
326  return ( mLastConsistencyAnswer == SOLUTION::FEAS );
327 }
