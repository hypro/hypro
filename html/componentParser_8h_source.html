105  Transition<Number>* createTransition(const std::pair<unsigned, unsigned>& _transition, const boost::optional<Aggregation>& _aggregation, const boost::optional<double>& _triggerTime, unsigned _dim, unsigned _discreteDim) {
106  Transition<Number>* res = new Transition<Number>(
107  mLocationManager.location(_transition.first),
108  mLocationManager.location(_transition.second));
109 
110  // setting guard
111  if(mContinuousGuards.size() + mDiscreteGuards.size() > 0) {
112  Condition<Number> g;
113  matrix_t<Number> matr = matrix_t<Number>(mContinuousGuards.size(), _dim);
114  vector_t<Number> vec = vector_t<Number>(mContinuousGuards.size());
115  unsigned rowCnt = 0;
116  for(const auto& row : mContinuousGuards) {
117  assert(row.rows() == 1);
118  matr.row(rowCnt) = row.block(0,0,1,row.cols()-1);
119  vec(rowCnt) = -row(0,_dim);
120  ++rowCnt;
121  }
122  g.setMatrix(matr);
123  g.setVector(vec);
124  // handle discrete guards
125  //g.discreteOffset = _dim;
126 
127  if(mDiscreteGuards.size() > 0) {
128  matrix_t<Number> guardMatrix = matrix_t<Number>::Zero(Eigen::Index(mDiscreteGuards.size()),_discreteDim-_dim+1);
129  vector_t<Number> guardVector = vector_t<Number>::Zero(Eigen::Index(mDiscreteGuards.size()));
130  unsigned row = 0;
131  for(const auto& guardPair : mDiscreteGuards) {
132  matrix_t<Number> reducedGuard = matrix_t<Number>(1,_discreteDim-_dim+1);
133  reducedGuard = guardPair.second.block(0,_dim,1,_discreteDim-_dim+1);
134  guardMatrix.row(row) = reducedGuard.block(0,0,1,reducedGuard.cols()-1);
135  guardVector(row) = reducedGuard(0,reducedGuard.cols()-1);
136 
137  ++row;
138  //g.discreteGuard.emplace_back(VariablePool::getInstance().carlVarByIndex(guardPair.first), reducedGuard);
139  }
140  g.setMatrix(guardMatrix,1);
141  g.setVector(guardVector,1);
142  }
143 
144  res->setGuard(g);
145  }
146 
147  // setting reset
148  Reset<Number> r;
149  //r.discreteOffset = _dim;
150  if(mContinuousResets.size() + mDiscreteResets.size() > 0) {
151  matrix_t<Number> matr = matrix_t<Number>::Identity(_dim,_dim);
152  vector_t<Number> vec = vector_t<Number>::Zero(_dim);
153  for(const auto& resetPair : mContinuousResets){
154  matr.row(resetPair.first) = resetPair.second.block(0,0,1,_dim);
155  vec(resetPair.first) = resetPair.second(0,_dim);
156  }
157  r.setMatrix(matr);
158  r.setVector(vec);
159  if(mDiscreteResets.size() > 0) {
160  matrix_t<Number> discreteMat = matrix_t<Number>::Identity(_discreteDim-_dim,_discreteDim-_dim);
161  vector_t<Number> discreteVec = vector_t<Number>::Zero(_discreteDim-_dim);
162  TRACE("hypro.parser","Identity reset: " << discreteMat );
163  for(const auto& resetPair : mDiscreteResets){
164  TRACE("hypro.parser", resetPair.second << " should fit in " << discreteMat );
165  TRACE("hypro.parser","Block start 0," << _dim << ", size: 1," << _discreteDim-_dim );
166  TRACE("hypro.parser","RowNum:" << resetPair.first-_dim );
167  discreteMat.row(resetPair.first-_dim) = resetPair.second.block(0,_dim,1,_discreteDim-_dim);
168  discreteVec(resetPair.first-_dim) = resetPair.second(0,_discreteDim);
169  }
170  r.setMatrix(discreteMat,1);
171  r.setVector(discreteVec,1);
172  }
173  }
174  res->setReset(r);
175 
176  // set aggregation settings
177  if(_aggregation){
178  res->setAggregation(*_aggregation);
179  }
180 
181  if(_triggerTime){
182  res->setTriggerTime(carl::rationalize<Number>(*_triggerTime));
183  }
184 
185  // update source location
186  mLocationManager.location(_transition.first)->addTransition(res);
187  TRACE("hypro.parser", "Parsed Transition: " << *res);
188  cleanup();
189  return res;
190  }

280  assert(_flow.rows() == _flow.cols());
281  if(mDiscreteInvariants.size() + mContinuousInvariants.size() > 0) {
282  Location<Number>* tmp = mLocationManager.create(_flow);

284  TRACE("hypro.parser","flow: " << tmp->getFlow() );
285  Condition<Number> inv;
286  matrix_t<Number> invariantMat = matrix_t<Number>(mContinuousInvariants.size(), _flow.cols()-1);
287  vector_t<Number> constants = vector_t<Number>(mContinuousInvariants.size());
288  unsigned rowCnt = 0;
289  for(const auto& invariantConstaint : mContinuousInvariants) {
290  assert(invariantConstaint.rows() == 1);
291  invariantMat.row(rowCnt) = invariantConstaint.block(0,0,1,_flow.cols()-1);
292  constants(rowCnt) = -invariantConstaint(0,_flow.cols()-1);
293  ++rowCnt;
294  }
295  inv.setMatrix(invariantMat);
296  inv.setVector(constants);
297  TRACE("hypro.parser","set continuous invariant to " << invariantMat );
298  for(const auto& invariantConstaintPair : mDiscreteInvariants) {
299  assert(invariantConstaintPair.second.rows() == 1);
300  // resize
301  TRACE("hypro.parser","new number of cols: " << _discreteDim-_flow.cols()+2 );
302  matrix_t<Number> resizedInvariant = matrix_t<Number>(1, _discreteDim-_flow.cols()+2);
303  resizedInvariant = invariantConstaintPair.second.block(0, _flow.cols()-1, 1, _discreteDim-_flow.cols()+2);
304

306  vector_t<Number> currentVector = inv.getVector(1);

316  currentVector.block(currentVector.rows()-resizedInvariant.rows(),0,resizedInvariant.rows(),1) = -resizedInvariant.block(0,resizedInvariant.cols()-1,resizedInvariant.rows(),1);
317  }
318  inv.setMatrix(currentInvariant,1);

322  }
