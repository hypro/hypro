89  std::vector<EvaluationResult<Number>> aggregate( std::vector<std::vector<EvaluationResult<Number>>>& resultStackBack, const matrix_t<Number>& ) const override {

94  //For all evaluation results in each direction in resultStackBack, iteratively look for the smallest evaluation result

96  EvaluationResult<Number> r = resultStackBack.front().at( i );

110  }
111  }

117  public:
118  /* Problem: Check emptiness of intersection of n convex objects
119  * Solution 1: Template Evaluation
120  * - For each template direction, evaluate every object except one in that direction and the one into the -direction
121  * - If -direction is bigger in absolute value, then intersection not empty
122  * - Needs n*t*LP worst time, since every evaluate needs to solve a LP
123  * - Can stop early if direction directly found -> best time possible time is n*LP
124  * - Scales bad in dimension
125  * Solution 2: Template Evaluation +
126  * - Like Solution 1, but parallel evaluation of 2 directions
127  * - Needs 0.5*n*t*LP time
128  * Solution 3: Center of Gravity (COG)
129  * - For each object, generate a point within the object
130  * - Either on a halfspace (needs an LP, how to choose which halfspace?)
131  * - Or the center of gravity ob the object (needs to compute vertices, then averaging -> expensive as hell)
132  * - Compute the COG of all points and determine the direction as the vector from the COG to the points
133  * - Evaluate all directions -> but how to determine emptiness from that?
134  * - Needs n*LP to generate points and n*LPs for evaluation
135  *
136  * All these solutions were not ideal for dealing with halfspaces. Algorithm now (for only 2 objects):
137  * - If both are not halfspaces: Use template evaluation: 2*#dir*LP
138  * - If one is a halfspace and the other not: Use satisfiesHalfspace: 2*LP
139  * - If none are halfspaces: Check linear dependency of halfspace normals: linear
140  */
141  bool empty( const std::vector<bool>& childrenEmpty ) const override {
142  //Current implementation uses Solution 1: template evaluation.
143 
144  //Quick check: If emptiness already computed, just return computation result
145  if ( mEmpty != TRIBOOL::NSET ) return ( mEmpty == TRIBOOL::TRUE );
146 
147  //Quick check: If not already computed, check if at least one child is empty
148  for ( const auto& child : childrenEmpty ) {
149  if ( child ) {
150  mEmpty = TRIBOOL::TRUE;
151  return true;
152  }
153  }
154 
155  //If no quick check worked: Turn children into SFs
156  std::vector<SupportFunctionNewT<Number, Converter, Setting>> sfChildren;
157  for ( std::size_t i = 0; i < this->getChildren().size(); ++i ) {
158  sfChildren.emplace_back( SupportFunctionNewT<Number, Converter, Setting>( this->getChildren().at( i ) ) );
159  }
160 
161  //Choose one child for which we will always evaluate in the negative direction
162  SupportFunctionNewT<Number, Converter, Setting> chosenOne = sfChildren.back();
163  sfChildren.pop_back();
164  std::vector<vector_t<Number>> directions = computeTemplate<Number>( mDimension, defaultTemplateDirectionCount );
165 
166  for ( const auto& direction : directions ) {
167  //Determine supportValue of direction and reverse direction for chosenOne
168  EvaluationResult<Number> chosenNegEval = chosenOne.evaluate( -direction, false );
169  EvaluationResult<Number> chosenPosEval = chosenOne.evaluate( direction, false );
170 
171  //Check if the supportValues don't overlap -> there is no intersection between at least two -> empty -> return true
172  //Check in both directions since directions and children do not have to be symmetrical
173  for ( const auto& child : sfChildren ) {
174  EvaluationResult<Number> childPosEval = child.evaluate( direction, false );
175  if ( childPosEval.supportValue < -chosenNegEval.supportValue && childPosEval.errorCode != SOLUTION::INFTY && chosenNegEval.errorCode != SOLUTION::INFTY ) {
176  mEmpty = TRIBOOL::TRUE;
177  return true;
178  }
179  EvaluationResult<Number> childNegEval = child.evaluate( -direction, false );
180  if ( -childNegEval.supportValue > chosenPosEval.supportValue && childNegEval.errorCode != SOLUTION::INFTY && chosenPosEval.errorCode != SOLUTION::INFTY ) {
181  mEmpty = TRIBOOL::TRUE;
182  return true;
183  }
184  }
185  }
186  mEmpty = TRIBOOL::FALSE;
187  return false;
188  }

209  //Erases all dimensions from a copy of dimensions that are denoted in dims
210  std::vector<std::size_t> intersectDims( const std::vector<std::vector<std::size_t>>& dims ) const override {
211  // we create the intersection of all results. Therefore we iterate over the first vector and check the other
212  // result vectors, if the respective element is contained. Iterating over the first is sufficient, as elements
213  // not in the first vector will not be in the intersection anyways.
214  std::vector<std::size_t> accumulatedResult;
215  for ( unsigned pos = 0; pos < dims.begin()->size(); ++pos ) {
216  std::size_t element = dims.begin()->at( pos );
217  bool elementInAllVectors = true;
218  for ( unsigned resIndex = 1; resIndex < dims.size(); ++resIndex ) {
219  if ( std::find( dims.at( resIndex ).begin(), dims.at( resIndex ).end(), element ) == dims.at( resIndex ).end() ) {
220  elementInAllVectors = false;
221  break;
222  }
223  }
224  if ( elementInAllVectors ) {
225  // if this is a projection, we need to remove non-projected dimensions -> we could not have been in a projection since
226  // since this part of the function was only executed if a node had more than 2 children
227  accumulatedResult.emplace_back( element );
228  }
229  }
230  return accumulatedResult;
231  }
