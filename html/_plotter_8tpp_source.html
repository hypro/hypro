459 template <typename Number>
460 std::vector<Point<Number>> Plotter<Number>::grahamScan( const std::vector<Point<Number>>& _points ) {
461  std::vector<Point<Number>> res;
462  if ( _points.size() < 3 ) {
463  return _points;
464  }
465 
466  // initialize -> find minimum Point
467  Point<Number> min = _points[0];
468  std::map<Number, Point<Number>> sortedPoints;
469  for ( const auto& point : _points ) {
470  assert( point.dimension() == 2 );
471  if ( point < min ) {
472  min = point;
473  }
474  }
475 
476  TRACE( "hypro", "points: " << _points );
477 
478  // std::cout << "Minimum: " << min.rawCoordinates().transpose() <<
479  // std::endl;
480 
481  // sort Points according to polar angle -> we have to insert manually
482  // (because of double imprecision)
483  for ( const auto& point : _points ) {
484  if ( point != min ) {
485  // std::cout << "Try to insert " << point << std::endl;
486  Number angle = point.polarCoordinates( min ).at( 1 );
487  // std::cout << "Computed polar angle: " << angle << std::endl;
488  if ( sortedPoints.empty() ) {
489  // std::cout << "points empty, simply insert." << std::endl;
490  sortedPoints.insert( std::make_pair( angle, point ) );
491  } else {
492  // std::cout << "Compare." << std::endl;
493  for ( auto pos = sortedPoints.begin(); pos != sortedPoints.end(); ) {
494  // if equal, take the one with bigger radial component
495  // std::cout << "Consider " << pos->second << ", angle: ";
496  Number newAngle = pos->second.polarCoordinates( min ).at( 1 );
497  // std::cout << newAngle << std::endl;
498  if ( angle == newAngle ) {
499  // if equal, compare radial coordinate (distance)
500  // std::cout << "AlmostEqual2sComplement" << std::endl;
501  if ( pos->second.polarCoordinates( min )[0] < point.polarCoordinates( min )[0] ) {
502  // std::cout << "sortedPoints erase " << pos->second <<
503  // std::endl;
504  pos = sortedPoints.erase( pos );
505  sortedPoints.insert( std::make_pair( angle, point ) );
506  }
507  break;
508  }
509  // we assume to be sorted, so check all angles, which are smaller or
510  // equal for equality -
511  // afterwards simply insert
512  else if ( angle > newAngle ) { // not equal and smaller -> continue
513  // search (at end, simply insert)
514  ++pos;
515  if ( pos == sortedPoints.end() ) {
516  sortedPoints.insert( std::make_pair( angle, point ) );
517  break;
518  }
519  } else { // larger and not equal until this point -> simply insert
520  sortedPoints.insert( std::make_pair( angle, point ) );
521  break;
522  }
523  }
524  // std::cout << "End Compare." << std::endl;
525  }
526  }
527  }
528 
529  // for(const auto& pair : sortedPoints)
530  // std::cout << "sorted: " << pair.first << ", " <<
531  // pair.second.rawCoordinates().transpose() << std::endl;
532 
533  // prepare stack -> initialize with 2 points
534  if ( sortedPoints.empty() ) {
535  res.emplace_back( min );
536  return res;
537  }
538  assert( sortedPoints.size() >= 1 );
539  std::stack<Point<Number>> stack;
540  stack.push( min );
541  stack.push( sortedPoints.begin()->second );
542  sortedPoints.erase( sortedPoints.begin() );
543  unsigned i = 0;
544  unsigned n = sortedPoints.size();
545 
546  // main loop -> check the two topmost elements of the stack and one third,
547  // new point
548  while ( i < n ) {
549  Point<Number> p1 = stack.top();
550  stack.pop();
551  Point<Number> p2 = stack.top();
552  stack.pop();
553  // std::cout << __func__ << ": " << p2.rawCoordinates().transpose() << "
554  // -- " <<
555  // p1.rawCoordinates().transpose() << " -- " <<
556  // sortedPoints.begin()->second.rawCoordinates().transpose() <<
557  // std::endl;
558  if ( isLeftTurn( p2, p1, sortedPoints.begin()->second ) ) {
559  // reinsert and add new point
560  stack.push( p2 );
561  stack.push( p1 );
562  stack.push( sortedPoints.begin()->second );
563  sortedPoints.erase( sortedPoints.begin() );
564  ++i;
565  } else {
566  // only reinsert second -> equal to removing the topmost object of the
567  // stack
568  // std::cout << "Drop " << p1.rawCoordinates().transpose() << std::endl;
569  stack.push( p2 );
570  if ( stack.size() < 2 ) {
571  // in this case simply insert, as the stack has to contain at least 2
572  // points
573  stack.push( sortedPoints.begin()->second );
574  sortedPoints.erase( sortedPoints.begin() );
575  ++i;
576  }
577  }
578  }
579 
580  // write result
581  while ( !stack.empty() ) {
582  res.push_back( stack.top() );
583  stack.pop();
584  }
585 
586  return res;
587 }

589 template <typename Number>
590 bool Plotter<Number>::isLeftTurn( const Point<Number>& a, const Point<Number>& b, const Point<Number>& c ) {
591  assert( a.dimension() == 2 );
592  assert( b.dimension() == 2 );
593  assert( c.dimension() == 2 );
594 
595  Number val = ( ( b.rawCoordinates()( 0 ) - a.rawCoordinates()( 0 ) ) *
596  ( c.rawCoordinates()( 1 ) - a.rawCoordinates()( 1 ) ) ) -
597  ( ( c.rawCoordinates()( 0 ) - a.rawCoordinates()( 0 ) ) *
598  ( b.rawCoordinates()( 1 ) - a.rawCoordinates()( 1 ) ) );
599  // Number val = c.polarCoordinates(a,false)[1] -
600  // b.polarCoordinates(a,false)[1];
601 
602  return ( val > 0 );
603 }
