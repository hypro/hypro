433 inline void _trim_names( const char *const *raw_names,
434  const char **trimmed_names,
435  char *storage, std::size_t count ) {
436  std::size_t offset = 0;
437 
438  for ( std::size_t index = 0; index < count; ++index ) {
439  trimmed_names[index] = storage + offset;
440

442  std::strcspn( raw_names[index], _name_enders );
443  storage[offset + trimmed_length] = '\0';
444 
445  std::size_t raw_length = std::strlen( raw_names[index] );
446  offset += raw_length + 1;
447  }
