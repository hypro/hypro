290 OrthogonalPolyhedronT<Number, Converter, Setting, Type> OrthogonalPolyhedronT<Number, Converter, Setting, Type>::hull() const {
291  if ( !mBoxUpToDate ) {
292  updateBoundaryBox();
293  }
294 
295  std::size_t nrofVertices = std::size_t( pow( 2, dimension() ) );
296  VertexContainer<Number> container;
297  Vertex<Number> vertex( vector_t<Number>( this->dimension() ) );
298 
299  for ( std::size_t vertexNr = 0; vertexNr < nrofVertices; vertexNr++ ) {
300  int i = 0;
301  for ( std::size_t p = 0; p < mGrid.dimension(); ++p ) {
302  // look if the bit of the current dimension is set
303  if ( ( vertexNr >> i++ ) & 1 ) {
304  vertex[p] = mBoundaryBox.interval( p ).upper();
305  } else {
306  vertex[p] = mBoundaryBox.interval( p ).lower();
307  }
308  }
309  vertex.setColor( vertexNr == 0 );
310  container.insert( vertex );
311  }
312  OrthogonalPolyhedronT<Number, Converter, Setting, Type> result = OrthogonalPolyhedronT<Number, Converter, Setting, Type>( container );
313 
314  return result;
