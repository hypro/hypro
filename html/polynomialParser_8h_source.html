207  std::pair<unsigned, std::vector<matrix_t<Number>>> createPairFromPoly( const unsigned& _d, const RELATION& _rel, const vector_t<double>& _constraint) {
208  std::vector<matrix_t<Number>> res;
209  // convert to Number and transpose to create a row.
210  matrix_t<Number> newRhs = convert<double,Number>(_constraint.transpose());
211  assert(newRhs.rows() == 1);
212  //std::cout << "create constraint for dimension " << _d << std::endl;
213  assert(_d < newRhs.cols()-1);
214  switch(_rel){
215  case RELATION::EQ: {
216  matrix_t<Number> resLower = matrix_t<Number>::Zero(1, newRhs.cols());
217  matrix_t<Number> resUpper = matrix_t<Number>::Zero(1, newRhs.cols());
218  resLower(0,_d) = -1;
219  resLower = resLower + newRhs;
220  resUpper(0,_d) = 1;
221  resUpper = resUpper - newRhs;
222  res.emplace_back(resLower);
223  res.emplace_back(resUpper);
224  //std::cout << "Created row from =: " << resLower << std::endl;
225  //std::cout << "Created row from =: " << resUpper << std::endl;
226  break;
227  }
228  case RELATION::GEQ: {
229  matrix_t<Number> resRow = matrix_t<Number>::Zero(1, newRhs.cols());
230  resRow(0,_d) = -1;
231  resRow = resRow + newRhs;
232  //std::cout << "GEQ: lhs:" << _d << " >= " << newRhs << std::endl;
233  res.emplace_back(resRow);
234  //std::cout << "Created row from >=: " << resRow << std::endl;
235  break;
236  }
237  case RELATION::LEQ: {
238  matrix_t<Number> resRow= matrix_t<Number>::Zero(1, newRhs.cols());
239  resRow(0,_d) = 1;
240  resRow = resRow - newRhs;
241  res.emplace_back(resRow);
242  //std::cout << "Created row from <=: " << resRow << std::endl;
243  break;
244  }
245  default:{
246  assert(false);
247  break;
248  }
249  }
250  return std::make_pair(_d, res);
