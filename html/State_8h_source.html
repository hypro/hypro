454  * @brief Comparison for equality operator.
455  * @details Note that this operator is very expensive in case both states are equal.
456  * @param[in] lhs The left hand side.
457  * @param[in] rhs The right hand side.
458  * @return True, if both states are equal, false otherwise.
459  */
460  friend bool operator==(const State<Number,Representation,Rargs...>& lhs, const State<Number,Representation,Rargs...>& rhs) {
461  // quick checks first
462  if (lhs.getNumberSets() != rhs.getNumberSets() || lhs.mTimestamp != rhs.mTimestamp) {
463  return false;
464  }
465  // location-based checks
466  if(lhs.mLoc != nullptr) {
467  if(rhs.mLoc != nullptr) {
468  if (*(lhs.mLoc) != *(rhs.mLoc)) {
469  return false;
470  }
471  } else {
472  return false;
473  }
474  } else {
475  if( rhs.mLoc != nullptr ) {
476  return false;
477  }
478  }
479 
480  // set-based checks
481  for(std::size_t i = 0; i < lhs.getNumberSets(); ++i) {
482  if( lhs.getSetType(i) != rhs.getSetType(i)) {
483  return false;
484  }
485  if(!boost::apply_visitor(genericCompareVisitor(), lhs.getSet(i), rhs.getSet(i))) {
486  return false;
487  }
488  }
489  return true;
490  }
